74  * This is a simple base class, which handles all of the action creation/etc
75  */
76 template <typename T>
77 class ActionBasedControllerHandle : public ActionBasedControllerHandleBase
78 {
79 public:
80  ActionBasedControllerHandle(const rclcpp::Node::SharedPtr& node, const std::string& name, const std::string& ns,
81  const std::string& logger_name)
82  : ActionBasedControllerHandleBase(name, logger_name), node_(node), done_(true), namespace_(ns)
83  {
84  controller_action_client_ = rclcpp_action::create_client<T>(node_, getActionName());
85 
86  unsigned int attempts = 0;
87  double timeout;
88  node_->get_parameter_or("trajectory_execution.controller_connection_timeout", timeout, 15.0);
89 
90  if (timeout == 0.0)
91  {
92  while (rclcpp::ok() && !controller_action_client_->wait_for_action_server(std::chrono::seconds(5)))
93  {
94  RCLCPP_WARN_STREAM(LOGGER, "Waiting for " << getActionName() << " to come up");
95  }
96  }
97  else
98  {
99  while (rclcpp::ok() &&
100  !controller_action_client_->wait_for_action_server(std::chrono::duration<double>(timeout / 3)) &&
101  ++attempts < 3)
102  {
103  RCLCPP_WARN_STREAM(LOGGER, "Waiting for " << getActionName() << " to come up");
104  }
105  }
106  if (!controller_action_client_->action_server_is_ready())
107  {
108  RCLCPP_ERROR_STREAM(LOGGER, "Action client not connected: " << getActionName());
109  controller_action_client_.reset();
110  }
111 
112  last_exec_ = moveit_controller_manager::ExecutionStatus::SUCCEEDED;
113  }
114 
115  bool isConnected() const
116  {
117  return static_cast<bool>(controller_action_client_);
118  }
119 
120  bool cancelExecution() override
121  {
122  if (!controller_action_client_)
123  return false;
124  if (!done_)
125  {
126  RCLCPP_INFO_STREAM(LOGGER, "Cancelling execution for " << name_);
127  auto cancel_result_future = controller_action_client_->async_cancel_goal(current_goal_);
128 
129  const auto& result = cancel_result_future.get();
130  if (!result)
131  RCLCPP_ERROR(LOGGER, "Failed to cancel goal");
132 
133  last_exec_ = moveit_controller_manager::ExecutionStatus::PREEMPTED;
134  done_ = true;
135  }
136  return true;
137  }

140  controllerDoneCallback(const typename rclcpp_action::ClientGoalHandle<T>::WrappedResult& wrapped_result) = 0;
141 
142  bool waitForExecution(const rclcpp::Duration& timeout = rclcpp::Duration(-1)) override
143  {
144  auto result_callback_done = std::make_shared<std::promise<bool>>();
145  auto result_future = controller_action_client_->async_get_result(
146  current_goal_, [this, result_callback_done](const auto& wrapped_result) {
147  controllerDoneCallback(wrapped_result);
148  result_callback_done->set_value(true);
149  });
150  if (timeout < std::chrono::nanoseconds(0))
151  {
152  result_future.wait();
153  }
154  else
155  {
156  std::future_status status = result_future.wait_for(timeout.to_chrono<std::chrono::duration<double>>());
157  if (status == std::future_status::timeout)
158  {
159  RCLCPP_WARN(LOGGER, "waitForExecution timed out");
160  return false;
161  }
162  }
163  // To accommodate for the delay after the future for the result is ready and the time controllerDoneCallback takes to finish
164  result_callback_done->get_future().wait();
165  return true;
166  }

193  void finishControllerExecution(const rclcpp_action::ResultCode& state)
194  {
195  RCLCPP_DEBUG_STREAM(LOGGER, "Controller " << name_ << " is done with state " << static_cast<int>(state));
196  if (state == rclcpp_action::ResultCode::SUCCEEDED)
197  last_exec_ = moveit_controller_manager::ExecutionStatus::SUCCEEDED;
198  else if (state == rclcpp_action::ResultCode::ABORTED)
199  last_exec_ = moveit_controller_manager::ExecutionStatus::ABORTED;
200  else if (state == rclcpp_action::ResultCode::CANCELED)
201  last_exec_ = moveit_controller_manager::ExecutionStatus::PREEMPTED;
202  else if (state == rclcpp_action::ResultCode::UNKNOWN)
203  last_exec_ = moveit_controller_manager::ExecutionStatus::UNKNOWN;
204  else
205  last_exec_ = moveit_controller_manager::ExecutionStatus::FAILED;
206  done_ = true;
207  }
