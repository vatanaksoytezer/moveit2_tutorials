62 void BackgroundProcessing::processingThread()
63 {
64  boost::unique_lock<boost::mutex> ulock(action_lock_);
65 
66  while (run_processing_thread_)
67  {
68  while (actions_.empty() && run_processing_thread_)
69  new_action_condition_.wait(ulock);
70 
71  while (!actions_.empty())
72  {
73  JobCallback fn = actions_.front();
74  std::string action_name = action_names_.front();
75  actions_.pop_front();
76  action_names_.pop_front();
77  processing_ = true;
78 
79  // make sure we are unlocked while we process the event
80  action_lock_.unlock();
81  try
82  {
83  RCLCPP_DEBUG(LOGGER, "Begin executing '%s'", action_name.c_str());
84  fn();
85  RCLCPP_DEBUG(LOGGER, "Done executing '%s'", action_name.c_str());
86  }
87  catch (std::exception& ex)
88  {
89  RCLCPP_ERROR(LOGGER, "Exception caught while processing action '%s': %s", action_name.c_str(), ex.what());
90  }
91  processing_ = false;
92  if (queue_change_event_)
93  queue_change_event_(COMPLETE, action_name);
94  action_lock_.lock();
95  }
96  }
97 }

111 void BackgroundProcessing::clear()
112 {
113  bool update = false;
114  std::deque<std::string> removed;
115  {
116  boost::mutex::scoped_lock _(action_lock_);
117  update = !actions_.empty();
118  actions_.clear();
119  action_names_.swap(removed);
120  }
121  if (update && queue_change_event_)
122  for (const std::string& it : removed)
123  queue_change_event_(REMOVE, it);
124 }
