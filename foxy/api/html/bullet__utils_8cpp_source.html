48 bool acmCheck(const std::string& body_1, const std::string& body_2,
49  const collision_detection::AllowedCollisionMatrix* acm)
50 {
51  collision_detection::AllowedCollision::Type allowed_type;
52 
53  if (acm != nullptr)
54  {
55  if (acm->getEntry(body_1, body_2, allowed_type))
56  {
57  if (allowed_type == collision_detection::AllowedCollision::Type::NEVER)
58  {
59  RCLCPP_DEBUG_STREAM(BULLET_LOGGER,
60  "Not allowed entry in ACM found, collision check between " << body_1 << " and " << body_2);
61  return false;
62  }
63  else
64  {
65  RCLCPP_DEBUG_STREAM(BULLET_LOGGER,
66  "Entry in ACM found, skipping collision check as allowed " << body_1 << " and " << body_2);
67  return true;
68  }
69  }
70  else
71  {
72  RCLCPP_DEBUG_STREAM(BULLET_LOGGER,
73  "No entry in ACM found, collision check between " << body_1 << " and " << body_2);
74  return false;
75  }
76  }
77  else
78  {
79  RCLCPP_DEBUG_STREAM(BULLET_LOGGER, "No ACM, collision check between " << body_1 << " and " << body_2);
80  return false;
81  }

117 btCollisionShape* createShapePrimitive(const shapes::Mesh* geom, const CollisionObjectType& collision_object_type,
118  CollisionObjectWrapper* cow)
119 {
120  assert(collision_object_type == CollisionObjectType::USE_SHAPE_TYPE ||
121  collision_object_type == CollisionObjectType::CONVEX_HULL ||
122  collision_object_type == CollisionObjectType::SDF);
123 
124  if (geom->vertex_count > 0 && geom->triangle_count > 0)
125  {
126  // convert the mesh to the assigned collision object type
127  switch (collision_object_type)
128  {
129  case CollisionObjectType::CONVEX_HULL:
130  {
131  // Create a convex hull shape to approximate Trimesh
132  collision_detection_bullet::AlignedVector<Eigen::Vector3d> input;
133  collision_detection_bullet::AlignedVector<Eigen::Vector3d> vertices;
134  std::vector<int> faces;
135 
136  input.reserve(geom->vertex_count);
137  for (unsigned int i = 0; i < geom->vertex_count; ++i)
138  input.push_back(Eigen::Vector3d(geom->vertices[3 * i], geom->vertices[3 * i + 1], geom->vertices[3 * i + 2]));
139 
140  if (collision_detection_bullet::createConvexHull(vertices, faces, input) < 0)
141  return nullptr;
142 
143  btConvexHullShape* subshape = new btConvexHullShape();
144  for (const Eigen::Vector3d& v : vertices)
145  subshape->addPoint(
146  btVector3(static_cast<btScalar>(v[0]), static_cast<btScalar>(v[1]), static_cast<btScalar>(v[2])));
147 
148  return subshape;
149  }
150  case CollisionObjectType::USE_SHAPE_TYPE:
151  {
152  btCompoundShape* compound =
153  new btCompoundShape(BULLET_COMPOUND_USE_DYNAMIC_AABB, static_cast<int>(geom->triangle_count));
154  compound->setMargin(
155  BULLET_MARGIN); // margin: compound seems to have no effect when positive but has an effect when negative
156 
157  for (unsigned i = 0; i < geom->triangle_count; ++i)
158  {
159  btVector3 v[3];
160  for (unsigned x = 0; x < 3; ++x)
161  {
162  unsigned idx = geom->triangles[3 * i + x];
163  for (unsigned y = 0; y < 3; ++y)
164  {
165  v[x][y] = static_cast<btScalar>(geom->vertices[3 * idx + y]);
166  }
167  }
168 
169  btCollisionShape* subshape = new btTriangleShapeEx(v[0], v[1], v[2]);
170  if (subshape != nullptr)
171  {
172  cow->manage(subshape);
173  subshape->setMargin(BULLET_MARGIN);
174  btTransform geom_trans;
175  geom_trans.setIdentity();
176  compound->addChildShape(geom_trans, subshape);
177  }
178  }
179 
180  return compound;
181  }
182  default:
183  {
184  RCLCPP_ERROR(BULLET_LOGGER, "This bullet shape type (%d) is not supported for geometry meshs",
185  static_cast<int>(collision_object_type));
186  return nullptr;
187  }
188  }
189  }
190  RCLCPP_ERROR(BULLET_LOGGER, "The mesh is empty!");
191  return nullptr;
192 }

194 btCollisionShape* createShapePrimitive(const shapes::OcTree* geom, const CollisionObjectType& collision_object_type,
195  CollisionObjectWrapper* cow)
196 {
197  assert(collision_object_type == CollisionObjectType::USE_SHAPE_TYPE ||
198  collision_object_type == CollisionObjectType::CONVEX_HULL ||
199  collision_object_type == CollisionObjectType::SDF ||
200  collision_object_type == CollisionObjectType::MULTI_SPHERE);
201 
202  btCompoundShape* subshape =
203  new btCompoundShape(BULLET_COMPOUND_USE_DYNAMIC_AABB, static_cast<int>(geom->octree->size()));
204  double occupancy_threshold = geom->octree->getOccupancyThres();
205 
206  // convert the mesh to the assigned collision object type
207  switch (collision_object_type)
208  {
209  case CollisionObjectType::USE_SHAPE_TYPE:
210  {
211  for (auto it = geom->octree->begin(static_cast<unsigned char>(geom->octree->getTreeDepth())),
212  end = geom->octree->end();
213  it != end; ++it)
214  {
215  if (it->getOccupancy() >= occupancy_threshold)
216  {
217  double size = it.getSize();
218  btTransform geom_trans;
219  geom_trans.setIdentity();
220  geom_trans.setOrigin(btVector3(static_cast<btScalar>(it.getX()), static_cast<btScalar>(it.getY()),
221  static_cast<btScalar>(it.getZ())));
222  btScalar l = static_cast<btScalar>(size / 2);
223  btBoxShape* childshape = new btBoxShape(btVector3(l, l, l));
224  childshape->setMargin(BULLET_MARGIN);
225  cow->manage(childshape);
226 
227  subshape->addChildShape(geom_trans, childshape);
228  }
229  }
230  return subshape;
231  }
232  case CollisionObjectType::MULTI_SPHERE:
233  {
234  for (auto it = geom->octree->begin(static_cast<unsigned char>(geom->octree->getTreeDepth())),
235  end = geom->octree->end();
236  it != end; ++it)
237  {
238  if (it->getOccupancy() >= occupancy_threshold)
239  {
240  double size = it.getSize();
241  btTransform geom_trans;
242  geom_trans.setIdentity();
243  geom_trans.setOrigin(btVector3(static_cast<btScalar>(it.getX()), static_cast<btScalar>(it.getY()),
244  static_cast<btScalar>(it.getZ())));
245  btSphereShape* childshape =
246  new btSphereShape(static_cast<btScalar>(std::sqrt(2 * ((size / 2) * (size / 2)))));
247  childshape->setMargin(BULLET_MARGIN);
248  cow->manage(childshape);
249 
250  subshape->addChildShape(geom_trans, childshape);
251  }
252  }
253  return subshape;
254  }
255  default:
256  {
257  RCLCPP_ERROR(BULLET_LOGGER, "This bullet shape type (%d) is not supported for geometry octree",
258  static_cast<int>(collision_object_type));
259  return nullptr;
260  }
261  }

287 CollisionObjectWrapperPtr makeCastCollisionObject(const CollisionObjectWrapperPtr& cow)
288 {
289  CollisionObjectWrapperPtr new_cow = cow->clone();
290 
291  btTransform tf;
292  tf.setIdentity();
293 
294  if (btBroadphaseProxy::isConvex(new_cow->getCollisionShape()->getShapeType()))
295  {
296  assert(dynamic_cast<btConvexShape*>(new_cow->getCollisionShape()) != nullptr);
297  btConvexShape* convex = static_cast<btConvexShape*>(new_cow->getCollisionShape());
298 
299  // This checks if the collision object is already a cast collision object
300  assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE);
301 
302  CastHullShape* shape = new CastHullShape(convex, tf);
303 
304  new_cow->manage(shape);
305  new_cow->setCollisionShape(shape);
306  }
307  else if (btBroadphaseProxy::isCompound(new_cow->getCollisionShape()->getShapeType()))
308  {
309  btCompoundShape* compound = static_cast<btCompoundShape*>(new_cow->getCollisionShape());
310  btCompoundShape* new_compound =
311  new btCompoundShape(BULLET_COMPOUND_USE_DYNAMIC_AABB, compound->getNumChildShapes());
312 
313  for (int i = 0; i < compound->getNumChildShapes(); ++i)
314  {
315  if (btBroadphaseProxy::isConvex(compound->getChildShape(i)->getShapeType()))
316  {
317  btConvexShape* convex = static_cast<btConvexShape*>(compound->getChildShape(i));
318  assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if already a cast collision object
319 
320  btTransform geom_trans = compound->getChildTransform(i);
321 
322  btCollisionShape* subshape = new CastHullShape(convex, tf);
323 
324  new_cow->manage(subshape);
325  subshape->setMargin(BULLET_MARGIN);
326  new_compound->addChildShape(geom_trans, subshape);
327  }
328  else if (btBroadphaseProxy::isCompound(compound->getChildShape(i)->getShapeType()))
329  {
330  btCompoundShape* second_compound = static_cast<btCompoundShape*>(compound->getChildShape(i));
331  btCompoundShape* new_second_compound =
332  new btCompoundShape(BULLET_COMPOUND_USE_DYNAMIC_AABB, second_compound->getNumChildShapes());
333  for (int j = 0; j < second_compound->getNumChildShapes(); ++j)
334  {
335  assert(!btBroadphaseProxy::isCompound(second_compound->getChildShape(j)->getShapeType()));
336 
337  btConvexShape* convex = static_cast<btConvexShape*>(second_compound->getChildShape(j));
338  assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if already a cast collision object
339 
340  btTransform geom_trans = second_compound->getChildTransform(j);
341 
342  btCollisionShape* subshape = new CastHullShape(convex, tf);
343 
344  new_cow->manage(subshape);
345  subshape->setMargin(BULLET_MARGIN);
346  new_second_compound->addChildShape(geom_trans, subshape);
347  }
348 
349  btTransform geom_trans = compound->getChildTransform(i);
350 
351  new_cow->manage(new_second_compound);
352 
353  // margin on compound seems to have no effect when positive but has an effect when negative
354  new_second_compound->setMargin(BULLET_MARGIN);
355  new_compound->addChildShape(geom_trans, new_second_compound);
356  }
357  else
358  {
359  RCLCPP_ERROR_STREAM(BULLET_LOGGER,
360  "I can only collision check convex shapes and compound shapes made of convex shapes");
361  throw std::runtime_error("I can only collision check convex shapes and compound shapes made of convex shapes");
362  }
363  }
364 
365  // margin on compound seems to have no effect when positive but has an effect when negative
366  new_compound->setMargin(BULLET_MARGIN);
367  new_cow->manage(new_compound);
368  new_cow->setCollisionShape(new_compound);
369  new_cow->setWorldTransform(cow->getWorldTransform());
370  }
371  else
372  {
373  RCLCPP_ERROR_STREAM(BULLET_LOGGER,
374  "I can only collision check convex shapes and compound shapes made of convex shapes");
375  throw std::runtime_error("I can only collision check convex shapes and compound shapes made of convex shapes");
376  }
377 
378  return new_cow;
379 }

432 bool BroadphaseFilterCallback::needBroadphaseCollision(btBroadphaseProxy* proxy0, btBroadphaseProxy* proxy1) const
433 {
434  bool cull = !(proxy0->m_collisionFilterMask & proxy1->m_collisionFilterGroup);
435  cull = cull || !(proxy1->m_collisionFilterMask & proxy0->m_collisionFilterGroup);
436 
437  if (cull)
438  return false;
439 
440  const CollisionObjectWrapper* cow0 = static_cast<const CollisionObjectWrapper*>(proxy0->m_clientObject);
441  const CollisionObjectWrapper* cow1 = static_cast<const CollisionObjectWrapper*>(proxy1->m_clientObject);
442 
443  if (!cow0->m_enabled)
444  return false;
445 
446  if (!cow1->m_enabled)
447  return false;
448 
449  if (cow0->getTypeID() == collision_detection::BodyType::ROBOT_ATTACHED &&
450  cow1->getTypeID() == collision_detection::BodyType::ROBOT_LINK)
451  if (cow0->m_touch_links.find(cow1->getName()) != cow0->m_touch_links.end())
452  return false;
453 
454  if (cow1->getTypeID() == collision_detection::BodyType::ROBOT_ATTACHED &&
455  cow0->getTypeID() == collision_detection::BodyType::ROBOT_LINK)
456  if (cow1->m_touch_links.find(cow0->getName()) != cow1->m_touch_links.end())
457  return false;
458 
459  if (cow0->getTypeID() == collision_detection::BodyType::ROBOT_ATTACHED &&
460  cow1->getTypeID() == collision_detection::BodyType::ROBOT_ATTACHED)
461  if (cow0->m_touch_links == cow1->m_touch_links)
462  return false;
463 
464  RCLCPP_DEBUG_STREAM(BULLET_LOGGER, "Broadphase pass " << cow0->getName() << " vs " << cow1->getName());
465  return true;
466 }

489 bool TesseractCollisionPairCallback::processOverlap(btBroadphasePair& pair)
490 {
491  results_callback_.collisions_.pair_done = false;
492 
493  if (results_callback_.collisions_.done)
494  {
495  return false;
496  }
497 
498  const CollisionObjectWrapper* cow0 = static_cast<const CollisionObjectWrapper*>(pair.m_pProxy0->m_clientObject);
499  const CollisionObjectWrapper* cow1 = static_cast<const CollisionObjectWrapper*>(pair.m_pProxy1->m_clientObject);
500 
501  std::pair<std::string, std::string> pair_names{ cow0->getName(), cow1->getName() };
502  if (results_callback_.needsCollision(cow0, cow1))
503  {
504  RCLCPP_DEBUG_STREAM(BULLET_LOGGER, "Processing " << cow0->getName() << " vs " << cow1->getName());
505  btCollisionObjectWrapper obj0_wrap(nullptr, cow0->getCollisionShape(), cow0, cow0->getWorldTransform(), -1, -1);
506  btCollisionObjectWrapper obj1_wrap(nullptr, cow1->getCollisionShape(), cow1, cow1->getWorldTransform(), -1, -1);
507 
508  // dispatcher will keep algorithms persistent in the collision pair
509  if (!pair.m_algorithm)
510  {
511  pair.m_algorithm = dispatcher_->findAlgorithm(&obj0_wrap, &obj1_wrap, nullptr, BT_CLOSEST_POINT_ALGORITHMS);
512  }
513 
514  if (pair.m_algorithm)
515  {
516  TesseractBroadphaseBridgedManifoldResult contact_point_result(&obj0_wrap, &obj1_wrap, results_callback_);
517  contact_point_result.m_closestPointDistanceThreshold = static_cast<btScalar>(results_callback_.contact_distance_);
518 
519  // discrete collision detection query
520  pair.m_algorithm->processCollision(&obj0_wrap, &obj1_wrap, dispatch_info_, &contact_point_result);
521  }
522  }
523  else
524  {
525  RCLCPP_DEBUG_STREAM(BULLET_LOGGER, "Not processing " << cow0->getName() << " vs " << cow1->getName());
526  }
527  return false;
528 }

530 CollisionObjectWrapper::CollisionObjectWrapper(const std::string& name, const collision_detection::BodyType& type_id,
531  const std::vector<shapes::ShapeConstPtr>& shapes,
532  const AlignedVector<Eigen::Isometry3d>& shape_poses,
533  const std::vector<CollisionObjectType>& collision_object_types,
534  bool active)
535  : m_name(name)
536  , m_type_id(type_id)
537  , m_shapes(shapes)
538  , m_shape_poses(shape_poses)
539  , m_collision_object_types(collision_object_types)
540 {
541  if (shapes.empty() || shape_poses.empty() ||
542  (shapes.size() != shape_poses.size() || collision_object_types.empty() ||
543  shapes.size() != collision_object_types.size()))
544  {
545  throw std::exception();
546  }
547 
548  this->setContactProcessingThreshold(BULLET_DEFAULT_CONTACT_DISTANCE);
549  assert(!name.empty());
550 
551  if (active)
552  {
553  m_collisionFilterGroup = btBroadphaseProxy::KinematicFilter;
554  m_collisionFilterMask = btBroadphaseProxy::KinematicFilter | btBroadphaseProxy::StaticFilter;
555  }
556  else
557  {
558  m_collisionFilterGroup = btBroadphaseProxy::StaticFilter;
559  m_collisionFilterMask = btBroadphaseProxy::KinematicFilter;
560  }
561 
562  if (shapes.size() == 1)
563  {
564  btCollisionShape* shape = createShapePrimitive(m_shapes[0], collision_object_types[0], this);
565  shape->setMargin(BULLET_MARGIN);
566  manage(shape);
567  setCollisionShape(shape);
568  setWorldTransform(convertEigenToBt(m_shape_poses[0]));
569  }
570  else
571  {
572  btCompoundShape* compound =
573  new btCompoundShape(BULLET_COMPOUND_USE_DYNAMIC_AABB, static_cast<int>(m_shapes.size()));
574  manage(compound);
