173  bool adaptAndPlan(const PlannerFn& planner, const planning_scene::PlanningSceneConstPtr& ps,
174  const planning_interface::MotionPlanRequest& req, planning_interface::MotionPlanResponse& res,
175  std::vector<std::size_t>& /*added_path_index*/) const override
176  {
177  RCLCPP_DEBUG(LOGGER, "CHOMP: adaptAndPlan ...");
178 
179  // following call to planner() calls the OMPL planner and stores the trajectory inside the MotionPlanResponse res
180  // variable which is then used by CHOMP for optimization of the computed trajectory
181  if (!planner(ps, req, res))
182  return false;
183 
184  // create a hybrid collision detector to set the collision checker as hybrid
185  collision_detection::CollisionDetectorAllocatorPtr hybrid_cd(
186  collision_detection::CollisionDetectorAllocatorHybrid::create());
187 
188  // create a writable planning scene
189  planning_scene::PlanningScenePtr planning_scene = ps->diff();
190  RCLCPP_DEBUG(LOGGER, "Configuring Planning Scene for CHOMP ...");
191  planning_scene->allocateCollisionDetector(hybrid_cd);
192 
193  chomp::ChompPlanner chomp_planner;
194  planning_interface::MotionPlanDetailedResponse res_detailed;
195  res_detailed.trajectory_.push_back(res.trajectory_);
196 
197  bool planning_success = chomp_planner.solve(planning_scene, req, params_, res_detailed);
198 
199  if (planning_success)
200  {
201  res.trajectory_ = res_detailed.trajectory_[0];
202  res.planning_time_ += res_detailed.processing_time_[0];
203  }
204  res.error_code_ = res_detailed.error_code_;
205 
206  return planning_success;
207  }
208 
209 private:
210  chomp::ChompParameters params_;
