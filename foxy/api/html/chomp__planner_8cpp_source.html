65 bool ChompPlanner::solve(const planning_scene::PlanningSceneConstPtr& planning_scene,
66  const planning_interface::MotionPlanRequest& req, const ChompParameters& params,
67  planning_interface::MotionPlanDetailedResponse& res) const
68 {
69  auto start_time = std::chrono::system_clock::now();
70  if (!planning_scene)
71  {
72  RCLCPP_ERROR(LOGGER, "No planning scene initialized.");
73  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
74  return false;
75  }
76 
77  // get the specified start state
78  moveit::core::RobotState start_state = planning_scene->getCurrentState();
79  moveit::core::robotStateMsgToRobotState(planning_scene->getTransforms(), req.start_state, start_state);
80 
81  if (!start_state.satisfiesBounds())
82  {
83  RCLCPP_ERROR(LOGGER, "Start state violates joint limits");
84  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_ROBOT_STATE;
85  return false;
86  }
87 
88  ChompTrajectory trajectory(planning_scene->getRobotModel(), 3.0, .03, req.group_name);
89  robotStateToArray(start_state, req.group_name, trajectory.getTrajectoryPoint(0));
90 
91  if (req.goal_constraints.size() != 1)
92  {
93  RCLCPP_ERROR(LOGGER, "Expecting exactly one goal constraint, got: %zd", req.goal_constraints.size());
94  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GOAL_CONSTRAINTS;
95  return false;
96  }
97 
98  if (req.goal_constraints[0].joint_constraints.empty() || !req.goal_constraints[0].position_constraints.empty() ||
99  !req.goal_constraints[0].orientation_constraints.empty())
100  {
101  RCLCPP_ERROR(LOGGER, "Only joint-space goals are supported");
102  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GOAL_CONSTRAINTS;
103  return false;
104  }
105 
106  const size_t goal_index = trajectory.getNumPoints() - 1;
107  moveit::core::RobotState goal_state(start_state);
108  for (const moveit_msgs::msg::JointConstraint& joint_constraint : req.goal_constraints[0].joint_constraints)
109  goal_state.setVariablePosition(joint_constraint.joint_name, joint_constraint.position);
110  if (!goal_state.satisfiesBounds())
111  {
112  RCLCPP_ERROR(LOGGER, "Goal state violates joint limits");
113  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_ROBOT_STATE;
114  return false;
115  }
116  robotStateToArray(goal_state, req.group_name, trajectory.getTrajectoryPoint(goal_index));
117 
118  const moveit::core::JointModelGroup* model_group =
119  planning_scene->getRobotModel()->getJointModelGroup(req.group_name);
120  // fix the goal to move the shortest angular distance for wrap-around joints:
121  for (size_t i = 0; i < model_group->getActiveJointModels().size(); i++)
122  {
123  const moveit::core::JointModel* model = model_group->getActiveJointModels()[i];
124  const moveit::core::RevoluteJointModel* revolute_joint =
125  dynamic_cast<const moveit::core::RevoluteJointModel*>(model);
126 
127  if (revolute_joint != nullptr)
128  {
129  if (revolute_joint->isContinuous())
130  {
131  double start = (trajectory)(0, i);
132  double end = (trajectory)(goal_index, i);
133  RCLCPP_INFO(LOGGER, "Start is %f end %f short %f", start, end, shortestAngularDistance(start, end));
134  (trajectory)(goal_index, i) = start + shortestAngularDistance(start, end);
135  }
136  }
137  }
138 
139  // fill in an initial trajectory based on user choice from the chomp_config.yaml file
140  if (params.trajectory_initialization_method_.compare("quintic-spline") == 0)
141  trajectory.fillInMinJerk();
142  else if (params.trajectory_initialization_method_.compare("linear") == 0)
143  trajectory.fillInLinearInterpolation();
144  else if (params.trajectory_initialization_method_.compare("cubic") == 0)
145  trajectory.fillInCubicInterpolation();
146  else if (params.trajectory_initialization_method_.compare("fillTrajectory") == 0)
147  {
148  if (!(trajectory.fillInFromTrajectory(*res.trajectory_[0])))
149  {
150  RCLCPP_ERROR(LOGGER, "Input trajectory has less than 2 points, "
151  "trajectory must contain at least start and goal state");
152  return false;
153  }
154  }
155  else
156  {
157  RCLCPP_ERROR(LOGGER, "invalid interpolation method specified in the chomp_planner file");
158  return false;
159  }
160 
161  RCLCPP_INFO(LOGGER, "CHOMP trajectory initialized using method: %s ",
162  (params.trajectory_initialization_method_).c_str());
163 
164  // optimize!
165  auto create_time = std::chrono::system_clock::now();
166 
167  int replan_count = 0;
168  bool replan_flag = false;
169  double org_learning_rate = 0.04, org_ridge_factor = 0.0, org_planning_time_limit = 10;
170  int org_max_iterations = 200;
171 
172  // storing the initial chomp parameters values
173  org_learning_rate = params.learning_rate_;
174  org_ridge_factor = params.ridge_factor_;
175  org_planning_time_limit = params.planning_time_limit_;
176  org_max_iterations = params.max_iterations_;
177 
178  std::unique_ptr<ChompOptimizer> optimizer;
179 
180  // create a non_const_params variable which stores the non constant version of the const params variable
181  ChompParameters params_nonconst = params;
182 
183  // while loop for replanning (recovery behaviour) if collision free optimized solution not found
184  while (true)
185  {
186  if (replan_flag)
187  {
188  // increase learning rate in hope to find a successful path; increase ridge factor to avoid obstacles; add 5
189  // additional secs in hope to find a solution; increase maximum iterations
190  params_nonconst.setRecoveryParams(params_nonconst.learning_rate_ + 0.02, params_nonconst.ridge_factor_ + 0.002,
191  params_nonconst.planning_time_limit_ + 5, params_nonconst.max_iterations_ + 50);
192  }
193 
194  // initialize a ChompOptimizer object to load up the optimizer with default parameters or with updated parameters in
195  // case of a recovery behaviour
196  optimizer.reset(new ChompOptimizer(&trajectory, planning_scene, req.group_name, &params_nonconst, start_state));
197  if (!optimizer->isInitialized())
198  {
199  RCLCPP_ERROR(LOGGER, "Could not initialize optimizer");
200  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED;
201  return false;
202  }
203 
204  RCLCPP_DEBUG(LOGGER, "Optimization took %ld sec to create",
205  (std::chrono::system_clock::now() - create_time).count());
206 
207  bool optimization_result = optimizer->optimize();
208 
209  // replan with updated parameters if no solution is found
210  if (params_nonconst.enable_failure_recovery_)
211  {
212  RCLCPP_INFO(LOGGER,
213  "Planned with Chomp Parameters (learning_rate, ridge_factor, "
214  "planning_time_limit, max_iterations), attempt: # %d ",
215  (replan_count + 1));
216  RCLCPP_INFO(LOGGER, "Learning rate: %f ridge factor: %f planning time limit: %f max_iterations %d ",
217  params_nonconst.learning_rate_, params_nonconst.ridge_factor_, params_nonconst.planning_time_limit_,
218  params_nonconst.max_iterations_);
219 
220  if (!optimization_result && replan_count < params_nonconst.max_recovery_attempts_)
221  {
222  replan_count++;
223  replan_flag = true;
224  }
225  else
226  {
227  break;
228  }
229  }
230  else
231  break;
232  } // end of while loop
233 
234  // resetting the CHOMP Parameters to the original values after a successful plan
235  params_nonconst.setRecoveryParams(org_learning_rate, org_ridge_factor, org_planning_time_limit, org_max_iterations);
236 
237  RCLCPP_DEBUG(LOGGER, "Optimization actually took %ld sec to run",
238  (std::chrono::system_clock::now() - create_time).count());
239  create_time = std::chrono::system_clock::now();
240  // assume that the trajectory is now optimized, fill in the output structure:
241 
242  RCLCPP_DEBUG(LOGGER, "Output trajectory has %zd joints", trajectory.getNumJoints());
243 
244  auto result = std::make_shared<robot_trajectory::RobotTrajectory>(planning_scene->getRobotModel(), req.group_name);
245  // fill in the entire trajectory
246  for (size_t i = 0; i < trajectory.getNumPoints(); i++)
247  {
248  const Eigen::MatrixXd::RowXpr source = trajectory.getTrajectoryPoint(i);
249  auto state = std::make_shared<moveit::core::RobotState>(start_state);
250  size_t joint_index = 0;
251  for (const moveit::core::JointModel* jm : result->getGroup()->getActiveJointModels())
252  {
253  assert(jm->getVariableCount() == 1);
254  state->setVariablePosition(jm->getFirstVariableIndex(), source[joint_index++]);
255  }
256  result->addSuffixWayPoint(state, 0.0);
257  }
258 
259  res.trajectory_.resize(1);
260  res.trajectory_[0] = result;
261 
262  RCLCPP_DEBUG(LOGGER, "Bottom took %ld sec to create", (std::chrono::system_clock::now() - create_time).count());
263  RCLCPP_DEBUG(LOGGER, "Serviced planning request in %ld wall-seconds",
264  (std::chrono::system_clock::now() - start_time).count());
265 
266  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
267  res.processing_time_.resize(1);
268  res.processing_time_[0] = std::chrono::duration<double>(std::chrono::system_clock::now() - start_time).count();
269 
270  // report planning failure if path has collisions
271  if (not optimizer->isCollisionFree())
272  {
273  RCLCPP_ERROR(LOGGER, "Motion plan is invalid.");
274  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_MOTION_PLAN;
275  return false;
276  }
277 
278  // check that final state is within goal tolerances
279  kinematic_constraints::JointConstraint jc(planning_scene->getRobotModel());
280  const moveit::core::RobotState& last_state = result->getLastWayPoint();
281  for (const moveit_msgs::msg::JointConstraint& constraint : req.goal_constraints[0].joint_constraints)
282  {
283  if (!jc.configure(constraint) || !jc.decide(last_state).satisfied)
284  {
285  RCLCPP_ERROR(LOGGER, "Goal constraints are violated: %s", constraint.joint_name.c_str());
286  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::GOAL_CONSTRAINTS_VIOLATED;
287  return false;
288  }
289  }
290 
291  res.processing_time_.resize(1);
292  res.processing_time_[0] = std::chrono::duration<double>(std::chrono::system_clock::now() - start_time).count();
293 
294  return true;
295 }
