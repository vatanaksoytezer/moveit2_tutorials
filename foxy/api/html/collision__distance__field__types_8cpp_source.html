50 std::vector<CollisionSphere> determineCollisionSpheres(const bodies::Body* body, Eigen::Isometry3d& relative_transform)
51 {
52  std::vector<CollisionSphere> css;
53 
54  bodies::BoundingCylinder cyl;
55  body->computeBoundingCylinder(cyl);
56  unsigned int num_points = ceil(cyl.length / (cyl.radius / 2.0));
57  double spacing = cyl.length / ((num_points * 1.0) - 1.0);
58  relative_transform = body->getPose().inverse() * cyl.pose;
59 
60  for (unsigned int i = 1; i < num_points - 1; i++)
61  {
62  CollisionSphere cs(relative_transform * Eigen::Vector3d(0, 0, (-cyl.length / 2.0) + i * spacing), cyl.radius);
63  css.push_back(cs);
64  }
65 
66  return css;
67 }

69 bool PosedDistanceField::getCollisionSphereGradients(const std::vector<CollisionSphere>& sphere_list,
70  const EigenSTL::vector_Vector3d& sphere_centers,
71  GradientInfo& gradient, const CollisionType& type,
72  double tolerance, bool subtract_radii, double maximum_value,
73  bool stop_at_first_collision)
74 {
75  // assumes gradient is properly initialized
76 
77  bool in_collision = false;
78  for (unsigned int i = 0; i < sphere_list.size(); i++)
79  {
80  Eigen::Vector3d p = sphere_centers[i];
81  Eigen::Vector3d grad(0, 0, 0);
82  bool in_bounds;
83  double dist = this->getDistanceGradient(p.x(), p.y(), p.z(), grad.x(), grad.y(), grad.z(), in_bounds);
84  if (!in_bounds && grad.norm() > 0)
85  {
86  // out of bounds
87  return true;
88  }
89 
90  if (dist < maximum_value)
91  {
92  if (subtract_radii)
93  {
94  dist -= sphere_list[i].radius_;
95 
96  if ((dist < 0) && (-dist >= tolerance))
97  {
98  in_collision = true;
99  }
100 
101  dist = std::abs(dist);
102  }
103  else
104  {
105  if (sphere_list[i].radius_ - dist > tolerance)
106  {
107  in_collision = true;
108  }
109  }
110 
111  if (dist < gradient.closest_distance)
112  {
113  gradient.closest_distance = dist;
114  }
115 
116  if (dist < gradient.distances[i])
117  {
118  gradient.types[i] = type;
119  gradient.distances[i] = dist;
120  gradient.gradients[i] = grad;
121  }
122  }
123 
124  if (stop_at_first_collision && in_collision)
125  {
126  return true;
127  }
128  }
129  return in_collision;
130 }

274 void BodyDecomposition::init(const std::vector<shapes::ShapeConstPtr>& shapes, const EigenSTL::vector_Isometry3d& poses,
275  double resolution, double padding)
276 {
277  bodies_.clear();
278  for (unsigned int i = 0; i < shapes.size(); i++)
279  {
280  bodies_.addBody(shapes[i].get(), poses[i], padding);
281  }
282 
283  // collecting collision spheres
284  collision_spheres_.clear();
285  relative_collision_points_.clear();
286  std::vector<CollisionSphere> body_spheres;
287  EigenSTL::vector_Vector3d body_collision_points;
288  for (unsigned int i = 0; i < bodies_.getCount(); i++)
289  {
290  body_spheres.clear();
291  body_collision_points.clear();
292 
293  body_spheres = determineCollisionSpheres(bodies_.getBody(i), relative_cylinder_pose_);
294  collision_spheres_.insert(collision_spheres_.end(), body_spheres.begin(), body_spheres.end());
295 
296  distance_field::findInternalPointsConvex(*bodies_.getBody(i), resolution, body_collision_points);
297  relative_collision_points_.insert(relative_collision_points_.end(), body_collision_points.begin(),
298  body_collision_points.end());
299  }
300 
301  sphere_radii_.resize(collision_spheres_.size());
302  for (unsigned int i = 0; i < collision_spheres_.size(); i++)
303  {
304  sphere_radii_[i] = collision_spheres_[i].radius_;
305  }
306 
307  // computing bounding sphere
308  std::vector<bodies::BoundingSphere> bounding_spheres(bodies_.getCount());
309  for (unsigned int i = 0; i < bodies_.getCount(); i++)
310  {
311  bodies_.getBody(i)->computeBoundingSphere(bounding_spheres[i]);
312  }
313  bodies::mergeBoundingSpheres(bounding_spheres, relative_bounding_sphere_);
314 
315  RCLCPP_DEBUG(LOGGER, "BodyDecomposition generated %i collision spheres out of %i shapes", collision_spheres_.size(),
316  shapes.size());
317 }

390 bool doBoundingSpheresIntersect(const PosedBodySphereDecompositionConstPtr& p1,
391  const PosedBodySphereDecompositionConstPtr& p2)
392 {
393  Eigen::Vector3d p1_sphere_center = p1->getBoundingSphereCenter();
394  Eigen::Vector3d p2_sphere_center = p2->getBoundingSphereCenter();
395  double p1_radius = p1->getBoundingSphereRadius();
396  double p2_radius = p2->getBoundingSphereRadius();
397 
398  double dist = (p1_sphere_center - p2_sphere_center).squaredNorm();
399  return dist < (p1_radius + p2_radius);
400 }

433 void getProximityGradientMarkers(const std::string& frame_id, const std::string& ns, const rclcpp::Duration& dur,
434  const std::vector<PosedBodySphereDecompositionPtr>& posed_decompositions,
435  const std::vector<PosedBodySphereDecompositionVectorPtr>& posed_vector_decompositions,
436  const std::vector<GradientInfo>& gradients, visualization_msgs::msg::MarkerArray& arr)
437 {
438  rclcpp::Clock ros_clock;
439  if (gradients.size() != posed_decompositions.size() + posed_vector_decompositions.size())
440  {
441  RCLCPP_WARN(LOGGER, "Size mismatch between gradients %u and decompositions %u", (unsigned int)gradients.size(),
442  (unsigned int)(posed_decompositions.size() + posed_vector_decompositions.size()));
443  return;
444  }
445  for (unsigned int i = 0; i < gradients.size(); i++)
446  {
447  for (unsigned int j = 0; j < gradients[i].distances.size(); j++)
448  {
449  visualization_msgs::msg::Marker arrow_mark;
450  arrow_mark.header.frame_id = frame_id;
451  arrow_mark.header.stamp = rclcpp::Clock(RCL_ROS_TIME).now();
452  if (ns.empty())
453  {
454  arrow_mark.ns = "self_coll_gradients";
455  }
456  else
457  {
458  arrow_mark.ns = ns;
459  }
460  arrow_mark.id = i * 1000 + j;
461  double xscale = 0.0;
462  double yscale = 0.0;
463  double zscale = 0.0;
464  if (gradients[i].distances[j] > 0.0 && gradients[i].distances[j] != DBL_MAX)
465  {
466  if (gradients[i].gradients[j].norm() > 0.0)
467  {
468  xscale = gradients[i].gradients[j].x() / gradients[i].gradients[j].norm();
469  yscale = gradients[i].gradients[j].y() / gradients[i].gradients[j].norm();
470  zscale = gradients[i].gradients[j].z() / gradients[i].gradients[j].norm();
471  }
472  else
473  {
474  RCLCPP_DEBUG(LOGGER, "Negative length for %u %d %lf", i, arrow_mark.id, gradients[i].gradients[j].norm());
475  }
476  }
477  else
478  {
479  RCLCPP_DEBUG(LOGGER, "Negative dist %lf for %u %d", gradients[i].distances[j], i, arrow_mark.id);
480  }
481  arrow_mark.points.resize(2);
482  if (i < posed_decompositions.size())
483  {
484  arrow_mark.points[1].x = posed_decompositions[i]->getSphereCenters()[j].x();
485  arrow_mark.points[1].y = posed_decompositions[i]->getSphereCenters()[j].y();
486  arrow_mark.points[1].z = posed_decompositions[i]->getSphereCenters()[j].z();
487  }
488  else
489  {
490  arrow_mark.points[1].x =
491  posed_vector_decompositions[i - posed_decompositions.size()]->getSphereCenters()[j].x();
492  arrow_mark.points[1].y =
493  posed_vector_decompositions[i - posed_decompositions.size()]->getSphereCenters()[j].y();
494  arrow_mark.points[1].z =
495  posed_vector_decompositions[i - posed_decompositions.size()]->getSphereCenters()[j].z();
496  }
497  arrow_mark.points[0] = arrow_mark.points[1];
498  arrow_mark.points[0].x -= xscale * gradients[i].distances[j];
499  arrow_mark.points[0].y -= yscale * gradients[i].distances[j];
500  arrow_mark.points[0].z -= zscale * gradients[i].distances[j];
501  arrow_mark.scale.x = 0.01;
502  arrow_mark.scale.y = 0.03;
503  arrow_mark.color.a = 1.0;
504  if (gradients[i].types[j] == SELF)
505  {
506  arrow_mark.color.r = 1.0;
507  arrow_mark.color.g = 0.2;
508  arrow_mark.color.b = .5;
509  }
510  else if (gradients[i].types[j] == INTRA)
511  {
512  arrow_mark.color.r = .2;
513  arrow_mark.color.g = 1.0;
514  arrow_mark.color.b = .5;
515  }
516  else if (gradients[i].types[j] == ENVIRONMENT)
517  {
518  arrow_mark.color.r = .2;
519  arrow_mark.color.g = .5;
520  arrow_mark.color.b = 1.0;
521  }
522  else if (gradients[i].types[j] == NONE)
523  {
524  arrow_mark.color.r = 1.0;
525  arrow_mark.color.g = .2;
526  arrow_mark.color.b = 1.0;
527  }
528  arr.markers.push_back(arrow_mark);
529  }
530  }
