55 QModelIndex CollisionLinearModel::mapFromSource(const QModelIndex& sourceIndex) const
56 {
57  // map (row,column) index to linear index k
58  // http://stackoverflow.com/questions/27086195/linear-index-upper-triangular-matrix
59  int r = sourceIndex.row(), c = sourceIndex.column();
60  int n = this->sourceModel()->columnCount();
61  if (r == c)
62  return QModelIndex(); // main diagonal elements are invalid
63  if (r > c) // only consider upper triagonal matrix
64  std::swap(r, c); // swap r,c if below diagonal
65 
66  int k = (n * (n - 1) / 2) - (n - r) * ((n - r) - 1) / 2 + c - r - 1;
67  return index(k, 2);
68 }

70 QModelIndex CollisionLinearModel::mapToSource(const QModelIndex& proxyIndex) const
71 {
72  // map linear index k to (row, column)
73  // http://stackoverflow.com/questions/27086195/linear-index-upper-triangular-matrix
74  int n = sourceModel()->columnCount();
75  int k = proxyIndex.row(); // linear (row) index
76  int r = n - 2 - (int)(sqrt(-8 * k + 4 * n * (n - 1) - 7) / 2.0 - 0.5);
77  int c = k + r + 1 - n * (n - 1) / 2 + (n - r) * ((n - r) - 1) / 2;
78  return sourceModel()->index(r, c);
79 }

102 QVariant CollisionLinearModel::data(const QModelIndex& index, int role) const
103 {
104  QModelIndex src_index = this->mapToSource(index);
105  switch (index.column())
106  {
107  case 0: // link name 1
108  if (role != Qt::DisplayRole)
109  return QVariant();
110  else
111  return this->sourceModel()->headerData(src_index.row(), Qt::Horizontal, Qt::DisplayRole);
112  case 1: // link name 2
113  if (role != Qt::DisplayRole)

115  return this->sourceModel()->headerData(src_index.column(), Qt::Vertical, Qt::DisplayRole);
116  case 2: // checkbox
117  if (role != Qt::CheckStateRole)
118  return QVariant();

121  case 3: // reason
122  if (role != Qt::DisplayRole)
123  return QVariant();

239 bool SortFilterProxyModel::filterAcceptsRow(int source_row, const QModelIndex& source_parent) const
240 {
241  CollisionLinearModel* m = qobject_cast<CollisionLinearModel*>(sourceModel());
242  if (!(show_all_ || m->reason(source_row) <= moveit_setup_assistant::ALWAYS ||
243  m->data(m->index(source_row, 2), Qt::CheckStateRole) == Qt::Checked))
244  return false; // not accepted due to check state

268 bool SortFilterProxyModel::lessThan(const QModelIndex& src_left, const QModelIndex& src_right) const
269 {
270  int row_left = src_left.row();
271  int row_right = src_right.row();
272  QAbstractItemModel* m = sourceModel();
273 
274  for (int i = 0, end = sort_columns_.size(); i < end && sort_columns_[i] >= 0; ++i)
275  {
276  int sc = sort_columns_[i];
277  int role = sc == 2 ? Qt::CheckStateRole : Qt::DisplayRole;
278  QVariant value_left = m->data(m->index(row_left, sc), role);
279  QVariant value_right = m->data(m->index(row_right, sc), role);
280 
281  if (value_left == value_right)
282  continue;
283 
284  bool smaller = (value_left < value_right);
285  if (sort_orders_[i] == Qt::DescendingOrder)
286  smaller = !smaller;
287  return smaller;
288  }
289  return false;
290 }

292 void SortFilterProxyModel::sort(int column, Qt::SortOrder order)
293 {
294  beginResetModel();
295  if (column < 0)
296  initSorting();
297  else
298  {
299  // remember sorting history
300  int prev_idx = sort_columns_.indexOf(column);
301  if (prev_idx < 0)
302  prev_idx = sort_columns_.size() - 1;
303  // remove old entries
304  sort_columns_.remove(prev_idx);
305  sort_orders_.remove(prev_idx);
306  // add new entries at front
307  sort_columns_.insert(0, column);
308  sort_orders_.insert(0, order);
309  }
310  QSortFilterProxyModel::sort(column, Qt::AscendingOrder);
311  endResetModel();
312 }
