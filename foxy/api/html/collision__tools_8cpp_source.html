86  mk.scale.y = cost_source.aabb_max[1] - cost_source.aabb_min[1];

121  mk.pose.position.z = contact.pos.z();

124  mk.pose.orientation.z = 0.0;
125  mk.pose.orientation.w = 1.0;

157 void intersectCostSources(std::set<CostSource>& cost_sources, const std::set<CostSource>& a,
158  const std::set<CostSource>& b)
159 {
160  cost_sources.clear();
161  CostSource tmp;
162  for (const auto& source_a : a)
163  for (const auto& source_b : b)
164  {
165  tmp.aabb_min[0] = std::max(source_a.aabb_min[0], source_b.aabb_min[0]);
166  tmp.aabb_min[1] = std::max(source_a.aabb_min[1], source_b.aabb_min[1]);
167  tmp.aabb_min[2] = std::max(source_a.aabb_min[2], source_b.aabb_min[2]);
168 
169  tmp.aabb_max[0] = std::min(source_a.aabb_max[0], source_b.aabb_max[0]);
170  tmp.aabb_max[1] = std::min(source_a.aabb_max[1], source_b.aabb_max[1]);
171  tmp.aabb_max[2] = std::min(source_a.aabb_max[2], source_b.aabb_max[2]);
172 
173  if (tmp.aabb_min[0] >= tmp.aabb_max[0] || tmp.aabb_min[1] >= tmp.aabb_max[1] || tmp.aabb_min[2] >= tmp.aabb_max[2])
174  continue;
175  tmp.cost = std::max(source_a.cost, source_b.cost);
176  cost_sources.insert(tmp);
177  }

180 void removeOverlapping(std::set<CostSource>& cost_sources, double overlap_fraction)
181 {
182  double p[3], q[3];
183  for (auto it = cost_sources.begin(); it != cost_sources.end(); ++it)
184  {
185  double vol = it->getVolume() * overlap_fraction;
186  std::vector<std::set<CostSource>::iterator> remove;
187  auto it1 = it;
188  for (auto jt = ++it1; jt != cost_sources.end(); ++jt)
189  {
190  p[0] = std::max(it->aabb_min[0], jt->aabb_min[0]);
191  p[1] = std::max(it->aabb_min[1], jt->aabb_min[1]);
192  p[2] = std::max(it->aabb_min[2], jt->aabb_min[2]);
193 
194  q[0] = std::min(it->aabb_max[0], jt->aabb_max[0]);
195  q[1] = std::min(it->aabb_max[1], jt->aabb_max[1]);
196  q[2] = std::min(it->aabb_max[2], jt->aabb_max[2]);
197 
198  if (p[0] >= q[0] || p[1] >= q[1] || p[2] >= q[2])
199  continue;
200 
201  double intersect_volume = (q[0] - p[0]) * (q[1] - p[1]) * (q[2] - p[2]);
202  if (intersect_volume >= vol)
203  remove.push_back(jt);
204  }
205  for (auto& r : remove)
206  cost_sources.erase(r);
207  }
208 }

210 void removeCostSources(std::set<CostSource>& cost_sources, const std::set<CostSource>& cost_sources_to_remove,
211  double overlap_fraction)
212 {
213  // remove all the boxes that overlap with the intersection previously computed in \e rem
214  double p[3], q[3];
215  for (const auto& source_remove : cost_sources_to_remove)
216  {
217  std::vector<std::set<CostSource>::iterator> remove;
218  std::set<CostSource> add;
219  for (auto it = cost_sources.begin(); it != cost_sources.end(); ++it)
220  {
221  p[0] = std::max(it->aabb_min[0], source_remove.aabb_min[0]);
222  p[1] = std::max(it->aabb_min[1], source_remove.aabb_min[1]);
223  p[2] = std::max(it->aabb_min[2], source_remove.aabb_min[2]);
224 
225  q[0] = std::min(it->aabb_max[0], source_remove.aabb_max[0]);
226  q[1] = std::min(it->aabb_max[1], source_remove.aabb_max[1]);
227  q[2] = std::min(it->aabb_max[2], source_remove.aabb_max[2]);
228 
229  if (p[0] >= q[0] || p[1] >= q[1] || p[2] >= q[2])
230  continue;
231 
232  double intersect_volume = (q[0] - p[0]) * (q[1] - p[1]) * (q[2] - p[2]);
233  if (intersect_volume >= it->getVolume() * overlap_fraction)
234  remove.push_back(it);
235  else
236  {
237  // there is some overlap, but not too large, so we split the cost source into multiple ones
238  for (int i = 0; i < 3; ++i)
239  {
240  // is there a box above axis i in the intersection?
241  if (it->aabb_max[i] > q[i])
242  {
243  CostSource cs = *it;
244  cs.aabb_min[i] = q[i];
245  add.insert(cs);
246  }
247  // is there a box below axis i in the intersection?
248  if (it->aabb_min[i] < p[i])
249  {
250  CostSource cs = *it;
251  cs.aabb_max[i] = p[i];
252  add.insert(cs);
253  }
254  }
255  }
256  }
257  for (auto& r : remove)
258  cost_sources.erase(r);
259  cost_sources.insert(add.begin(), add.end());
260  }
261 }
