55 CommandListManager::CommandListManager(const ros::NodeHandle& nh, const moveit::core::RobotModelConstPtr& model)
56  : nh_(nh), model_(model)
57 {
58  // Obtain the aggregated joint limits
59  pilz_industrial_motion_planner::JointLimitsContainer aggregated_limit_active_joints;
60 
61  aggregated_limit_active_joints = pilz_industrial_motion_planner::JointLimitsAggregator::getAggregatedLimits(
62  ros::NodeHandle(PARAM_NAMESPACE_LIMITS), model_->getActiveJointModels());
63 
64  // Obtain cartesian limits
65  pilz_industrial_motion_planner::CartesianLimit cartesian_limit =
66  pilz_industrial_motion_planner::CartesianLimitsAggregator::getAggregatedLimits(
67  ros::NodeHandle(PARAM_NAMESPACE_LIMITS));
68 
69  pilz_industrial_motion_planner::LimitsContainer limits;
70  limits.setJointLimits(aggregated_limit_active_joints);
71  limits.setCartesianLimits(cartesian_limit);
72 
73  plan_comp_builder_.setModel(model);
74  plan_comp_builder_.setBlender(std::unique_ptr<pilz_industrial_motion_planner::TrajectoryBlender>(
75  new pilz_industrial_motion_planner::TrajectoryBlenderTransitionWindow(limits)));
76 }

78 RobotTrajCont CommandListManager::solve(const planning_scene::PlanningSceneConstPtr& planning_scene,
79  const planning_pipeline::PlanningPipelinePtr& planning_pipeline,
80  const moveit_msgs::MotionSequenceRequest& req_list)
81 {
82  if (req_list.items.empty())
83  {
84  return RobotTrajCont();
85  }
86 
87  checkForNegativeRadii(req_list);
88  checkLastBlendRadiusZero(req_list);
89  checkStartStates(req_list);
90 
91  MotionResponseCont resp_cont{ solveSequenceItems(planning_scene, planning_pipeline, req_list) };
92 
93  assert(model_);
94  RadiiCont radii{ extractBlendRadii(*model_, req_list) };
95  checkForOverlappingRadii(resp_cont, radii);
96 
97  plan_comp_builder_.reset();
98  for (MotionResponseCont::size_type i = 0; i < resp_cont.size(); ++i)
99  {
100  plan_comp_builder_.append(resp_cont.at(i).trajectory_,
101  // The blend radii has to be "attached" to
102  // the second part of a blend trajectory,
103  // therefore: "i-1".
104  (i > 0 ? radii.at(i - 1) : 0.));
105  }
106  return plan_comp_builder_.build();
107 }

109 bool CommandListManager::checkRadiiForOverlap(const robot_trajectory::RobotTrajectory& traj_A, const double radii_A,
110  const robot_trajectory::RobotTrajectory& traj_B,
111  const double radii_B) const
112 {
113  // No blending between trajectories from different groups
114  if (traj_A.getGroupName() != traj_B.getGroupName())
115  {
116  return false;
117  }
118 
119  auto sum_radii{ radii_A + radii_B };
120  if (sum_radii == 0.)
121  {
122  return false;
123  }
124 
125  const std::string& blend_frame{ getSolverTipFrame(model_->getJointModelGroup(traj_A.getGroupName())) };
126  auto distance_endpoints = (traj_A.getLastWayPoint().getFrameTransform(blend_frame).translation() -
127  traj_B.getLastWayPoint().getFrameTransform(blend_frame).translation())
128  .norm();
129  return distance_endpoints <= sum_radii;
130 }

132 void CommandListManager::checkForOverlappingRadii(const MotionResponseCont& resp_cont, const RadiiCont& radii) const
133 {
134  if (resp_cont.empty())
135  {
136  return;
137  }
138  if (resp_cont.size() < 3)
139  {
140  return;
141  }
142 
143  for (MotionResponseCont::size_type i = 0; i < resp_cont.size() - 2; ++i)
144  {
145  if (checkRadiiForOverlap(*(resp_cont.at(i).trajectory_), radii.at(i), *(resp_cont.at(i + 1).trajectory_),
146  radii.at(i + 1)))
147  {
148  std::ostringstream os;
149  os << "Overlapping blend radii between command [" << i << "] and [" << i + 1 << "].";
150  throw OverlappingBlendRadiiException(os.str());
151  }
152  }
153 }

179 bool CommandListManager::isInvalidBlendRadii(const moveit::core::RobotModel& model,
180  const moveit_msgs::MotionSequenceItem& item_A,
181  const moveit_msgs::MotionSequenceItem& item_B)
182 {
183  // Zero blend radius is always valid
184  if (item_A.blend_radius == 0.)
185  {
186  return false;
187  }
188 
189  // No blending between different groups
190  if (item_A.req.group_name != item_B.req.group_name)
191  {
192  ROS_WARN_STREAM("Blending between different groups (in this case: \""
193  << item_A.req.group_name << "\" and \"" << item_B.req.group_name << "\") not allowed");
194  return true;
195  }
196 
197  // No blending for groups without solver
198  if (!hasSolver(model.getJointModelGroup(item_A.req.group_name)))
199  {
200  ROS_WARN_STREAM("Blending for groups without solver not allowed");
201  return true;
202  }
203 
204  return false;
205 }

207 CommandListManager::RadiiCont CommandListManager::extractBlendRadii(const moveit::core::RobotModel& model,
208  const moveit_msgs::MotionSequenceRequest& req_list)
209 {
210  RadiiCont radii(req_list.items.size(), 0.);
211  for (RadiiCont::size_type i = 0; i < (radii.size() - 1); ++i)
212  {
213  if (isInvalidBlendRadii(model, req_list.items.at(i), req_list.items.at(i + 1)))
214  {
215  ROS_WARN_STREAM("Invalid blend radii between commands: [" << i << "] and [" << i + 1
216  << "] => Blend radii set to zero");
217  continue;
218  }
219  radii.at(i) = req_list.items.at(i).blend_radius;
220  }
221  return radii;
222 }

224 CommandListManager::MotionResponseCont
225 CommandListManager::solveSequenceItems(const planning_scene::PlanningSceneConstPtr& planning_scene,
226  const planning_pipeline::PlanningPipelinePtr& planning_pipeline,
227  const moveit_msgs::MotionSequenceRequest& req_list) const
228 {
229  MotionResponseCont motion_plan_responses;
230  size_t curr_req_index{ 0 };
231  const size_t num_req{ req_list.items.size() };
232  for (const auto& seq_item : req_list.items)
233  {
234  planning_interface::MotionPlanRequest req{ seq_item.req };
235  setStartState(motion_plan_responses, req.group_name, req.start_state);
236 
237  planning_interface::MotionPlanResponse res;
238  planning_pipeline->generatePlan(planning_scene, req, res);
239  if (res.error_code_.val != res.error_code_.SUCCESS)
240  {
241  std::ostringstream os;
242  os << "Could not solve request\n---\n" << req << "\n---\n";
243  throw PlanningPipelineException(os.str(), res.error_code_.val);
244  }
245  motion_plan_responses.emplace_back(res);
246  ROS_DEBUG_STREAM("Solved [" << ++curr_req_index << "/" << num_req << "]");
247  }
248  return motion_plan_responses;
249 }

260 void CommandListManager::checkStartStatesOfGroup(const moveit_msgs::MotionSequenceRequest& req_list,
261  const std::string& group_name)
262 {
263  bool first_elem{ true };
264  for (const moveit_msgs::MotionSequenceItem& item : req_list.items)
265  {
266  if (item.req.group_name != group_name)
267  {
268  continue;
269  }
270 
271  if (first_elem)
272  {
273  first_elem = false;
274  continue;
275  }
276 
277  if (!(item.req.start_state.joint_state.position.empty() && item.req.start_state.joint_state.velocity.empty() &&
278  item.req.start_state.joint_state.effort.empty() && item.req.start_state.joint_state.name.empty()))
279  {
280  std::ostringstream os;
281  os << "Only the first request is allowed to have a start state, but"
282  << " the requests for group: \"" << group_name << "\" violate the rule";
283  throw StartStateSetException(os.str());
284  }
285  }

288 void CommandListManager::checkStartStates(const moveit_msgs::MotionSequenceRequest& req_list)
289 {
290  if (req_list.items.size() <= 1)
291  {
292  return;
293  }
294 
295  GroupNamesCont group_names{ getGroupNames(req_list) };

297  {
298  checkStartStatesOfGroup(req_list, curr_group_name);

302 CommandListManager::GroupNamesCont CommandListManager::getGroupNames(const moveit_msgs::MotionSequenceRequest& req_list)
303 {
304  GroupNamesCont group_names;
305  std::for_each(req_list.items.cbegin(), req_list.items.cend(),
306  [&group_names](const moveit_msgs::MotionSequenceItem& item) {
307  if (std::find(group_names.cbegin(), group_names.cend(), item.req.group_name) == group_names.cend())
308  {
309  group_names.emplace_back(item.req.group_name);
310  }
311  });
312  return group_names;
313 }
