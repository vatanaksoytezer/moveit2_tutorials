90 bool ompl_interface::ConstrainedGoalSampler::sampleUsingConstraintSampler(const ob::GoalLazySamples* gls,
91  ob::State* new_goal)
92 {
93  // moveit::Profiler::ScopedBlock sblock("ConstrainedGoalSampler::sampleUsingConstraintSampler");
94 
95  unsigned int max_attempts = planning_context_->getMaximumGoalSamplingAttempts();
96  unsigned int attempts_so_far = gls->samplingAttemptsCount();
97 
98  // terminate after too many attempts
99  if (attempts_so_far >= max_attempts)
100  return false;
101 
102  // terminate after a maximum number of samples
103  if (gls->getStateCount() >= planning_context_->getMaximumGoalSamples())
104  return false;
105 
106  // terminate the sampling thread when a solution has been found
107  if (planning_context_->getOMPLSimpleSetup()->getProblemDefinition()->hasSolution())
108  return false;
109 
110  unsigned int max_attempts_div2 = max_attempts / 2;
111  for (unsigned int a = gls->samplingAttemptsCount(); a < max_attempts && gls->isSampling(); ++a)
112  {
113  bool verbose = false;
114  if (gls->getStateCount() == 0 && a >= max_attempts_div2)
115  if (verbose_display_ < 1)
116  {
117  verbose = true;
118  verbose_display_++;
119  }
120 
121  if (constraint_sampler_)
122  {
123  // makes the constraint sampler also perform a validity callback
124  moveit::core::GroupStateValidityCallbackFn gsvcf =
125  std::bind(&ompl_interface::ConstrainedGoalSampler::stateValidityCallback, this, new_goal,
126  std::placeholders::_1, // pointer to state
127  std::placeholders::_2, // const* joint model group
128  std::placeholders::_3, // double* of joint positions
129  verbose);
130  constraint_sampler_->setGroupStateValidityCallback(gsvcf);
131 
132  if (constraint_sampler_->project(work_state_, planning_context_->getMaximumStateSamplingAttempts()))
133  {
134  work_state_.update();
135  if (kinematic_constraint_set_->decide(work_state_, verbose).satisfied)
136  {
137  if (checkStateValidity(new_goal, work_state_, verbose))
138  return true;
139  }
140  else
141  {
142  invalid_sampled_constraints_++;
143  if (!warned_invalid_samples_ && invalid_sampled_constraints_ >= (attempts_so_far * 8) / 10)
144  {
145  warned_invalid_samples_ = true;
146  RCLCPP_WARN(LOGGER, "More than 80%% of the sampled goal states "
147  "fail to satisfy the constraints imposed on the goal sampler. "
148  "Is the constrained sampler working correctly?");
149  }
150  }
151  }
152  }
153  else
154  {
155  default_sampler_->sampleUniform(new_goal);
156  if (static_cast<const StateValidityChecker*>(si_->getStateValidityChecker().get())->isValid(new_goal, verbose))
157  {
158  planning_context_->getOMPLStateSpace()->copyToRobotState(work_state_, new_goal);
159  if (kinematic_constraint_set_->decide(work_state_, verbose).satisfied)
160  return true;
161  }
162  }
163  }
164  return false;
165 }
