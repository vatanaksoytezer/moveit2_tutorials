42 collision_detection::Contact* processResult(ContactTestData& cdata, collision_detection::Contact& contact,
43  const std::pair<std::string, std::string>& key, bool found)
44 {
45  // add deepest penetration / smallest distance to result
46  if (cdata.req.distance)
47  {
48  if (contact.depth < cdata.res.distance)
49  {
50  cdata.res.distance = contact.depth;
51  }
52  }
53 
54  RCLCPP_DEBUG_STREAM(BULLET_LOGGER, "Contact btw " << key.first << " and " << key.second << " dist: " << contact.depth);
55  // case if pair hasn't a contact yet
56  if (!found)
57  {
58  if (contact.depth <= 0)
59  {
60  cdata.res.collision = true;
61  }
62 
63  std::vector<collision_detection::Contact> data;
64 
65  // if we dont want contacts we are done here
66  if (!cdata.req.contacts)
67  {
68  if (!cdata.req.distance)
69  {
70  cdata.done = true;
71  }
72  return nullptr;
73  }
74  else
75  {
76  data.reserve(cdata.req.max_contacts_per_pair);
77  data.emplace_back(contact);
78  cdata.res.contact_count++;
79  }
80 
81  if (cdata.res.contact_count >= cdata.req.max_contacts)
82  {
83  if (!cdata.req.distance)
84  {
85  cdata.done = true;
86  }
87  }
88 
89  if (cdata.req.max_contacts_per_pair == 1u)
90  {
91  cdata.pair_done = true;
92  }
93 
94  return &(cdata.res.contacts.insert(std::make_pair(key, data)).first->second.back());
95  }
96  else
97  {
98  std::vector<collision_detection::Contact>& dr = cdata.res.contacts[key];
99  dr.emplace_back(contact);
100  cdata.res.contact_count++;
101 
102  if (dr.size() >= cdata.req.max_contacts_per_pair)
103  {
104  cdata.pair_done = true;
105  }
106 
107  if (cdata.res.contact_count >= cdata.req.max_contacts)
108  {
109  if (!cdata.req.distance)
110  {
111  cdata.done = true;
112  }
113  }
114 
115  return &(dr.back());
116  }
117 
118  return nullptr;
119 }

121 int createConvexHull(AlignedVector<Eigen::Vector3d>& vertices, std::vector<int>& faces,
122  const AlignedVector<Eigen::Vector3d>& input, double shrink, double shrinkClamp)
123 {
124  vertices.clear();
125  faces.clear();
126 
127  btConvexHullComputer conv;
128  btAlignedObjectArray<btVector3> points;
129  points.reserve(static_cast<int>(input.size()));
130  for (const Eigen::Vector3d& v : input)
131  {
132  points.push_back(btVector3(static_cast<btScalar>(v[0]), static_cast<btScalar>(v[1]), static_cast<btScalar>(v[2])));
133  }
134 
135  btScalar val = conv.compute(&points[0].getX(), sizeof(btVector3), points.size(), static_cast<btScalar>(shrink),
136  static_cast<btScalar>(shrinkClamp));
137  if (val < 0)
138  {
139  RCLCPP_ERROR(BULLET_LOGGER, "Failed to create convex hull");
140  return -1;
141  }
142 
143  int num_verts = conv.vertices.size();
144  vertices.reserve(static_cast<size_t>(num_verts));
145  for (int i = 0; i < num_verts; i++)
146  {
147  btVector3& v = conv.vertices[i];
148  vertices.push_back(Eigen::Vector3d(v.getX(), v.getY(), v.getZ()));
149  }
150 
151  int num_faces = conv.faces.size();
152  faces.reserve(static_cast<size_t>(3 * num_faces));
153  for (int i = 0; i < num_faces; i++)
154  {
155  std::vector<int> face;
156  face.reserve(3);
157 
158  const btConvexHullComputer::Edge* source_edge = &(conv.edges[conv.faces[i]]);
159  int a = source_edge->getSourceVertex();
160  face.push_back(a);
161 
162  int b = source_edge->getTargetVertex();
163  face.push_back(b);
164 
165  const btConvexHullComputer::Edge* edge = source_edge->getNextEdgeOfFace();
166  int c = edge->getTargetVertex();
167  face.push_back(c);
168 
169  edge = edge->getNextEdgeOfFace();
170  c = edge->getTargetVertex();
171  while (c != a)
172  {
173  face.push_back(c);
174 
175  edge = edge->getNextEdgeOfFace();
176  c = edge->getTargetVertex();
177  }
178  faces.push_back(static_cast<int>(face.size()));
179  faces.insert(faces.end(), face.begin(), face.end());
180  }
181 
182  return num_faces;
183 }
