58 bool JointConstraintSampler::configure(const std::vector<kinematic_constraints::JointConstraint>& jc)
59 {
60  clear();
61 
62  if (!jmg_)
63  {
64  RCLCPP_ERROR(LOGGER, "NULL group specified for constraint sampler");
65  return false;
66  }
67 
68  // find and keep the constraints that operate on the group we sample
69  // also keep bounds for joints as convenient
70  std::map<std::string, JointInfo> bound_data;
71  bool some_valid_constraint = false;
72  for (const kinematic_constraints::JointConstraint& joint_constraint : jc)
73  {
74  if (!joint_constraint.enabled())
75  continue;
76 
77  const moveit::core::JointModel* jm = joint_constraint.getJointModel();
78  if (!jmg_->hasJointModel(jm->getName()))
79  continue;
80 
81  some_valid_constraint = true;
82 
83  const moveit::core::VariableBounds& joint_bounds = jm->getVariableBounds(joint_constraint.getJointVariableName());
84  JointInfo ji;
85  std::map<std::string, JointInfo>::iterator it = bound_data.find(joint_constraint.getJointVariableName());
86  if (it != bound_data.end())
87  ji = it->second;
88  else
89  ji.index_ = jmg_->getVariableGroupIndex(joint_constraint.getJointVariableName());
90  ji.potentiallyAdjustMinMaxBounds(
91  std::max(joint_bounds.min_position_,
92  joint_constraint.getDesiredJointPosition() - joint_constraint.getJointToleranceBelow()),
93  std::min(joint_bounds.max_position_,
94  joint_constraint.getDesiredJointPosition() + joint_constraint.getJointToleranceAbove()));
95 
96  RCLCPP_DEBUG(LOGGER, "Bounds for %s JointConstraint are %g %g", joint_constraint.getJointVariableName().c_str(),
97  ji.min_bound_, ji.max_bound_);
98 
99  if (ji.min_bound_ > ji.max_bound_ + std::numeric_limits<double>::epsilon())
100  {
101  std::stringstream cs;
102  joint_constraint.print(cs);
103  RCLCPP_ERROR(LOGGER,
104  "The constraints for joint '%s' are such that "
105  "there are no possible values for the joint: min_bound: %g, max_bound: %g. Failing.\n",
106  jm->getName().c_str(), ji.min_bound_, ji.max_bound_);
107  clear();
108  return false;
109  }
110  bound_data[joint_constraint.getJointVariableName()] = ji;
111  }
112 
113  if (!some_valid_constraint)
114  {
115  RCLCPP_WARN(LOGGER, "No valid joint constraints");
116  return false;
117  }
118 
119  for (std::pair<const std::string, JointInfo>& it : bound_data)
120  bounds_.push_back(it.second);
121 
122  // get a separate list of joints that are not bounded; we will sample these randomly
123  const std::vector<const moveit::core::JointModel*>& joints = jmg_->getJointModels();
124  for (const moveit::core::JointModel* joint : joints)
125  if (bound_data.find(joint->getName()) == bound_data.end() && joint->getVariableCount() > 0 &&
126  joint->getMimic() == nullptr)
127  {
128  // check if all the vars of the joint are found in bound_data instead
129  const std::vector<std::string>& vars = joint->getVariableNames();
130  if (vars.size() > 1)
131  {
132  bool all_found = true;
133  for (const std::string& var : vars)
134  if (bound_data.find(var) == bound_data.end())
135  {
136  all_found = false;
137  break;
138  }
139  if (all_found)
140  continue;
141  }
142  unbounded_.push_back(joint);
143  // Get the first variable name of this joint and find its index position in the planning group
144  uindex_.push_back(jmg_->getVariableGroupIndex(vars[0]));
145  }
146  values_.resize(jmg_->getVariableCount());
147  is_valid_ = true;
148  return true;
149 }

151 bool JointConstraintSampler::sample(moveit::core::RobotState& state,
152  const moveit::core::RobotState& /* reference_state */,
153  unsigned int /* max_attempts */)
154 {
155  if (!is_valid_)
156  {
157  RCLCPP_WARN(LOGGER, "JointConstraintSampler not configured, won't sample");
158  return false;
159  }
160 
161  // sample the unbounded joints first (in case some joint variables are bounded)
162  std::vector<double> v;
163  for (std::size_t i = 0; i < unbounded_.size(); ++i)
164  {
165  v.resize(unbounded_[i]->getVariableCount());
166  unbounded_[i]->getVariableRandomPositions(random_number_generator_, &v[0]);
167  for (std::size_t j = 0; j < v.size(); ++j)
168  values_[uindex_[i] + j] = v[j];
169  }
170 
171  // enforce the constraints for the constrained components (could be all of them)
172  for (const JointInfo& bound : bounds_)
173  values_[bound.index_] = random_number_generator_.uniformReal(bound.min_bound_, bound.max_bound_);
174 
175  state.setJointGroupPositions(jmg_, values_);
176 
177  // we are always successful
178  return true;
179 }

240 bool IKConstraintSampler::configure(const IKSamplingPose& sp)
241 {
242  clear();
243  if (!sp.position_constraint_ && !sp.orientation_constraint_)
244  return false;
245  if ((!sp.orientation_constraint_ && !sp.position_constraint_->enabled()) ||
246  (!sp.position_constraint_ && !sp.orientation_constraint_->enabled()) ||
247  (sp.position_constraint_ && sp.orientation_constraint_ && !sp.position_constraint_->enabled() &&
248  !sp.orientation_constraint_->enabled()))
249  {
250  RCLCPP_WARN(LOGGER, "No enabled constraints in sampling pose");
251  return false;
252  }
253 
254  sampling_pose_ = sp;
255  ik_timeout_ = jmg_->getDefaultIKTimeout();
256  if (sampling_pose_.position_constraint_ && sampling_pose_.orientation_constraint_)
257  if (sampling_pose_.position_constraint_->getLinkModel()->getName() !=
258  sampling_pose_.orientation_constraint_->getLinkModel()->getName())
259  {
260  RCLCPP_ERROR(LOGGER, "Position and orientation constraints need to be specified for the same link "
261  "in order to use IK-based sampling");
262  return false;
263  }
264 
265  if (sampling_pose_.position_constraint_ && sampling_pose_.position_constraint_->mobileReferenceFrame())
266  frame_depends_.push_back(sampling_pose_.position_constraint_->getReferenceFrame());
267  if (sampling_pose_.orientation_constraint_ && sampling_pose_.orientation_constraint_->mobileReferenceFrame())
268  frame_depends_.push_back(sampling_pose_.orientation_constraint_->getReferenceFrame());
269  kb_ = jmg_->getSolverInstance();
270  if (!kb_)
271  {
272  RCLCPP_WARN(LOGGER, "No solver instance in setup");
273  is_valid_ = false;
274  return false;
275  }
276  is_valid_ = loadIKSolver();
277  return is_valid_;
278 }

280 bool IKConstraintSampler::configure(const moveit_msgs::msg::Constraints& constr)
281 {
282  for (std::size_t p = 0; p < constr.position_constraints.size(); ++p)
283  for (std::size_t o = 0; o < constr.orientation_constraints.size(); ++o)
284  if (constr.position_constraints[p].link_name == constr.orientation_constraints[o].link_name)
285  {
286  kinematic_constraints::PositionConstraintPtr pc(
287  new kinematic_constraints::PositionConstraint(scene_->getRobotModel()));
288  kinematic_constraints::OrientationConstraintPtr oc(
289  new kinematic_constraints::OrientationConstraint(scene_->getRobotModel()));
290  if (pc->configure(constr.position_constraints[p], scene_->getTransforms()) &&
291  oc->configure(constr.orientation_constraints[o], scene_->getTransforms()))
292  return configure(IKSamplingPose(pc, oc));
293  }
294 
295  for (const moveit_msgs::msg::PositionConstraint& position_constraint : constr.position_constraints)
296  {
297  kinematic_constraints::PositionConstraintPtr pc(
298  new kinematic_constraints::PositionConstraint(scene_->getRobotModel()));
299  if (pc->configure(position_constraint, scene_->getTransforms()))
300  return configure(IKSamplingPose(pc));
301  }
302 
303  for (const moveit_msgs::msg::OrientationConstraint& orientation_constraint : constr.orientation_constraints)
304  {
305  kinematic_constraints::OrientationConstraintPtr oc(
306  new kinematic_constraints::OrientationConstraint(scene_->getRobotModel()));
307  if (oc->configure(orientation_constraint, scene_->getTransforms()))
308  return configure(IKSamplingPose(oc));
309  }
310  return false;
311 }

313 double IKConstraintSampler::getSamplingVolume() const
314 {
315  double v = 1.0;
316  if (sampling_pose_.position_constraint_)
317  {
318  const std::vector<bodies::BodyPtr>& constraint_regions =
319  sampling_pose_.position_constraint_->getConstraintRegions();
320  double vol = 0;
321  for (const bodies::BodyPtr& constraint_region : constraint_regions)
322  vol += constraint_region->computeVolume();
323  if (!constraint_regions.empty())
324  v *= vol;
325  }
326 
327  if (sampling_pose_.orientation_constraint_)
328  v *= sampling_pose_.orientation_constraint_->getXAxisTolerance() *
329  sampling_pose_.orientation_constraint_->getYAxisTolerance() *
330  sampling_pose_.orientation_constraint_->getZAxisTolerance();
331  return v;
332 }

341 bool IKConstraintSampler::loadIKSolver()
342 {
343  if (!kb_)
344  {
345  RCLCPP_ERROR(LOGGER, "No IK solver");
346  return false;
347  }
348 
349  // check if we need to transform the request into the coordinate frame expected by IK
350  ik_frame_ = kb_->getBaseFrame();
351  transform_ik_ = !moveit::core::Transforms::sameFrame(ik_frame_, jmg_->getParentModel().getModelFrame());
352  if (!ik_frame_.empty() && ik_frame_[0] == '/')
353  ik_frame_.erase(ik_frame_.begin());
354  if (transform_ik_)
355  if (!jmg_->getParentModel().hasLinkModel(ik_frame_))
356  {
357  RCLCPP_ERROR(LOGGER,
358  "The IK solver expects requests in frame '%s' but this frame is not known to the sampler. "
359  "Ignoring transformation (IK may fail)",
360  ik_frame_.c_str());
361  transform_ik_ = false;
362  }
363 
364  // check if IK is performed for the desired link
365  bool wrong_link = false;
366  if (sampling_pose_.position_constraint_)
367  {
368  const moveit::core::LinkModel* lm = sampling_pose_.position_constraint_->getLinkModel();
369  if (!moveit::core::Transforms::sameFrame(kb_->getTipFrame(), lm->getName()))
370  {
371  wrong_link = true;
372  const moveit::core::LinkTransformMap& fixed_links = lm->getAssociatedFixedTransforms();
373  for (const std::pair<const moveit::core::LinkModel* const, Eigen::Isometry3d>& fixed_link : fixed_links)
374  if (moveit::core::Transforms::sameFrame(fixed_link.first->getName(), kb_->getTipFrame()))
375  {
376  eef_to_ik_tip_transform_ = fixed_link.second; // valid isometry by contract
377  need_eef_to_ik_tip_transform_ = true;
378  wrong_link = false;
379  break;
380  }
381  }
382  }
383 
384  if (!wrong_link && sampling_pose_.orientation_constraint_)
385  {
386  const moveit::core::LinkModel* lm = sampling_pose_.orientation_constraint_->getLinkModel();
387  if (!moveit::core::Transforms::sameFrame(kb_->getTipFrame(), lm->getName()))
388  {
389  wrong_link = true;
390  const moveit::core::LinkTransformMap& fixed_links = lm->getAssociatedFixedTransforms();
391  for (const std::pair<const moveit::core::LinkModel* const, Eigen::Isometry3d>& fixed_link : fixed_links)
392  if (moveit::core::Transforms::sameFrame(fixed_link.first->getName(), kb_->getTipFrame()))
393  {
394  eef_to_ik_tip_transform_ = fixed_link.second; // valid isometry by contract
395  need_eef_to_ik_tip_transform_ = true;
396  wrong_link = false;
397  break;
398  }
399  }
400  }
401 
402  if (wrong_link)
403  {
404  RCLCPP_ERROR(LOGGER, "IK cannot be performed for link '%s'. The solver can report IK solutions for link '%s'.",
405  sampling_pose_.position_constraint_ ?
406  sampling_pose_.position_constraint_->getLinkModel()->getName().c_str() :
407  sampling_pose_.orientation_constraint_->getLinkModel()->getName().c_str(),
408  kb_->getTipFrame().c_str());
409  return false;
410  }
411  return true;
412 }

414 bool IKConstraintSampler::samplePose(Eigen::Vector3d& pos, Eigen::Quaterniond& quat, const moveit::core::RobotState& ks,
415  unsigned int max_attempts)
416 {
417  if (ks.dirtyLinkTransforms())
418  {
419  // samplePose below requires accurate transforms
420  RCLCPP_ERROR(LOGGER, "IKConstraintSampler received dirty robot state, but valid transforms are required. "
421  "Failing.");
422  return false;
423  }
424 
425  if (sampling_pose_.position_constraint_)
426  {
427  const std::vector<bodies::BodyPtr>& b = sampling_pose_.position_constraint_->getConstraintRegions();
428  if (!b.empty())
429  {
430  bool found = false;
431  std::size_t k = random_number_generator_.uniformInteger(0, b.size() - 1);
432  for (std::size_t i = 0; i < b.size(); ++i)
433  if (b[(i + k) % b.size()]->samplePointInside(random_number_generator_, max_attempts, pos))
434  {
435  found = true;
436  break;
437  }
438  if (!found)
439  {
440  RCLCPP_ERROR(LOGGER, "Unable to sample a point inside the constraint region");
441  return false;
442  }
443  }
444  else
445  {
446  RCLCPP_ERROR(LOGGER, "Unable to sample a point inside the constraint region. "
447  "Constraint region is empty when it should not be.");
448  return false;
449  }
450 
451  // if this constraint is with respect a mobile frame, we need to convert this rotation to the root frame of the
452  // model
453  if (sampling_pose_.position_constraint_->mobileReferenceFrame())
454  pos = ks.getFrameTransform(sampling_pose_.position_constraint_->getReferenceFrame()) * pos;
455  }
456  else
457  {
458  // do FK for rand state
459  moveit::core::RobotState temp_state(ks);
460  temp_state.setToRandomPositions(jmg_);
461  pos = temp_state.getGlobalLinkTransform(sampling_pose_.orientation_constraint_->getLinkModel()).translation();
462  }
463 
464  if (sampling_pose_.orientation_constraint_)
465  {
466  // sample a rotation matrix within the allowed bounds
467  double angle_x =
468  2.0 * (random_number_generator_.uniform01() - 0.5) *
469  (sampling_pose_.orientation_constraint_->getXAxisTolerance() - std::numeric_limits<double>::epsilon());
470  double angle_y =
471  2.0 * (random_number_generator_.uniform01() - 0.5) *
472  (sampling_pose_.orientation_constraint_->getYAxisTolerance() - std::numeric_limits<double>::epsilon());
473  double angle_z =
474  2.0 * (random_number_generator_.uniform01() - 0.5) *
475  (sampling_pose_.orientation_constraint_->getZAxisTolerance() - std::numeric_limits<double>::epsilon());
476  Eigen::Isometry3d diff(Eigen::AngleAxisd(angle_x, Eigen::Vector3d::UnitX()) *
477  Eigen::AngleAxisd(angle_y, Eigen::Vector3d::UnitY()) *
478  Eigen::AngleAxisd(angle_z, Eigen::Vector3d::UnitZ()));
479  // diff is isometry by construction
480  // getDesiredRotationMatrix() returns a valid rotation matrix by contract
481  // reqr has thus to be a valid isometry
482  Eigen::Isometry3d reqr(sampling_pose_.orientation_constraint_->getDesiredRotationMatrix() * diff.linear());
483  quat = Eigen::Quaterniond(reqr.linear()); // reqr is isometry, so quat has to be normalized
484 
485  // if this constraint is with respect a mobile frame, we need to convert this rotation to the root frame of the
486  // model
487  if (sampling_pose_.orientation_constraint_->mobileReferenceFrame())
488  {
489  // getFrameTransform() returns a valid isometry by contract
490  const Eigen::Isometry3d& t = ks.getFrameTransform(sampling_pose_.orientation_constraint_->getReferenceFrame());
491  // rt is isometry by construction
492  Eigen::Isometry3d rt(t.linear() * quat);
493  quat = Eigen::Quaterniond(rt.linear()); // rt is isometry, so quat has to be normalized
494  }
495  }
496  else
497  {
498  // sample a random orientation
499  double q[4];
500  random_number_generator_.quaternion(q);
501  quat = Eigen::Quaterniond(q[3], q[0], q[1], q[2]); // quat is normalized by contract
502  }
503 
504  // if there is an offset, we need to undo the induced rotation in the sampled transform origin (point)
505  if (sampling_pose_.position_constraint_ && sampling_pose_.position_constraint_->hasLinkOffset())
506  // the rotation matrix that corresponds to the desired orientation
507  pos = pos - quat * sampling_pose_.position_constraint_->getLinkOffset();
508 
509  return true;
510 }

536 bool IKConstraintSampler::sampleHelper(moveit::core::RobotState& state, const moveit::core::RobotState& reference_state,
537  unsigned int max_attempts, bool project)
538 {
539  if (!is_valid_)
540  {
541  RCLCPP_WARN(LOGGER, "IKConstraintSampler not configured, won't sample");
542  return false;
543  }
544 
545  kinematics::KinematicsBase::IKCallbackFn adapted_ik_validity_callback;
546  if (group_state_validity_callback_)
547  adapted_ik_validity_callback =
548  boost::bind(&samplingIkCallbackFnAdapter, &state, jmg_, group_state_validity_callback_, _1, _2, _3);
549 
550  for (unsigned int a = 0; a < max_attempts; ++a)
551  {
552  // sample a point in the constraint region
553  Eigen::Vector3d point;
554  Eigen::Quaterniond quat; // quat is normalized by contract
555  if (!samplePose(point, quat, reference_state, max_attempts))
556  {
557  if (verbose_)
558  RCLCPP_INFO(LOGGER, "IK constraint sampler was unable to produce a pose to run IK for");
559  return false;
560  }
561 
562  // we now have the transform we wish to perform IK for, in the planning frame
563  if (transform_ik_)
564  {
565  // we need to convert this transform to the frame expected by the IK solver
566  // both the planning frame and the frame for the IK are assumed to be robot links
567  Eigen::Isometry3d ikq(Eigen::Translation3d(point) * quat); // valid isometry by construction
568  // getFrameTransform() returns a valid isometry by contract
569  ikq = reference_state.getFrameTransform(ik_frame_).inverse() * ikq; // valid isometry * valid isometry
570  point = ikq.translation();
571  quat = Eigen::Quaterniond(ikq.linear()); // ikq is isometry, so quat is normalized
572  }
573 
574  if (need_eef_to_ik_tip_transform_)
575  {
576  // After sampling the pose needs to be transformed to the ik chain tip
577  Eigen::Isometry3d ikq(Eigen::Translation3d(point) * quat); // valid isometry by construction
578  ikq = ikq * eef_to_ik_tip_transform_; // eef_to_ik_tip_transform_ is valid isometry (checked in loadIKSolver())
579  point = ikq.translation();
580  quat = Eigen::Quaterniond(ikq.linear()); // ikq is isometry, so quat is normalized
581  }
582 
583  geometry_msgs::msg::Pose ik_query;
584  ik_query.position.x = point.x();
585  ik_query.position.y = point.y();
586  ik_query.position.z = point.z();
587  ik_query.orientation.x = quat.x();
588  ik_query.orientation.y = quat.y();
589  ik_query.orientation.z = quat.z();

612 bool IKConstraintSampler::callIK(const geometry_msgs::msg::Pose& ik_query,
613  const kinematics::KinematicsBase::IKCallbackFn& adapted_ik_validity_callback,
614  double timeout, moveit::core::RobotState& state, bool use_as_seed)
615 {
616  const std::vector<unsigned int>& ik_joint_bijection = jmg_->getKinematicsSolverJointBijection();
617  std::vector<double> seed(ik_joint_bijection.size(), 0.0);
618  std::vector<double> vals;
619 
620  if (use_as_seed)
621  state.copyJointGroupPositions(jmg_, vals);
622  else
623  // sample a seed value
624  jmg_->getVariableRandomPositions(random_number_generator_, vals);
625 
626  assert(vals.size() == ik_joint_bijection.size());
627  for (std::size_t i = 0; i < ik_joint_bijection.size(); ++i)
628  seed[i] = vals[ik_joint_bijection[i]];
629 
630  std::vector<double> ik_sol;
631  moveit_msgs::msg::MoveItErrorCodes error;
632 
633  if (adapted_ik_validity_callback ?
634  kb_->searchPositionIK(ik_query, seed, timeout, ik_sol, adapted_ik_validity_callback, error) :
635  kb_->searchPositionIK(ik_query, seed, timeout, ik_sol, error))
636  {
637  assert(ik_sol.size() == ik_joint_bijection.size());
638  std::vector<double> solution(ik_joint_bijection.size());
639  for (std::size_t i = 0; i < ik_joint_bijection.size(); ++i)
640  solution[ik_joint_bijection[i]] = ik_sol[i];
641  state.setJointGroupPositions(jmg_, solution);
642 
643  return validate(state);
644  }
645  else
646  {
647  if (error.val != moveit_msgs::msg::MoveItErrorCodes::NO_IK_SOLUTION &&
648  error.val != moveit_msgs::msg::MoveItErrorCodes::INVALID_ROBOT_STATE &&
649  error.val != moveit_msgs::msg::MoveItErrorCodes::TIMED_OUT)
650  {
651  RCLCPP_ERROR(LOGGER, "IK solver failed with error %d", error.val);
652  }
653  else if (verbose_)
654  {
655  RCLCPP_INFO(LOGGER, "IK failed");
656  }
657  }
658  return false;
659 }
