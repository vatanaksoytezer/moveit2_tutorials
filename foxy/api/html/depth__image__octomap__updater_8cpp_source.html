214 void DepthImageOctomapUpdater::depthImageCallback(const sensor_msgs::msg::Image::ConstSharedPtr& depth_msg,
215  const sensor_msgs::msg::CameraInfo::ConstSharedPtr& info_msg)
216 {
217  RCLCPP_DEBUG(LOGGER, "Received a new depth image message (frame = '%s', encoding='%s')",
218  depth_msg->header.frame_id.c_str(), depth_msg->encoding.c_str());
219  rclcpp::Time start = node_->now();
220 
221  if (max_update_rate_ > 0)
222  {
223  // ensure we are not updating the octomap representation too often
224  if (node_->now() - last_update_time_ <= rclcpp::Duration(std::chrono::duration<double>(1.0 / max_update_rate_)))
225  return;
226  last_update_time_ = node_->now();
227  }
228 
229  // measure the frequency at which we receive updates
230  if (image_callback_count_ < 1000)
231  {
232  if (image_callback_count_ > 0)
233  {
234  const double dt_start = (start - last_depth_callback_start_).seconds();
235  if (image_callback_count_ < 2)
236  average_callback_dt_ = dt_start;
237  else
238  average_callback_dt_ =
239  ((image_callback_count_ - 1) * average_callback_dt_ + dt_start) / (double)image_callback_count_;
240  }
241  }
242  else
243  // every 1000 updates we reset the counter almost to the beginning (use 2 so we don't have so much of a ripple in
244  // the measured average)
245  image_callback_count_ = 2;
246  last_depth_callback_start_ = start;
247  ++image_callback_count_;
248 
249  if (monitor_->getMapFrame().empty())
250  monitor_->setMapFrame(depth_msg->header.frame_id);
251 
252  /* get transform for cloud into map frame */
253  tf2::Stamped<tf2::Transform> map_h_sensor;
254  if (monitor_->getMapFrame() == depth_msg->header.frame_id)
255  map_h_sensor.setIdentity();
256  else
257  {
258  if (tf_buffer_)
259  {
260  // wait at most 50ms
261  static const double TEST_DT = 0.005;
262  const int nt = (int)(0.5 + average_callback_dt_ / TEST_DT) * std::max(1, ((int)queue_size_ / 2));
263  bool found = false;
264  std::string err;
265  for (int t = 0; t < nt; ++t)
266  try
267  {
268  tf2::fromMsg(tf_buffer_->lookupTransform(monitor_->getMapFrame(), depth_msg->header.frame_id,
269  depth_msg->header.stamp),
270  map_h_sensor);
271  found = true;
272  break;
273  }
274  catch (tf2::TransformException& ex)
275  {
276  std::chrono::duration<double, std::nano> tmp_duration(TEST_DT);
277  static const rclcpp::Duration D(tmp_duration.count());
278  err = ex.what();
279  std::this_thread::sleep_for(D.to_chrono<std::chrono::seconds>());
280  }
281  static const unsigned int MAX_TF_COUNTER = 1000; // so we avoid int overflow
282  if (found)
283  {
284  good_tf_++;
285  if (good_tf_ > MAX_TF_COUNTER)
286  {
287  const unsigned int div = MAX_TF_COUNTER / 10;
288  good_tf_ /= div;
289  failed_tf_ /= div;
290  }
291  }
292  else
293  {
294  failed_tf_++;
295  if (failed_tf_ > good_tf_)
296  RCLCPP_WARN_THROTTLE(LOGGER, *node_->get_clock(), 1000,
297  "More than half of the image messages discared due to TF being unavailable (%u%%). "
298  "Transform error of sensor data: %s; quitting callback.",
299  (100 * failed_tf_) / (good_tf_ + failed_tf_), err.c_str());
300  else
301  RCLCPP_DEBUG_THROTTLE(LOGGER, *node_->get_clock(), 1000,
302  "Transform error of sensor data: %s; quitting callback", err.c_str());
303  if (failed_tf_ > MAX_TF_COUNTER)
304  {
305  const unsigned int div = MAX_TF_COUNTER / 10;
306  good_tf_ /= div;
307  failed_tf_ /= div;
308  }
309  return;
310  }
311  }
312  else
313  return;
314  }
315 
316  if (!updateTransformCache(depth_msg->header.frame_id, depth_msg->header.stamp))
317  return;
318 
319  if (depth_msg->is_bigendian && !HOST_IS_BIG_ENDIAN)
320  RCLCPP_ERROR_THROTTLE(LOGGER, *node_->get_clock(), 1000, "endian problem: received image data does not match host");
321 
322  const int w = depth_msg->width;
323  const int h = depth_msg->height;
324 
325  // call the mesh filter
326  mesh_filter::StereoCameraModel::Parameters& params = mesh_filter_->parameters();
327  params.setCameraParameters(info_msg->k[0], info_msg->k[4], info_msg->k[2], info_msg->k[5]);
328  params.setImageSize(w, h);
329 
330  const bool is_u_short = depth_msg->encoding == sensor_msgs::image_encodings::TYPE_16UC1;
331  if (is_u_short)
332  mesh_filter_->filter(&depth_msg->data[0], GL_UNSIGNED_SHORT);
333  else
334  {
335  if (depth_msg->encoding != sensor_msgs::image_encodings::TYPE_32FC1)
336  {
337  RCLCPP_ERROR_THROTTLE(LOGGER, *node_->get_clock(), 1000, "Unexpected encoding type: '%s'. Ignoring input.",
338  depth_msg->encoding.c_str());
339  return;
340  }
341  mesh_filter_->filter(&depth_msg->data[0], GL_FLOAT);
342  }
343 
344  // the mesh filter runs in background; compute extra things in the meantime
345 
346  // Use correct principal point from calibration
347  const double px = info_msg->k[2];
348  const double py = info_msg->k[5];
349 
350  // if the camera parameters have changed at all, recompute the cache we had
351  if (w >= static_cast<int>(x_cache_.size()) || h >= static_cast<int>(y_cache_.size()) || K2_ != px || K5_ != py ||
352  K0_ != info_msg->k[0] || K4_ != info_msg->k[4])
353  {
354  K2_ = px;
355  K5_ = py;
356  K0_ = info_msg->k[0];
357  K4_ = info_msg->k[4];
358 
359  inv_fx_ = 1.0 / K0_;
360  inv_fy_ = 1.0 / K4_;
361 
362  // if there are any NaNs, discard data
363  if (!(px == px && py == py && inv_fx_ == inv_fx_ && inv_fy_ == inv_fy_))
364  return;
365 
366  // Pre-compute some constants
367  if (static_cast<int>(x_cache_.size()) < w)
368  x_cache_.resize(w);
369  if (static_cast<int>(y_cache_.size()) < h)
370  y_cache_.resize(h);
371 
372  for (int x = 0; x < w; ++x)
373  x_cache_[x] = (x - px) * inv_fx_;
374 
375  for (int y = 0; y < h; ++y)
376  y_cache_[y] = (y - py) * inv_fy_;
377  }
378 
379  const octomap::point3d sensor_origin(map_h_sensor.getOrigin().getX(), map_h_sensor.getOrigin().getY(),
380  map_h_sensor.getOrigin().getZ());
381 
382  octomap::KeySet* occupied_cells_ptr = new octomap::KeySet();
383  octomap::KeySet* model_cells_ptr = new octomap::KeySet();
384  octomap::KeySet& occupied_cells = *occupied_cells_ptr;
385  octomap::KeySet& model_cells = *model_cells_ptr;
386 
387  // allocate memory if needed
388  std::size_t img_size = h * w;
389  if (filtered_labels_.size() < img_size)
390  filtered_labels_.resize(img_size);
391 
392  // get the labels of the filtered data
393  const unsigned int* labels_row = &filtered_labels_[0];
394  mesh_filter_->getFilteredLabels(&filtered_labels_[0]);
395 
396  // publish debug information if needed
397  if (debug_info_)
398  {
399  sensor_msgs::msg::Image debug_msg;
400  debug_msg.header = depth_msg->header;
401  debug_msg.height = h;
402  debug_msg.width = w;
403  debug_msg.is_bigendian = HOST_IS_BIG_ENDIAN;
404  debug_msg.encoding = sensor_msgs::image_encodings::TYPE_32FC1;
405  debug_msg.step = w * sizeof(float);
406  debug_msg.data.resize(img_size * sizeof(float));
407  mesh_filter_->getModelDepth(reinterpret_cast<float*>(&debug_msg.data[0]));
408  pub_model_depth_image_.publish(debug_msg, *info_msg);
409 
410  sensor_msgs::msg::Image filtered_depth_msg;
411  filtered_depth_msg.header = depth_msg->header;
412  filtered_depth_msg.height = h;
413  filtered_depth_msg.width = w;
414  filtered_depth_msg.is_bigendian = HOST_IS_BIG_ENDIAN;
415  filtered_depth_msg.encoding = sensor_msgs::image_encodings::TYPE_32FC1;
416  filtered_depth_msg.step = w * sizeof(float);
417  filtered_depth_msg.data.resize(img_size * sizeof(float));
418  mesh_filter_->getFilteredDepth(reinterpret_cast<float*>(&filtered_depth_msg.data[0]));
419  pub_filtered_depth_image_.publish(filtered_depth_msg, *info_msg);
420 
421  sensor_msgs::msg::Image label_msg;
422  label_msg.header = depth_msg->header;
423  label_msg.height = h;
424  label_msg.width = w;
425  label_msg.is_bigendian = HOST_IS_BIG_ENDIAN;
426  label_msg.encoding = sensor_msgs::image_encodings::RGBA8;
427  label_msg.step = w * sizeof(unsigned int);
428  label_msg.data.resize(img_size * sizeof(unsigned int));
429  mesh_filter_->getFilteredLabels(reinterpret_cast<unsigned int*>(&label_msg.data[0]));
430 
431  pub_filtered_label_image_.publish(label_msg, *info_msg);
432  }
433 
434  if (!filtered_cloud_topic_.empty())
435  {
436  sensor_msgs::msg::Image filtered_msg;
437  filtered_msg.header = depth_msg->header;
438  filtered_msg.height = h;
439  filtered_msg.width = w;
440  filtered_msg.is_bigendian = HOST_IS_BIG_ENDIAN;
441  filtered_msg.encoding = sensor_msgs::image_encodings::TYPE_16UC1;
442  filtered_msg.step = w * sizeof(unsigned short);
443  filtered_msg.data.resize(img_size * sizeof(unsigned short));
444 
445  // reuse float buffer across callbacks
446  static std::vector<float> filtered_data;
447  if (filtered_data.size() < img_size)
448  filtered_data.resize(img_size);
449 
450  mesh_filter_->getFilteredDepth(reinterpret_cast<float*>(&filtered_data[0]));
451  unsigned short* msg_data = reinterpret_cast<unsigned short*>(&filtered_msg.data[0]);
452  for (std::size_t i = 0; i < img_size; ++i)
453  {
454  // rescale depth to millimeter to work with `unsigned short`
455  msg_data[i] = static_cast<unsigned short>(filtered_data[i] * 1000 + 0.5);
456  }
457  pub_filtered_depth_image_.publish(filtered_msg, *info_msg);
458  }
459 
460  // figure out occupied cells and model cells
461  tree_->lockRead();
462 
463  try
464  {
465  const int h_bound = h - skip_vertical_pixels_;
466  const int w_bound = w - skip_horizontal_pixels_;
467 
468  if (is_u_short)
469  {
470  const uint16_t* input_row = reinterpret_cast<const uint16_t*>(&depth_msg->data[0]);
471 
472  for (int y = skip_vertical_pixels_; y < h_bound; ++y, labels_row += w, input_row += w)
473  for (int x = skip_horizontal_pixels_; x < w_bound; ++x)
474  {
475  // not filtered
476  if (labels_row[x] == mesh_filter::MeshFilterBase::BACKGROUND)
477  {
478  float zz = (float)input_row[x] * 1e-3; // scale from mm to m
479  float yy = y_cache_[y] * zz;
480  float xx = x_cache_[x] * zz;
481  /* transform to map frame */
482  tf2::Vector3 point_tf = map_h_sensor * tf2::Vector3(xx, yy, zz);
483  occupied_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
484  }
485  // on far plane or a model point -> remove
486  else if (labels_row[x] >= mesh_filter::MeshFilterBase::FAR_CLIP)
487  {
488  float zz = input_row[x] * 1e-3;
489  float yy = y_cache_[y] * zz;
490  float xx = x_cache_[x] * zz;
491  /* transform to map frame */
492  tf2::Vector3 point_tf = map_h_sensor * tf2::Vector3(xx, yy, zz);
493  // add to the list of model cells
494  model_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
495  }
496  }
497  }
498  else
499  {
500  const float* input_row = reinterpret_cast<const float*>(&depth_msg->data[0]);
501 
502  for (int y = skip_vertical_pixels_; y < h_bound; ++y, labels_row += w, input_row += w)
503  for (int x = skip_horizontal_pixels_; x < w_bound; ++x)
504  {
505  if (labels_row[x] == mesh_filter::MeshFilterBase::BACKGROUND)
506  {
507  float zz = input_row[x];
508  float yy = y_cache_[y] * zz;
509  float xx = x_cache_[x] * zz;
510  /* transform to map frame */
511  tf2::Vector3 point_tf = map_h_sensor * tf2::Vector3(xx, yy, zz);
512  occupied_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
513  }
514  else if (labels_row[x] >= mesh_filter::MeshFilterBase::FAR_CLIP)
515  {
516  float zz = input_row[x];
517  float yy = y_cache_[y] * zz;
518  float xx = x_cache_[x] * zz;
519  /* transform to map frame */
520  tf2::Vector3 point_tf = map_h_sensor * tf2::Vector3(xx, yy, zz);
521  // add to the list of model cells
522  model_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
523  }
524  }
525  }
526  }
527  catch (...)
528  {
529  tree_->unlockRead();
530  RCLCPP_ERROR(LOGGER, "Internal error while parsing depth data");
531  return;
532  }
533  tree_->unlockRead();
534 
535  /* cells that overlap with the model are not occupied */
536  for (const octomap::OcTreeKey& model_cell : model_cells)
537  occupied_cells.erase(model_cell);
538 
539  // mark occupied cells
540  tree_->lockWrite();
541  try
542  {
543  /* now mark all occupied cells */
544  for (const octomap::OcTreeKey& occupied_cell : occupied_cells)
545  tree_->updateNode(occupied_cell, true);
546  }
547  catch (...)
548  {
549  RCLCPP_ERROR(LOGGER, "Internal error while updating octree");
550  }
551  tree_->unlockWrite();
552  tree_->triggerUpdateCallback();
553 
554  // at this point we still have not freed the space
555  free_space_updater_->pushLazyUpdate(occupied_cells_ptr, model_cells_ptr, sensor_origin);
556 
557  RCLCPP_DEBUG(LOGGER, "Processed depth image in %lf ms", (node_->now() - start).seconds() * 1000.0);
558 }
