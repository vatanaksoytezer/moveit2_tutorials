83  pub_model_depth_image_ =
84  model_depth_transport_->advertiseCamera("/model/depth", queue_size_, itssc, itssc, rssc, rssc);

105 void mesh_filter::DepthSelfFiltering::filter(const sensor_msgs::ImageConstPtr& depth_msg,
106  const sensor_msgs::CameraInfoConstPtr& info_msg)
107 {
108  transform_provider_.setFrame(depth_msg->header.frame_id);
109  // do filtering here
110  mesh_filter::StereoCameraModel::Parameters& params = mesh_filter_->parameters();
111  params.setCameraParameters(info_msg->K[0], info_msg->K[4], info_msg->K[2], info_msg->K[5]);
112  params.setImageSize(depth_msg->width, depth_msg->height);
113 
114  const float* src = (const float*)&depth_msg->data[0];
115  //*
116  static unsigned data_size = 0;
117  static unsigned short* data = nullptr;
118  if (data_size < depth_msg->width * depth_msg->height)
119  data = new unsigned short[depth_msg->width * depth_msg->height];
120  for (unsigned idx = 0; idx < depth_msg->width * depth_msg->height; ++idx)
121  data[idx] = (unsigned short)(src[idx] * 1000.0);
122 
123  mesh_filter_->filter(data, GL_UNSIGNED_SHORT);
124  // delete[] data;
125  /*/
126  mesh_filter_->filter ((void*) &depth_msg->data[0], GL_FLOAT);
127  //*/
128  if (pub_filtered_depth_image_.getNumSubscribers() > 0)
129  {
130  filtered_depth_ptr_->encoding = sensor_msgs::image_encodings::TYPE_32FC1;
131  filtered_depth_ptr_->header = depth_msg->header;
132 
133  if (static_cast<uint32_t>(filtered_depth_ptr_->image.cols) != depth_msg->width ||
134  static_cast<uint32_t>(filtered_depth_ptr_->image.rows) != depth_msg->height)
135  filtered_depth_ptr_->image = cv::Mat(depth_msg->height, depth_msg->width, CV_32FC1);
136  mesh_filter_->getFilteredDepth((float*)filtered_depth_ptr_->image.data);
137  pub_filtered_depth_image_.publish(filtered_depth_ptr_->toImageMsg(), info_msg);
138  }
139 
140  // this is from rendering of the model
141  if (pub_model_depth_image_.getNumSubscribers() > 0)
142  {
143  model_depth_ptr_->encoding = sensor_msgs::image_encodings::TYPE_32FC1;
144  model_depth_ptr_->header = depth_msg->header;
145 
146  if (static_cast<uint32_t>(model_depth_ptr_->image.cols) != depth_msg->width ||
147  static_cast<uint32_t>(model_depth_ptr_->image.rows) != depth_msg->height)
148  model_depth_ptr_->image = cv::Mat(depth_msg->height, depth_msg->width, CV_32FC1);
149  mesh_filter_->getModelDepth((float*)model_depth_ptr_->image.data);
150  pub_model_depth_image_.publish(model_depth_ptr_->toImageMsg(), info_msg);
151  }
152 
153  if (pub_filtered_label_image_.getNumSubscribers() > 0)
154  {
155  filtered_label_ptr_->encoding = sensor_msgs::image_encodings::RGBA8;

177 void mesh_filter::DepthSelfFiltering::addMeshes(MeshFilter<StereoCameraModel>& mesh_filter)
178 {
179  robot_model_loader::RobotModelLoader robot_model_loader("robot_description");
180  moveit::core::RobotModelConstPtr robot_model = robot_model_loader.getModel();
181  const auto& links = robot_model->getLinkModelsWithCollisionGeometry();
182  for (auto link : links)
183  {
184  const auto& shapes = link->getShapes();
185  for (const auto& shape : shapes)
186  {
187  if (shape->type == shapes::MESH)
188  {
189  const shapes::Mesh& m = static_cast<const shapes::Mesh&>(*shape);
190  MeshHandle mesh_handle = mesh_filter.addMesh(m);
191  transform_provider_.addHandle(mesh_handle, link->getName());
192  }
193  }
194  }
195 }
