65 double DistanceField::getDistanceGradient(double x, double y, double z, double& gradient_x, double& gradient_y,
66  double& gradient_z, bool& in_bounds) const
67 {
68  int gx, gy, gz;
69 
70  worldToGrid(x, y, z, gx, gy, gz);
71 
72  // if out of bounds, return max_distance, and 0 gradient
73  // we need extra padding of 1 to get gradients
74  if (gx < 1 || gy < 1 || gz < 1 || gx >= getXNumCells() - 1 || gy >= getYNumCells() - 1 || gz >= getZNumCells() - 1)
75  {
76  gradient_x = 0.0;
77  gradient_y = 0.0;
78  gradient_z = 0.0;
79  in_bounds = false;
80  return getUninitializedDistance();
81  }
82 
83  gradient_x = (getDistance(gx + 1, gy, gz) - getDistance(gx - 1, gy, gz)) * inv_twice_resolution_;
84  gradient_y = (getDistance(gx, gy + 1, gz) - getDistance(gx, gy - 1, gz)) * inv_twice_resolution_;
85  gradient_z = (getDistance(gx, gy, gz + 1) - getDistance(gx, gy, gz - 1)) * inv_twice_resolution_;
86 
87  in_bounds = true;
88  return getDistance(gx, gy, gz);
89 }

240 void DistanceField::getOcTreePoints(const octomap::OcTree* octree, EigenSTL::vector_Vector3d* points)
241 {
242  // lower extent
243  double min_x, min_y, min_z;
244  gridToWorld(0, 0, 0, min_x, min_y, min_z);
245 
246  octomap::point3d bbx_min(min_x, min_y, min_z);
247 
248  int num_x = getXNumCells();
249  int num_y = getYNumCells();
250  int num_z = getZNumCells();
251 
252  // upper extent
253  double max_x, max_y, max_z;
254  gridToWorld(num_x, num_y, num_z, max_x, max_y, max_z);
255 
256  octomap::point3d bbx_max(max_x, max_y, max_z);
257 
258  for (octomap::OcTree::leaf_bbx_iterator it = octree->begin_leafs_bbx(bbx_min, bbx_max), end = octree->end_leafs_bbx();
259  it != end; ++it)
260  {
261  if (octree->isNodeOccupied(*it))
262  {
263  if (it.getSize() <= resolution_)
264  {
265  Eigen::Vector3d point(it.getX(), it.getY(), it.getZ());
266  points->push_back(point);
267  }
268  else
269  {
270  double ceil_val = ceil(it.getSize() / resolution_) * resolution_ / 2.0;
271  for (double x = it.getX() - ceil_val; x <= it.getX() + ceil_val; x += resolution_)
272  {
273  for (double y = it.getY() - ceil_val; y <= it.getY() + ceil_val; y += resolution_)
274  {
275  for (double z = it.getZ() - ceil_val; z <= it.getZ() + ceil_val; z += resolution_)
276  {
277  points->push_back(Eigen::Vector3d(x, y, z));
278  }
279  }
280  }
281  }
282  }
283  }
284 }

293 void DistanceField::moveShapeInField(const shapes::Shape* shape, const Eigen::Isometry3d& old_pose,
294  const Eigen::Isometry3d& new_pose)
295 {
296  if (shape->type == shapes::OCTREE)
297  {
298  RCLCPP_WARN(LOGGER, "Move shape not supported for Octree");
299  return;
300  }
301  bodies::Body* body = bodies::createEmptyBodyFromShapeType(shape->type);
302  body->setDimensionsDirty(shape);
303  body->setPoseDirty(old_pose);
304  body->updateInternalData();
305  EigenSTL::vector_Vector3d old_point_vec;
306  findInternalPointsConvex(*body, resolution_, old_point_vec);
307  body->setPose(new_pose);
308  EigenSTL::vector_Vector3d new_point_vec;
309  findInternalPointsConvex(*body, resolution_, new_point_vec);
310  delete body;
311  updatePointsInField(old_point_vec, new_point_vec);
312 }

344 void DistanceField::getPlaneMarkers(PlaneVisualizationType type, double length, double width, double height,
345  const Eigen::Vector3d& origin, const std::string& frame_id,
346  const rclcpp::Time& stamp, visualization_msgs::msg::Marker& plane_marker) const
347 {
348  plane_marker.header.frame_id = frame_id;
349  plane_marker.header.stamp = stamp;
350  plane_marker.ns = "distance_field_plane";

354  plane_marker.scale.x = resolution_;
355  plane_marker.scale.y = resolution_;
356  plane_marker.scale.z = resolution_;
357  // plane_marker.lifetime = ros::Duration(30.0);

361 
362  double min_x = 0;
363  double max_x = 0;
364  double min_y = 0;
365  double max_y = 0;
366  double min_z = 0;
367  double max_z = 0;
368 
369  switch (type)
370  {
371  case XY_PLANE:
372  min_z = height;
373  max_z = height;
374 
375  min_x = -length / 2.0;
376  max_x = length / 2.0;
377  min_y = -width / 2.0;

392 
393  min_y = -length / 2.0;
394  max_y = length / 2.0;

396  max_z = width / 2.0;
397  break;

400  min_x += origin.x();
401  min_y += origin.y();
402  min_z += origin.z();
403 
404  max_x += origin.x();
405  max_y += origin.y();
406  max_z += origin.z();
407 
408  int min_x_cell = 0;
409  int max_x_cell = 0;
410  int min_y_cell = 0;
411  int max_y_cell = 0;
412  int min_z_cell = 0;
413  int max_z_cell = 0;

415  worldToGrid(min_x, min_y, min_z, min_x_cell, min_y_cell, min_z_cell);

418  for (int x = min_x_cell; x <= max_x_cell; ++x)
419  {
420  for (int y = min_y_cell; y <= max_y_cell; ++y)
421  {
422  for (int z = min_z_cell; z <= max_z_cell; ++z)
423  {

425  {
426  continue;
427  }

429  int last = plane_marker.points.size();
430  plane_marker.points.resize(last + 1);
431  plane_marker.colors.resize(last + 1);

436  plane_marker.points[last].y = vec.y();
437  plane_marker.points[last].z = vec.z();

441  plane_marker.colors[last].g = fmax(fmin(0.05 / fabs(dist), 1.0), 0.0);
442  plane_marker.colors[last].b = fmax(fmin(0.01 / fabs(dist), 1.0), 0.0);
443  }
444  else

446  plane_marker.colors[last].b = fmax(fmin(0.1 / (dist + 0.001), 1.0), 0.0);
447  plane_marker.colors[last].g = fmax(fmin(0.05 / (dist + 0.001), 1.0), 0.0);
448  plane_marker.colors[last].r = fmax(fmin(0.01 / (dist + 0.001), 1.0), 0.0);
449  }
450  }
451  }
452  }

470 void DistanceField::getProjectionPlanes(const std::string& frame_id, const rclcpp::Time& stamp, double max_dist,
471  visualization_msgs::msg::Marker& marker) const
472 {
473  int max_x_cell = getXNumCells();
474  int max_y_cell = getYNumCells();
475  int max_z_cell = getZNumCells();
476 
477  double* x_projection = new double[max_y_cell * max_z_cell];
478  double* y_projection = new double[max_z_cell * max_x_cell];
479  double* z_projection = new double[max_x_cell * max_y_cell];
480  double initial_val = sqrt(INT_MAX);
481 
482  // Initialize
483  for (int y = 0; y < max_y_cell; y++)
484  for (int x = 0; x < max_x_cell; x++)
485  z_projection[x + y * max_x_cell] = initial_val;
486 
487  for (int z = 0; z < max_z_cell; z++)
488  for (int y = 0; y < max_y_cell; y++)
489  x_projection[y + z * max_y_cell] = initial_val;
490 
491  for (int z = 0; z < max_z_cell; z++)
492  for (int x = 0; x < max_x_cell; x++)
493  y_projection[x + z * max_x_cell] = initial_val;
494 
495  // Calculate projections
496  for (int z = 0; z < max_z_cell; z++)
497  {
498  for (int y = 0; y < max_y_cell; y++)
499  {
500  for (int x = 0; x < max_x_cell; x++)
501  {
502  double dist = getDistance(x, y, z);
503  z_projection[x + y * max_x_cell] = std::min(dist, z_projection[x + y * max_x_cell]);
504  x_projection[y + z * max_y_cell] = std::min(dist, x_projection[y + z * max_y_cell]);
505  y_projection[x + z * max_x_cell] = std::min(dist, y_projection[x + z * max_x_cell]);
506  }
507  }
508  }
509 
510  // Make markers
511  marker.points.clear();
512  marker.header.frame_id = frame_id;
513  marker.header.stamp = stamp;
514  marker.ns = "distance_field_projection_plane";
515  marker.id = 1;
516  marker.type = visualization_msgs::msg::Marker::CUBE_LIST;
517  marker.action = visualization_msgs::msg::Marker::MODIFY;
518  marker.scale.x = getResolution();
519  marker.scale.y = getResolution();
520  marker.scale.z = getResolution();
521  marker.color.a = 1.0;
522  // marker.lifetime = ros::Duration(30.0);
523 
524  int x, y, z;
525  int index = 0;
526  marker.points.resize(max_x_cell * max_y_cell + max_y_cell * max_z_cell + max_z_cell * max_x_cell);
527  marker.colors.resize(max_x_cell * max_y_cell + max_y_cell * max_z_cell + max_z_cell * max_x_cell);
528 
529  z = 0;
530  for (y = 0; y < max_y_cell; y++)
531  {
532  for (x = 0; x < max_x_cell; x++)
533  {
534  double dist = z_projection[x + y * max_x_cell];
535  setPoint(x, y, z, dist, marker.points[index], marker.colors[index], max_dist);
536  index++;
537  }
538  }
539 
540  x = 0;
541  for (z = 0; z < max_z_cell; z++)
542  {
543  for (y = 0; y < max_y_cell; y++)
544  {
545  double dist = x_projection[y + z * max_y_cell];
546  setPoint(x, y, z, dist, marker.points[index], marker.colors[index], max_dist);
547  index++;
548  }
549  }
550 
551  y = 0;
552  for (z = 0; z < max_z_cell; z++)
553  {
554  for (x = 0; x < max_x_cell; x++)
555  {
556  double dist = y_projection[x + z * max_x_cell];
557  setPoint(x, y, z, dist, marker.points[index], marker.colors[index], max_dist);
558  index++;
559  }
560  }
561 
562  if (x_projection)
563  delete[] x_projection;
564  if (y_projection)
565  delete[] y_projection;
566  if (z_projection)
567  delete[] z_projection;
568 }
