313 inline void EnvironmentChain3D::convertJointAnglesToCoord(const std::vector<double>& angle, std::vector<int>& coord)
314 {
315  coord.resize(angle.size());
316  for (unsigned int i = 0; i < angle.size(); i++)
317  {
318  // NOTE: Added 3/1/09
319  double pos_angle = angle[i];
320  if (pos_angle < 0.0)
321  pos_angle += 2 * M_PI;
322 
323  coord[i] = (int)((pos_angle + angle_discretization_ * 0.5) / angle_discretization_);
324  }
325 }
