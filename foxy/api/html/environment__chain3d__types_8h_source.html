47 static inline unsigned int intHash(unsigned int key)
48 {
49  key += (key << 12);
50  key ^= (key >> 22);
51  key += (key << 4);
52  key ^= (key >> 9);
53  key += (key << 10);
54  key ^= (key >> 2);
55  key += (key << 7);
56  key ^= (key >> 12);
57  return key;
58 }

106  unsigned int getHashBin(const std::vector<int>& coord)
107  {
108  unsigned int val = 0;
109 
110  for (size_t i = 0; i < coord.size(); i++)
111  val += intHash(coord[i]) << i;
112 
113  return intHash(val) & (hash_table_size_ - 1);
114  }

116  EnvChain3DHashEntry* addHashEntry(const std::vector<int>& coord, const std::vector<double>& angles,
117  const int (&xyz)[3], int action)
118  {
119  EnvChain3DHashEntry* new_hash_entry = new EnvChain3DHashEntry();
120  new_hash_entry->stateID = state_ID_to_coord_table_.size();
121  new_hash_entry->coord = coord;
122  new_hash_entry->angles = angles;
123  memcpy(new_hash_entry->xyz, xyz, sizeof(int) * 3);
124  new_hash_entry->action = action;
125  state_ID_to_coord_table_.push_back(new_hash_entry);
126  unsigned int bin = getHashBin(coord);
127  coord_to_state_ID_table_[bin].push_back(new_hash_entry);
128 
129  // have to do for DiscreteSpaceInformation
130  // insert into and initialize the mappings
131  int* entry = new int[NUMOFINDICES_STATEID2IND];
132  memset(entry, -1, NUMOFINDICES_STATEID2IND * sizeof(int));
133  state_ID_to_index_mapping_.push_back(entry);
134  if (new_hash_entry->stateID != (int)state_ID_to_index_mapping_.size() - 1)
135  {
136  ROS_ERROR_STREAM("Size mismatch between state mappings " << new_hash_entry->stateID << " "
137  << state_ID_to_index_mapping_.size());

184 class JointMotionWrapper
185 {
186 public:
187  JointMotionWrapper(const planning_models::RobotModel::JointModel* joint_model) : joint_model_(joint_model)
188  {
189  std::vector<moveit_msgs::msg::JointLimits> limits = joint_model->getLimits();
190  joint_limit_ = limits.front();
191  }
192 
193  bool getSuccessorValue(double start, double delta, double& end)
194  {
195  end = start + delta;
196  if (joint_limit_.has_position_limits)
197  {
198  if (end < joint_limit_.min_position)
199  {
200  if (fabs(start - joint_limit_.min_position) > std::numeric_limits<double>::epsilon())
201  {
202  // start not at min limit, so peg the end position
203  end = joint_limit_.min_position;
204  }
205  else
206  {
207  // start already at min limit
208  return false;
209  }
210  }
211  else if (end > joint_limit_.max_position)
212  {
213  if (fabs(start - joint_limit_.max_position) > std::numeric_limits<double>::epsilon())
214  {
215  // start not at max limit, so peg the end position
216  end = joint_limit_.max_position;
217  }
218  else
219  {
220  // start already at max limit
221  return false;
222  }
223  }
224  }
225  else
226  {
227  end = normalizeAngle(end);
228  }
229  return true;
230  }
231 
232  bool canGetCloser(double start, double goal, double delta)
233  {
234  double start_dist = getDoubleDistance(start, goal);
235  double plus_value = 0.0;
236  double minus_value = 0.0;
237  if (getSuccessorValue(start, delta, plus_value))
238  {
239  double succ_dist = getDoubleDistance(plus_value, goal);
240  if (succ_dist < start_dist)
241  {
242  return true;
243  }
244  }
245  if (getSuccessorValue(start, -delta, minus_value))
246  {
247  double succ_dist = getDoubleDistance(minus_value, goal);
248  if (succ_dist < start_dist)
249  {
250  return true;
251  }
252  }
253  return false;
254  }
255 
256  double getDoubleDistance(double start, double end)
257  {
258  if (joint_limit_.has_position_limits)
259  {
260  return fabs(end - start);
261  }
262  else
263  {
264  return fabs(planning_models::shortestAngularDistance(start, end));
265  }
266  }
267  int getIntegerDistance(double start, double end, double delta)
268  {
269  double val;
270  if (joint_limit_.has_position_limits)
271  {
272  val = fabs(end - start) / delta;
273  }
274  else
275  {
276  val = fabs(planning_models::shortestAngularDistance(start, end)) / delta;
277  }
278  if (val - floor(val) > std::numeric_limits<double>::epsilon())
279  {
280  return ceil(val);
281  }
282  else
283  {
284  return floor(val);
285  }
286  }
287 
288 protected:
289  const planning_models::RobotModel::JointModel* joint_model_;
290  moveit_msgs::msg::JointLimits joint_limit_;
291 };
