58 bool collisionCallback(fcl::CollisionObjectd* o1, fcl::CollisionObjectd* o2, void* data)
59 {
60  CollisionData* cdata = reinterpret_cast<CollisionData*>(data);
61  if (cdata->done_)
62  return true;
63  const CollisionGeometryData* cd1 = static_cast<const CollisionGeometryData*>(o1->collisionGeometry()->getUserData());
64  const CollisionGeometryData* cd2 = static_cast<const CollisionGeometryData*>(o2->collisionGeometry()->getUserData());
65 
66  // do not collision check geoms part of the same object / link / attached body
67  if (cd1->sameObject(*cd2))
68  return false;
69 
70  // If active components are specified
71  if (cdata->active_components_only_)
72  {
73  const moveit::core::LinkModel* l1 =
74  cd1->type == BodyTypes::ROBOT_LINK ?
75  cd1->ptr.link :
76  (cd1->type == BodyTypes::ROBOT_ATTACHED ? cd1->ptr.ab->getAttachedLink() : nullptr);
77  const moveit::core::LinkModel* l2 =
78  cd2->type == BodyTypes::ROBOT_LINK ?
79  cd2->ptr.link :
80  (cd2->type == BodyTypes::ROBOT_ATTACHED ? cd2->ptr.ab->getAttachedLink() : nullptr);
81 
82  // If neither of the involved components is active
83  if ((!l1 || cdata->active_components_only_->find(l1) == cdata->active_components_only_->end()) &&
84  (!l2 || cdata->active_components_only_->find(l2) == cdata->active_components_only_->end()))
85  return false;
86  }
87 
88  // use the collision matrix (if any) to avoid certain collision checks
89  DecideContactFn dcf;
90  bool always_allow_collision = false;
91  if (cdata->acm_)
92  {
93  AllowedCollision::Type type;
94  bool found = cdata->acm_->getAllowedCollision(cd1->getID(), cd2->getID(), type);
95  if (found)
96  {
97  // if we have an entry in the collision matrix, we read it
98  if (type == AllowedCollision::ALWAYS)
99  {
100  always_allow_collision = true;
101  if (cdata->req_->verbose)
102  RCLCPP_DEBUG(LOGGER,
103  "Collision between '%s' (type '%s') and '%s' (type '%s') is always allowed. "
104  "No contacts are computed.",
105  cd1->getID().c_str(), cd1->getTypeString().c_str(), cd2->getID().c_str(),
106  cd2->getTypeString().c_str());
107  }
108  else if (type == AllowedCollision::CONDITIONAL)
109  {
110  cdata->acm_->getAllowedCollision(cd1->getID(), cd2->getID(), dcf);
111  if (cdata->req_->verbose)
112  RCLCPP_DEBUG(LOGGER, "Collision between '%s' and '%s' is conditionally allowed", cd1->getID().c_str(),
113  cd2->getID().c_str());
114  }
115  }
116  }
117 
118  // check if a link is touching an attached object
119  if (cd1->type == BodyTypes::ROBOT_LINK && cd2->type == BodyTypes::ROBOT_ATTACHED)
120  {
121  const std::set<std::string>& tl = cd2->ptr.ab->getTouchLinks();
122  if (tl.find(cd1->getID()) != tl.end())
123  {
124  always_allow_collision = true;
125  if (cdata->req_->verbose)
126  RCLCPP_DEBUG(LOGGER, "Robot link '%s' is allowed to touch attached object '%s'. No contacts are computed.",
127  cd1->getID().c_str(), cd2->getID().c_str());
128  }
129  }
130  else if (cd2->type == BodyTypes::ROBOT_LINK && cd1->type == BodyTypes::ROBOT_ATTACHED)
131  {
132  const std::set<std::string>& tl = cd1->ptr.ab->getTouchLinks();
133  if (tl.find(cd2->getID()) != tl.end())
134  {
135  always_allow_collision = true;
136  if (cdata->req_->verbose)
137  RCLCPP_DEBUG(LOGGER, "Robot link '%s' is allowed to touch attached object '%s'. No contacts are computed.",
138  cd2->getID().c_str(), cd1->getID().c_str());
139  }
140  }
141  // bodies attached to the same link should not collide
142  if (cd1->type == BodyTypes::ROBOT_ATTACHED && cd2->type == BodyTypes::ROBOT_ATTACHED)
143  {
144  if (cd1->ptr.ab->getAttachedLink() == cd2->ptr.ab->getAttachedLink())
145  always_allow_collision = true;
146  }
147 
148  // if collisions are always allowed, we are done
149  if (always_allow_collision)
150  return false;
151 
152  if (cdata->req_->verbose)
153  RCLCPP_DEBUG(LOGGER, "Actually checking collisions between %s and %s", cd1->getID().c_str(), cd2->getID().c_str());
154 
155  // see if we need to compute a contact
156  std::size_t want_contact_count = 0;
157  if (cdata->req_->contacts)
158  if (cdata->res_->contact_count < cdata->req_->max_contacts)
159  {
160  std::size_t have;
161  if (cd1->getID() < cd2->getID())
162  {
163  std::pair<std::string, std::string> cp(cd1->getID(), cd2->getID());
164  have = cdata->res_->contacts.find(cp) != cdata->res_->contacts.end() ? cdata->res_->contacts[cp].size() : 0;
165  }
166  else
167  {
168  std::pair<std::string, std::string> cp(cd2->getID(), cd1->getID());
169  have = cdata->res_->contacts.find(cp) != cdata->res_->contacts.end() ? cdata->res_->contacts[cp].size() : 0;
170  }
171  if (have < cdata->req_->max_contacts_per_pair)
172  want_contact_count =
173  std::min(cdata->req_->max_contacts_per_pair - have, cdata->req_->max_contacts - cdata->res_->contact_count);
174  }
175 
176  if (dcf)
177  {
178  // if we have a decider for allowed contacts, we need to look at all the contacts
179  bool enable_cost = cdata->req_->cost;
180  std::size_t num_max_cost_sources = cdata->req_->max_cost_sources;
181  bool enable_contact = true;
182  fcl::CollisionResultd col_result;
183  int num_contacts = fcl::collide(o1, o2,
184  fcl::CollisionRequestd(std::numeric_limits<size_t>::max(), enable_contact,
185  num_max_cost_sources, enable_cost),
186  col_result);
187  if (num_contacts > 0)
188  {
189  if (cdata->req_->verbose)
190  RCLCPP_INFO(LOGGER,
191  "Found %d contacts between '%s' and '%s'. "
192  "These contacts will be evaluated to check if they are accepted or not",
193  num_contacts, cd1->getID().c_str(), cd2->getID().c_str());
194  Contact c;
195  const std::pair<std::string, std::string>& pc = cd1->getID() < cd2->getID() ?
196  std::make_pair(cd1->getID(), cd2->getID()) :
197  std::make_pair(cd2->getID(), cd1->getID());
198  for (int i = 0; i < num_contacts; ++i)
199  {
200  fcl2contact(col_result.getContact(i), c);
201  // if the contact is not allowed, we have a collision
202  if (!dcf(c))
203  {
204  // store the contact, if it is needed
205  if (want_contact_count > 0)
206  {
207  --want_contact_count;
208  cdata->res_->contacts[pc].push_back(c);
209  cdata->res_->contact_count++;
210  if (cdata->req_->verbose)
211  RCLCPP_INFO(LOGGER, "Found unacceptable contact between '%s' and '%s'. Contact was stored.",
212  cd1->getID().c_str(), cd2->getID().c_str());
213  }
214  else if (cdata->req_->verbose)
215  RCLCPP_INFO(LOGGER,
216  "Found unacceptable contact between '%s' (type '%s') and '%s' "
217  "(type '%s'). Contact was stored.",
218  cd1->getID().c_str(), cd1->getTypeString().c_str(), cd2->getID().c_str(),
219  cd2->getTypeString().c_str());
220  cdata->res_->collision = true;
221  if (want_contact_count == 0)
222  break;
223  }
224  }
225  }
226 
227  if (enable_cost)
228  {
229  std::vector<fcl::CostSourced> cost_sources;
230  col_result.getCostSources(cost_sources);
231 
232  CostSource cs;
233  for (auto& cost_source : cost_sources)
234  {
235  fcl2costsource(cost_source, cs);
236  cdata->res_->cost_sources.insert(cs);
237  while (cdata->res_->cost_sources.size() > cdata->req_->max_cost_sources)
238  cdata->res_->cost_sources.erase(--cdata->res_->cost_sources.end());
239  }
240  }
241  }
242  else
243  {
244  if (want_contact_count > 0)
245  {
246  // otherwise, we need to compute more things
247  bool enable_cost = cdata->req_->cost;
248  std::size_t num_max_cost_sources = cdata->req_->max_cost_sources;
249  bool enable_contact = true;
250 
251  fcl::CollisionResultd col_result;
252  int num_contacts =
253  fcl::collide(o1, o2,
254  fcl::CollisionRequestd(want_contact_count, enable_contact, num_max_cost_sources, enable_cost),
255  col_result);
256  if (num_contacts > 0)
257  {
258  int num_contacts_initial = num_contacts;
259 
260  // make sure we don't get more contacts than we want
261  if (want_contact_count >= (std::size_t)num_contacts)
262  want_contact_count -= num_contacts;
263  else
264  {
265  num_contacts = want_contact_count;
266  want_contact_count = 0;
267  }
268 
269  if (cdata->req_->verbose)
270  RCLCPP_INFO(LOGGER,
271  "Found %d contacts between '%s' (type '%s') and '%s' (type '%s'), "
272  "which constitute a collision. %d contacts will be stored",
273  num_contacts_initial, cd1->getID().c_str(), cd1->getTypeString().c_str(), cd2->getID().c_str(),
274  cd2->getTypeString().c_str(), num_contacts);
275 
276  const std::pair<std::string, std::string>& pc = cd1->getID() < cd2->getID() ?
277  std::make_pair(cd1->getID(), cd2->getID()) :
278  std::make_pair(cd2->getID(), cd1->getID());
279  cdata->res_->collision = true;
280  for (int i = 0; i < num_contacts; ++i)
281  {
282  Contact c;
283  fcl2contact(col_result.getContact(i), c);
284  cdata->res_->contacts[pc].push_back(c);
285  cdata->res_->contact_count++;
286  }
287  }
288 
289  if (enable_cost)
290  {
291  std::vector<fcl::CostSourced> cost_sources;
292  col_result.getCostSources(cost_sources);
293 
294  CostSource cs;
295  for (auto& cost_source : cost_sources)
296  {
297  fcl2costsource(cost_source, cs);
298  cdata->res_->cost_sources.insert(cs);
299  while (cdata->res_->cost_sources.size() > cdata->req_->max_cost_sources)
300  cdata->res_->cost_sources.erase(--cdata->res_->cost_sources.end());
301  }
302  }
303  }
304  else
305  {
306  bool enable_cost = cdata->req_->cost;
307  std::size_t num_max_cost_sources = cdata->req_->max_cost_sources;
308  bool enable_contact = false;
309  fcl::CollisionResultd col_result;
310  int num_contacts = fcl::collide(
311  o1, o2, fcl::CollisionRequestd(1, enable_contact, num_max_cost_sources, enable_cost), col_result);
312  if (num_contacts > 0)
313  {
314  cdata->res_->collision = true;
315  if (cdata->req_->verbose)
316  RCLCPP_INFO(LOGGER,
317  "Found a contact between '%s' (type '%s') and '%s' (type '%s'), "
318  "which constitutes a collision. "
319  "Contact information is not stored.",
320  cd1->getID().c_str(), cd1->getTypeString().c_str(), cd2->getID().c_str(),
321  cd2->getTypeString().c_str());
322  }
323 
324  if (enable_cost)
325  {
326  std::vector<fcl::CostSourced> cost_sources;
327  col_result.getCostSources(cost_sources);
328 
329  CostSource cs;
330  for (auto& cost_source : cost_sources)
331  {
332  fcl2costsource(cost_source, cs);
333  cdata->res_->cost_sources.insert(cs);
334  while (cdata->res_->cost_sources.size() > cdata->req_->max_cost_sources)
335  cdata->res_->cost_sources.erase(--cdata->res_->cost_sources.end());
336  }
337  }
338  }
339  }
340 
341  if (cdata->res_->collision)
342  if (!cdata->req_->contacts || cdata->res_->contact_count >= cdata->req_->max_contacts)
343  {
344  if (!cdata->req_->cost)
345  cdata->done_ = true;
346  if (cdata->req_->verbose)
347  RCLCPP_INFO(LOGGER,
348  "Collision checking is considered complete (collision was found and %u contacts are stored)",
349  (unsigned int)cdata->res_->contact_count);
350  }
351 
352  if (!cdata->done_ && cdata->req_->is_done)
353  {
354  cdata->done_ = cdata->req_->is_done(*cdata->res_);
355  if (cdata->done_ && cdata->req_->verbose)
356  RCLCPP_INFO(LOGGER,
357  "Collision checking is considered complete due to external callback. "
358  "%s was found. %u contacts are stored.",
359  cdata->res_->collision ? "Collision" : "No collision", (unsigned int)cdata->res_->contact_count);
360  }
361 
362  return cdata->done_;
363 }

397  }
398  }

401  // executed (this is only removal of expired entries)

404 
406  unsigned int clean_count_;
407 };

409 bool distanceCallback(fcl::CollisionObjectd* o1, fcl::CollisionObjectd* o2, void* data, double& min_dist)
410 {
411  DistanceData* cdata = reinterpret_cast<DistanceData*>(data);
412 
413  const CollisionGeometryData* cd1 = static_cast<const CollisionGeometryData*>(o1->collisionGeometry()->getUserData());
414  const CollisionGeometryData* cd2 = static_cast<const CollisionGeometryData*>(o2->collisionGeometry()->getUserData());
415 
416  // do not distance check for geoms part of the same object / link / attached body
417  if (cd1->sameObject(*cd2))
418  return false;
419 
420  // If active components are specified
421  if (cdata->req->active_components_only)
422  {
423  const moveit::core::LinkModel* l1 =
424  cd1->type == BodyTypes::ROBOT_LINK ?
425  cd1->ptr.link :
426  (cd1->type == BodyTypes::ROBOT_ATTACHED ? cd1->ptr.ab->getAttachedLink() : nullptr);
427  const moveit::core::LinkModel* l2 =
428  cd2->type == BodyTypes::ROBOT_LINK ?
429  cd2->ptr.link :
430  (cd2->type == BodyTypes::ROBOT_ATTACHED ? cd2->ptr.ab->getAttachedLink() : nullptr);
431 
432  // If neither of the involved components is active
433  if ((!l1 || cdata->req->active_components_only->find(l1) == cdata->req->active_components_only->end()) &&
434  (!l2 || cdata->req->active_components_only->find(l2) == cdata->req->active_components_only->end()))
435  {
436  return false;
437  }
438  }
439 
440  // use the collision matrix (if any) to avoid certain distance checks
441  bool always_allow_collision = false;
442  if (cdata->req->acm)
443  {
444  AllowedCollision::Type type;
445  bool found = cdata->req->acm->getAllowedCollision(cd1->getID(), cd2->getID(), type);
446  if (found)
447  {
448  // if we have an entry in the collision matrix, we read it
449  if (type == AllowedCollision::ALWAYS)
450  {
451  always_allow_collision = true;
452  if (cdata->req->verbose)
453  RCLCPP_DEBUG(LOGGER, "Collision between '%s' and '%s' is always allowed. No distances are computed.",
454  cd1->getID().c_str(), cd2->getID().c_str());
455  }
456  }
457  }
458 
459  // check if a link is touching an attached object
460  if (cd1->type == BodyTypes::ROBOT_LINK && cd2->type == BodyTypes::ROBOT_ATTACHED)
461  {
462  const std::set<std::string>& tl = cd2->ptr.ab->getTouchLinks();
463  if (tl.find(cd1->getID()) != tl.end())
464  {
465  always_allow_collision = true;
466  if (cdata->req->verbose)
467  RCLCPP_DEBUG(LOGGER, "Robot link '%s' is allowed to touch attached object '%s'. No distances are computed.",
468  cd1->getID().c_str(), cd2->getID().c_str());
469  }
470  }
471  else if (cd2->type == BodyTypes::ROBOT_LINK && cd1->type == BodyTypes::ROBOT_ATTACHED)
472  {
473  const std::set<std::string>& tl = cd1->ptr.ab->getTouchLinks();
474  if (tl.find(cd2->getID()) != tl.end())
475  {
476  always_allow_collision = true;
477  if (cdata->req->verbose)
478  RCLCPP_DEBUG(LOGGER, "Robot link '%s' is allowed to touch attached object '%s'. No distances are computed.",
479  cd2->getID().c_str(), cd1->getID().c_str());
480  }
481  }
482 
483  if (always_allow_collision)
484  {
485  return false;
486  }
487  if (cdata->req->verbose)
488  RCLCPP_DEBUG(LOGGER, "Actually checking collisions between %s and %s", cd1->getID().c_str(), cd2->getID().c_str());
489 
490  double dist_threshold = cdata->req->distance_threshold;
491 
492  const std::pair<const std::string&, const std::string&> pc =
493  cd1->getID() < cd2->getID() ? std::make_pair(std::cref(cd1->getID()), std::cref(cd2->getID())) :
494  std::make_pair(std::cref(cd2->getID()), std::cref(cd1->getID()));
495 
496  DistanceMap::iterator it = cdata->res->distances.find(pc);
497 
498  // GLOBAL search: for efficiency, distance_threshold starts at the smallest distance between any pairs found so far
499  if (cdata->req->type == DistanceRequestType::GLOBAL)
500  {
501  dist_threshold = cdata->res->minimum_distance.distance;
502  }
503  // Check if a distance between this pair has been found yet. Decrease threshold_distance if so, to narrow the search
504  else if (it != cdata->res->distances.end())
505  {
506  if (cdata->req->type == DistanceRequestType::LIMITED)
507  {
508  // If at the limit for a given pair just return
509  if (it->second.size() >= cdata->req->max_contacts_per_body)
510  {
511  return cdata->done;
512  }
513  }
514  else if (cdata->req->type == DistanceRequestType::SINGLE)
515  {
516  dist_threshold = it->second[0].distance;
517  }
518  }
519 
520  fcl::DistanceResultd fcl_result;
521  fcl_result.min_distance = dist_threshold;
522  // fcl::distance segfaults when given an octree with a null root pointer (using FCL 0.6.1)
523  if ((o1->getObjectType() == fcl::OT_OCTREE &&
524  !std::static_pointer_cast<const fcl::OcTreed>(o1->collisionGeometry())->getRoot()) ||
525  (o2->getObjectType() == fcl::OT_OCTREE &&
526  !std::static_pointer_cast<const fcl::OcTreed>(o2->collisionGeometry())->getRoot()))
527  {
528  return false;
529  }
530  double distance = fcl::distance(o1, o2, fcl::DistanceRequestd(cdata->req->enable_nearest_points), fcl_result);
531 
532  // Check if either object is already in the map. If not add it or if present
533  // check to see if the new distance is closer. If closer remove the existing
534  // one and add the new distance information.
535  if (distance < dist_threshold)
536  {
537  // thread_local storage makes this variable persistent. We do not clear it at every iteration because all members
538  // get overwritten.
539  thread_local DistanceResultsData dist_result;
540  dist_result.distance = fcl_result.min_distance;
541 
542  // Careful here: Get the collision geometry data again, since FCL might
543  // swap o1 and o2 in the result.
544  const CollisionGeometryData* res_cd1 = static_cast<const CollisionGeometryData*>(fcl_result.o1->getUserData());
545  const CollisionGeometryData* res_cd2 = static_cast<const CollisionGeometryData*>(fcl_result.o2->getUserData());
546 
547 #if (MOVEIT_FCL_VERSION >= FCL_VERSION_CHECK(0, 6, 0))
548  dist_result.nearest_points[0] = fcl_result.nearest_points[0];
549  dist_result.nearest_points[1] = fcl_result.nearest_points[1];
550 #else
551  dist_result.nearest_points[0] = Eigen::Map<const Eigen::Vector3d>(fcl_result.nearest_points[0].data.vs);
552  dist_result.nearest_points[1] = Eigen::Map<const Eigen::Vector3d>(fcl_result.nearest_points[1].data.vs);
553 #endif
554  dist_result.link_names[0] = res_cd1->getID();
555  dist_result.link_names[1] = res_cd2->getID();
556  dist_result.body_types[0] = res_cd1->type;
557  dist_result.body_types[1] = res_cd2->type;
558  if (cdata->req->enable_nearest_points)
559  {
560  dist_result.normal = (dist_result.nearest_points[1] - dist_result.nearest_points[0]).normalized();
561  }
562 
563  if (distance <= 0 && cdata->req->enable_signed_distance)
564  {
565  dist_result.nearest_points[0].setZero();
566  dist_result.nearest_points[1].setZero();
567  dist_result.normal.setZero();
568 
569  fcl::CollisionRequestd coll_req;
570  thread_local fcl::CollisionResultd coll_res;
571  coll_res.clear(); // thread_local storage makes the variable persistent. Ensure that it is cleared!
572  coll_req.enable_contact = true;
573  coll_req.num_max_contacts = 200;
574  std::size_t contacts = fcl::collide(o1, o2, coll_req, coll_res);
575  if (contacts > 0)
576  {
577  double max_dist = 0;
578  int max_index = 0;
579  for (std::size_t i = 0; i < contacts; ++i)
580  {
581  const fcl::Contactd& contact = coll_res.getContact(i);
582  if (contact.penetration_depth > max_dist)
583  {
584  max_dist = contact.penetration_depth;
585  max_index = i;
586  }
587  }
588 
589  const fcl::Contactd& contact = coll_res.getContact(max_index);
590  dist_result.distance = -contact.penetration_depth;
591 
592 #if (MOVEIT_FCL_VERSION >= FCL_VERSION_CHECK(0, 6, 0))
593  dist_result.nearest_points[0] = contact.pos;
594  dist_result.nearest_points[1] = contact.pos;
595 #else
596  dist_result.nearest_points[0] = Eigen::Map<const Eigen::Vector3d>(contact.pos.data.vs);
597  dist_result.nearest_points[1] = Eigen::Map<const Eigen::Vector3d>(contact.pos.data.vs);
598 #endif
599 
600  if (cdata->req->enable_nearest_points)
601  {
602 #if (MOVEIT_FCL_VERSION >= FCL_VERSION_CHECK(0, 6, 0))
603  Eigen::Vector3d normal(contact.normal);
604 #else
605  Eigen::Vector3d normal(contact.normal.data.vs);
606 #endif
607 
608  // Check order of o1/o2 again, we might need to flip the normal
609  if (contact.o1 == o1->collisionGeometry().get())
610  dist_result.normal = normal;
611  else
612  dist_result.normal = -normal;
613  }
614  }
615  }
616 
617  if (dist_result.distance < cdata->res->minimum_distance.distance)
618  {
619  cdata->res->minimum_distance = dist_result;
620  }
621 
622  if (dist_result.distance <= 0)
623  {
624  cdata->res->collision = true;
625  }
626 
627  if (cdata->req->type != DistanceRequestType::GLOBAL)
628  {
629  if (it == cdata->res->distances.end())
630  {
631  std::vector<DistanceResultsData> data;
632  data.reserve(cdata->req->type == DistanceRequestType::SINGLE ? 1 : cdata->req->max_contacts_per_body);
633  data.push_back(dist_result);
634  cdata->res->distances.insert(std::make_pair(pc, data));
635  }
636  else
637  {
638  if (cdata->req->type == DistanceRequestType::ALL)
639  {
640  it->second.push_back(dist_result);
641  }
642  else if (cdata->req->type == DistanceRequestType::SINGLE)
643  {
644  if (dist_result.distance < it->second[0].distance)
645  it->second[0] = dist_result;
646  }
647  else if (cdata->req->type == DistanceRequestType::LIMITED)
648  {
649  assert(it->second.size() < cdata->req->max_contacts_per_body);
650  it->second.push_back(dist_result);
651  }
652  }
653  }
654 
655  if (!cdata->req->enable_signed_distance && cdata->res->collision)
656  {
657  cdata->done = true;
658  }
659  }
660 
661  return cdata->done;
662 }

664 /* Templated function to get a different cache for each of the template arguments combinations.
665  *
666  * The returned cache is a quasi-singleton for each thread as it is created \e thread_local. */
667 template <typename BV, typename T>
668 FCLShapeCache& GetShapeCache()
669 {
670  /* The cache is created thread_local, that is each thread calling
671  * this quasi-singleton function will get its own instance. Once
672  * the thread joins/exits, the cache gets deleted.
673  * Reasoning is that multi-threaded planners (eg OMPL) or user-code
674  * will often need to do collision checks with the same object
675  * simultaneously (especially true for attached objects). Having only
676  * one global cache leads to many cache misses. Also as the cache can
677  * only be accessed by one thread we don't need any locking.
678  */
679  static thread_local FCLShapeCache cache;
680  return cache;
681 }

711 
712  FCLShapeCache& cache = GetShapeCache<BV, T>();

714  shapes::ShapeConstWeakPtr wptr(shape);
715  {
716  ShapeMap::const_iterator cache_it = cache.map_.find(wptr);
717  if (cache_it != cache.map_.end())
718  {
719  if (cache_it->second->collision_geometry_data_->ptr.raw == data)
720  {
721  // RCLCPP_DEBUG(LOGGER, "Collision data structures for object %s retrieved from
722  // cache.",
723  // cache_it->second->collision_geometry_data_->getID().c_str());
724  return cache_it->second;
725  }
726  else if (cache_it->second.unique())
727  {
728  const_cast<FCLGeometry*>(cache_it->second.get())->updateCollisionGeometryData(data, shape_index, false);
729  // RCLCPP_DEBUG(LOGGER, "Collision data structures for object %s retrieved from
730  // cache after updating
731  // the source
732  // object.", cache_it->second->collision_geometry_data_->getID().c_str());
733  return cache_it->second;
734  }
735  }
736  }
737 
738  // attached objects could have previously been World::Object; we try to move them
739  // from their old cache to the new one, if possible. the code is not pretty, but should help
740  // when we attach/detach objects that are in the world
741  if (IfSameType<T, moveit::core::AttachedBody>::VALUE == 1)
742  {
743  // get the cache that corresponds to objects; maybe this attached object used to be a world object
744  FCLShapeCache& othercache = GetShapeCache<BV, World::Object>();
745 
746  // attached bodies could be just moved from the environment.
747  auto cache_it = othercache.map_.find(wptr);
748  if (cache_it != othercache.map_.end())
749  {
750  if (cache_it->second.unique())
751  {
752  // remove from old cache
753  FCLGeometryConstPtr obj_cache = cache_it->second;
754  othercache.map_.erase(cache_it);
755  // update the CollisionGeometryData; nobody has a pointer to this, so we can safely modify it
756  const_cast<FCLGeometry*>(obj_cache.get())->updateCollisionGeometryData(data, shape_index, true);
757 
758  // RCLCPP_DEBUG(LOGGER, "Collision data structures for attached body %s retrieved
759  // from the cache for
760  // world objects.",
761  // obj_cache->collision_geometry_data_->getID().c_str());
762 
763  // add to the new cache
764  cache.map_[wptr] = obj_cache;
765  cache.bumpUseCount();
766  return obj_cache;
767  }
768  }
769  }
770  else
771  // world objects could have previously been attached objects; we try to move them
772  // from their old cache to the new one, if possible. the code is not pretty, but should help
773  // when we attach/detach objects that are in the world
774  if (IfSameType<T, World::Object>::VALUE == 1)
775  {
776  // get the cache that corresponds to objects; maybe this attached object used to be a world object
777  FCLShapeCache& othercache = GetShapeCache<BV, moveit::core::AttachedBody>();
778 
779  // attached bodies could be just moved from the environment.
780  auto cache_it = othercache.map_.find(wptr);
781  if (cache_it != othercache.map_.end())
782  {
783  if (cache_it->second.unique())
784  {
785  // remove from old cache
786  FCLGeometryConstPtr obj_cache = cache_it->second;
787  othercache.map_.erase(cache_it);
788 
789  // update the CollisionGeometryData; nobody has a pointer to this, so we can safely modify it
790  const_cast<FCLGeometry*>(obj_cache.get())->updateCollisionGeometryData(data, shape_index, true);
791 
792  // RCLCPP_DEBUG(LOGGER, "Collision data structures for world object %s retrieved
793  // from the cache for
794  // attached
795  // bodies.",
796  // obj_cache->collision_geometry_data_->getID().c_str());
797 
798  // add to the new cache
799  cache.map_[wptr] = obj_cache;
800  cache.bumpUseCount();
801  return obj_cache;
802  }
803  }
804  }
805 
806  fcl::CollisionGeometryd* cg_g = nullptr;
807  // handle cases individually
808  switch (shape->type)
809  {
810  case shapes::PLANE:
811  {
812  const shapes::Plane* p = static_cast<const shapes::Plane*>(shape.get());
813  cg_g = new fcl::Planed(p->a, p->b, p->c, p->d);
814  }
815  break;
816  case shapes::SPHERE:
817  {
818  const shapes::Sphere* s = static_cast<const shapes::Sphere*>(shape.get());
819  cg_g = new fcl::Sphered(s->radius);
820  }
821  break;
822  case shapes::BOX:
823  {
824  const shapes::Box* s = static_cast<const shapes::Box*>(shape.get());
825  const double* size = s->size;
826  cg_g = new fcl::Boxd(size[0], size[1], size[2]);
827  }
828  break;
829  case shapes::CYLINDER:
830  {
831  const shapes::Cylinder* s = static_cast<const shapes::Cylinder*>(shape.get());
832  cg_g = new fcl::Cylinderd(s->radius, s->length);
833  }
834  break;
835  case shapes::CONE:
836  {
837  const shapes::Cone* s = static_cast<const shapes::Cone*>(shape.get());
838  cg_g = new fcl::Coned(s->radius, s->length);
839  }
840  break;
841  case shapes::MESH:
842  {
843  auto g = new fcl::BVHModel<BV>();
844  const shapes::Mesh* mesh = static_cast<const shapes::Mesh*>(shape.get());
845  if (mesh->vertex_count > 0 && mesh->triangle_count > 0)
846  {
847  std::vector<fcl::Triangle> tri_indices(mesh->triangle_count);
848  for (unsigned int i = 0; i < mesh->triangle_count; ++i)
849  tri_indices[i] =
850  fcl::Triangle(mesh->triangles[3 * i], mesh->triangles[3 * i + 1], mesh->triangles[3 * i + 2]);
851 
852  std::vector<fcl::Vector3d> points(mesh->vertex_count);
853  for (unsigned int i = 0; i < mesh->vertex_count; ++i)
854  points[i] = fcl::Vector3d(mesh->vertices[3 * i], mesh->vertices[3 * i + 1], mesh->vertices[3 * i + 2]);
855 
856  g->beginModel();
857  g->addSubModel(points, tri_indices);
858  g->endModel();
859  }
860  cg_g = g;
861  }
862  break;
863  case shapes::OCTREE:
864  {
865  const shapes::OcTree* g = static_cast<const shapes::OcTree*>(shape.get());
866  cg_g = new fcl::OcTreed(g->octree);
867  }
868  break;
869  default:
870  RCLCPP_ERROR(LOGGER, "This shape type (%d) is not supported using FCL yet", (int)shape->type);
871  cg_g = nullptr;
872  }
873 
874  if (cg_g)
875  {
876  cg_g->computeLocalAABB();
877  FCLGeometryConstPtr res(new FCLGeometry(cg_g, data, shape_index));
878  cache.map_[wptr] = res;
879  cache.bumpUseCount();
880  return res;
881  }
882  return FCLGeometryConstPtr();
883 }
