60  {

64  bool adaptAndPlan(const PlannerFn& planner, const planning_scene::PlanningSceneConstPtr& planning_scene,
65  const planning_interface::MotionPlanRequest& req, planning_interface::MotionPlanResponse& res,
66  std::vector<std::size_t>& added_path_index) const override
67  {
68  RCLCPP_DEBUG(LOGGER, "Running '%s'", getDescription().c_str());
69 
70  // get the specified start state
71  moveit::core::RobotState start_state = planning_scene->getCurrentState();
72  moveit::core::robotStateMsgToRobotState(planning_scene->getTransforms(), req.start_state, start_state);
73 
74  // if the start state is otherwise valid but does not meet path constraints
75  if (planning_scene->isStateValid(start_state, req.group_name) &&
76  !planning_scene->isStateValid(start_state, req.path_constraints, req.group_name))
77  {
78  RCLCPP_INFO(LOGGER, "Path constraints not satisfied for start state...");
79  planning_scene->isStateValid(start_state, req.path_constraints, req.group_name, true);
80  RCLCPP_INFO(LOGGER, "Planning to path constraints...");
81 
82  planning_interface::MotionPlanRequest req2 = req;
83  req2.goal_constraints.resize(1);
84  req2.goal_constraints[0] = req.path_constraints;
85  req2.path_constraints = moveit_msgs::msg::Constraints();
86  planning_interface::MotionPlanResponse res2;
87  // we call the planner for this additional request, but we do not want to include potential
88  // index information from that call
89  std::vector<std::size_t> added_path_index_temp;
90  added_path_index_temp.swap(added_path_index);
91  bool solved1 = planner(planning_scene, req2, res2);
92  added_path_index_temp.swap(added_path_index);
93 
94  if (solved1)
95  {
96  planning_interface::MotionPlanRequest req3 = req;
97  RCLCPP_INFO(LOGGER, "Planned to path constraints. Resuming original planning request.");
98 
99  // extract the last state of the computed motion plan and set it as the new start state
100  moveit::core::robotStateToRobotStateMsg(res2.trajectory_->getLastWayPoint(), req3.start_state);
101  bool solved2 = planner(planning_scene, req3, res);
102  res.planning_time_ += res2.planning_time_;
103 
104  if (solved2)
105  {
106  // since we add a prefix, we need to correct any existing index positions
107  for (std::size_t& added_index : added_path_index)
108  added_index += res2.trajectory_->getWayPointCount();
109 
110  // we mark the fact we insert a prefix path (we specify the index position we just added)
111  for (std::size_t i = 0; i < res2.trajectory_->getWayPointCount(); ++i)
112  added_path_index.push_back(i);
113 
114  // we need to append the solution paths.
115  res2.trajectory_->append(*res.trajectory_, 0.0);
116  res2.trajectory_->swap(*res.trajectory_);
117  return true;
118  }
119  else
120  return false;
121  }
122  else
123  {
124  RCLCPP_WARN(LOGGER, "Unable to plan to path constraints. Running usual motion plan.");
125  bool result = planner(planning_scene, req, res);
126  res.planning_time_ += res2.planning_time_;
127  return result;
128  }
129  }
130  else
131  {
132  RCLCPP_DEBUG(LOGGER, "Path constraints are OK. Running usual motion plan.");
133  return planner(planning_scene, req, res);
134  }
135  }
136 };
137 } // namespace default_planner_request_adapters
