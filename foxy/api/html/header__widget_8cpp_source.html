106  // Horizontal layout splitter
107  QHBoxLayout* hlayout = new QHBoxLayout();
108 
109  // Widget Title
110  QLabel* widget_title = new QLabel(this);

125  // Line Edit for path

135  hlayout->addWidget(browse_button);
136 
137  // Add horizontal layer to verticle layer
138  layout->addLayout(hlayout);
139 
140  setLayout(layout);
141 }

146 void LoadPathWidget::btnFileDialog()
147 {
148  QString path;
149  if (dir_only_) // only allow user to select a directory
150  {
151  path = QFileDialog::getExistingDirectory(this, "Open Package Directory", path_box_->text(),
152  QFileDialog::ShowDirsOnly | QFileDialog::DontResolveSymlinks);
153  }
154  else // only allow user to select file
155  {
156  QString start_path;
157 
158  start_path = path_box_->text();
159 
160  if (load_only_)
161  {
162  path = QFileDialog::getOpenFileName(this, "Open File", start_path, "");
163  }
164  else
165  {
166  path = QFileDialog::getSaveFileName(this, "Create/Load File", start_path, "");
167  }
168  }
169 
170  // check they did not press cancel
171  if (!path.isNull())
172  path_box_->setText(path);
173 }
