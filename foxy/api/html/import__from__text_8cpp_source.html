54 void parseStart(std::istream& in, planning_scene_monitor::PlanningSceneMonitor* psm,
55  moveit_warehouse::RobotStateStorage* rs)
56 {
57  int count;
58  in >> count;
59  if (in.good() && !in.eof())
60  {
61  for (int i = 0; i < count; ++i)
62  {
63  std::map<std::string, double> v;
64  std::string name;
65  in >> name;
66  if (in.good() && !in.eof())
67  {
68  std::string joint;
69  std::string marker;
70  double value;
71  in >> joint;
72  while (joint != "." && in.good() && !in.eof())
73  {
74  in >> marker;
75  if (marker != "=")
76  joint = ".";
77  else
78  {
79  in >> value;
80  v[joint] = value;
81  }
82  if (joint != ".")
83  in >> joint;
84  }
85  }
86  if (!v.empty())
87  {
88  moveit::core::RobotState st = psm->getPlanningScene()->getCurrentState();
89  st.setVariablePositions(v);
90  moveit_msgs::msg::RobotState msg;
91  moveit::core::robotStateToRobotStateMsg(st, msg);
92  RCLCPP_INFO(LOGGER, "Parsed start state '%s'", name.c_str());
93  rs->addRobotState(msg, name);
94  }
95  }
96  }
97 }

99 void parseLinkConstraint(std::istream& in, planning_scene_monitor::PlanningSceneMonitor* psm,
100  moveit_warehouse::ConstraintsStorage* cs)
101 {
102  Eigen::Translation3d pos(Eigen::Vector3d::Zero());
103  Eigen::Quaterniond rot(Eigen::Quaterniond::Identity());
104 
105  bool have_position = false;
106  bool have_orientation = false;
107 
108  std::string name;
109  std::getline(in, name);
110 
111  // The link name is optional, in which case there would be a blank line
112  std::string link_name;
113  std::getline(in, link_name);
114 
115  std::string type;
116  in >> type;
117 
118  while (type != "." && in.good() && !in.eof())
119  {
120  if (type == "xyz")
121  {
122  have_position = true;
123  double x, y, z;
124  in >> x >> y >> z;
125  pos = Eigen::Translation3d(x, y, z);
126  }
127  else if (type == "rpy")
128  {
129  have_orientation = true;
130  double r, p, y;
131  in >> r >> p >> y;
132  rot = Eigen::Quaterniond(Eigen::AngleAxisd(r, Eigen::Vector3d::UnitX()) *
133  Eigen::AngleAxisd(p, Eigen::Vector3d::UnitY()) *
134  Eigen::AngleAxisd(y, Eigen::Vector3d::UnitZ()));
135  }
136  else
137  RCLCPP_ERROR(LOGGER, "Unknown link constraint element: '%s'", type.c_str());
138  in >> type;
139  }
140 
141  // Convert to getLine method by eating line break
142  std::string end_link;
143  std::getline(in, end_link);
144 
145  if (have_position && have_orientation)
146  {
147  geometry_msgs::msg::PoseStamped pose;
148  pose.pose = tf2::toMsg(pos * rot);
149  pose.header.frame_id = psm->getRobotModel()->getModelFrame();
150  moveit_msgs::msg::Constraints constr = kinematic_constraints::constructGoalConstraints(link_name, pose);
151  constr.name = name;
152  RCLCPP_INFO(LOGGER, "Parsed link constraint '%s'", name.c_str());
153  cs->addConstraints(constr);
154  }
155 }

157 void parseGoal(std::istream& in, planning_scene_monitor::PlanningSceneMonitor* psm,
158  moveit_warehouse::ConstraintsStorage* cs)
159 {
160  int count;
161  in >> count;
162 
163  // Convert to getLine method from here-on, so eat the line break.
164  std::string end_link;
165  std::getline(in, end_link);
166 
167  if (in.good() && !in.eof())
168  {
169  for (int i = 0; i < count; ++i)
170  {
171  std::string type;
172  std::getline(in, type);
173 
174  if (in.good() && !in.eof())
175  {
176  if (type == "link_constraint")
177  parseLinkConstraint(in, psm, cs);
178  else
179  RCLCPP_INFO(LOGGER, "Unknown goal type: '%s'", type.c_str());
180  }
181  }
182  }

185 void parseQueries(std::istream& in, planning_scene_monitor::PlanningSceneMonitor* psm,
186  moveit_warehouse::RobotStateStorage* rs, moveit_warehouse::ConstraintsStorage* cs)
187 {
188  std::string scene_name;
189  in >> scene_name;
190  while (in.good() && !in.eof())
191  {
192  std::string type;
193  in >> type;
194 
195  if (in.good() && !in.eof())
196  {
197  if (type == "start")
198  parseStart(in, psm, rs);
199  else if (type == "goal")
200  parseGoal(in, psm, cs);
201  else
202  RCLCPP_ERROR(LOGGER, "Unknown query type: '%s'", type.c_str());
203  }
204  }
205 }

207 int main(int argc, char** argv)
208 {
209  rclcpp::init(argc, argv);
210  rclcpp::NodeOptions node_options;
211  node_options.allow_undeclared_parameters(true);
212  node_options.automatically_declare_parameters_from_overrides(true);
213  rclcpp::Node::SharedPtr node = rclcpp::Node::make_shared("import_from_text_to_warehouse", node_options);
214 
215  boost::program_options::options_description desc;
216  desc.add_options()("help", "Show help message")("queries", boost::program_options::value<std::string>(),
217  "Name of file containing motion planning queries.")(
218  "scene", boost::program_options::value<std::string>(), "Name of file containing motion planning scene.")(
219  "host", boost::program_options::value<std::string>(),
220  "Host for the DB.")("port", boost::program_options::value<std::size_t>(), "Port for the DB.");
221 
222  boost::program_options::variables_map vm;
223  boost::program_options::store(boost::program_options::parse_command_line(argc, argv, desc), vm);
224  boost::program_options::notify(vm);
225 
226  if (vm.count("help") || argc == 1) // show help if no parameters passed
227  {
228  std::cout << desc << std::endl;
229  return 1;
230  }
231  // Set up db
232  warehouse_ros::DatabaseConnection::Ptr conn = moveit_warehouse::loadDatabase(node);
233  if (vm.count("host") && vm.count("port"))
234  conn->setParams(vm["host"].as<std::string>(), vm["port"].as<std::size_t>());
235  if (!conn->connect())
236  return 1;
237 
238  planning_scene_monitor::PlanningSceneMonitor psm(node, ROBOT_DESCRIPTION);
239  if (!psm.getPlanningScene())
240  {
241  RCLCPP_ERROR(LOGGER, "Unable to initialize PlanningSceneMonitor");
242  return 1;
243  }
244 
245  moveit_warehouse::PlanningSceneStorage pss(conn);
246  moveit_warehouse::ConstraintsStorage cs(conn);
247  moveit_warehouse::RobotStateStorage rs(conn);
248 
249  if (vm.count("scene"))
250  {
251  std::ifstream fin(vm["scene"].as<std::string>().c_str());
252  psm.getPlanningScene()->loadGeometryFromStream(fin);
253  fin.close();
254  moveit_msgs::msg::PlanningScene psmsg;
255  psm.getPlanningScene()->getPlanningSceneMsg(psmsg);
256  pss.addPlanningScene(psmsg);
257  }
258 
259  if (vm.count("queries"))
260  {
261  std::ifstream fin(vm["queries"].as<std::string>().c_str());
262  if (fin.good() && !fin.eof())
263  parseQueries(fin, &psm, &rs, &cs);
264  fin.close();
265  }
266 
267  rclcpp::spin(node);
268  return 0;
269 }
