48 bool validateFloats(const visualization_msgs::msg::InteractiveMarker& msg)
49 {
50  bool valid = true;
51  valid = valid && rviz_common::validateFloats(msg.pose);
52  valid = valid && rviz_common::validateFloats(msg.scale);
53  for (const auto& control : msg.controls)
54  {
55  valid = valid && rviz_common::validateFloats(control.orientation);
56  for (const auto& marker : control.markers)
57  {
58  valid = valid && rviz_common::validateFloats(marker.pose);
59  valid = valid && rviz_common::validateFloats(marker.scale);
60  valid = valid && rviz_common::validateFloats(marker.color);
61  valid = valid && rviz_common::validateFloats(marker.points);
62  }
63  }
64  return valid;
65 }

192 void InteractiveMarkerDisplay::updateMarkers(const std::vector<visualization_msgs::msg::InteractiveMarker>& markers)
193 {
194  for (const visualization_msgs::msg::InteractiveMarker& marker : markers)
195  {
196  if (!validateFloats(marker))
197  {
198  setStatusStd(rviz_common::properties::StatusProperty::Error, marker.name, "Marker contains invalid floats!");
199  continue;
200  }
201  RVIZ_COMMON_LOG_DEBUG_STREAM("Processing interactive marker '" << marker.name << "'. " << marker.controls.size());
202 
203  auto int_marker_entry = interactive_markers_map_.find(marker.name);
204 
205  if (int_marker_entry == interactive_markers_map_.end())
206  {
207  int_marker_entry =
208  interactive_markers_map_
209  .insert(std::make_pair(marker.name, std::make_shared<InteractiveMarker>(getSceneNode(), context_)))
210  .first;
211  connect(int_marker_entry->second.get(), SIGNAL(userFeedback(visualization_msgs::msg::InteractiveMarkerFeedback&)),
212  this, SLOT(publishFeedback(visualization_msgs::msg::InteractiveMarkerFeedback&)));
213  connect(
214  int_marker_entry->second.get(),
215  SIGNAL(statusUpdate(rviz_common::properties::StatusProperty::Level, const std::string&, const std::string&)),
216  this,
217  SLOT(onStatusUpdate(rviz_common::properties::StatusProperty::Level, const std::string&, const std::string&)));
218  }
219 
220  if (int_marker_entry->second->processMessage(marker))
221  {
222  int_marker_entry->second->setShowAxes(show_axes_property_->getBool());
223  int_marker_entry->second->setShowVisualAids(show_visual_aids_property_->getBool());
224  int_marker_entry->second->setShowDescription(show_descriptions_property_->getBool());
225  }
226  else
227  {
228  unsubscribe();
229  return;
230  }
231  }
232 }

249 void InteractiveMarkerDisplay::updatePoses(
250  const std::vector<visualization_msgs::msg::InteractiveMarkerPose>& marker_poses)
251 {
252  for (const visualization_msgs::msg::InteractiveMarkerPose& marker_pose : marker_poses)
253  {
254  if (!rviz_common::validateFloats(marker_pose.pose))
255  {
256  setStatusStd(rviz_common::properties::StatusProperty::Error, marker_pose.name,
257  "Pose message contains invalid floats!");
258  return;
259  }
260 
261  auto int_marker_entry = interactive_markers_map_.find(marker_pose.name);
262 
263  if (int_marker_entry != interactive_markers_map_.end())
264  {
265  int_marker_entry->second->processMessage(marker_pose);
266  }
267  else
268  {
269  setStatusStd(rviz_common::properties::StatusProperty::Error, marker_pose.name,
270  "Pose received for non-existing marker '" + marker_pose.name);
271  unsubscribe();
272  return;
273  }
274  }

296 void InteractiveMarkerDisplay::statusCallback(interactive_markers::InteractiveMarkerClient::Status status,
297  const std::string& message)
298 {
299  rviz_common::properties::StatusProperty::Level rviz_level;
300  switch (status)
301  {
302  case interactive_markers::InteractiveMarkerClient::STATUS_DEBUG:
303  rviz_level = rviz_common::properties::StatusProperty::Ok;
304  break;
305  case interactive_markers::InteractiveMarkerClient::STATUS_INFO:
306  rviz_level = rviz_common::properties::StatusProperty::Ok;
307  break;
308  case interactive_markers::InteractiveMarkerClient::STATUS_WARN:
309  rviz_level = rviz_common::properties::StatusProperty::Warn;
310  break;
311  case interactive_markers::InteractiveMarkerClient::STATUS_ERROR:
312  rviz_level = rviz_common::properties::StatusProperty::Error;
313  break;
314  default:
315  RVIZ_COMMON_LOG_WARNING("Unexpected status level from interactive marker client received");
316  rviz_level = rviz_common::properties::StatusProperty::Error;
317  }
318  setStatusStd(rviz_level, "Interactive Marker Client", message);
319 }
