43 pilz_industrial_motion_planner::JointLimitsContainer
44 pilz_industrial_motion_planner::JointLimitsAggregator::getAggregatedLimits(
45  const ros::NodeHandle& nh, const std::vector<const moveit::core::JointModel*>& joint_models)
46 {
47  JointLimitsContainer container;
48 
49  ROS_INFO_STREAM("Reading limits from namespace " << nh.getNamespace());
50 
51  // Iterate over all joint models and generate the map
52  for (auto joint_model : joint_models)
53  {
54  JointLimit joint_limit;
55 
56  // If there is something defined for the joint on the parameter server
57  if (joint_limits_interface::getJointLimits(joint_model->getName(), nh, joint_limit))
58  {
59  if (joint_limit.has_position_limits)
60  {
61  checkPositionBoundsThrowing(joint_model, joint_limit);
62  }
63  else
64  {
65  updatePositionLimitFromJointModel(joint_model, joint_limit);
66  }
67 
68  if (joint_limit.has_velocity_limits)
69  {
70  checkVelocityBoundsThrowing(joint_model, joint_limit);
71  }
72  else
73  {
74  updateVelocityLimitFromJointModel(joint_model, joint_limit);
75  }
76  }
77  else
78  {
79  // If there is nothing defined for this joint on the parameter server just
80  // update the values by the values of
81  // the urdf
82 
83  updatePositionLimitFromJointModel(joint_model, joint_limit);
84  updateVelocityLimitFromJointModel(joint_model, joint_limit);
85  }
86 
87  // Update max_deceleration if no max_acceleration has been set
88  if (joint_limit.has_acceleration_limits && !joint_limit.has_deceleration_limits)
89  {
90  joint_limit.max_deceleration = -joint_limit.max_acceleration;
91  joint_limit.has_deceleration_limits = true;
92  }
93 
94  // Insert the joint limit into the map
95  container.addLimit(joint_model->getName(), joint_limit);
96  }
97 
98  return container;
99 }

101 void pilz_industrial_motion_planner::JointLimitsAggregator::updatePositionLimitFromJointModel(
102  const moveit::core::JointModel* joint_model, JointLimit& joint_limit)
103 {
104  switch (joint_model->getVariableBounds().size())
105  {
106  // LCOV_EXCL_START
107  case 0:
108  ROS_ERROR_STREAM("no bounds set for joint " << joint_model->getName());
109  break;
110  // LCOV_EXCL_STOP
111  case 1:
112  joint_limit.has_position_limits = joint_model->getVariableBounds()[0].position_bounded_;
113  joint_limit.min_position = joint_model->getVariableBounds()[0].min_position_;
114  joint_limit.max_position = joint_model->getVariableBounds()[0].max_position_;
115  break;
116  // LCOV_EXCL_START
117  default:
118  ROS_ERROR_STREAM("Multi-DOF-Joints not supported. The robot won't move. " << joint_model->getName());
119  joint_limit.has_position_limits = true;
120  joint_limit.min_position = 0;
121  joint_limit.max_position = 0;
122  break;
123  // LCOV_EXCL_STOP
124  }
125 
126  ROS_DEBUG_STREAM("Limit(" << joint_model->getName() << " min:" << joint_limit.min_position
127  << " max:" << joint_limit.max_position);
128 }

154 void pilz_industrial_motion_planner::JointLimitsAggregator::checkPositionBoundsThrowing(
155  const moveit::core::JointModel* joint_model, const JointLimit& joint_limit)
156 {
157  // Check min position
158  if (!joint_model->satisfiesPositionBounds(&joint_limit.min_position))
159  {
160  throw AggregationBoundsViolationException("min_position of " + joint_model->getName() +
161  " violates min limit from URDF");
162  }
163 
164  // Check max position
165  if (!joint_model->satisfiesPositionBounds(&joint_limit.max_position))
166  {
167  throw AggregationBoundsViolationException("max_position of " + joint_model->getName() +
168  " violates max limit from URDF");
169  }
