59 static double normalizeAngle(double angle)
60 {
61  double v = fmod(angle, 2.0 * boost::math::constants::pi<double>());
62  if (v < -boost::math::constants::pi<double>())
63  v += 2.0 * boost::math::constants::pi<double>();
64  else if (v > boost::math::constants::pi<double>())
65  v -= 2.0 * boost::math::constants::pi<double>();
66  return v;
67 }

84 template <typename Derived>
85 std::tuple<Eigen::Matrix<typename Eigen::MatrixBase<Derived>::Scalar, 3, 1>, bool>
86 CalcEulerAngles(const Eigen::MatrixBase<Derived>& R)
87 {
88  using std::atan2;
89  using std::sqrt;
90  EIGEN_STATIC_ASSERT_MATRIX_SPECIFIC_SIZE(Derived, 3, 3)
91  using Index = EIGEN_DEFAULT_DENSE_INDEX_TYPE;
92  using Scalar = typename Eigen::MatrixBase<Derived>::Scalar;
93  const Index i = 0;
94  const Index j = 1;
95  const Index k = 2;
96  Eigen::Matrix<Scalar, 3, 1> res;
97  const Scalar rsum = sqrt((R(i, i) * R(i, i) + R(i, j) * R(i, j) + R(j, k) * R(j, k) + R(k, k) * R(k, k)) / 2);
98  res[1] = atan2(R(i, k), rsum);
99  // There is a singularity when cos(beta) == 0
100  if (rsum > 4 * Eigen::NumTraits<Scalar>::epsilon())
101  { // cos(beta) != 0
102  res[0] = atan2(-R(j, k), R(k, k));
103  res[2] = atan2(-R(i, j), R(i, i));
104  return { res, true };
105  }
106  else if (R(i, k) > 0)
107  { // cos(beta) == 0 and sin(beta) == 1
108  const Scalar spos = R(j, i) + R(k, j); // 2*sin(alpha + gamma)
109  const Scalar cpos = R(j, j) - R(k, i); // 2*cos(alpha + gamma)
110  res[0] = atan2(spos, cpos);
111  res[2] = 0;
112  return { res, false };
113  } // cos(beta) == 0 and sin(beta) == -1
114  const Scalar sneg = R(k, j) - R(j, i); // 2*sin(alpha + gamma)
115  const Scalar cneg = R(j, j) + R(k, i); // 2*cos(alpha + gamma)
116  res[0] = atan2(sneg, cneg);
117  res[2] = 0;
118  return { res, false };
119 }

128 bool JointConstraint::configure(const moveit_msgs::msg::JointConstraint& jc)
129 {
130  // clearing before we configure to get rid of any old data
131  clear();
132 
133  // testing tolerances first
134  if (jc.tolerance_above < 0.0 || jc.tolerance_below < 0.0)
135  {
136  RCLCPP_WARN(LOGGER, "JointConstraint tolerance values must be positive.");
137  joint_model_ = nullptr;
138  return false;
139  }
140 
141  joint_variable_name_ = jc.joint_name;
142  local_variable_name_.clear();
143  if (robot_model_->hasJointModel(joint_variable_name_))
144  joint_model_ = robot_model_->getJointModel(joint_variable_name_);
145  else
146  {
147  std::size_t pos = jc.joint_name.find_last_of('/');
148  if (pos != std::string::npos)
149  {
150  joint_model_ = robot_model_->getJointModel(jc.joint_name.substr(0, pos));
151  if (pos + 1 < jc.joint_name.length())
152  local_variable_name_ = jc.joint_name.substr(pos + 1);
153  }
154  else
155  joint_model_ = robot_model_->getJointModel(jc.joint_name);
156  }
157 
158  if (joint_model_)
159  {
160  if (local_variable_name_.empty())
161  {
162  // check if the joint has 1 DOF (the only kind we can handle)
163  if (joint_model_->getVariableCount() == 0)
164  {
165  RCLCPP_ERROR(LOGGER, "Joint '%s' has no parameters to constrain", jc.joint_name.c_str());
166  joint_model_ = nullptr;
167  }
168  else if (joint_model_->getVariableCount() > 1)
169  {
170  RCLCPP_ERROR(LOGGER,
171  "Joint '%s' has more than one parameter to constrain. "
172  "This type of constraint is not supported.",
173  jc.joint_name.c_str());
174  joint_model_ = nullptr;
175  }
176  }
177  else
178  {
179  int found = -1;
180  const std::vector<std::string>& local_var_names = joint_model_->getLocalVariableNames();
181  for (std::size_t i = 0; i < local_var_names.size(); ++i)
182  if (local_var_names[i] == local_variable_name_)
183  {
184  found = i;
185  break;
186  }
187  if (found < 0)
188  {
189  RCLCPP_ERROR(LOGGER, "Local variable name '%s' is not known to joint '%s'", local_variable_name_.c_str(),
190  joint_model_->getName().c_str());
191  joint_model_ = nullptr;
192  }
193  }
194  }
195 
196  if (joint_model_)
197  {
198  joint_is_continuous_ = false;
199  joint_tolerance_above_ = jc.tolerance_above;
200  joint_tolerance_below_ = jc.tolerance_below;
201  joint_variable_index_ = robot_model_->getVariableIndex(joint_variable_name_);
202 
203  // check if we have to wrap angles when computing distances
204  joint_is_continuous_ = false;
205  if (joint_model_->getType() == moveit::core::JointModel::REVOLUTE)
206  {
207  const moveit::core::RevoluteJointModel* rjoint =
208  static_cast<const moveit::core::RevoluteJointModel*>(joint_model_);
209  if (rjoint->isContinuous())
210  joint_is_continuous_ = true;
211  }
212  else if (joint_model_->getType() == moveit::core::JointModel::PLANAR)
213  {
214  if (local_variable_name_ == "theta")
215  joint_is_continuous_ = true;
216  }
217 
218  if (joint_is_continuous_)
219  {
220  joint_position_ = normalizeAngle(jc.position);
221  }
222  else
223  {
224  joint_position_ = jc.position;
225  const moveit::core::VariableBounds& bounds = joint_model_->getVariableBounds(joint_variable_name_);
226 
227  if (bounds.min_position_ > joint_position_ + joint_tolerance_above_)
228  {
229  joint_position_ = bounds.min_position_;
230  joint_tolerance_above_ = std::numeric_limits<double>::epsilon();
231  RCLCPP_WARN(LOGGER,
232  "Joint %s is constrained to be below the minimum bounds. "
233  "Assuming minimum bounds instead.",
234  jc.joint_name.c_str());
235  }
236  else if (bounds.max_position_ < joint_position_ - joint_tolerance_below_)
237  {
238  joint_position_ = bounds.max_position_;
239  joint_tolerance_below_ = std::numeric_limits<double>::epsilon();
240  RCLCPP_WARN(LOGGER,
241  "Joint %s is constrained to be above the maximum bounds. "
242  "Assuming maximum bounds instead.",
243  jc.joint_name.c_str());
244  }
245  }
246 
247  if (jc.weight <= std::numeric_limits<double>::epsilon())
248  {
249  RCLCPP_WARN(LOGGER, "The weight on constraint for joint '%s' is very near zero. Setting to 1.0.",
250  jc.joint_name.c_str());
251  constraint_weight_ = 1.0;
252  }
253  else
254  constraint_weight_ = jc.weight;
255  }
256  return joint_model_ != nullptr;
257 }

271 ConstraintEvaluationResult JointConstraint::decide(const moveit::core::RobotState& state, bool verbose) const
272 {
273  if (!joint_model_)
274  return ConstraintEvaluationResult(true, 0.0);
275 
276  double current_joint_position = state.getVariablePosition(joint_variable_index_);
277  double dif = 0.0;
278 
279  // compute signed shortest distance for continuous joints
280  if (joint_is_continuous_)
281  {
282  dif = normalizeAngle(current_joint_position) - joint_position_;
283 
284  if (dif > boost::math::constants::pi<double>())
285  dif = 2.0 * boost::math::constants::pi<double>() - dif;
286  else if (dif < -boost::math::constants::pi<double>())
287  dif += 2.0 * boost::math::constants::pi<double>(); // we include a sign change to have dif > 0
288  }
289  else
290  dif = current_joint_position - joint_position_;
291 
292  // check bounds
293  bool result = dif <= (joint_tolerance_above_ + 2.0 * std::numeric_limits<double>::epsilon()) &&
294  dif >= (-joint_tolerance_below_ - 2.0 * std::numeric_limits<double>::epsilon());
295  if (verbose)
296  RCLCPP_INFO(LOGGER,
297  "Constraint %s:: Joint name: '%s', actual value: %f, desired value: %f, "
298  "tolerance_above: %f, tolerance_below: %f",
299  result ? "satisfied" : "violated", joint_variable_name_.c_str(), current_joint_position,
300  joint_position_, joint_tolerance_above_, joint_tolerance_below_);
301  return ConstraintEvaluationResult(result, constraint_weight_ * fabs(dif));
302 }

336 bool PositionConstraint::configure(const moveit_msgs::msg::PositionConstraint& pc, const moveit::core::Transforms& tf)
337 {
338  // clearing before we configure to get rid of any old data
339  clear();
340 
341  link_model_ = robot_model_->getLinkModel(pc.link_name);
342  if (link_model_ == nullptr)
343  {
344  RCLCPP_WARN(LOGGER, "Position constraint link model %s not found in kinematic model. Constraint invalid.",
345  pc.link_name.c_str());
346  return false;
347  }
348 
349  if (pc.header.frame_id.empty())
350  {
351  RCLCPP_WARN(LOGGER, "No frame specified for position constraint on link '%s'!", pc.link_name.c_str());
352  return false;
353  }
354 
355  offset_ = Eigen::Vector3d(pc.target_point_offset.x, pc.target_point_offset.y, pc.target_point_offset.z);
356  has_offset_ = offset_.squaredNorm() > std::numeric_limits<double>::epsilon();
357 
358  if (tf.isFixedFrame(pc.header.frame_id))
359  {
360  constraint_frame_id_ = tf.getTargetFrame();
361  mobile_frame_ = false;
362  }
363  else
364  {
365  constraint_frame_id_ = pc.header.frame_id;
366  mobile_frame_ = true;
367  }
368 
369  // load primitive shapes, first clearing any we already have
370  for (std::size_t i = 0; i < pc.constraint_region.primitives.size(); ++i)
371  {
372  std::unique_ptr<shapes::Shape> shape(shapes::constructShapeFromMsg(pc.constraint_region.primitives[i]));
373  if (shape)
374  {
375  if (pc.constraint_region.primitive_poses.size() <= i)
376  {
377  RCLCPP_WARN(LOGGER, "Constraint region message does not contain enough primitive poses");
378  continue;
379  }
380  Eigen::Isometry3d t;
381  tf2::fromMsg(pc.constraint_region.primitive_poses[i], t);
382  ASSERT_ISOMETRY(t) // unsanitized input, could contain a non-isometry
383  constraint_region_pose_.push_back(t);
384  if (!mobile_frame_)
385  tf.transformPose(pc.header.frame_id, constraint_region_pose_.back(), constraint_region_pose_.back());
386 
387  const bodies::BodyPtr body(bodies::createEmptyBodyFromShapeType(shape->type));
388  body->setDimensionsDirty(shape.get());
389  body->setPoseDirty(constraint_region_pose_.back());
390  body->updateInternalData();
391  constraint_region_.push_back(body);
392  }
393  else
394  RCLCPP_WARN(LOGGER, "Could not construct primitive shape %zu", i);
395  }
396 
397  // load meshes
398  for (std::size_t i = 0; i < pc.constraint_region.meshes.size(); ++i)
399  {
400  std::unique_ptr<shapes::Shape> shape(shapes::constructShapeFromMsg(pc.constraint_region.meshes[i]));
401  if (shape)
402  {
403  if (pc.constraint_region.mesh_poses.size() <= i)
404  {
405  RCLCPP_WARN(LOGGER, "Constraint region message does not contain enough primitive poses");
406  continue;
407  }
408  Eigen::Isometry3d t;
409  tf2::fromMsg(pc.constraint_region.mesh_poses[i], t);
410  ASSERT_ISOMETRY(t) // unsanitized input, could contain a non-isometry
411  constraint_region_pose_.push_back(t);
412  if (!mobile_frame_)
413  tf.transformPose(pc.header.frame_id, constraint_region_pose_.back(), constraint_region_pose_.back());
414  const bodies::BodyPtr body(bodies::createEmptyBodyFromShapeType(shape->type));
415  body->setDimensionsDirty(shape.get());
416  body->setPoseDirty(constraint_region_pose_.back());
417  body->updateInternalData();
418  constraint_region_.push_back(body);
419  }
420  else
421  {
422  RCLCPP_WARN(LOGGER, "Could not construct mesh shape %zu", i);
423  }
424  }
425 
426  if (pc.weight <= std::numeric_limits<double>::epsilon())
427  {
428  RCLCPP_WARN(LOGGER, "The weight on position constraint for link '%s' is near zero. Setting to 1.0.",
429  pc.link_name.c_str());
430  constraint_weight_ = 1.0;
431  }
432  else
433  constraint_weight_ = pc.weight;
434 
435  return !constraint_region_.empty();
436 }

438 bool PositionConstraint::equal(const KinematicConstraint& other, double margin) const
439 {
440  if (other.getType() != type_)
441  return false;
442  const PositionConstraint& o = static_cast<const PositionConstraint&>(other);
443 
444  if (link_model_ == o.link_model_ && moveit::core::Transforms::sameFrame(constraint_frame_id_, o.constraint_frame_id_))
445  {
446  if ((offset_ - o.offset_).norm() > margin)
447  return false;
448  std::vector<bool> other_region_matches_this(constraint_region_.size(), false);
449  for (std::size_t i = 0; i < constraint_region_.size(); ++i)
450  {
451  bool some_match = false;
452  // need to check against all other regions
453  for (std::size_t j = 0; j < o.constraint_region_.size(); ++j)
454  {
455  // constraint_region_pose_ contain only valid isometries, so diff is also a valid isometry
456  Eigen::Isometry3d diff = constraint_region_pose_[i].inverse() * o.constraint_region_pose_[j];
457  if (diff.translation().norm() < margin && diff.linear().isIdentity(margin) &&
458  constraint_region_[i]->getType() == o.constraint_region_[j]->getType() &&
459  fabs(constraint_region_[i]->computeVolume() - o.constraint_region_[j]->computeVolume()) < margin)
460  {
461  some_match = true;
462  // can't break, as need to do matches the other way as well
463  other_region_matches_this[j] = true;
464  }
465  }
466  if (!some_match)
467  return false;
468  }
469  for (std::size_t i = 0; i < o.constraint_region_.size(); ++i)
470  if (!other_region_matches_this[i])
471  return false;
472  return true;
473  }
474  return false;
475 }

477 // helper function to avoid code duplication
478 static inline ConstraintEvaluationResult finishPositionConstraintDecision(const Eigen::Vector3d& pt,
479  const Eigen::Vector3d& desired,
480  const std::string& name, double weight,
481  bool result, bool verbose)
482 {
483  double dx = desired.x() - pt.x();
484  double dy = desired.y() - pt.y();
485  double dz = desired.z() - pt.z();
486  if (verbose)
487  {
488  RCLCPP_INFO(LOGGER, "Position constraint %s on link '%s'. Desired: %f, %f, %f, current: %f, %f, %f",
489  result ? "satisfied" : "violated", name.c_str(), desired.x(), desired.y(), desired.z(), pt.x(), pt.y(),
490  pt.z());
491  RCLCPP_INFO(LOGGER, "Differences %g %g %g", dx, dy, dz);
492  }
493  return ConstraintEvaluationResult(result, weight * sqrt(dx * dx + dy * dy + dz * dz));
494 }

496 ConstraintEvaluationResult PositionConstraint::decide(const moveit::core::RobotState& state, bool verbose) const
497 {
498  if (!link_model_ || constraint_region_.empty())
499  return ConstraintEvaluationResult(true, 0.0);
500 
501  Eigen::Vector3d pt = state.getGlobalLinkTransform(link_model_) * offset_;
502  if (mobile_frame_)
503  {
504  for (std::size_t i = 0; i < constraint_region_.size(); ++i)
505  {
506  Eigen::Isometry3d tmp = state.getFrameTransform(constraint_frame_id_) * constraint_region_pose_[i];
507  bool result = constraint_region_[i]->cloneAt(tmp)->containsPoint(pt, verbose);
508  if (result || (i + 1 == constraint_region_pose_.size()))
509  return finishPositionConstraintDecision(pt, tmp.translation(), link_model_->getName(), constraint_weight_,
510  result, verbose);
511  else
512  finishPositionConstraintDecision(pt, tmp.translation(), link_model_->getName(), constraint_weight_, result,
513  verbose);
514  }
515  }
516  else
517  {
518  for (std::size_t i = 0; i < constraint_region_.size(); ++i)
519  {
520  bool result = constraint_region_[i]->containsPoint(pt, true);
521  if (result || (i + 1 == constraint_region_.size()))
522  return finishPositionConstraintDecision(pt, constraint_region_[i]->getPose().translation(),
523  link_model_->getName(), constraint_weight_, result, verbose);
524  else
525  finishPositionConstraintDecision(pt, constraint_region_[i]->getPose().translation(), link_model_->getName(),
526  constraint_weight_, result, verbose);
527  }
528  }
529  return ConstraintEvaluationResult(false, 0.0);
530 }

556 bool OrientationConstraint::configure(const moveit_msgs::msg::OrientationConstraint& oc,
557  const moveit::core::Transforms& tf)
558 {
559  // clearing out any old data
560  clear();
561 
562  link_model_ = robot_model_->getLinkModel(oc.link_name);
563  if (!link_model_)
564  {
565  RCLCPP_WARN(LOGGER, "Could not find link model for link name %s", oc.link_name.c_str());
566  return false;
567  }
568  Eigen::Quaterniond q;
569  tf2::fromMsg(oc.orientation, q);
570  if (fabs(q.norm() - 1.0) > 1e-3)
571  {
572  RCLCPP_WARN(LOGGER,
573  "Orientation constraint for link '%s' is probably incorrect: %f, %f, %f, "
574  "%f. Assuming identity instead.",
575  oc.link_name.c_str(), oc.orientation.x, oc.orientation.y, oc.orientation.z, oc.orientation.w);
576  q = Eigen::Quaterniond(1.0, 0.0, 0.0, 0.0);
577  }
578 
579  if (oc.header.frame_id.empty())
580  RCLCPP_WARN(LOGGER, "No frame specified for position constraint on link '%s'!", oc.link_name.c_str());
581 
582  if (tf.isFixedFrame(oc.header.frame_id))
583  {
584  tf.transformQuaternion(oc.header.frame_id, q, q);
585  desired_rotation_frame_id_ = tf.getTargetFrame();
586  desired_rotation_matrix_ = Eigen::Matrix3d(q);
587  desired_rotation_matrix_inv_ = desired_rotation_matrix_.transpose();
588  mobile_frame_ = false;
589  }
590  else
591  {
592  desired_rotation_frame_id_ = oc.header.frame_id;
593  desired_rotation_matrix_ = Eigen::Matrix3d(q);
594  mobile_frame_ = true;
595  }
596  std::stringstream matrix_str;
597  matrix_str << desired_rotation_matrix_;

655 ConstraintEvaluationResult OrientationConstraint::decide(const moveit::core::RobotState& state, bool verbose) const
656 {
657  if (!link_model_)
658  return ConstraintEvaluationResult(true, 0.0);
659 
660  std::tuple<Eigen::Vector3d, bool> euler_angles_error;
661  if (mobile_frame_)
662  {
663  // getFrameTransform() returns a valid isometry by contract
664  Eigen::Matrix3d tmp = state.getFrameTransform(desired_rotation_frame_id_).linear() * desired_rotation_matrix_;
665  // getGlobalLinkTransform() returns a valid isometry by contract
666  Eigen::Isometry3d diff(tmp.transpose() * state.getGlobalLinkTransform(link_model_).linear()); // valid isometry
667  euler_angles_error = CalcEulerAngles(diff.linear());
668  }
669  else
670  {
671  // diff is valid isometry by construction
672  Eigen::Isometry3d diff(desired_rotation_matrix_inv_ * state.getGlobalLinkTransform(link_model_).linear());
673  euler_angles_error = CalcEulerAngles(diff.linear());
674  }
675 
676  // Converting from a rotation matrix to an intrinsic XYZ euler angles have 2 singularities:
677  // pitch ~= pi/2 ==> roll + yaw = theta
678  // pitch ~= -pi/2 ==> roll - yaw = theta
679  // in those cases CalcEulerAngles will set roll (xyz(0)) to theta and yaw (xyz(2)) to zero, so for us to be able to
680  // capture yaw tolerance violation we do the following, if theta violate the absolute yaw tolerance we think of it as
681  // pure yaw rotation and set roll to zero
682  auto& xyz = std::get<Eigen::Vector3d>(euler_angles_error);
683  if (!std::get<bool>(euler_angles_error))
684  {
685  if (normalizeAbsoluteAngle(xyz(0)) > absolute_z_axis_tolerance_ + std::numeric_limits<double>::epsilon())
686  {
687  xyz(2) = xyz(0);
688  xyz(0) = 0;
689  }
690  }
691  // Account for angle wrapping
692  xyz = xyz.unaryExpr(&normalizeAbsoluteAngle);
693 
694  // 0,1,2 corresponds to XYZ, the convention used in sampling constraints
695  bool result = xyz(2) < absolute_z_axis_tolerance_ + std::numeric_limits<double>::epsilon() &&
696  xyz(1) < absolute_y_axis_tolerance_ + std::numeric_limits<double>::epsilon() &&
697  xyz(0) < absolute_x_axis_tolerance_ + std::numeric_limits<double>::epsilon();
698 
699  if (verbose)
700  {
701  Eigen::Quaterniond q_act(state.getGlobalLinkTransform(link_model_).linear());
702  Eigen::Quaterniond q_des(desired_rotation_matrix_);
703  RCLCPP_INFO(LOGGER,
704  "Orientation constraint %s for link '%s'. Quaternion desired: %f %f %f %f, quaternion "
705  "actual: %f %f %f %f, error: x=%f, y=%f, z=%f, tolerance: x=%f, y=%f, z=%f",
706  result ? "satisfied" : "violated", link_model_->getName().c_str(), q_des.x(), q_des.y(), q_des.z(),
707  q_des.w(), q_act.x(), q_act.y(), q_act.z(), q_act.w(), xyz(0), xyz(1), xyz(2),
708  absolute_x_axis_tolerance_, absolute_y_axis_tolerance_, absolute_z_axis_tolerance_);
709  }
710 
711  return ConstraintEvaluationResult(result, constraint_weight_ * (xyz(0) + xyz(1) + xyz(2)));
712 }

748 bool VisibilityConstraint::configure(const moveit_msgs::msg::VisibilityConstraint& vc,
749  const moveit::core::Transforms& tf)
750 {
751  clear();
752  target_radius_ = fabs(vc.target_radius);
753 
754  if (vc.target_radius <= std::numeric_limits<double>::epsilon())
755  RCLCPP_WARN(LOGGER, "The radius of the target disc that must be visible should be strictly positive");
756 
757  if (vc.cone_sides < 3)
758  {
759  RCLCPP_WARN(LOGGER,
760  "The number of sides for the visibility region must be 3 or more. "
761  "Assuming 3 sides instead of the specified %d",
762  vc.cone_sides);
763  cone_sides_ = 3;
764  }
765  else
766  cone_sides_ = vc.cone_sides;
767 
768  // compute the points on the base circle of the cone that make up the cone sides
769  points_.clear();
770  double delta = 2.0 * boost::math::constants::pi<double>() / (double)cone_sides_;
771  double a = 0.0;
772  for (unsigned int i = 0; i < cone_sides_; ++i, a += delta)
773  {
774  double x = sin(a) * target_radius_;
775  double y = cos(a) * target_radius_;
776  points_.push_back(Eigen::Vector3d(x, y, 0.0));
777  }
778 
779  tf2::fromMsg(vc.target_pose.pose, target_pose_);
780  ASSERT_ISOMETRY(target_pose_) // unsanitized input, could contain a non-isometry
781 
782  if (tf.isFixedFrame(vc.target_pose.header.frame_id))
783  {
784  tf.transformPose(vc.target_pose.header.frame_id, target_pose_, target_pose_);
785  target_frame_id_ = tf.getTargetFrame();
786  mobile_target_frame_ = false;
787  // transform won't change, so apply it now
788  for (Eigen::Vector3d& point : points_)
789  point = target_pose_ * point;
790  }
791  else
792  {
793  target_frame_id_ = vc.target_pose.header.frame_id;
794  mobile_target_frame_ = true;
795  }
796 
797  tf2::fromMsg(vc.sensor_pose.pose, sensor_pose_);
798  ASSERT_ISOMETRY(sensor_pose_) // unsanitized input, could contain a non-isometry
799 
800  if (tf.isFixedFrame(vc.sensor_pose.header.frame_id))
801  {
802  tf.transformPose(vc.sensor_pose.header.frame_id, sensor_pose_, sensor_pose_);
803  sensor_frame_id_ = tf.getTargetFrame();
804  mobile_sensor_frame_ = false;
805  }
806  else
807  {
808  sensor_frame_id_ = vc.sensor_pose.header.frame_id;
809  mobile_sensor_frame_ = true;
810  }
811 
812  if (vc.weight <= std::numeric_limits<double>::epsilon())
813  {
814  RCLCPP_WARN(LOGGER, "The weight of visibility constraint is near zero. Setting to 1.0.");
815  constraint_weight_ = 1.0;
816  }
817  else
818  constraint_weight_ = vc.weight;
819 
820  max_view_angle_ = vc.max_view_angle;
821  max_range_angle_ = vc.max_range_angle;
822  sensor_view_direction_ = vc.sensor_view_direction;
823 
824  return target_radius_ > std::numeric_limits<double>::epsilon();
825 }

861 shapes::Mesh* VisibilityConstraint::getVisibilityCone(const moveit::core::RobotState& state) const
862 {
863  // the current pose of the sensor
864 
865  const Eigen::Isometry3d& sp =
866  mobile_sensor_frame_ ? state.getFrameTransform(sensor_frame_id_) * sensor_pose_ : sensor_pose_;
867  const Eigen::Isometry3d& tp =
868  mobile_target_frame_ ? state.getFrameTransform(target_frame_id_) * target_pose_ : target_pose_;
869 
870  // transform the points on the disc to the desired target frame
871  const EigenSTL::vector_Vector3d* points = &points_;
872  std::unique_ptr<EigenSTL::vector_Vector3d> temp_points;
873  if (mobile_target_frame_)
874  {
875  temp_points.reset(new EigenSTL::vector_Vector3d(points_.size()));
876  for (std::size_t i = 0; i < points_.size(); ++i)
877  temp_points->at(i) = tp * points_[i];
878  points = temp_points.get();
879  }
880 
881  // allocate memory for a mesh to represent the visibility cone
882  shapes::Mesh* m = new shapes::Mesh();
883  m->vertex_count = cone_sides_ + 2;
884  m->vertices = new double[m->vertex_count * 3];
885  m->triangle_count = cone_sides_ * 2;
886  m->triangles = new unsigned int[m->triangle_count * 3];
887  // we do NOT allocate normals because we do not compute them
888 
889  // the sensor origin
890  m->vertices[0] = sp.translation().x();
891  m->vertices[1] = sp.translation().y();
892  m->vertices[2] = sp.translation().z();
893 
894  // the center of the base of the cone approximation
895  m->vertices[3] = tp.translation().x();
896  m->vertices[4] = tp.translation().y();
897  m->vertices[5] = tp.translation().z();
898 
899  // the points that approximate the base disc
900  for (std::size_t i = 0; i < points->size(); ++i)
901  {
902  m->vertices[i * 3 + 6] = points->at(i).x();
903  m->vertices[i * 3 + 7] = points->at(i).y();
904  m->vertices[i * 3 + 8] = points->at(i).z();
905  }
906 
907  // add the triangles
908  std::size_t p3 = points->size() * 3;
909  for (std::size_t i = 1; i < points->size(); ++i)
910  {
911  // triangle forming a side of the cone, using the sensor origin
912  std::size_t i3 = (i - 1) * 3;
913  m->triangles[i3] = i + 1;
914  m->triangles[i3 + 1] = 0;
915  m->triangles[i3 + 2] = i + 2;
916  // triangle forming a part of the base of the cone, using the center of the base
917  std::size_t i6 = p3 + i3;
918  m->triangles[i6] = i + 1;
919  m->triangles[i6 + 1] = 1;
920  m->triangles[i6 + 2] = i + 2;
921  }
922 
923  // last triangles
924  m->triangles[p3 - 3] = points->size() + 1;
925  m->triangles[p3 - 2] = 0;
926  m->triangles[p3 - 1] = 2;
927  p3 *= 2;
928  m->triangles[p3 - 3] = points->size() + 1;
929  m->triangles[p3 - 2] = 1;
930  m->triangles[p3 - 1] = 2;
931 
932  return m;
933 }

1010 ConstraintEvaluationResult VisibilityConstraint::decide(const moveit::core::RobotState& state, bool verbose) const
1011 {
1012  if (target_radius_ <= std::numeric_limits<double>::epsilon())
1013  return ConstraintEvaluationResult(true, 0.0);
1014 
1015  if (max_view_angle_ > 0.0 || max_range_angle_ > 0.0)
1016  {
1017  // getFrameTransform() returns a valid isometry by contract
1018  // sensor_pose_ is valid isometry (checked in configure())
1019  const Eigen::Isometry3d& sp =
1020  mobile_sensor_frame_ ? state.getFrameTransform(sensor_frame_id_) * sensor_pose_ : sensor_pose_;
1021  // target_pose_ is valid isometry (checked in configure())
1022  const Eigen::Isometry3d& tp =
1023  mobile_target_frame_ ? state.getFrameTransform(target_frame_id_) * target_pose_ : target_pose_;
1024 
1025  // necessary to do subtraction as SENSOR_Z is 0 and SENSOR_X is 2
1026  const Eigen::Vector3d& normal2 = sp.linear().col(2 - sensor_view_direction_);
1027 
1028  if (max_view_angle_ > 0.0)
1029  {
1030  const Eigen::Vector3d& normal1 = tp.linear().col(2) * -1.0; // along Z axis and inverted
1031  double dp = normal2.dot(normal1);
1032  double ang = acos(dp);
1033  if (dp < 0.0)
1034  {
1035  if (verbose)
1036  RCLCPP_INFO(LOGGER, "Visibility constraint is violated because the sensor is looking at "
1037  "the wrong side");
1038  return ConstraintEvaluationResult(false, 0.0);
1039  }
1040  if (max_view_angle_ < ang)
1041  {
1042  if (verbose)
1043  RCLCPP_INFO(LOGGER,
1044  "Visibility constraint is violated because the view angle is %lf "
1045  "(above the maximum allowed of %lf)",
1046  ang, max_view_angle_);
1047  return ConstraintEvaluationResult(false, 0.0);
1048  }
1049  }
1050  if (max_range_angle_ > 0.0)
1051  {
1052  const Eigen::Vector3d& dir = (tp.translation() - sp.translation()).normalized();
1053  double dp = normal2.dot(dir);
1054  if (dp < 0.0)
1055  {
1056  if (verbose)
1057  RCLCPP_INFO(LOGGER, "Visibility constraint is violated because the sensor is looking at "
1058  "the wrong side");
1059  return ConstraintEvaluationResult(false, 0.0);
1060  }
1061 
1062  double ang = acos(dp);
1063  if (max_range_angle_ < ang)
1064  {
1065  if (verbose)
1066  RCLCPP_INFO(LOGGER,
1067  "Visibility constraint is violated because the range angle is %lf "
1068  "(above the maximum allowed of %lf)",
1069  ang, max_range_angle_);
1070  return ConstraintEvaluationResult(false, 0.0);
1071  }
1072  }
1073  }
1074 
1075  shapes::Mesh* m = getVisibilityCone(state);
1076  if (!m)
1077  return ConstraintEvaluationResult(false, 0.0);
1078 
1079  // add the visibility cone as an object
1080  collision_env_->getWorld()->addToObject("cone", shapes::ShapeConstPtr(m), Eigen::Isometry3d::Identity());
1081 
1082  // check for collisions between the robot and the cone
1083  collision_detection::CollisionRequest req;
1084  collision_detection::CollisionResult res;
1085  collision_detection::AllowedCollisionMatrix acm;
1086  collision_detection::DecideContactFn fn =
1087  std::bind(&VisibilityConstraint::decideContact, this, std::placeholders::_1);
1088  acm.setDefaultEntry(std::string("cone"), fn);
1089 
1090  req.contacts = true;
1091  req.verbose = verbose;
1092  req.max_contacts = 1;
1093  collision_env_->checkRobotCollision(req, res, state, acm);
1094 
1095  if (verbose)
1096  {
1097  std::stringstream ss;
1098  m->print(ss);
1099  RCLCPP_INFO(LOGGER, "Visibility constraint %ssatisfied. Visibility cone approximation:\n %s",
1100  res.collision ? "not " : "", ss.str().c_str());
1101  }
1102 
1103  collision_env_->getWorld()->removeObject("cone");
1104 
1105  return ConstraintEvaluationResult(!res.collision, res.collision ? res.contacts.begin()->second.front().depth : 0.0);
1106 }
