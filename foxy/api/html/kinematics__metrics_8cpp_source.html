47 double KinematicsMetrics::getJointLimitsPenalty(const moveit::core::RobotState& state,
48  const moveit::core::JointModelGroup* joint_model_group) const
49 {
50  if (fabs(penalty_multiplier_) <= boost::math::tools::epsilon<double>())
51  return 1.0;
52  double joint_limits_multiplier(1.0);
53  const std::vector<const moveit::core::JointModel*>& joint_model_vector = joint_model_group->getJointModels();
54  for (const moveit::core::JointModel* joint_model : joint_model_vector)
55  {
56  if (joint_model->getType() == moveit::core::JointModel::REVOLUTE)
57  {
58  const moveit::core::RevoluteJointModel* revolute_model =
59  static_cast<const moveit::core::RevoluteJointModel*>(joint_model);
60  if (revolute_model->isContinuous())
61  continue;
62  }
63  if (joint_model->getType() == moveit::core::JointModel::PLANAR)
64  {
65  const moveit::core::JointModel::Bounds& bounds = joint_model->getVariableBounds();
66  if (bounds[0].min_position_ == -std::numeric_limits<double>::max() ||
67  bounds[0].max_position_ == std::numeric_limits<double>::max() ||
68  bounds[1].min_position_ == -std::numeric_limits<double>::max() ||
69  bounds[1].max_position_ == std::numeric_limits<double>::max() ||
70  bounds[2].min_position_ == -boost::math::constants::pi<double>() ||
71  bounds[2].max_position_ == boost::math::constants::pi<double>())
72  continue;
73  }
74  if (joint_model->getType() == moveit::core::JointModel::FLOATING)
75  {
76  // Joint limits are not well-defined for floating joints
77  continue;
78  }
79  const double* joint_values = state.getJointPositions(joint_model);
80  const moveit::core::JointModel::Bounds& bounds = joint_model->getVariableBounds();
81  std::vector<double> lower_bounds, upper_bounds;
82  for (const moveit::core::VariableBounds& bound : bounds)
83  {
84  lower_bounds.push_back(bound.min_position_);
85  upper_bounds.push_back(bound.max_position_);
86  }
87  double lower_bound_distance = joint_model->distance(joint_values, &lower_bounds[0]);
88  double upper_bound_distance = joint_model->distance(joint_values, &upper_bounds[0]);
89  double range = lower_bound_distance + upper_bound_distance;
90  if (range <= boost::math::tools::epsilon<double>())
91  continue;
92  joint_limits_multiplier *= (lower_bound_distance * upper_bound_distance / (range * range));
93  }
94  return (1.0 - exp(-penalty_multiplier_ * joint_limits_multiplier));
95 }

107 bool KinematicsMetrics::getManipulabilityIndex(const moveit::core::RobotState& state,
108  const moveit::core::JointModelGroup* joint_model_group,
109  double& manipulability_index, bool translation) const
110 {
111  // state.getJacobian() only works for chain groups.
112  if (!joint_model_group->isChain())
113  {
114  return false;
115  }
116 
117  Eigen::MatrixXd jacobian = state.getJacobian(joint_model_group);
118  // Get joint limits penalty
119  double penalty = getJointLimitsPenalty(state, joint_model_group);
120  if (translation)
121  {
122  if (jacobian.cols() < 6)
123  {
124  Eigen::JacobiSVD<Eigen::MatrixXd> svdsolver(jacobian.topLeftCorner(3, jacobian.cols()));
125  Eigen::MatrixXd singular_values = svdsolver.singularValues();
126  manipulability_index = 1.0;
127  for (unsigned int i = 0; i < singular_values.rows(); ++i)
128  {
129  RCLCPP_DEBUG(LOGGER, "Singular value: %d %f", i, singular_values(i, 0));
130  manipulability_index *= singular_values(i, 0);
131  }
132  // Get manipulability index
133  manipulability_index = penalty * manipulability_index;
134  }
135  else
136  {
137  Eigen::MatrixXd jacobian_2 = jacobian.topLeftCorner(3, jacobian.cols());
138  Eigen::MatrixXd matrix = jacobian_2 * jacobian_2.transpose();
139  // Get manipulability index
140  manipulability_index = penalty * sqrt(matrix.determinant());
141  }
142  }
143  else
144  {
145  if (jacobian.cols() < 6)
146  {
147  Eigen::JacobiSVD<Eigen::MatrixXd> svdsolver(jacobian);
148  Eigen::MatrixXd singular_values = svdsolver.singularValues();
149  manipulability_index = 1.0;
150  for (unsigned int i = 0; i < singular_values.rows(); ++i)
151  {
152  RCLCPP_DEBUG(LOGGER, "Singular value: %d %f", i, singular_values(i, 0));
153  manipulability_index *= singular_values(i, 0);
154  }
155  // Get manipulability index
156  manipulability_index = penalty * manipulability_index;
157  }
158  else
159  {
160  Eigen::MatrixXd matrix = jacobian * jacobian.transpose();
161  // Get manipulability index
162  manipulability_index = penalty * sqrt(matrix.determinant());
163  }
164  }
165  return true;
166 }

179 bool KinematicsMetrics::getManipulabilityEllipsoid(const moveit::core::RobotState& state,
180  const moveit::core::JointModelGroup* joint_model_group,
181  Eigen::MatrixXcd& eigen_values,
182  Eigen::MatrixXcd& eigen_vectors) const
183 {
184  // state.getJacobian() only works for chain groups.
185  if (!joint_model_group->isChain())
186  {
187  return false;
188  }
189 
190  Eigen::MatrixXd jacobian = state.getJacobian(joint_model_group);
191  Eigen::MatrixXd matrix = jacobian * jacobian.transpose();
192  Eigen::EigenSolver<Eigen::MatrixXd> eigensolver(matrix.block(0, 0, 3, 3));

208 bool KinematicsMetrics::getManipulability(const moveit::core::RobotState& state,
209  const moveit::core::JointModelGroup* joint_model_group,
210  double& manipulability, bool translation) const
211 {
212  // state.getJacobian() only works for chain groups.
213  if (!joint_model_group->isChain())
214  {
215  return false;
216  }
217  // Get joint limits penalty
218  double penalty = getJointLimitsPenalty(state, joint_model_group);
219  if (translation)
220  {
221  Eigen::MatrixXd jacobian = state.getJacobian(joint_model_group);
222  Eigen::JacobiSVD<Eigen::MatrixXd> svdsolver(jacobian.topLeftCorner(3, jacobian.cols()));
223  Eigen::MatrixXd singular_values = svdsolver.singularValues();
224  for (int i = 0; i < singular_values.rows(); ++i)
225  {
226  RCLCPP_DEBUG(LOGGER, "Singular value: %d %f", i, singular_values(i, 0));
227  }
228 
229  manipulability = penalty * singular_values.minCoeff() / singular_values.maxCoeff();
230  }
231  else
232  {
233  Eigen::MatrixXd jacobian = state.getJacobian(joint_model_group);
234  Eigen::JacobiSVD<Eigen::MatrixXd> svdsolver(jacobian);
235  Eigen::MatrixXd singular_values = svdsolver.singularValues();
236  for (int i = 0; i < singular_values.rows(); ++i)
237  {
238  RCLCPP_DEBUG(LOGGER, "Singular value: %d %f", i, singular_values(i, 0));
239  }
240  manipulability = penalty * singular_values.minCoeff() / singular_values.maxCoeff();
241  }
242  return true;
243 }
