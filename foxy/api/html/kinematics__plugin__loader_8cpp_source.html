98  std::vector<std::string> chooseTipFrames(const moveit::core::JointModelGroup* jmg)
99  {
100  std::vector<std::string> tips;
101  std::map<std::string, std::vector<std::string>>::const_iterator ik_it = iksolver_to_tip_links_.find(jmg->getName());
102 
103  // Check if tips were loaded onto the rosparam server previously
104  if (ik_it != iksolver_to_tip_links_.end())
105  {
106  // the tip is being chosen based on a corresponding rosparam ik link
107  RCLCPP_DEBUG(LOGGER,
108  "Choosing tip frame of kinematic solver for group %s"
109  "based on values in rosparam server.",
110  jmg->getName().c_str());
111  tips = ik_it->second;
112  }
113  else
114  {
115  // get the last link in the chain
116  RCLCPP_DEBUG(LOGGER,
117  "Choosing tip frame of kinematic solver for group %s"
118  "based on last link in chain",
119  jmg->getName().c_str());
120 
121  tips.push_back(jmg->getLinkModels().back()->getName());
122  }
123 
124  // Error check
125  if (tips.empty())
126  {
127  RCLCPP_ERROR(LOGGER, "Error choosing kinematic solver tip frame(s).");
128  }
129 
130  // Debug tip choices
131  std::stringstream tip_debug;
132  tip_debug << "Planning group '" << jmg->getName() << "' has tip(s): ";
133  for (const auto& tip : tips)
134  tip_debug << tip << ", ";
135  RCLCPP_DEBUG_STREAM(LOGGER, tip_debug.str());
136 
137  return tips;
138  }

140  kinematics::KinematicsBasePtr allocKinematicsSolver(const moveit::core::JointModelGroup* jmg)
141  {
142  kinematics::KinematicsBasePtr result;
143  if (!kinematics_loader_)
144  {
145  RCLCPP_ERROR(LOGGER, "Invalid kinematics loader.");
146  return result;
147  }
148  if (!jmg)
149  {
150  RCLCPP_ERROR(LOGGER, "Specified group is NULL. Cannot allocate kinematics solver.");
151  return result;
152  }
153  const std::vector<const moveit::core::LinkModel*>& links = jmg->getLinkModels();
154  if (links.empty())
155  {
156  RCLCPP_ERROR(LOGGER, "No links specified for group '%s'. Cannot allocate kinematics solver.",
157  jmg->getName().c_str());
158  return result;
159  }
160 
161  RCLCPP_DEBUG(LOGGER, "Trying to allocate kinematics solver for group '%s'", jmg->getName().c_str());
162 
163  std::map<std::string, std::vector<std::string>>::const_iterator it =
164  possible_kinematics_solvers_.find(jmg->getName());
165  if (it == possible_kinematics_solvers_.end())
166  {
167  RCLCPP_DEBUG(LOGGER, "No kinematics solver available for this group");
168  return result;
169  }
170 
171  const std::string& base = links.front()->getParentJointModel()->getParentLinkModel() ?
172  links.front()->getParentJointModel()->getParentLinkModel()->getName() :
173  jmg->getParentModel().getModelFrame();
174 
175  // just to be sure, do not call the same pluginlib instance allocation function in parallel
176  boost::mutex::scoped_lock slock(lock_);
177  for (std::size_t i = 0; !result && i < it->second.size(); ++i)
178  {
179  try
180  {
181  result = kinematics_loader_->createUniqueInstance(it->second[i]);
182  if (result)
183  {
184  // choose the tip of the IK solver
185  const std::vector<std::string> tips = chooseTipFrames(jmg);
186 
187  // choose search resolution
188  double search_res = search_res_.find(jmg->getName())->second[i]; // we know this exists, by construction
189 
190  if (!result->initialize(node_, jmg->getParentModel(), jmg->getName(),
191  (base.empty() || base[0] != '/') ? base : base.substr(1), tips, search_res))
192  {
193  RCLCPP_ERROR(LOGGER, "Kinematics solver of type '%s' could not be initialized for group '%s'",
194  it->second[i].c_str(), jmg->getName().c_str());
195  result.reset();
196  continue;
197  }
198 
199  result->setDefaultTimeout(jmg->getDefaultIKTimeout());
200  RCLCPP_DEBUG(LOGGER,
201  "Successfully allocated and initialized a kinematics solver of type '%s' with search "
202  "resolution %lf for group '%s' at address %p",
203  it->second[i].c_str(), search_res, jmg->getName().c_str(), result.get());
204  break;
205  }
206  }
207  catch (pluginlib::PluginlibException& e)
208  {
209  RCLCPP_ERROR(LOGGER, "The kinematics plugin (%s) failed to load. Error: %s", it->first.c_str(), e.what());
210  }
211  }
212 
213  if (!result)
214  {
215  RCLCPP_DEBUG(LOGGER, "No usable kinematics solver was found for this group.\n"
216  "Did you load kinematics.yaml into your node's namespace?");
217  }
218  return result;
219  }
220 
221  // cache solver between two consecutive calls
222  // first call in RobotModelLoader::loadKinematicsSolvers() is just to check suitability for jmg
223  // second call in JointModelGroup::setSolverAllocators() is to actually retrieve the instance for use
224  kinematics::KinematicsBasePtr allocKinematicsSolverWithCache(const moveit::core::JointModelGroup* jmg)
225  {
226  boost::mutex::scoped_lock slock(cache_lock_);
227  kinematics::KinematicsBasePtr& cached = instances_[jmg];
228  if (cached.unique())
229  return std::move(cached); // pass on unique instance
230 
231  // create a new instance and store in instances_
232  cached = allocKinematicsSolver(jmg);
233  return cached;
234  }

283 moveit::core::SolverAllocatorFn KinematicsPluginLoader::getLoaderFunction(const srdf::ModelSharedPtr& srdf_model)
284 {
285  moveit::tools::Profiler::ScopedStart prof_start;
286  moveit::tools::Profiler::ScopedBlock prof_block("KinematicsPluginLoader::getLoaderFunction(SRDF)");
287 
288  if (!loader_)
289  {
290  RCLCPP_DEBUG(LOGGER, "Configuring kinematics solvers");
291  groups_.clear();
292 
293  std::map<std::string, std::vector<std::string>> possible_kinematics_solvers;
294  std::map<std::string, std::vector<double>> search_res;
295  std::map<std::string, std::vector<std::string>> iksolver_to_tip_links;
296 
297  if (srdf_model)
298  {
299  const std::vector<srdf::Model::Group>& known_groups = srdf_model->getGroups();
300  if (default_search_resolution_ <= std::numeric_limits<double>::epsilon())
301  default_search_resolution_ = kinematics::KinematicsBase::DEFAULT_SEARCH_DISCRETIZATION;
302 
303  if (default_solver_plugin_.empty())
304  {
305  RCLCPP_DEBUG(LOGGER, "Loading settings for kinematics solvers from the ROS param server ...");
306  // read the list of plugin names for possible kinematics solvers
307  for (const srdf::Model::Group& known_group : known_groups)
308  {
309  std::string base_param_name = known_group.name_;
310  std::string ksolver_param_name = base_param_name + ".kinematics_solver";
311  RCLCPP_DEBUG(LOGGER, "Looking for param %s ", ksolver_param_name.c_str());
312  rclcpp::Parameter ksolver_param = declare_parameter(node_, ksolver_param_name);
313  if (ksolver_param.get_type() == rclcpp::ParameterType::PARAMETER_NOT_SET)
314  {
315  base_param_name = robot_description_ + "_kinematics." + known_group.name_;
316  ksolver_param_name = base_param_name + ".kinematics_solver";
317  RCLCPP_DEBUG(LOGGER, "Looking for param %s ", ksolver_param_name.c_str());
318  ksolver_param = declare_parameter(node_, ksolver_param_name);
319  }
320  if (ksolver_param.get_type() != rclcpp::ParameterType::PARAMETER_NOT_SET)
321  {
322  RCLCPP_DEBUG(LOGGER, "Found param %s", ksolver_param_name.c_str());
323 
324  const auto& ksolver = ksolver_param.as_string();
325  std::stringstream ss(ksolver);
326  bool first = true;
327  while (ss.good() && !ss.eof())
328  {
329  if (first)
330  {
331  first = false;
332  groups_.push_back(known_group.name_);
333  }
334  std::string solver;
335  ss >> solver >> std::ws;
336  possible_kinematics_solvers[known_group.name_].push_back(solver);
337  RCLCPP_DEBUG(LOGGER, "Using kinematics solver '%s' for group '%s'.", solver.c_str(),
338  known_group.name_.c_str());
339  }
340  }
341 
342  std::string ksolver_timeout_param_name = base_param_name + ".kinematics_solver_timeout";
343  rclcpp::Parameter ksolver_timeout_param = declare_parameter(node_, ksolver_timeout_param_name);
344  if (ksolver_timeout_param.get_type() != rclcpp::ParameterType::PARAMETER_NOT_SET)
345  {
346  if (ksolver_timeout_param.get_type() == rclcpp::ParameterType::PARAMETER_DOUBLE)
347  {
348  ik_timeout_[known_group.name_] = ksolver_timeout_param.as_double();
349  }
350  else if (ksolver_timeout_param.get_type() == rclcpp::ParameterType::PARAMETER_INTEGER)
351  { // just in case this is an int
352  ik_timeout_[known_group.name_] = ksolver_timeout_param.as_int();
353  }
354  }
355 
356  std::string ksolver_res_param_name = base_param_name + ".kinematics_solver_search_resolution";
357  rclcpp::Parameter ksolver_res_param = declare_parameter(node_, ksolver_res_param_name);
358  if (ksolver_res_param.get_type() != rclcpp::ParameterType::PARAMETER_NOT_SET)
359  {
360  if (ksolver_res_param.get_type() == rclcpp::ParameterType::PARAMETER_STRING)
361  {
362  const auto& ksolver_res = ksolver_res_param.as_string();
363  std::stringstream ss(ksolver_res);
364  while (ss.good() && !ss.eof())
365  {
366  double res;
367  ss >> res >> std::ws;
368  search_res[known_group.name_].push_back(res);
369  }
370  }
371  // handle the case this param is just one value and parsed as a double
372  else if (ksolver_res_param.get_type() == rclcpp::ParameterType::PARAMETER_DOUBLE)
373  {
374  search_res[known_group.name_].push_back(ksolver_res_param.as_double());
375  }
376  else if (ksolver_res_param.get_type() == rclcpp::ParameterType::PARAMETER_INTEGER)
377  {
378  search_res[known_group.name_].push_back(ksolver_res_param.as_int());
379  }
380  }
381 
382  // Allow a kinematic solver's tip links to be specified on the rosparam server as an array
383  std::string ksolver_ik_links_param_name = base_param_name + ".kinematics_solver_ik_links";
384  rclcpp::Parameter ksolver_ik_links_param = declare_parameter(node_, ksolver_ik_links_param_name);
385  if (ksolver_ik_links_param.get_type() != rclcpp::ParameterType::PARAMETER_NOT_SET)
386  {
387  if (ksolver_ik_links_param.get_type() == rclcpp::ParameterType::PARAMETER_STRING_ARRAY)
388  {
389  const auto& ksolver_ik_links = ksolver_ik_links_param.as_string_array();
390  for (auto& ksolver_ik_link : ksolver_ik_links)
391  {
392  RCLCPP_DEBUG(LOGGER, "found tip %s for group %s", ksolver_ik_link.c_str(), known_group.name_.c_str());
393  iksolver_to_tip_links[known_group.name_].push_back(ksolver_ik_link);
394  }
395  }
396  else
397  {
398  RCLCPP_WARN(LOGGER, "the parameter '%s' needs to be of type 'STRING_ARRAY'",
399  ksolver_ik_links_param_name.c_str());
400  }
401  }
402 
403  // make sure there is a default resolution at least specified for every solver (in case it was not specified
404  // on the param server)
405  while (search_res[known_group.name_].size() < possible_kinematics_solvers[known_group.name_].size())
406  search_res[known_group.name_].push_back(default_search_resolution_);
407  }
408  }
409  else
410  {
411  RCLCPP_DEBUG(LOGGER, "Using specified default settings for kinematics solvers ...");
412  for (const srdf::Model::Group& known_group : known_groups)
413  {
414  possible_kinematics_solvers[known_group.name_].resize(1, default_solver_plugin_);
415  search_res[known_group.name_].resize(1, default_search_resolution_);
416  ik_timeout_[known_group.name_] = default_solver_timeout_;
417  groups_.push_back(known_group.name_);
418  }
419  }
420  }
421 
422  loader_.reset(new KinematicsLoaderImpl(node_, robot_description_, possible_kinematics_solvers, search_res,
423  iksolver_to_tip_links));
424  }
425 
426  return boost::bind(&KinematicsPluginLoader::KinematicsLoaderImpl::allocKinematicsSolverWithCache, loader_.get(),
427  boost::placeholders::_1);
428 }
