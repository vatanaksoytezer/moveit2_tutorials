52 LowPassFilter::LowPassFilter(double low_pass_filter_coeff)
53  : previous_measurements_{ 0., 0. }
54  , previous_filtered_measurement_(0.)
55  , scale_term_(1. / (1. + low_pass_filter_coeff))
56  , feedback_term_(1. - low_pass_filter_coeff)
57 {
58  // guarantee this doesn't change because the logic below depends on this length implicity
59  static_assert(LowPassFilter::FILTER_LENGTH == 2, "moveit_servo::LowPassFilter::FILTER_LENGTH should be 2");
60 
61  // Make sure input values are ok
62  if (std::isinf(feedback_term_))
63  throw std::length_error("moveit_servo::LowPassFilter: infinite feedback_term_");
64 
65  if (std::isinf(scale_term_))
66  throw std::length_error("moveit_servo::LowPassFilter: infinite scale_term_");
67 
68  if (low_pass_filter_coeff < 1)
69  throw std::length_error("moveit_servo::LowPassFilter: Filter coefficient < 1. makes the lowpass filter unstable");
70 
71  if (std::abs(feedback_term_) < EPSILON)
72  throw std::length_error("moveit_servo::LowPassFilter: Filter coefficient value resulted in feedback term of 0");
73 }

83 double LowPassFilter::filter(double new_measurement)
84 {
85  // Push in the new measurement
86  previous_measurements_[1] = previous_measurements_[0];
87  previous_measurements_[0] = new_measurement;
88 
89  double new_filtered_measurement = scale_term_ * (previous_measurements_[1] + previous_measurements_[0] -
90  feedback_term_ * previous_filtered_measurement_);
91 
92  // Store the new filtered measurement
93  previous_filtered_measurement_ = new_filtered_measurement;
94 
95  return new_filtered_measurement;
96 }
