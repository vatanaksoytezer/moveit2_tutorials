117 void MeshShape::endTriangles()
118 {
119  if (started_)
120  {
121  started_ = false;
122  manual_object_->end();
123  static uint32_t count = 0;
124  std::string name = "ConvertedMeshShape@" + boost::lexical_cast<std::string>(count++);
125  manual_object_->convertToMesh(name);
126  entity_ = scene_manager_->createEntity(name);
127  if (entity_)
128  {
129  entity_->setMaterialName(material_name_, "rviz_rendering");
130  offset_node_->attachObject(entity_);
131  }
132  else
133  RVIZ_COMMON_LOG_ERROR("Unable to construct triangle mesh");
134  }
135  else
136  RVIZ_COMMON_LOG_ERROR("No triangles added");
137 }
