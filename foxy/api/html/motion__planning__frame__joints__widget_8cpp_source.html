85 QVariant JMGItemModel::data(const QModelIndex& index, int role) const
86 {
87  if (!index.isValid())
88  return QVariant();
89 
90  int idx = jmg_ ? jmg_->getVariableIndexList()[index.row()] : index.row();
91  switch (index.column())
92  {
93  case 0: // joint name column
94  switch (role)
95  {
96  case Qt::DisplayRole:
97  return QString::fromStdString(robot_state_.getVariableNames()[idx]);
98  case Qt::TextAlignmentRole:
99  return Qt::AlignLeft;
100  }
101  break;
102  case 1: // joint value column
103  {
104  double value = robot_state_.getVariablePosition(idx);
105  const moveit::core::JointModel* jm = robot_state_.getRobotModel()->getJointOfVariable(idx);
106  switch (role)
107  {
108  case Qt::DisplayRole:
109  return value;
110  case Qt::EditRole:
111  if (jm)
112  return jm->getType() == moveit::core::JointModel::REVOLUTE ? value * 180 / M_PI : value;
113  break;
114  case ProgressBarDelegate::JointTypeRole:
115  if (jm)
116  return jm->getType();
117  break;
118  case ProgressBarDelegate::VariableBoundsRole:
119  if (const moveit::core::VariableBounds* bounds = getVariableBounds(jm, index))
120  return QPointF(bounds->min_position_, bounds->max_position_);
121  break;
122  case Qt::TextAlignmentRole:
123  return Qt::AlignLeft;
124  }
125  }
126  }
127  return QVariant();
128 }

137 bool JMGItemModel::setData(const QModelIndex& index, const QVariant& value, int role)
138 {
139  if (index.column() != 1 || role != Qt::EditRole)
140  return false;
141 
142  int var_idx = jmg_ ? jmg_->getVariableIndexList()[index.row()] : index.row();
143  const moveit::core::JointModel* jm = robot_state_.getRobotModel()->getJointOfVariable(var_idx);
144  if (!value.canConvert<double>())
145  return false;
146 
147  bool ok;
148  double v = value.toDouble(&ok);
149  if (!ok)
150  return false;
151 
152  // for revolute joints, we convert degrees to radians
153  if (jm && jm->getType() == moveit::core::JointModel::REVOLUTE)
154  v *= M_PI / 180;
155 
156  robot_state_.setVariablePosition(var_idx, v);
157  jm->enforcePositionBounds(robot_state_.getVariablePositions() + jm->getFirstVariableIndex());
158  dataChanged(index, index);
159  return true;

272 // Find matching key vector in columns of haystack and return the best-aligned column index.
273 // Only consider available indexes (> 0). If no match is found, return largest available index.
274 // sign becomes -1 if key has changed sign compared to the matching haystack column.
275 static Eigen::Index findMatching(const Eigen::VectorXd& key, const Eigen::MatrixXd& haystack,
276  const Eigen::VectorXi& available, double& sign)
277 {
278  Eigen::Index result = available.array().maxCoeff();
279  double best_match = 0.0;
280  for (unsigned int i = 0; i < available.rows(); ++i)
281  {
282  int index = available[i];
283  if (index < 0) // index already taken
284  continue;
285  if (index >= haystack.cols())
286  return result;
287  double match = haystack.col(available[i]).transpose() * key;
288  double abs_match = std::abs(match);
289  if (abs_match > 0.5 && abs_match > best_match)
290  {
291  best_match = abs_match;
292  result = index;
293  sign = match > 0 ? 1.0 : -1.0;
294  }
295  }
296  return result;
297 }

299 void MotionPlanningFrameJointsWidget::updateNullspaceSliders()
300 {
301  JMGItemModel* model = dynamic_cast<JMGItemModel*>(ui_->joints_view_->model());
302  std::size_t i = 0;
303  if (model && model->getJointModelGroup() && model->getJointModelGroup()->isChain())
304  {
305  model->getRobotState().updateLinkTransforms();
306  Eigen::MatrixXd jacobian;
307  if (!model->getRobotState().getJacobian(model->getJointModelGroup(),
308  model->getJointModelGroup()->getLinkModels().back(),
309  Eigen::Vector3d::Zero(), jacobian, false))
310  goto cleanup;
311 
312  svd_.compute(jacobian, Eigen::ComputeFullV);
313  Eigen::Index rank = svd_.rank();
314  std::size_t ns_dim = svd_.cols() - rank;
315  Eigen::MatrixXd ns(svd_.cols(), ns_dim);
316  Eigen::VectorXi available(ns_dim);
317  for (std::size_t j = 0; j < ns_dim; ++j)
318  available[j] = j;
319 
320  ns_sliders_.reserve(ns_dim);
321  // create/unhide sliders
322  for (; i < ns_dim; ++i)
323  {
324  if (i >= ns_sliders_.size())
325  ns_sliders_.push_back(createNSSlider(i + 1));
326  ns_sliders_[i]->show();
327 
328  // Find matching null-space basis vector in previous nullspace_
329  double sign = 1.0;
330  const Eigen::VectorXd& current = svd_.matrixV().col(rank + i);
331  Eigen::Index index = findMatching(current, nullspace_, available, sign);
332  ns.col(index).noalias() = sign * current;
333  available[index] = -1; // mark index as taken
334  }
335  nullspace_ = ns;
336  }
337 
338 cleanup:
339  if (i == 0)
340  nullspace_.resize(0, 0);
341 
342  // hide remaining sliders
343  for (; i < ns_sliders_.size(); ++i)
344  ns_sliders_[i]->hide();
345 }

358 void MotionPlanningFrameJointsWidget::jogNullspace(double value)
359 {
360  if (value == 0)
361  return;
362 
363  std::size_t index = std::find(ns_sliders_.begin(), ns_sliders_.end(), sender()) - ns_sliders_.begin();
364  if (static_cast<int>(index) >= nullspace_.cols())
365  return;
366 
367  JMGItemModel* model = dynamic_cast<JMGItemModel*>(ui_->joints_view_->model());
368  if (!model)
369  return;
370 
371  Eigen::VectorXd values;
372  model->getRobotState().copyJointGroupPositions(model->getJointModelGroup(), values);
373  values += value * nullspace_.col(index);
374  model->getRobotState().setJointGroupPositions(model->getJointModelGroup(), values);
375  model->getRobotState().harmonizePositions(model->getJointModelGroup());
376  triggerUpdate(model);
377 }

380 {

383  QStyleOptionViewItem style_option = option;
384  initStyleOption(&style_option, index);

410  const float max = bounds.y();

412  QStyleOptionProgressBar opt;
413  opt.rect = option.rect;
414  opt.minimum = 0;
415  opt.maximum = 1000;
416  opt.progress = 1000. * (value - min) / (max - min);
417  opt.text = style_option.text;
418  opt.textAlignment = style_option.displayAlignment;
419  opt.textVisible = true;
420  style->drawControl(QStyle::CE_ProgressBar, &opt, painter);
421  return;
422  }
423  }
