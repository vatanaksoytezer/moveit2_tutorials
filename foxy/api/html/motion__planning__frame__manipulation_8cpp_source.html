72 void MotionPlanningFrame::processDetectedObjects()
73 {
74  pick_object_name_.clear();
75 
76  std::vector<std::string> object_ids;
77  double min_x = ui_->roi_center_x->value() - ui_->roi_size_x->value() / 2.0;
78  double min_y = ui_->roi_center_y->value() - ui_->roi_size_y->value() / 2.0;
79  double min_z = ui_->roi_center_z->value() - ui_->roi_size_z->value() / 2.0;
80 
81  double max_x = ui_->roi_center_x->value() + ui_->roi_size_x->value() / 2.0;
82  double max_y = ui_->roi_center_y->value() + ui_->roi_size_y->value() / 2.0;
83  double max_z = ui_->roi_center_z->value() + ui_->roi_size_z->value() / 2.0;
84 
85  rclcpp::Time start_time = rclcpp::Clock().now();
86  while (object_ids.empty() && (rclcpp::Clock().now() - start_time) <= rclcpp::Duration(3.0))
87  {
88  // collect all objects in region of interest
89  {
90  const planning_scene_monitor::LockedPlanningSceneRO& ps = planning_display_->getPlanningSceneRO();
91  const collision_detection::WorldConstPtr& world = ps->getWorld();
92  object_ids.clear();
93  for (const auto& object : *world)
94  {
95  if (!object.second->shape_poses_.empty())
96  {
97  const auto& position = object.second->shape_poses_[0].translation();
98  if (position.x() >= min_x && position.x() <= max_x && position.y() >= min_y && position.y() <= max_y &&
99  position.z() >= min_z && position.z() <= max_z)
100  {
101  object_ids.push_back(object.first);
102  }
103  }
104  }
105  if (!object_ids.empty())
106  break;
107  }
108  rclcpp::sleep_for(std::chrono::milliseconds(500));
109  }
110 
111  RCLCPP_DEBUG(LOGGER, "Found %d objects", (int)object_ids.size());
112  updateDetectedObjectsList(object_ids);
113 }

143 void MotionPlanningFrame::triggerObjectDetection()
144 {
145  if (!object_recognition_client_)
146  {
147  object_recognition_client_ = rclcpp_action::create_client<object_recognition_msgs::action::ObjectRecognition>(
148  node_, OBJECT_RECOGNITION_ACTION);
149  if (!object_recognition_client_->wait_for_action_server(std::chrono::seconds(3)))
150  {
151  RCLCPP_ERROR(LOGGER, "Object recognition action server not responsive");
152  return;
153  }
154  }
155 
156  object_recognition_msgs::action::ObjectRecognition::Goal goal;
157  auto goal_handle_future = object_recognition_client_->async_send_goal(goal);
158  goal_handle_future.wait();
159  if (goal_handle_future.get()->get_status() != rclcpp_action::GoalStatus::STATUS_SUCCEEDED)
160  {
161  RCLCPP_ERROR(LOGGER, "ObjectRecognition client: send goal call failed");
162  return;
163  }
164 }
