96  void status()

99  {

119 private:
120  void configureCapabilities()
121  {
122  try
123  {
124  capability_plugin_loader_.reset(
125  new pluginlib::ClassLoader<MoveGroupCapability>("moveit_ros_move_group", "move_group::MoveGroupCapability"));
126  }
127  catch (pluginlib::PluginlibException& ex)
128  {
129  RCLCPP_FATAL_STREAM(LOGGER, "Exception while creating plugin loader for move_group capabilities: " << ex.what());
130  return;
131  }
132 
133  std::set<std::string> capabilities;
134 
135  // add default capabilities
136  for (const char* capability : DEFAULT_CAPABILITIES)
137  capabilities.insert(capability);
138 
139  // add capabilities listed in ROS parameter
140  std::string capability_plugins;
141  if (context_->moveit_cpp_->getNode()->get_parameter("capabilities", capability_plugins))
142  {
143  boost::char_separator<char> sep(" ");
144  boost::tokenizer<boost::char_separator<char> > tok(capability_plugins, sep);
145  capabilities.insert(tok.begin(), tok.end());
146  }
147 
148  // add capabilities configured for planning pipelines
149  for (const auto& pipeline_entry : context_->moveit_cpp_->getPlanningPipelines())
150  {
151  const auto& pipeline_name = pipeline_entry.first;
152  std::string pipeline_capabilities;
153  if (context_->moveit_cpp_->getNode()->get_parameter("planning_pipelines/" + pipeline_name + "/capabilities",
154  pipeline_capabilities))
155  {
156  boost::char_separator<char> sep(" ");
157  boost::tokenizer<boost::char_separator<char> > tok(pipeline_capabilities, sep);
158  capabilities.insert(tok.begin(), tok.end());
159  }
160  }
161 
162  // drop capabilities that have been explicitly disabled
163  if (context_->moveit_cpp_->getNode()->get_parameter("disable_capabilities", capability_plugins))
164  {
165  boost::char_separator<char> sep(" ");
166  boost::tokenizer<boost::char_separator<char> > tok(capability_plugins, sep);
167  for (boost::tokenizer<boost::char_separator<char> >::iterator cap_name = tok.begin(); cap_name != tok.end();
168  ++cap_name)
169  capabilities.erase(*cap_name);
170  }
171 
172  for (const std::string& capability : capabilities)
173  {
174  try
175  {
176  printf(MOVEIT_CONSOLE_COLOR_CYAN "Loading '%s'...\n" MOVEIT_CONSOLE_COLOR_RESET, capability.c_str());
177  MoveGroupCapabilityPtr cap = capability_plugin_loader_->createUniqueInstance(capability);
178  cap->setContext(context_);
179  cap->initialize();
180  capabilities_.push_back(cap);
181  }
182  catch (pluginlib::PluginlibException& ex)
183  {
184  RCLCPP_ERROR_STREAM(LOGGER,
185  "Exception while loading move_group capability '" << capability << "': " << ex.what());
186  }
187  }
188 
189  std::stringstream ss;
190  ss << std::endl;
191  ss << std::endl;
192  ss << "********************************************************" << std::endl;
193  ss << "* MoveGroup using: " << std::endl;
194  for (const MoveGroupCapabilityPtr& cap : capabilities_)
195  ss << "* - " << cap->getName() << std::endl;
196  ss << "********************************************************" << std::endl;
197  RCLCPP_INFO(LOGGER, "%s", ss.str().c_str());
198  }
199 
200  MoveGroupContextPtr context_;
201  std::shared_ptr<pluginlib::ClassLoader<MoveGroupCapability> > capability_plugin_loader_;
202  std::vector<MoveGroupCapabilityPtr> capabilities_;
203 };

206 int main(int argc, char** argv)
207 {
208  rclcpp::init(argc, argv);
209 
210  rclcpp::NodeOptions opt;
211  opt.allow_undeclared_parameters(true);
212  opt.automatically_declare_parameters_from_overrides(true);
213  rclcpp::Node::SharedPtr nh = rclcpp::Node::make_shared("move_group", opt);
214  moveit_cpp::MoveItCpp::Options moveit_cpp_options(nh);
215 
216  // Prepare PlanningPipelineOptions
217  moveit_cpp_options.planning_pipeline_options.parent_namespace = nh->get_effective_namespace() + ".planning_pipelines";
218  std::vector<std::string> planning_pipeline_configs;
219  if (nh->get_parameter("planning_pipelines", planning_pipeline_configs))
220  {
221  if (planning_pipeline_configs.empty())
222  {
223  RCLCPP_ERROR(LOGGER, "Failed to read parameter 'move_group.planning_pipelines'");
224  }
225  else
226  {
227  for (const auto& config : planning_pipeline_configs)
228  {
229  moveit_cpp_options.planning_pipeline_options.pipeline_names.push_back(config);
230  }
231  }
232  }
233 
234  // Retrieve default planning pipeline
235  auto& pipeline_names = moveit_cpp_options.planning_pipeline_options.pipeline_names;
236  std::string default_planning_pipeline;
237  if (nh->get_parameter("default_planning_pipeline", default_planning_pipeline))
238  {
239  // Ignore default_planning_pipeline if there is no known entry in pipeline_names
240  if (std::find(pipeline_names.begin(), pipeline_names.end(), default_planning_pipeline) == pipeline_names.end())
241  {
242  RCLCPP_WARN(LOGGER,
243  "MoveGroup launched with ~default_planning_pipeline '%s' not configured in ~/planning_pipelines",
244  default_planning_pipeline.c_str());
245  default_planning_pipeline = ""; // reset invalid pipeline id
246  }
247  }
248  else
249  {
250  // Handle deprecated move_group.launch
251  RCLCPP_WARN(LOGGER,
252  "MoveGroup launched without ~default_planning_pipeline specifying the namespace for the default "
253  "planning pipeline configuration");
254  }
255 
256  // If there is no valid default pipeline, either pick the first available one, or fall back to old behavior
257  if (default_planning_pipeline.empty())
258  {
259  if (!pipeline_names.empty())
260  {
261  RCLCPP_WARN(LOGGER, "Using default pipeline '%s'", pipeline_names[0].c_str());
262  default_planning_pipeline = pipeline_names[0];
263  }
264  else
265  {
266  RCLCPP_WARN(LOGGER, "Falling back to using the the move_group node namespace (deprecated behavior).");
267  default_planning_pipeline = "move_group";
268  moveit_cpp_options.planning_pipeline_options.pipeline_names = { default_planning_pipeline };
269  moveit_cpp_options.planning_pipeline_options.parent_namespace = nh->get_effective_namespace();
270  }
271 
272  // Reset invalid pipeline parameter for MGI requests
273  nh->set_parameter(rclcpp::Parameter("default_planning_pipeline", default_planning_pipeline));
274  }
275 
276  // Initialize MoveItCpp
277  const auto tf_buffer = std::make_shared<tf2_ros::Buffer>(nh->get_clock(), tf2::durationFromSec(10.0));
278  const auto moveit_cpp = std::make_shared<moveit_cpp::MoveItCpp>(nh, moveit_cpp_options, tf_buffer);
279  const auto planning_scene_monitor = moveit_cpp->getPlanningSceneMonitor();
280 
281  if (planning_scene_monitor->getPlanningScene())
282  {
283  bool debug = false;
284  for (int i = 1; i < argc; ++i)
285  if (strncmp(argv[i], "--debug", 7) == 0)
286  {
287  debug = true;
288  break;
289  }
290  debug = true;
291  if (debug)
292  RCLCPP_INFO(LOGGER, "MoveGroup debug mode is ON");
293  else
294  RCLCPP_INFO(LOGGER, "MoveGroup debug mode is OFF");
295 
296  rclcpp::executors::MultiThreadedExecutor executor;
297 
298  move_group::MoveGroupExe mge(moveit_cpp, default_planning_pipeline, debug);
299 
300  planning_scene_monitor->publishDebugInformation(debug);
301 
302  mge.status();
303  executor.add_node(nh);
304  executor.spin();
305 
306  rclcpp::shutdown();
307  }
308  else
309  RCLCPP_ERROR(LOGGER, "Planning scene not configured");
310 
311  return 0;
312 }
