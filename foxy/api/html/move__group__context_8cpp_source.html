46 move_group::MoveGroupContext::MoveGroupContext(const moveit_cpp::MoveItCppPtr& moveit_cpp,
47  const std::string& default_planning_pipeline,
48  bool allow_trajectory_execution, bool debug)
49  : moveit_cpp_(moveit_cpp)
50  , planning_scene_monitor_(moveit_cpp->getPlanningSceneMonitor())
51  , allow_trajectory_execution_(allow_trajectory_execution)
52  , debug_(debug)
53 {
54  // Check if default planning pipeline has been initialized successfully
55  const auto& pipelines = moveit_cpp->getPlanningPipelines();
56  const auto default_pipeline_it = pipelines.find(default_planning_pipeline);
57  if (default_pipeline_it != pipelines.end())
58  {
59  planning_pipeline_ = default_pipeline_it->second;
60 
61  // configure the planning pipeline
62  planning_pipeline_->displayComputedMotionPlans(true);
63  planning_pipeline_->checkSolutionPaths(true);
64 
65  if (debug_)
66  planning_pipeline_->publishReceivedRequests(true);
67  }
68  else
69  {
70  RCLCPP_ERROR(
71  LOGGER,
72  "Failed to find default PlanningPipeline '%s' - please check MoveGroup's planning pipeline configuration.",
73  default_planning_pipeline.c_str());
74  }
75 
76  if (allow_trajectory_execution_)
77  {
78  trajectory_execution_manager_ = moveit_cpp_->getTrajectoryExecutionManager();
79  plan_execution_.reset(new plan_execution::PlanExecution(moveit_cpp_->getNode(), planning_scene_monitor_,
80  trajectory_execution_manager_));
81  plan_with_sensing_.reset(new plan_execution::PlanWithSensing(moveit_cpp_->getNode(), trajectory_execution_manager_));
82  if (debug)
83  plan_with_sensing_->displayCostSources(true);
84  }
85 }
