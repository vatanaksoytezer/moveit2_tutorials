74 void MoveGroupSequenceAction::executeSequenceCallback(const moveit_msgs::MoveGroupSequenceGoalConstPtr& goal)
75 {
76  setMoveState(move_group::PLANNING);
77 
78  // Handle empty requests
79  if (goal->request.items.empty())
80  {
81  ROS_WARN("Received empty request. That's ok but maybe not what you intended.");
82  setMoveState(move_group::IDLE);
83  moveit_msgs::MoveGroupSequenceResult action_res;
84  action_res.response.error_code.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
85  move_action_server_->setSucceeded(action_res, "Received empty request.");
86  return;
87  }
88 
89  // before we start planning, ensure that we have the latest robot state
90  // received...
91  context_->planning_scene_monitor_->waitForCurrentRobotState(ros::Time::now());
92  context_->planning_scene_monitor_->updateFrameTransforms();
93 
94  moveit_msgs::MoveGroupSequenceResult action_res;
95  if (goal->planning_options.plan_only || !context_->allow_trajectory_execution_)
96  {
97  if (!goal->planning_options.plan_only)
98  {
99  ROS_WARN("Only plan will be calculated, although plan_only == false."); // LCOV_EXCL_LINE
100  }
101  executeMoveCallbackPlanOnly(goal, action_res);
102  }
103  else
104  {
105  executeSequenceCallbackPlanAndExecute(goal, action_res);
106  }
107 
108  switch (action_res.response.error_code.val)
109  {
110  case moveit_msgs::MoveItErrorCodes::SUCCESS:
111  move_action_server_->setSucceeded(action_res, "Success");
112  break;
113  case moveit_msgs::MoveItErrorCodes::PREEMPTED:
114  move_action_server_->setPreempted(action_res, "Preempted");
115  break;
116  default:
117  move_action_server_->setAborted(action_res, "See error code for more information");
118  break;
119  }
120 
121  setMoveState(move_group::IDLE);
122 }

124 void MoveGroupSequenceAction::executeSequenceCallbackPlanAndExecute(
125  const moveit_msgs::MoveGroupSequenceGoalConstPtr& goal, moveit_msgs::MoveGroupSequenceResult& action_res)
126 {
127  ROS_INFO("Combined planning and execution request received for "
128  "MoveGroupSequenceAction.");
129 
130  plan_execution::PlanExecution::Options opt;
131 
132  const moveit_msgs::PlanningScene& planning_scene_diff =
133  moveit::core::isEmpty(goal->planning_options.planning_scene_diff.robot_state) ?
134  goal->planning_options.planning_scene_diff :
135  clearSceneRobotState(goal->planning_options.planning_scene_diff);
136 
137  opt.replan_ = goal->planning_options.replan;
138  opt.replan_attempts_ = goal->planning_options.replan_attempts;
139  opt.replan_delay_ = goal->planning_options.replan_delay;
140  opt.before_execution_callback_ = boost::bind(&MoveGroupSequenceAction::startMoveExecutionCallback, this);
141 
142  opt.plan_callback_ =
143  boost::bind(&MoveGroupSequenceAction::planUsingSequenceManager, this, boost::cref(goal->request), _1);
144 
145  if (goal->planning_options.look_around && context_->plan_with_sensing_)
146  {
147  ROS_WARN("Plan with sensing not yet implemented/tested. This option is "
148  "ignored."); // LCOV_EXCL_LINE
149  }
150 
151  plan_execution::ExecutableMotionPlan plan;
152  context_->plan_execution_->planAndExecute(plan, planning_scene_diff, opt);
153 
154  StartStatesMsg start_states_msg;
155  convertToMsg(plan.plan_components_, start_states_msg, action_res.response.planned_trajectories);
156  try
157  {
158  action_res.response.sequence_start = start_states_msg.at(0);
159  }
160  catch (std::out_of_range&)
161  {
162  ROS_WARN("Can not determine start state from empty sequence.");
163  }
164  action_res.response.error_code = plan.error_code_;
165 }

168  PlannedTrajMsgs& planned_trajs_msgs)

179 void MoveGroupSequenceAction::executeMoveCallbackPlanOnly(const moveit_msgs::MoveGroupSequenceGoalConstPtr& goal,
180  moveit_msgs::MoveGroupSequenceResult& res)
181 {
182  ROS_INFO("Planning request received for MoveGroupSequenceAction action.");
183 
184  // lock the scene so that it does not modify the world representation while
185  // diff() is called
186  planning_scene_monitor::LockedPlanningSceneRO lscene(context_->planning_scene_monitor_);
187 
188  const planning_scene::PlanningSceneConstPtr& the_scene =
189  (moveit::core::isEmpty(goal->planning_options.planning_scene_diff)) ?
190  static_cast<const planning_scene::PlanningSceneConstPtr&>(lscene) :
191  lscene->diff(goal->planning_options.planning_scene_diff);
192 
193  ros::Time planning_start = ros::Time::now();
194  RobotTrajCont traj_vec;
195  try
196  {
197  traj_vec = command_list_manager_->solve(the_scene, context_->planning_pipeline_, goal->request);
198  }
199  catch (const MoveItErrorCodeException& ex)
200  {
201  ROS_ERROR_STREAM("> Planning pipeline threw an exception (error code: " << ex.getErrorCode() << "): " << ex.what());
202  res.response.error_code.val = ex.getErrorCode();
203  return;
204  }
205  // LCOV_EXCL_START // Keep moveit up even if lower parts throw
206  catch (const std::exception& ex)
207  {
208  ROS_ERROR("Planning pipeline threw an exception: %s", ex.what());
209  res.response.error_code.val = moveit_msgs::MoveItErrorCodes::FAILURE;
210  return;
211  }
212  // LCOV_EXCL_STOP
213 
214  StartStatesMsg start_states_msg;
215  start_states_msg.resize(traj_vec.size());
216  res.response.planned_trajectories.resize(traj_vec.size());
217  for (RobotTrajCont::size_type i = 0; i < traj_vec.size(); ++i)
218  {
219  move_group::MoveGroupCapability::convertToMsg(traj_vec.at(i), start_states_msg.at(i),
220  res.response.planned_trajectories.at(i));
221  }
222  try
223  {
224  res.response.sequence_start = start_states_msg.at(0);
225  }
226  catch (std::out_of_range&)
227  {
228  ROS_WARN("Can not determine start state from empty sequence.");
229  }
230 
231  res.response.error_code.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
232  res.response.planning_time = ros::Time::now().toSec() - planning_start.toSec();
233 }

235 bool MoveGroupSequenceAction::planUsingSequenceManager(const moveit_msgs::MotionSequenceRequest& req,
236  plan_execution::ExecutableMotionPlan& plan)
237 {
238  setMoveState(move_group::PLANNING);
239 
240  planning_scene_monitor::LockedPlanningSceneRO lscene(plan.planning_scene_monitor_);
241  RobotTrajCont traj_vec;
242  try
243  {
244  // Select planning_pipeline to handle request
245  // All motions in the SequenceRequest need to use the same planning pipeline (but can use different planners)
246  const planning_pipeline::PlanningPipelinePtr planning_pipeline =
247  resolvePlanningPipeline(req.items[0].req.pipeline_id);
248  if (!planning_pipeline)
249  {
250  ROS_ERROR_STREAM("Could not load planning pipeline " << req.items[0].req.pipeline_id);
251  return false;
252  }
253 
254  traj_vec = command_list_manager_->solve(plan.planning_scene_, planning_pipeline, req);
255  }
256  catch (const MoveItErrorCodeException& ex)
257  {
258  ROS_ERROR_STREAM("Planning pipeline threw an exception (error code: " << ex.getErrorCode() << "): " << ex.what());
259  plan.error_code_.val = ex.getErrorCode();
260  return false;
261  }
262  // LCOV_EXCL_START // Keep MoveIt up even if lower parts throw
263  catch (const std::exception& ex)
264  {
265  ROS_ERROR_STREAM("Planning pipeline threw an exception: " << ex.what());
266  plan.error_code_.val = moveit_msgs::MoveItErrorCodes::FAILURE;
267  return false;
268  }
269  // LCOV_EXCL_STOP
270 
271  if (!traj_vec.empty())
272  {
273  plan.plan_components_.resize(traj_vec.size());
274  for (size_t i = 0; i < traj_vec.size(); ++i)
275  {
276  plan.plan_components_.at(i).trajectory_ = traj_vec.at(i);
277  plan.plan_components_.at(i).description_ = "plan";
278  }
279  }
280  plan.error_code_.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
281  return true;
282 }
