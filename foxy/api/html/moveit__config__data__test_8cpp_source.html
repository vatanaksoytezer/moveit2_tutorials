67 TEST_F(MoveItConfigData, ReadingControllers)
68 {
69  // Contains all the configuration data for the setup assistant
70  moveit_setup_assistant::MoveItConfigDataPtr config_data;
71 
72  config_data.reset(new moveit_setup_assistant::MoveItConfigData());
73  config_data->srdf_->srdf_model_ = srdf_model_;
74  config_data->setRobotModel(robot_model_);
75 
76  // Initially no controllers
77  EXPECT_EQ(config_data->getROSControllers().size(), 0u);
78 
79  // Adding default controllers, a controller for each planning group
80  config_data->addDefaultControllers();
81 
82  // Number of the planning groups defined in the model srdf
83  size_t group_count = config_data->srdf_->srdf_model_->getGroups().size();
84 
85  // Test that addDefaultControllers() did accually add a controller for the new_group
86  EXPECT_EQ(config_data->getROSControllers().size(), group_count);
87 
88  // Temporary file used during the test and is deleted when the test is finished
89  char test_file[] = "/tmp/msa_unittest_ros_controller.yaml";
90 
91  // ros_controller.yaml written correctly
92  EXPECT_EQ(config_data->outputROSControllersYAML(test_file), true);
93 
94  // Reset MoveIt config MoveItConfigData
95  config_data.reset(new moveit_setup_assistant::MoveItConfigData());
96 
97  // Initially no controllers
98  EXPECT_EQ(config_data->getROSControllers().size(), 0u);
99 
100  // ros_controllers.yaml read correctly
101  EXPECT_EQ(config_data->inputROSControllersYAML(test_file), true);
102 
103  // ros_controllers.yaml parsed correctly
104  EXPECT_EQ(config_data->getROSControllers().size(), group_count);
105 
106  // Remove ros_controllers.yaml temp file which was used in testing
107  boost::filesystem::remove(test_file);
108 }

138 TEST_F(MoveItConfigData, WritingSensorsConfig)
139 {
140  // Contains all the config data for the setup assistant
141  moveit_setup_assistant::MoveItConfigDataPtr config_data;
142  config_data.reset(new moveit_setup_assistant::MoveItConfigData());
143 
144  // Empty Config Should have No Sensors
145  EXPECT_EQ(config_data->getSensorPluginConfig().size(), 0u);
146 
147  // Temporary file used during the test and is deleted when the test is finished
148  char test_file[] = "/tmp/msa_unittest_sensors.yaml";
149 
150  // sensors yaml written correctly
151  EXPECT_EQ(config_data->output3DSensorPluginYAML(test_file), true);
152 
153  // Set default file
154  boost::filesystem::path setup_assistant_path(config_data->setup_assistant_path_);
155  std::string default_file_path =
156  (setup_assistant_path / "templates/moveit_config_pkg_template/config/sensors_3d.yaml").string();
157 
158  // Read from the written file
159  config_data.reset(new moveit_setup_assistant::MoveItConfigData());
160  EXPECT_EQ(config_data->input3DSensorsYAML(test_file), true);
161 
162  // Should still have No Sensors
163  EXPECT_EQ(config_data->getSensorPluginConfig().size(), 0u);
164 
165  // Now load the default file and write it to a file
166  config_data.reset(new moveit_setup_assistant::MoveItConfigData());
167  EXPECT_EQ(config_data->input3DSensorsYAML(default_file_path), true);
168  EXPECT_EQ(config_data->getSensorPluginConfig().size(), 2u);
169  EXPECT_EQ(config_data->output3DSensorPluginYAML(test_file), true);
170 
171  // Read from the written file
172  config_data.reset(new moveit_setup_assistant::MoveItConfigData());
173  EXPECT_EQ(config_data->input3DSensorsYAML(test_file), true);
174 
175  // Should now have two sensors
176  EXPECT_EQ(config_data->getSensorPluginConfig().size(), 2u);
177 }
