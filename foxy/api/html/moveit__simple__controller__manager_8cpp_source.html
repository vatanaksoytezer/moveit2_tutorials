50 static const std::string PARAM_BASE_NAME = "moveit_simple_controller_manager";
51 class MoveItSimpleControllerManager : public moveit_controller_manager::MoveItControllerManager
52 {
53 public:

58  void initialize(const rclcpp::Node::SharedPtr& node) override
59  {
60  node_ = node;
61  if (!node_->has_parameter(PARAM_BASE_NAME + ".controller_names"))
62  {
63  RCLCPP_ERROR_STREAM(LOGGER, "No controller_names specified.");
64  return;
65  }
66  rclcpp::Parameter controller_names_param;
67  node_->get_parameter(PARAM_BASE_NAME + ".controller_names", controller_names_param);
68  if (controller_names_param.get_type() != rclcpp::ParameterType::PARAMETER_STRING_ARRAY)
69  {
70  RCLCPP_ERROR(LOGGER, "Parameter controller_names should be specified as a string array");
71  return;
72  }
73  std::vector<std::string> controller_names = controller_names_param.as_string_array();
74  /* actually create each controller */
75  for (const std::string& controller_name : controller_names)
76  {
77  try
78  {
79  std::string action_ns;
80  if (!node_->get_parameter(PARAM_BASE_NAME + "." + controller_name + ".action_ns", action_ns))
81  {
82  RCLCPP_ERROR_STREAM(LOGGER, "No action namespace specified for controller " << controller_name);
83  RCLCPP_ERROR_STREAM(LOGGER, "Parameter: " << (PARAM_BASE_NAME + "." + controller_name + ".action"));
84  continue;
85  }
86 
87  std::string type;
88  if (!node_->get_parameter(PARAM_BASE_NAME + "." + controller_name + ".type", type))
89  {
90  RCLCPP_ERROR_STREAM(LOGGER, "No type specified for controller " << controller_name);
91  continue;
92  }
93 
94  std::vector<std::string> controller_joints;
95  if (!node_->get_parameter(PARAM_BASE_NAME + "." + controller_name + ".joints", controller_joints) ||
96  controller_joints.empty())
97  {
98  RCLCPP_ERROR_STREAM(LOGGER, "No joints specified for controller " << controller_name);
99  continue;
100  }
101 
102  ActionBasedControllerHandleBasePtr new_handle;
103  if (type == "GripperCommand")
104  {
105  double max_effort;
106  if (!node->get_parameter(PARAM_BASE_NAME + "." + controller_name + ".max_effort", max_effort))
107  {
108  RCLCPP_INFO_STREAM(LOGGER, "Max effort set to 0.0");
109  max_effort = 0.0;
110  }
111 
112  new_handle = std::make_shared<GripperControllerHandle>(node_, controller_name, action_ns, max_effort);
113  if (static_cast<GripperControllerHandle*>(new_handle.get())->isConnected())
114  {
115  bool parallel_gripper = false;
116  if (node_->get_parameter(PARAM_BASE_NAME + ".parallel", parallel_gripper) && parallel_gripper)
117  {
118  if (controller_joints.size() != 2)
119  {
120  RCLCPP_ERROR_STREAM(LOGGER, "Parallel Gripper requires exactly two joints, " << controller_joints.size()
121  << " are specified");
122  continue;
123  }
124  static_cast<GripperControllerHandle*>(new_handle.get())
125  ->setParallelJawGripper(controller_joints[0], controller_joints[1]);
126  }
127  else
128  {
129  std::string command_joint;
130  if (!node_->get_parameter(PARAM_BASE_NAME + ".command_joint", command_joint))
131  command_joint = controller_joints[0];
132 
133  static_cast<GripperControllerHandle*>(new_handle.get())->setCommandJoint(command_joint);
134  }
135 
136  bool allow_failure;
137  node_->get_parameter_or(PARAM_BASE_NAME + ".allow_failure", allow_failure, false);
138  static_cast<GripperControllerHandle*>(new_handle.get())->allowFailure(allow_failure);
139 
140  RCLCPP_INFO_STREAM(LOGGER, "Added GripperCommand controller for " << controller_name);
141  controllers_[controller_name] = new_handle;
142  }
143  }
144  else if (type == "FollowJointTrajectory")
145  {
146  auto h = new FollowJointTrajectoryControllerHandle(node_, controller_name, action_ns);
147  new_handle.reset(h);
148  if (h->isConnected())
149  {
150  RCLCPP_INFO_STREAM(LOGGER, "Added FollowJointTrajectory controller for " << controller_name);
151  controllers_[controller_name] = new_handle;
152  }
153  }
154  else
155  {
156  RCLCPP_ERROR_STREAM(LOGGER, "Unknown controller type: " << type);
157  continue;
158  }
159  if (!controllers_[controller_name])
160  {
161  controllers_.erase(controller_name);
162  continue;
163  }
164 
165  moveit_controller_manager::MoveItControllerManager::ControllerState state;
166  node_->get_parameter_or(PARAM_BASE_NAME + "." + controller_name + ".default", state.default_, false);
167  state.active_ = true;
168 
169  controller_states_[controller_name] = state;
170 
171  /* add list of joints, used by controller manager and MoveIt */
172  for (const std::string& controller_joint : controller_joints)
173  new_handle->addJoint(controller_joint);
174  }
175  catch (...)
176  {
177  RCLCPP_ERROR_STREAM(LOGGER, "Caught unknown exception while parsing controller information");
178  }
179  }
180  }
181  /*
182  * Get a controller, by controller name (which was specified in the controllers.yaml
183  */
184  moveit_controller_manager::MoveItControllerHandlePtr getControllerHandle(const std::string& name) override
185  {
186  std::map<std::string, ActionBasedControllerHandleBasePtr>::const_iterator it = controllers_.find(name);
187  if (it != controllers_.end())
188  return static_cast<moveit_controller_manager::MoveItControllerHandlePtr>(it->second);
189  else
190  RCLCPP_FATAL_STREAM(LOGGER, "No such controller: " << name);
191  return moveit_controller_manager::MoveItControllerHandlePtr();
192  }
193 
194  /*
195  * Get the list of controller names.
196  */
197  void getControllersList(std::vector<std::string>& names) override
198  {
199  for (std::map<std::string, ActionBasedControllerHandleBasePtr>::const_iterator it = controllers_.begin();
200  it != controllers_.end(); ++it)
201  names.push_back(it->first);
202  RCLCPP_INFO_STREAM(LOGGER, "Returned " << names.size() << " controllers in list");
203  }

205  /*
206  * This plugin assumes that all controllers are already active -- and if they are not, well, it has no way to deal
207  * with it anyways!
208  */
209  void getActiveControllers(std::vector<std::string>& names) override
210  {
211  getControllersList(names);
212  }

222  /*
223  * Get the list of joints that a controller can control.
224  */
225  void getControllerJoints(const std::string& name, std::vector<std::string>& joints) override
226  {
227  std::map<std::string, ActionBasedControllerHandleBasePtr>::const_iterator it = controllers_.find(name);
228  if (it != controllers_.end())
229  {
230  it->second->getJoints(joints);
231  }
232  else
233  {
234  RCLCPP_WARN(LOGGER,
235  "The joints for controller '%s' are not known. Perhaps the controller configuration is "
236  "not loaded on the param server?",
237  name.c_str());
238  joints.clear();
239  }
240  }
241 
242  moveit_controller_manager::MoveItControllerManager::ControllerState
243  getControllerState(const std::string& name) override
244  {
245  return controller_states_[name];
246  }
247 
248  /* Cannot switch our controllers */
249  bool switchControllers(const std::vector<std::string>& /* activate */,
250  const std::vector<std::string>& /* deactivate */) override
251  {
252  return false;
253  }
254 
255 protected:
256  rclcpp::Node::SharedPtr node_;
257  std::map<std::string, ActionBasedControllerHandleBasePtr> controllers_;
258  std::map<std::string, moveit_controller_manager::MoveItControllerManager::ControllerState> controller_states_;
259 };
