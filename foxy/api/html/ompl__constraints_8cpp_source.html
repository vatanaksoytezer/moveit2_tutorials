59 Eigen::VectorXd Bounds::penalty(const Eigen::Ref<const Eigen::VectorXd>& x) const
60 {
61  assert((long)lower_.size() == x.size());
62  Eigen::VectorXd penalty(x.size());
63 
64  for (unsigned int i = 0; i < x.size(); i++)
65  {
66  if (x[i] < lower_.at(i))
67  {
68  penalty[i] = lower_.at(i) - x[i];
69  }
70  else if (x[i] > upper_.at(i))
71  {
72  penalty[i] = x[i] - upper_.at(i);
73  }
74  else
75  {
76  penalty[i] = 0.0;
77  }
78  }
79  return penalty;
80 }

82 Eigen::VectorXd Bounds::derivative(const Eigen::Ref<const Eigen::VectorXd>& x) const
83 {
84  assert((long)lower_.size() == x.size());
85  Eigen::VectorXd derivative(x.size());
86 
87  for (unsigned int i = 0; i < x.size(); i++)
88  {
89  if (x[i] < lower_.at(i))
90  {
91  derivative[i] = -1.0;
92  }
93  else if (x[i] > upper_.at(i))
94  {
95  derivative[i] = 1.0;
96  }
97  else
98  {
99  derivative[i] = 0.0;
100  }
101  }
102  return derivative;
103 }

144 void BaseConstraint::jacobian(const Eigen::Ref<const Eigen::VectorXd>& joint_values,
145  Eigen::Ref<Eigen::MatrixXd> out) const
146 {
147  const Eigen::VectorXd constraint_error = calcError(joint_values);
148  const Eigen::VectorXd constraint_derivative = bounds_.derivative(constraint_error);
149  const Eigen::MatrixXd robot_jacobian = calcErrorJacobian(joint_values);
150  for (std::size_t i = 0; i < bounds_.size(); i++)
151  {
152  out.row(i) = constraint_derivative[i] * robot_jacobian.row(i);
153  }
154 }

197 void BoxConstraint::parseConstraintMsg(const moveit_msgs::msg::Constraints& constraints)
198 {
199  RCLCPP_DEBUG(LOGGER, "Parsing box position constraint for OMPL constrained state space.");
200  assert(bounds_.size() == 0);
201  bounds_ = positionConstraintMsgToBoundVector(constraints.position_constraints.at(0));
202  RCLCPP_DEBUG(LOGGER, "Parsed Box constraints");
203  RCLCPP_DEBUG_STREAM(LOGGER, "Bounds: " << bounds_);
204 
205  // extract target position and orientation
206  geometry_msgs::msg::Point position =
207  constraints.position_constraints.at(0).constraint_region.primitive_poses.at(0).position;
208 
209  target_position_ << position.x, position.y, position.z;
210 
211  tf2::fromMsg(constraints.position_constraints.at(0).constraint_region.primitive_poses.at(0).orientation,
212  target_orientation_);
213 
214  link_name_ = constraints.position_constraints.at(0).link_name;
215  RCLCPP_DEBUG_STREAM(LOGGER, "Position constraints applied to link: " << link_name_);
216 }

237 void EqualityPositionConstraint::parseConstraintMsg(const moveit_msgs::msg::Constraints& constraints)
238 {
239  const auto dims = constraints.position_constraints.at(0).constraint_region.primitives.at(0).dimensions;
240 
241  is_dim_constrained_ = { false, false, false };
242  for (std::size_t i = 0; i < dims.size(); i++)
243  {
244  if (dims.at(i) < EQUALITY_CONSTRAINT_THRESHOLD)
245  {
246  if (dims.at(i) < getTolerance())
247  {
248  RCLCPP_ERROR_STREAM(
249  LOGGER,
250  "Dimension: " << i
251  << " of position constraint is smaller than the tolerance used to evaluate the constraints. "
252  "This will make all states invalid and planning will fail. Please use a value between: "
253  << getTolerance() << " and " << EQUALITY_CONSTRAINT_THRESHOLD);
254  }
255 
256  is_dim_constrained_.at(i) = true;
257  }
258  }
259 
260  // extract target position and orientation
261  geometry_msgs::msg::Point position =
262  constraints.position_constraints.at(0).constraint_region.primitive_poses.at(0).position;
263 
264  target_position_ << position.x, position.y, position.z;
265 
266  tf2::fromMsg(constraints.position_constraints.at(0).constraint_region.primitive_poses.at(0).orientation,
267  target_orientation_);
268 
269  RCLCPP_DEBUG_STREAM(LOGGER, "Equality constraint on x-position? " << (is_dim_constrained_[0] ? "yes" : "no"));
270  RCLCPP_DEBUG_STREAM(LOGGER, "Equality constraint on y-position? " << (is_dim_constrained_[1] ? "yes" : "no"));
271  RCLCPP_DEBUG_STREAM(LOGGER, "Equality constraint on z-position? " << (is_dim_constrained_[2] ? "yes" : "no"));
272 
273  link_name_ = constraints.position_constraints.at(0).link_name;
274  RCLCPP_DEBUG_STREAM(LOGGER, "Position constraints applied to link: " << link_name_);
275 }

277 void EqualityPositionConstraint::function(const Eigen::Ref<const Eigen::VectorXd>& joint_values,
278  Eigen::Ref<Eigen::VectorXd> out) const
279 {
280  Eigen::Vector3d error =
281  target_orientation_.matrix().transpose() * (forwardKinematics(joint_values).translation() - target_position_);
282  for (std::size_t dim = 0; dim < 3; dim++)
283  {
284  if (is_dim_constrained_.at(dim))
285  {
286  out[dim] = error[dim]; // equality constraint dimension
287  }
288  else
289  {
290  out[dim] = 0.0; // unbounded dimension
291  }
292  }
293 }

295 void EqualityPositionConstraint::jacobian(const Eigen::Ref<const Eigen::VectorXd>& joint_values,
296  Eigen::Ref<Eigen::MatrixXd> out) const
297 {
298  out.setZero();
299  Eigen::MatrixXd jac = target_orientation_.matrix().transpose() * robotGeometricJacobian(joint_values).topRows(3);
300  for (std::size_t dim = 0; dim < 3; dim++)
301  {
302  if (is_dim_constrained_.at(dim))
303  {
304  out.row(dim) = jac.row(dim); // equality constraint dimension
305  }
306  }
307 }

309 /************************************
310  * MoveIt constraint message parsing
311  * **********************************/
312 Bounds positionConstraintMsgToBoundVector(const moveit_msgs::msg::PositionConstraint& pos_con)
313 {
314  auto dims = pos_con.constraint_region.primitives.at(0).dimensions;
315 
316  // dimension of -1 signifies unconstrained parameter, so set to infinity
317  for (auto& dim : dims)
318  {
319  if (dim == -1)
320  {
321  dim = std::numeric_limits<double>::infinity();
322  }
323  }
324 
325  return { { -dims.at(0) / 2.0, -dims.at(1) / 2.0, -dims.at(2) / 2.0 },
326  { dims.at(0) / 2.0, dims.at(1) / 2.0, dims.at(2) / 2.0 } };
327 }

329 /******************************************
330  * OMPL Constraints Factory
331  * ****************************************/
332 std::shared_ptr<BaseConstraint> createOMPLConstraint(const moveit::core::RobotModelConstPtr& robot_model,
333  const std::string& group,
334  const moveit_msgs::msg::Constraints& constraints)
335 {
336  // TODO(bostoncleek): does this reach the end w/o a return ?
337 
338  const std::size_t num_dofs = robot_model->getJointModelGroup(group)->getVariableCount();
339  const std::size_t num_pos_con = constraints.position_constraints.size();
340  const std::size_t num_ori_con = constraints.orientation_constraints.size();
341 
342  // This factory method contains template code to support different constraints, but only position constraints are
343  // currently supported. The other options return a nullptr for now and should not be used.
344 
345  if (num_pos_con > 1)
346  {
347  RCLCPP_WARN(LOGGER, "Only a single position constraints supported. Using the first one.");
348  }
349  if (num_ori_con > 1)
350  {
351  RCLCPP_WARN(LOGGER, "Only a single orientation constraints supported. Using the first one.");
352  }
353 
354  if (num_pos_con > 0 && num_ori_con > 0)
355  {
356  RCLCPP_ERROR(LOGGER, "Combining position and orientation constraints not implemented yet for OMPL's constrained "
357  "state space.");
358  return nullptr;
359  }
360  else if (num_pos_con > 0)
361  {
362  RCLCPP_DEBUG_STREAM(LOGGER, "Constraint name: " << constraints.name);
363  BaseConstraintPtr pos_con;
364  if (constraints.name == "use_equality_constraints")
365  {
366  RCLCPP_DEBUG(LOGGER, "OMPL is using equality position constraints.");
367  pos_con = std::make_shared<EqualityPositionConstraint>(robot_model, group, num_dofs);
368  }
369  else
370  {
371  RCLCPP_DEBUG(LOGGER, "OMPL is using box position constraints.");
372  pos_con = std::make_shared<BoxConstraint>(robot_model, group, num_dofs);
373  }
374  pos_con->init(constraints);
375  return pos_con;
376  }
377  else if (num_ori_con > 0)
378  {
379  RCLCPP_ERROR(LOGGER, "Orientation constraints are not yet supported.");
380  return nullptr;
381  }
382  else
383  {
384  RCLCPP_ERROR(LOGGER, "No path constraints found in planning request.");
385  return nullptr;
386  }
387 }
