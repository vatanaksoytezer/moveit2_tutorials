55  public:
56  void log(const std::string& text, ompl::msg::LogLevel level, const char* filename, int line) override
57  {
58  switch (level)
59  {
60  case ompl::msg::LOG_DEV2:
61  case ompl::msg::LOG_DEV1:
62  case ompl::msg::LOG_DEBUG:
63  RCLCPP_DEBUG(OMPL_LOGGER, "%s:%i - %s", filename, line, text.c_str());
64  break;
65  case ompl::msg::LOG_INFO:
66  RCLCPP_INFO(OMPL_LOGGER, "%s:%i - %s", filename, line, text.c_str());
67  break;
68  case ompl::msg::LOG_WARN:
69  RCLCPP_WARN(OMPL_LOGGER, "%s:%i - %s", filename, line, text.c_str());
70  break;
71  case ompl::msg::LOG_ERROR:
72  RCLCPP_ERROR(OMPL_LOGGER, "%s:%i - %s", filename, line, text.c_str());
73  break;
74  case ompl::msg::LOG_NONE:
75  default:
76  /* ignore */
77  break;
78  }
79  }
80  };
81

83  ompl::msg::useOutputHandler(output_handler_.get());
