54 void PlanComponentsBuilder::appendWithStrictTimeIncrease(robot_trajectory::RobotTrajectory& result,
55  const robot_trajectory::RobotTrajectory& source)
56 {
57  if (result.empty() ||
58  !pilz_industrial_motion_planner::isRobotStateEqual(result.getLastWayPoint(), source.getFirstWayPoint(),
59  result.getGroupName(), ROBOT_STATE_EQUALITY_EPSILON))
60  {
61  result.append(source, 0.0);
62  return;
63  }
64 
65  for (size_t i = 1; i < source.getWayPointCount(); ++i)
66  {
67  result.addSuffixWayPoint(source.getWayPoint(i), source.getWayPointDurationFromPrevious(i));
68  }

71 void PlanComponentsBuilder::blend(const robot_trajectory::RobotTrajectoryPtr& other, const double blend_radius)
72 {
73  if (!blender_)
74  {
75  throw NoBlenderSetException("No blender set");
76  }
77 
78  assert(other->getGroupName() == traj_tail_->getGroupName());
79 
80  pilz_industrial_motion_planner::TrajectoryBlendRequest blend_request;
81 
82  blend_request.first_trajectory = traj_tail_;
83  blend_request.second_trajectory = other;
84  blend_request.blend_radius = blend_radius;
85  blend_request.group_name = traj_tail_->getGroupName();
86  blend_request.link_name = getSolverTipFrame(model_->getJointModelGroup(blend_request.group_name));
87 
88  pilz_industrial_motion_planner::TrajectoryBlendResponse blend_response;
89  if (!blender_->blend(blend_request, blend_response))
90  {
91  throw BlendingFailedException("Blending failed");
92  }
93 
94  // Append the new trajectory elements
95  appendWithStrictTimeIncrease(*(traj_cont_.back()), *blend_response.first_trajectory);
96  traj_cont_.back()->append(*blend_response.blend_trajectory, 0.0);
97  // Store the last new trajectory element for future processing
98  traj_tail_ = blend_response.second_trajectory; // first for next blending segment
99 }

101 void PlanComponentsBuilder::append(const robot_trajectory::RobotTrajectoryPtr& other, const double blend_radius)
102 {
103  if (!model_)
104  {
105  throw NoRobotModelSetException("No robot model set");
106  }
107 
108  if (!traj_tail_)
109  {
110  traj_tail_ = other;
111  // Reserve space in container for new trajectory
112  traj_cont_.emplace_back(new robot_trajectory::RobotTrajectory(model_, other->getGroupName()));
113  return;
114  }
115 
116  // Create new trajectory for every group change
117  if (other->getGroupName() != traj_tail_->getGroupName())
118  {
119  appendWithStrictTimeIncrease(*(traj_cont_.back()), *traj_tail_);
120  traj_tail_ = other;
121  // Create new container element
122  traj_cont_.emplace_back(new robot_trajectory::RobotTrajectory(model_, other->getGroupName()));
123  return;
124  }
125 
126  // No blending
127  if (blend_radius <= 0.0)
128  {
129  appendWithStrictTimeIncrease(*(traj_cont_.back()), *traj_tail_);
130  traj_tail_ = other;
131  return;
132  }
133 
134  blend(other, blend_radius);
135 }
