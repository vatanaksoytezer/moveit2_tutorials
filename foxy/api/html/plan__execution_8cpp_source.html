105 std::string plan_execution::PlanExecution::getErrorCodeString(const moveit_msgs::msg::MoveItErrorCodes& error_code)
106 {
107  if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
108  return "Success";
109  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME)
110  return "Invalid group name";
111  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED)
112  return "Planning failed.";
113  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::INVALID_MOTION_PLAN)
114  return "Invalid motion plan";
115  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::UNABLE_TO_AQUIRE_SENSOR_DATA)
116  return "Unable to aquire sensor data";
117  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::MOTION_PLAN_INVALIDATED_BY_ENVIRONMENT_CHANGE)
118  return "Motion plan invalidated by environment change";
119  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::CONTROL_FAILED)
120  return "Controller failed during execution";
121  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::TIMED_OUT)
122  return "Timeout reached";
123  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::PREEMPTED)
124  return "Preempted";
125  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::INVALID_GOAL_CONSTRAINTS)
126  return "Invalid goal constraints";
127  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::INVALID_OBJECT_NAME)
128  return "Invalid object name";
129  else if (error_code.val == moveit_msgs::msg::MoveItErrorCodes::FAILURE)
130  return "Catastrophic failure";
131  return "Unknown event";
132 }

161 void plan_execution::PlanExecution::planAndExecuteHelper(ExecutableMotionPlan& plan, const Options& opt)
162 {
163  // perform initial configuration steps & various checks
164  preempt_.checkAndClear(); // clear any previous preempt_ request
165 
166  bool preempt_requested = false;
167 
168  // run the actual motion plan & execution
169  unsigned int max_replan_attempts =
170  opt.replan_ ? (opt.replan_attempts_ > 0 ? opt.replan_attempts_ : default_max_replan_attempts_) : 1;
171  unsigned int replan_attempts = 0;
172  bool previously_solved = false;
173 
174  // run a planning loop for at most the maximum replanning attempts;
175  // re-planning is executed only in case of known types of failures (e.g., environment changed)
176  do
177  {
178  replan_attempts++;
179  RCLCPP_INFO(LOGGER, "Planning attempt %u of at most %u", replan_attempts, max_replan_attempts);
180 
181  if (opt.before_plan_callback_)
182  opt.before_plan_callback_();
183 
184  new_scene_update_ = false; // we clear any scene updates to be evaluated because we are about to compute a new
185  // plan, which should consider most recent updates already
186 
187  // if we never had a solved plan, or there is no specified way of fixing plans, just call the planner; otherwise,
188  // try to repair the plan we previously had;
189  bool solved =
190  (!previously_solved || !opt.repair_plan_callback_) ?
191  opt.plan_callback_(plan) :
192  opt.repair_plan_callback_(plan, trajectory_execution_manager_->getCurrentExpectedTrajectoryIndex());
193 
194  preempt_requested = preempt_.checkAndClear();
195  if (preempt_requested)
196  break;
197 
198  // if planning fails in a manner that is not recoverable, we exit the loop,
199  // otherwise, we attempt to continue, if replanning attempts are left
200  if (plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED ||
201  plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::INVALID_MOTION_PLAN ||
202  plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::UNABLE_TO_AQUIRE_SENSOR_DATA)
203  {
204  if (plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::UNABLE_TO_AQUIRE_SENSOR_DATA &&
205  opt.replan_delay_ > 0.0)
206  {
207  auto replan_delay_seconds = std::chrono::duration<double>(opt.replan_delay_);
208  rclcpp::sleep_for(std::chrono::duration_cast<std::chrono::nanoseconds>(replan_delay_seconds));
209  }
210  continue;
211  }
212 
213  // abort if no plan was found
214  if (solved)
215  previously_solved = true;
216  else
217  break;
218 
219  if (plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
220  {
221  if (opt.before_execution_callback_)
222  opt.before_execution_callback_();
223 
224  preempt_requested = preempt_.checkAndClear();
225  if (preempt_requested)
226  break;
227 
228  // execute the trajectory, and monitor its execution
229  plan.error_code_ = executeAndMonitor(plan, false);
230  }
231 
232  if (plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::PREEMPTED)
233  preempt_requested = true;
234 
235  // if execution succeeded or failed in a manner that we do not consider recoverable, we exit the loop (with failure)
236  if (plan.error_code_.val != moveit_msgs::msg::MoveItErrorCodes::MOTION_PLAN_INVALIDATED_BY_ENVIRONMENT_CHANGE)
237  break;
238  else
239  {
240  // otherwise, we wait (if needed)
241  if (opt.replan_delay_ > 0.0)
242  {
243  RCLCPP_INFO(LOGGER, "Waiting for a %lf seconds before attempting a new plan ...", opt.replan_delay_);
244  auto replan_delay_seconds = std::chrono::duration<double>(opt.replan_delay_);
245  rclcpp::sleep_for(std::chrono::duration_cast<std::chrono::nanoseconds>(replan_delay_seconds));
246  RCLCPP_INFO(node_->get_logger(), "Done waiting");
247  }
248  }
249 
250  preempt_requested = preempt_.checkAndClear();
251  if (preempt_requested)
252  break;
253 
254  } while (replan_attempts < max_replan_attempts);
255 
256  if (preempt_requested)
257  {
258  RCLCPP_DEBUG(LOGGER, "PlanExecution was preempted");
259  plan.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::PREEMPTED;
260  }
261 
262  if (opt.done_callback_)
263  opt.done_callback_();
264 
265  if (plan.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
266  {
267  RCLCPP_DEBUG(LOGGER, "PlanExecution finished successfully.");
268  }
269  else
270  {
271  RCLCPP_DEBUG(LOGGER, "PlanExecution terminating with error code %d - '%s'", plan.error_code_.val,
272  getErrorCodeString(plan.error_code_).c_str());
273  }
274 }

276 bool plan_execution::PlanExecution::isRemainingPathValid(const ExecutableMotionPlan& plan,
277  const std::pair<int, int>& path_segment)
278 {
279  if (path_segment.first >= 0 &&
280  plan.plan_components_[path_segment.first].trajectory_monitoring_) // If path_segment.second <= 0, the function
281  // will fallback to check the entire trajectory
282  {
283  planning_scene_monitor::LockedPlanningSceneRO lscene(plan.planning_scene_monitor_); // lock the scene so that it
284  // does not modify the world
285  // representation while
286  // isStateValid() is called
287  const robot_trajectory::RobotTrajectory& t = *plan.plan_components_[path_segment.first].trajectory_;
288  const collision_detection::AllowedCollisionMatrix* acm =
289  plan.plan_components_[path_segment.first].allowed_collision_matrix_.get();
290  std::size_t wpc = t.getWayPointCount();
291  collision_detection::CollisionRequest req;
292  req.group_name = t.getGroupName();
293  for (std::size_t i = std::max(path_segment.second - 1, 0); i < wpc; ++i)
294  {
295  collision_detection::CollisionResult res;
296  if (acm)
297  plan.planning_scene_->checkCollisionUnpadded(req, res, t.getWayPoint(i), *acm);
298  else
299  plan.planning_scene_->checkCollisionUnpadded(req, res, t.getWayPoint(i));
300 
301  if (res.collision || !plan.planning_scene_->isStateFeasible(t.getWayPoint(i), false))
302  {
303  // Dave's debacle
304  RCLCPP_INFO(LOGGER, "Trajectory component '%s' is invalid",
305  plan.plan_components_[path_segment.first].description_.c_str());
306 
307  // call the same functions again, in verbose mode, to show what issues have been detected
308  plan.planning_scene_->isStateFeasible(t.getWayPoint(i), true);
309  req.verbose = true;
310  res.clear();
311  if (acm)
312  plan.planning_scene_->checkCollisionUnpadded(req, res, t.getWayPoint(i), *acm);
313  else
314  plan.planning_scene_->checkCollisionUnpadded(req, res, t.getWayPoint(i));
315  return false;
316  }
317  }
318  }
319  return true;
320 }

322 moveit_msgs::msg::MoveItErrorCodes plan_execution::PlanExecution::executeAndMonitor(ExecutableMotionPlan& plan,
323  bool reset_preempted)
324 {
325  if (reset_preempted)
326  preempt_.checkAndClear();
327 
328  if (!plan.planning_scene_monitor_)
329  plan.planning_scene_monitor_ = planning_scene_monitor_;
330  if (!plan.planning_scene_)
331  plan.planning_scene_ = planning_scene_monitor_->getPlanningScene();
332 
333  moveit_msgs::msg::MoveItErrorCodes result;
334 
335  // try to execute the trajectory
336  execution_complete_ = true;
337 
338  if (!trajectory_execution_manager_)
339  {
340  RCLCPP_ERROR(LOGGER, "No trajectory execution manager");
341  result.val = moveit_msgs::msg::MoveItErrorCodes::CONTROL_FAILED;
342  return result;
343  }
344 
345  if (plan.plan_components_.empty())
346  {
347  result.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
348  return result;
349  }
350 
351  execution_complete_ = false;
352 
353  // push the trajectories we have slated for execution to the trajectory execution manager
354  int prev = -1;
355  for (std::size_t i = 0; i < plan.plan_components_.size(); ++i)
356  {
357  // \todo should this be in trajectory_execution ? Maybe. Then that will have to use kinematic_trajectory too;
358  // spliting trajectories for controllers becomes interesting: tied to groups instead of joints. this could cause
359  // some problems
360  // in the meantime we do a hack:
361 
362  bool unwound = false;
363  for (std::size_t j = 0; j < i; ++j)
364  // if we ran unwind on a path for the same group
365  if (plan.plan_components_[j].trajectory_ &&
366  plan.plan_components_[j].trajectory_->getGroup() == plan.plan_components_[i].trajectory_->getGroup() &&
367  !plan.plan_components_[j].trajectory_->empty())
368  {
369  plan.plan_components_[i].trajectory_->unwind(plan.plan_components_[j].trajectory_->getLastWayPoint());
370  unwound = true;
371  break;
372  }
373 
374  if (!unwound)
375  {
376  // unwind the path to execute based on the current state of the system
377  if (prev < 0)
378  plan.plan_components_[i].trajectory_->unwind(
379  plan.planning_scene_monitor_ && plan.planning_scene_monitor_->getStateMonitor() ?
380  *plan.planning_scene_monitor_->getStateMonitor()->getCurrentState() :
381  plan.planning_scene_->getCurrentState());
382  else
383  plan.plan_components_[i].trajectory_->unwind(plan.plan_components_[prev].trajectory_->getLastWayPoint());
384  }
385 
386  if (plan.plan_components_[i].trajectory_ && !plan.plan_components_[i].trajectory_->empty())
387  prev = i;
388 
389  // convert to message, pass along
390  moveit_msgs::msg::RobotTrajectory msg;
391  plan.plan_components_[i].trajectory_->getRobotTrajectoryMsg(msg);
392  if (!trajectory_execution_manager_->push(msg, plan.plan_components_[i].controller_names_))
393  {
394  trajectory_execution_manager_->clear();
395  RCLCPP_ERROR(LOGGER, "Apparently trajectory initialization failed");
396  execution_complete_ = true;
397  result.val = moveit_msgs::msg::MoveItErrorCodes::CONTROL_FAILED;
398  return result;
399  }
400  }
401 
402  if (!trajectory_monitor_ && planning_scene_monitor_->getStateMonitor())
403  {
404  // Pass current value of reconfigurable parameter plan_execution/record_trajectory_state_frequency
405  double sampling_frequency = 0.0;
406  node_->get_parameter_or("plan_execution.record_trajectory_state_frequency", sampling_frequency, 0.0);
407  trajectory_monitor_ = std::make_shared<planning_scene_monitor::TrajectoryMonitor>(
408  planning_scene_monitor_->getStateMonitor(), sampling_frequency);
409  }
410 
411  // start recording trajectory states
412  if (trajectory_monitor_)
413  trajectory_monitor_->startTrajectoryMonitor();
414 
415  // start a trajectory execution thread
416  trajectory_execution_manager_->execute(
417  boost::bind(&PlanExecution::doneWithTrajectoryExecution, this, boost::placeholders::_1),
418  boost::bind(&PlanExecution::successfulTrajectorySegmentExecution, this, &plan, boost::placeholders::_1));
419  // wait for path to be done, while checking that the path does not become invalid
420  rclcpp::WallRate r(100);
421  path_became_invalid_ = false;
422  bool preempt_requested = false;
423 
424  while (rclcpp::ok() && !execution_complete_ && !path_became_invalid_)
425  {
426  r.sleep();
427  // check the path if there was an environment update in the meantime
428  if (new_scene_update_)
429  {
430  new_scene_update_ = false;
431  std::pair<int, int> current_index = trajectory_execution_manager_->getCurrentExpectedTrajectoryIndex();
432  if (!isRemainingPathValid(plan, current_index))
433  {
434  RCLCPP_INFO(LOGGER, "Trajectory component '%s' is invalid after scene update",
435  plan.plan_components_[current_index.first].description_.c_str());
436  path_became_invalid_ = true;
437  break;
438  }
439  }
440 
441  preempt_requested = preempt_.checkAndClear();
442  if (preempt_requested)
443  break;
444  }
445 
446  // stop execution if needed
447  if (preempt_requested)
448  {
449  RCLCPP_INFO(LOGGER, "Stopping execution due to preempt request");
450  trajectory_execution_manager_->stopExecution();
451  }
452  else if (path_became_invalid_)
453  {
454  RCLCPP_INFO(LOGGER, "Stopping execution because the path to execute became invalid"
455  "(probably the environment changed)");
456  trajectory_execution_manager_->stopExecution();
457  }
458  else if (!execution_complete_)
459  {
460  RCLCPP_WARN(LOGGER, "Stopping execution due to unknown reason."
461  "Possibly the node is about to shut down.");
462  trajectory_execution_manager_->stopExecution();
463  }
464 
465  // stop recording trajectory states
466  if (trajectory_monitor_)
467  {
468  trajectory_monitor_->stopTrajectoryMonitor();
469  plan.executed_trajectory_ =
470  std::make_shared<robot_trajectory::RobotTrajectory>(planning_scene_monitor_->getRobotModel(), "");
471  trajectory_monitor_->swapTrajectory(*plan.executed_trajectory_);
472  }
473 
474  // decide return value
475  if (path_became_invalid_)
476  result.val = moveit_msgs::msg::MoveItErrorCodes::MOTION_PLAN_INVALIDATED_BY_ENVIRONMENT_CHANGE;
477  else
478  {
479  if (preempt_requested)
480  {
481  result.val = moveit_msgs::msg::MoveItErrorCodes::PREEMPTED;
482  }
483  else
484  {
485  if (trajectory_execution_manager_->getLastExecutionStatus() ==
486  moveit_controller_manager::ExecutionStatus::SUCCEEDED)
487  result.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
488  else if (trajectory_execution_manager_->getLastExecutionStatus() ==
489  moveit_controller_manager::ExecutionStatus::TIMED_OUT)
490  result.val = moveit_msgs::msg::MoveItErrorCodes::TIMED_OUT;
491  else
492  result.val = moveit_msgs::msg::MoveItErrorCodes::CONTROL_FAILED;
493  }
494  }
495  return result;
496 }

512 void plan_execution::PlanExecution::successfulTrajectorySegmentExecution(const ExecutableMotionPlan* plan,
513  std::size_t index)
514 {
515  if (plan->plan_components_.empty())
516  {
517  RCLCPP_WARN(LOGGER, "Length of provided motion plan is zero.");
518  return;
519  }
520 
521  // if any side-effects are associated to the trajectory part that just completed, execute them
522  RCLCPP_DEBUG(LOGGER, "Completed '%s'", plan->plan_components_[index].description_.c_str());
523  if (plan->plan_components_[index].effect_on_success_)
524  if (!plan->plan_components_[index].effect_on_success_(plan))
525  {
526  // execution of side-effect failed
527  RCLCPP_ERROR(LOGGER, "Execution of path-completion side-effect failed. Preempting.");
528  preempt_.request();
529  return;
530  }
531 
532  // if there is a next trajectory, check it for validity, before we start execution
533  ++index;
534  if (index < plan->plan_components_.size() && plan->plan_components_[index].trajectory_ &&
535  !plan->plan_components_[index].trajectory_->empty())
536  {
537  std::pair<int, int> next_index(static_cast<int>(index), 0);
538  if (!isRemainingPathValid(*plan, next_index))
539  {
540  RCLCPP_INFO(LOGGER, "Upcoming trajectory component '%s' is invalid",
541  plan->plan_components_[next_index.first].description_.c_str());
542  path_became_invalid_ = true;
543  }
544  }
