56 // Plan the arm movement to the approach location
57 bool PlanStage::evaluate(const ManipulationPlanPtr& plan) const
58 {
59  planning_interface::MotionPlanRequest req;
60  planning_interface::MotionPlanResponse res;
61  req.group_name = plan->shared_data_->planning_group_->getName();
62  req.num_planning_attempts = 1;
63  req.allowed_planning_time = (plan->shared_data_->timeout_ - ros::WallTime::now()).toSec();
64  req.path_constraints = plan->shared_data_->path_constraints_;
65  req.planner_id = plan->shared_data_->planner_id_;
66  req.start_state.is_diff = true;
67 
68  req.goal_constraints.resize(1, kinematic_constraints::constructGoalConstraints(*plan->approach_state_,
69  plan->shared_data_->planning_group_));
70  unsigned int attempts = 0;
71  do // give the planner two chances
72  {
73  attempts++;
74  if (!signal_stop_ && planning_pipeline_->generatePlan(planning_scene_, req, res) &&
75  res.error_code_.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS && res.trajectory_ &&
76  !res.trajectory_->empty())
77  {
78  // We have a valid motion plan, now apply pre-approach end effector posture (open gripper) if applicable
79  if (!plan->approach_posture_.joint_names.empty())
80  {
81  moveit::core::RobotStatePtr pre_approach_state(new moveit::core::RobotState(res.trajectory_->getLastWayPoint()));
82  robot_trajectory::RobotTrajectoryPtr pre_approach_traj(new robot_trajectory::RobotTrajectory(
83  pre_approach_state->getRobotModel(), plan->shared_data_->end_effector_group_->getName()));
84  pre_approach_traj->setRobotTrajectoryMsg(*pre_approach_state, plan->approach_posture_);
85 
86  // Apply the open gripper state to the waypoint
87  // If user has defined a time for it's gripper movement time, don't add the
88  // DEFAULT_GRASP_POSTURE_COMPLETION_DURATION
89  if (!plan->approach_posture_.points.empty() &&
90  plan->approach_posture_.points.back().time_from_start > ros::Duration(0.0))
91  {
92  pre_approach_traj->addPrefixWayPoint(pre_approach_state, 0.0);
93  }
94  else
95  { // Do what was done before
96  ROS_INFO_STREAM("Adding default duration of " << PickPlace::DEFAULT_GRASP_POSTURE_COMPLETION_DURATION
97  << " seconds to the grasp closure time. Assign time_from_start "
98  << "to your trajectory to avoid this.");
99  pre_approach_traj->addPrefixWayPoint(pre_approach_state, PickPlace::DEFAULT_GRASP_POSTURE_COMPLETION_DURATION);
100  }
101 
102  // Add the open gripper trajectory to the plan
103  plan_execution::ExecutableTrajectory et(pre_approach_traj, "pre_grasp");
104  plan->trajectories_.insert(plan->trajectories_.begin(), et);
105  }
106 
107  // Add the pre-approach trajectory to the plan
108  plan_execution::ExecutableTrajectory et(res.trajectory_, name_);
109  plan->trajectories_.insert(plan->trajectories_.begin(), et);
110 
111  plan->error_code_ = res.error_code_;
112 
113  return true;
114  }
115  else
116  plan->error_code_ = res.error_code_;
117  }
118  // if the planner reported an invalid plan, give it a second chance
119  while (!signal_stop_ && plan->error_code_.val == moveit_msgs::msg::MoveItErrorCodes::INVALID_MOTION_PLAN &&
120  attempts < 2);
121 
122  return false;
123 }
