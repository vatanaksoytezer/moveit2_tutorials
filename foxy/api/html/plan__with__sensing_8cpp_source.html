155 bool plan_execution::PlanWithSensing::computePlan(ExecutableMotionPlan& plan,
156  const ExecutableMotionPlanComputationFn& motion_planner,
157  unsigned int max_look_attempts, double max_safe_path_cost)
158 {
159  if (max_safe_path_cost <= std::numeric_limits<double>::epsilon())
160  max_safe_path_cost = default_max_safe_path_cost_;
161 
162  if (max_look_attempts == 0)
163  max_look_attempts = default_max_look_attempts_;
164 
165  double previous_cost = 0.0;
166  unsigned int look_attempts = 0;
167 
168  // this flag is set to true when all conditions for looking around are met, and the command is sent.
169  // the intention is for the planning loop not to terminate when having just looked around
170  bool just_looked_around = false;
171 
172  // this flag indicates whether the last lookAt() operation failed. If this operation fails once, we assume that
173  // maybe some information was gained anyway (the sensor moved part of the way) and we try to plan one more time.
174  // If we have two sensor pointing failures in a row, we fail
175  bool look_around_failed = false;
176 
177  // there can be a maximum number of looking attempts as well that lead to replanning, if the cost
178  // of the path is above a maximum threshold.
179  do
180  {
181  bool solved = motion_planner(plan);
182  if (!solved || plan.error_code_.val != moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
183  return solved;
184 
185  // determine the sources of cost for this path
186  std::set<collision_detection::CostSource> cost_sources;
187  {
188  planning_scene_monitor::LockedPlanningSceneRO lscene(plan.planning_scene_monitor_); // it is ok if
189  // planning_scene_monitor_ is
190  // null; there just will be
191  // no locking done
192  for (std::size_t i = 0; i < plan.plan_components_.size(); ++i)
193  {
194  std::set<collision_detection::CostSource> cost_sources_i;
195  plan.planning_scene_->getCostSources(*plan.plan_components_[i].trajectory_, max_cost_sources_,
196  plan.plan_components_[i].trajectory_->getGroupName(), cost_sources_i,
197  discard_overlapping_cost_sources_);
198  cost_sources.insert(cost_sources_i.begin(), cost_sources_i.end());
199  if (cost_sources.size() > max_cost_sources_)
200  {
201  std::set<collision_detection::CostSource> other;
202  other.swap(cost_sources);
203  std::size_t j = 0;
204  for (std::set<collision_detection::CostSource>::iterator it = other.begin(); j < max_cost_sources_; ++it, ++j)
205  cost_sources.insert(*it);
206  }
207  }
208  }
209 
210  // display the costs if needed
211  if (display_cost_sources_)
212  {
213  visualization_msgs::msg::MarkerArray arr;
214  collision_detection::getCostMarkers(arr, plan.planning_scene_->getPlanningFrame(), cost_sources);
215  cost_sources_publisher_->publish(arr);
216  }
217 
218  double cost = collision_detection::getTotalCost(cost_sources);
219  RCLCPP_DEBUG(LOGGER, "The total cost of the trajectory is %lf.", cost);
220  if (previous_cost > 0.0)
221  RCLCPP_DEBUG(LOGGER, "The change in the trajectory cost is %lf after the perception step.", cost - previous_cost);
222  if (cost > max_safe_path_cost && look_attempts < max_look_attempts)
223  {
224  ++look_attempts;
225  RCLCPP_INFO(
226  LOGGER,
227  "The cost of the trajectory is %lf, which is above the maximum safe cost of %lf. Attempt %u (of at most "
228  "%u) at looking around.",
229  cost, max_safe_path_cost, look_attempts, max_look_attempts);
230 
231  bool looked_at_result = lookAt(cost_sources, plan.planning_scene_->getPlanningFrame());
232  if (looked_at_result)
233  RCLCPP_INFO(LOGGER, "Sensor was successfully actuated. Attempting to recompute a motion plan.");
234  else
235  {
236  if (look_around_failed)
237  {
238  RCLCPP_WARN(LOGGER, "Looking around seems to keep failing. Giving up.");
239  }
240  else
241  {
242  RCLCPP_WARN(LOGGER, "Looking around seems to have failed. Attempting to recompute a motion plan anyway.");
243  }
244  }
245  if (looked_at_result || !look_around_failed)
246  {
247  previous_cost = cost;
248  just_looked_around = true;
249  }
250  look_around_failed = !looked_at_result;
251  // if we are unable to look, let this loop continue into the next if statement
252  if (just_looked_around)
253  continue;
254  }
255 
256  if (cost > max_safe_path_cost)
257  {
258  plan.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::UNABLE_TO_AQUIRE_SENSOR_DATA;
259  return true;
260  }
261  else
262  return true;
263  } while (true);
264 
265  return false;
266 }

268 bool plan_execution::PlanWithSensing::lookAt(const std::set<collision_detection::CostSource>& cost_sources,
269  const std::string& frame_id)
270 {
271  if (!sensor_manager_)
272  {
273  RCLCPP_WARN(LOGGER, "It seems looking around would be useful, but no MoveIt! Sensor Manager is loaded. Did you set "
274  "~moveit_sensor_manager ?");
275  return false;
276  }
277 
278  if (before_look_callback_)
279  before_look_callback_();
280 
281  std::vector<std::string> names;
282  sensor_manager_->getSensorsList(names);
283  geometry_msgs::msg::PointStamped point;
284  for (const std::string& name : names)
285  if (collision_detection::getSensorPositioning(point.point, cost_sources))
286  {
287  point.header.stamp = node_->now();
288  point.header.frame_id = frame_id;
289  RCLCPP_DEBUG(LOGGER, "Pointing sensor %s to: %s\n", name.c_str(), point.header.frame_id.c_str());
290  moveit_msgs::msg::RobotTrajectory sensor_trajectory;
291  if (sensor_manager_->pointSensorTo(name, point, sensor_trajectory))
292  {
293  if (!trajectory_processing::isTrajectoryEmpty(sensor_trajectory))
294  return trajectory_execution_manager_->push(sensor_trajectory) &&
295  trajectory_execution_manager_->executeAndWait();
296  else
297  return true;
298  }
299  }
300  return false;
301 }
