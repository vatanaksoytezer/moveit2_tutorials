116 PlanningComponent::PlanSolution PlanningComponent::plan(const PlanRequestParameters& parameters)
117 {
118  last_plan_solution_.reset(new PlanSolution());
119  if (!joint_model_group_)
120  {
121  RCLCPP_ERROR(LOGGER, "Failed to retrieve joint model group for name '%s'.", group_name_.c_str());
122  last_plan_solution_->error_code = MoveItErrorCode(moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME);
123  return *last_plan_solution_;
124  }
125 
126  // Clone current planning scene
127  planning_scene_monitor::PlanningSceneMonitorPtr planning_scene_monitor =
128  moveit_cpp_->getPlanningSceneMonitorNonConst();
129  planning_scene_monitor->updateFrameTransforms();
130  planning_scene_monitor->lockSceneRead(); // LOCK planning scene
131  planning_scene::PlanningScenePtr planning_scene =
132  planning_scene::PlanningScene::clone(planning_scene_monitor->getPlanningScene());
133  planning_scene_monitor->unlockSceneRead(); // UNLOCK planning scene
134  planning_scene_monitor.reset(); // release this pointer
135 
136  // Init MotionPlanRequest
137  ::planning_interface::MotionPlanRequest req;
138  req.group_name = group_name_;
139  req.planner_id = parameters.planner_id;
140  req.num_planning_attempts = std::max(1, parameters.planning_attempts);
141  req.allowed_planning_time = parameters.planning_time;
142  req.max_velocity_scaling_factor = parameters.max_velocity_scaling_factor;
143  req.max_acceleration_scaling_factor = parameters.max_acceleration_scaling_factor;
144  if (workspace_parameters_set_)
145  req.workspace_parameters = workspace_parameters_;
146 
147  // Set start state
148  moveit::core::RobotStatePtr start_state = considered_start_state_;
149  if (!start_state)
150  start_state = moveit_cpp_->getCurrentState();
151  start_state->update();
152  moveit::core::robotStateToRobotStateMsg(*start_state, req.start_state);
153  planning_scene->setCurrentState(*start_state);
154 
155  // Set goal constraints
156  if (current_goal_constraints_.empty())
157  {
158  RCLCPP_ERROR(LOGGER, "No goal constraints set for planning request");
159  last_plan_solution_->error_code = MoveItErrorCode(moveit_msgs::msg::MoveItErrorCodes::INVALID_GOAL_CONSTRAINTS);
160  return *last_plan_solution_;
161  }
162  req.goal_constraints = current_goal_constraints_;
163 
164  req.path_constraints = current_path_constraints_;
165 
166  // Run planning attempt
167  ::planning_interface::MotionPlanResponse res;
168  if (planning_pipeline_names_.find(parameters.planning_pipeline) == planning_pipeline_names_.end())
169  {
170  RCLCPP_ERROR(LOGGER, "No planning pipeline available for name '%s'", parameters.planning_pipeline.c_str());
171  last_plan_solution_->error_code = MoveItErrorCode(moveit_msgs::msg::MoveItErrorCodes::FAILURE);
172  return *last_plan_solution_;
173  }
174  const planning_pipeline::PlanningPipelinePtr pipeline =
175  moveit_cpp_->getPlanningPipelines().at(parameters.planning_pipeline);
176  pipeline->generatePlan(planning_scene, req, res);
177  last_plan_solution_->error_code = res.error_code_.val;
178  if (res.error_code_.val != res.error_code_.SUCCESS)
179  {
180  RCLCPP_ERROR(LOGGER, "Could not compute plan successfully");
181  return *last_plan_solution_;
182  }
183  last_plan_solution_->start_state = req.start_state;
184  last_plan_solution_->trajectory = res.trajectory_;
185  // TODO(henningkayser): Visualize trajectory
186  // std::vector<const moveit::core::LinkModel*> eef_links;
187  // if (joint_model_group->getEndEffectorTips(eef_links))
188  //{
189  // for (const auto& eef_link : eef_links)
190  // {
191  // RCLCPP_INFO_STREAM("Publishing trajectory for end effector " << eef_link->getName());
192  // visual_tools_->publishTrajectoryLine(last_solution_trajectory_, eef_link);
193  // visual_tools_->publishTrajectoryPath(last_solution_trajectory_, false);
194  // visual_tools_->publishRobotState(last_solution_trajectory_->getLastWayPoint(), rviz_visual_tools::TRANSLUCENT);
195  // }
196  //}
197  return *last_plan_solution_;
198 }

223 bool PlanningComponent::setStartState(const std::string& start_state_name)
224 {
225  const auto& named_targets = getNamedTargetStates();
226  if (std::find(named_targets.begin(), named_targets.end(), start_state_name) == named_targets.end())
227  {
228  RCLCPP_ERROR(LOGGER, "No predefined joint state found for target name '%s'", start_state_name.c_str());
229  return false;
230  }
231  moveit::core::RobotState start_state(moveit_cpp_->getRobotModel());
232  start_state.setToDefaultValues(joint_model_group_, start_state_name);
