119 template <typename GeneratorT>
120 bool pilz_industrial_motion_planner::PlanningContextBase<GeneratorT>::solve(planning_interface::MotionPlanResponse& res)
121 {
122  if (!terminated_)
123  {
124  // Use current state as start state if not set
125  if (request_.start_state.joint_state.name.empty())
126  {
127  moveit_msgs::RobotState current_state;
128  moveit::core::robotStateToRobotStateMsg(getPlanningScene()->getCurrentState(), current_state);
129  request_.start_state = current_state;
130  }
131  bool result = generator_.generate(request_, res);
132  return result;
133  // res.error_code_.val = moveit_msgs::MoveItErrorCodes::INVALID_MOTION_PLAN;
134  // return false; // TODO
135  }
136 
137  ROS_ERROR("Using solve on a terminated planning context!");
138  res.error_code_.val = moveit_msgs::MoveItErrorCodes::PLANNING_FAILED;
139  return false;
140 }

142 template <typename GeneratorT>
143 bool pilz_industrial_motion_planner::PlanningContextBase<GeneratorT>::solve(
144  planning_interface::MotionPlanDetailedResponse& res)
145 {
146  // delegate to regular response
147  planning_interface::MotionPlanResponse undetailed_response;
148  bool result = solve(undetailed_response);
149 
150  res.description_.push_back("plan");
151  res.trajectory_.push_back(undetailed_response.trajectory_);
152  res.processing_time_.push_back(undetailed_response.planning_time_);
153 
154  res.description_.push_back("simplify");
155  res.trajectory_.push_back(undetailed_response.trajectory_);
156  res.processing_time_.push_back(0);
157 
158  res.description_.push_back("interpolate");
159  res.trajectory_.push_back(undetailed_response.trajectory_);
160  res.processing_time_.push_back(0);
161 
162  res.error_code_ = undetailed_response.error_code_;
163  return result;
164 }
