50 bool pilz_industrial_motion_planner::PlanningContextLoaderPTP::loadContext(
51  planning_interface::PlanningContextPtr& planning_context, const std::string& name, const std::string& group) const
52 {
53  if (limits_set_ && model_set_)
54  {
55  planning_context.reset(new PlanningContextPTP(name, group, model_, limits_));
56  return true;
57  }
58  else
59  {
60  if (!limits_set_)
61  {
62  ROS_ERROR_STREAM("Joint Limits are not defined. Cannot load planning "
63  "context. Call setLimits loadContext");
64  }
65  if (!model_set_)
66  {
67  ROS_ERROR_STREAM("Robot model was not set");
68  }
69  return false;
70  }
71 }
