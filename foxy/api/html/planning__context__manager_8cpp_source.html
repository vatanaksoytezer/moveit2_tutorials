104 template <typename T>
105 ompl::base::PlannerPtr MultiQueryPlannerAllocator::allocatePlanner(const ob::SpaceInformationPtr& si,
106  const std::string& new_name,
107  const ModelBasedPlanningContextSpecification& spec)
108 {
109  // Store planner instance if multi-query planning is enabled
110  auto cfg = spec.config_;
111  auto it = cfg.find("multi_query_planning_enabled");
112  bool multi_query_planning_enabled = false;
113  if (it != cfg.end())
114  {
115  multi_query_planning_enabled = boost::lexical_cast<bool>(it->second);
116  cfg.erase(it);
117  }
118  if (multi_query_planning_enabled)
119  {
120  // If we already have an instance, use that one
121  auto planner_map_it = planners_.find(new_name);
122  if (planner_map_it != planners_.end())
123  {
124  return planner_map_it->second;
125  }
126 
127  // Certain multi-query planners allow loading and storing the generated planner data. This feature can be
128  // selectively enabled for loading and storing using the bool parameters 'load_planner_data' and
129  // 'store_planner_data'. The storage file path is set using the parameter 'planner_data_path'.
130  // File read and write access are handled by the PlannerDataStorage class. If the file path is invalid
131  // an error message is printed and the planner is constructed/destructed with default values.
132  it = cfg.find("load_planner_data");
133  bool load_planner_data = false;
134  if (it != cfg.end())
135  {
136  load_planner_data = boost::lexical_cast<bool>(it->second);
137  cfg.erase(it);
138  }
139  it = cfg.find("store_planner_data");
140  bool store_planner_data = false;
141  if (it != cfg.end())
142  {
143  store_planner_data = boost::lexical_cast<bool>(it->second);
144  cfg.erase(it);
145  }
146  it = cfg.find("planner_data_path");
147  std::string planner_data_path;
148  if (it != cfg.end())
149  {
150  planner_data_path = it->second;
151  cfg.erase(it);
152  }
153  // Store planner instance for multi-query use
154  planners_[new_name] =
155  allocatePlannerImpl<T>(si, new_name, spec, load_planner_data, store_planner_data, planner_data_path);
156  return planners_[new_name];
157  }
158  else
159  {
160  // Return single-shot planner instance
161  return allocatePlannerImpl<T>(si, new_name, spec);
162  }
163 }

165 template <typename T>
166 ompl::base::PlannerPtr
167 MultiQueryPlannerAllocator::allocatePlannerImpl(const ob::SpaceInformationPtr& si, const std::string& new_name,
168  const ModelBasedPlanningContextSpecification& spec,
169  bool load_planner_data, bool store_planner_data,
170  const std::string& file_path)
171 {
172  ob::PlannerPtr planner;
173  // Try to initialize planner with loaded planner data
174  if (load_planner_data)
175  {
176  RCLCPP_INFO(LOGGER, "Loading planner data");
177  ob::PlannerData data(si);
178  storage_.load(file_path.c_str(), data);
179  planner.reset(allocatePersistentPlanner<T>(data));
180  if (!planner)
181  {
182  RCLCPP_ERROR(LOGGER,
183  "Creating a '%s' planner from persistent data is not supported. Going to create a new instance.",
184  new_name.c_str());
185  }
186  }
187 
188  if (!planner)
189  {
190  planner.reset(new T(si));
191  }
192 
193  if (!new_name.empty())
194  {
195  planner->setName(new_name);
196  }
197 
198  planner->params().setParams(spec.config_, true);
199  // Remember which planner instances to store when the destructor is called
200  if (store_planner_data)
201  {
202  planner_data_storage_paths_[new_name] = file_path;
203  }
204 
205  return planner;
206 }

326 ModelBasedPlanningContextPtr
327 PlanningContextManager::getPlanningContext(const planning_interface::PlannerConfigurationSettings& config,
328  const StateSpaceFactoryTypeSelector& factory_selector,
329  const moveit_msgs::msg::MotionPlanRequest& req) const
330 {
331  const ModelBasedStateSpaceFactoryPtr& factory = factory_selector(config.group);
332 
333  // Check for a cached planning context
334  ModelBasedPlanningContextPtr context;
335 
336  {
337  std::unique_lock<std::mutex> slock(cached_contexts_->lock_);
338  auto cached_contexts = cached_contexts_->contexts_.find(std::make_pair(config.name, factory->getType()));
339  if (cached_contexts != cached_contexts_->contexts_.end())
340  {
341  for (const ModelBasedPlanningContextPtr& cached_context : cached_contexts->second)
342  if (cached_context.unique())
343  {
344  RCLCPP_DEBUG(LOGGER, "Reusing cached planning context");
345  context = cached_context;
346  break;
347  }
348  }
349  }
350 
351  // Create a new planning context
352  if (!context)
353  {
354  ModelBasedStateSpaceSpecification space_spec(robot_model_, config.group);
355  ModelBasedPlanningContextSpecification context_spec;
356  context_spec.config_ = config.config;
357  context_spec.planner_selector_ = getPlannerSelector();
358  context_spec.constraint_sampler_manager_ = constraint_sampler_manager_;
359  context_spec.state_space_ = factory->getNewStateSpace(space_spec);
360 
361  if (factory->getType() == ConstrainedPlanningStateSpace::PARAMETERIZATION_TYPE)
362  {
363  RCLCPP_DEBUG_STREAM(LOGGER, "planning_context_manager: Using OMPL's constrained state space for planning.");
364 
365  // Select the correct type of constraints based on the path constraints in the planning request.
366  ompl::base::ConstraintPtr ompl_constraint =
367  createOMPLConstraint(robot_model_, config.group, req.path_constraints);
368 
369  // Create a constrained state space of type "projected state space".
370  // Other types are available, so we probably should add another setting to ompl_planning.yaml
371  // to choose between them.
372  context_spec.constrained_state_space_ =
373  std::make_shared<ob::ProjectedStateSpace>(context_spec.state_space_, ompl_constraint);
374 
375  // Pass the constrained state space to ompl simple setup through the creation of a
376  // ConstrainedSpaceInformation object. This makes sure the state space is properly initialized.
377  context_spec.ompl_simple_setup_ = std::make_shared<ompl::geometric::SimpleSetup>(
378  std::make_shared<ob::ConstrainedSpaceInformation>(context_spec.constrained_state_space_));
379  }
380  else
381  {
382  // Choose the correct simple setup type to load
383  context_spec.ompl_simple_setup_.reset(new ompl::geometric::SimpleSetup(context_spec.state_space_));
384  }
385 
386  RCLCPP_DEBUG(LOGGER, "Creating new planning context");
387  context.reset(new ModelBasedPlanningContext(config.name, context_spec));
388 
389  // Do not cache a constrained planning context, as the constraints could be changed
390  // and need to be parsed again.
391  if (factory->getType() != ConstrainedPlanningStateSpace::PARAMETERIZATION_TYPE)
392  {
393  {
394  std::lock_guard<std::mutex> slock(cached_contexts_->lock_);
395  cached_contexts_->contexts_[std::make_pair(config.name, factory->getType())].push_back(context);
396  }
397  }
398  }
399 
400  context->setMaximumPlanningThreads(max_planning_threads_);
401  context->setMaximumGoalSamples(max_goal_samples_);
402  context->setMaximumStateSamplingAttempts(max_state_sampling_attempts_);
403  context->setMaximumGoalSamplingAttempts(max_goal_sampling_attempts_);
404 
405  if (max_solution_segment_length_ > std::numeric_limits<double>::epsilon())
406  {
407  context->setMaximumSolutionSegmentLength(max_solution_segment_length_);
408  }
409 
410  context->setMinimumWaypointCount(minimum_waypoint_count_);
411  context->setSpecificationConfig(config.config);
412 
413  return context;
414 }

433 const ModelBasedStateSpaceFactoryPtr&
434 PlanningContextManager::getStateSpaceFactory2(const std::string& group,
435  const moveit_msgs::msg::MotionPlanRequest& req) const
436 {
437  // find the problem representation to use
438  auto best = state_space_factories_.end();
439  int prev_priority = 0;
440  for (auto it = state_space_factories_.begin(); it != state_space_factories_.end(); ++it)
441  {
442  int priority = it->second->canRepresentProblem(group, req, robot_model_);
443  if (priority > prev_priority)
444  {
445  best = it;
446  prev_priority = priority;
447  }
448  }
449 
450  if (best == state_space_factories_.end())
451  {
452  RCLCPP_ERROR(LOGGER, "There are no known state spaces that can represent the given planning "
453  "problem");
454  static const ModelBasedStateSpaceFactoryPtr EMPTY;
455  return EMPTY;
456  }
457  else
458  {
459  RCLCPP_DEBUG(LOGGER, "Using '%s' parameterization for solving problem", best->first.c_str());
460  return best->second;
461  }
462 }

464 ModelBasedPlanningContextPtr PlanningContextManager::getPlanningContext(
465  const planning_scene::PlanningSceneConstPtr& planning_scene, const moveit_msgs::msg::MotionPlanRequest& req,
466  moveit_msgs::msg::MoveItErrorCodes& error_code, const rclcpp::Node::SharedPtr& node,
467  bool use_constraints_approximation) const
468 {
469  if (req.group_name.empty())
470  {
471  RCLCPP_ERROR(LOGGER, "No group specified to plan for");
472  error_code.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
473  return ModelBasedPlanningContextPtr();
474  }
475 
476  error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
477 
478  if (!planning_scene)
479  {
480  RCLCPP_ERROR(LOGGER, "No planning scene supplied as input");
481  return ModelBasedPlanningContextPtr();
482  }
483 
484  // identify the correct planning configuration
485  auto pc = planner_configs_.end();
486  if (!req.planner_id.empty())
487  {
488  pc = planner_configs_.find(req.planner_id.find(req.group_name) == std::string::npos ?
489  req.group_name + "[" + req.planner_id + "]" :
490  req.planner_id);
491  if (pc == planner_configs_.end())
492  {
493  RCLCPP_WARN(LOGGER,
494  "Cannot find planning configuration for group '%s' using planner '%s'. Will use defaults instead.",
495  req.group_name.c_str(), req.planner_id.c_str());
496  }
497  }
498 
499  if (pc == planner_configs_.end())
500  {
501  pc = planner_configs_.find(req.group_name);
502  if (pc == planner_configs_.end())
503  {
504  RCLCPP_ERROR(LOGGER, "Cannot find planning configuration for group '%s'", req.group_name.c_str());
505  return ModelBasedPlanningContextPtr();
506  }
507  }
508 
509  // State space selection process
510  // ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
511  // There are 3 options for the factory_selector
512  // 1) enforce_constrained_state_space = true AND there are path constraints in the planning request
513  // Overrides all other settings and selects a ConstrainedPlanningStateSpace factory
514  // 2) enforce_joint_model_state_space = true
515  // If 1) is false, then this one overrides the remaining settings and returns a JointModelStateSpace factory
516  // 3) Not 1) or 2), then the factory is selected based on the priority that each one returns.
517  // See PoseModelStateSpaceFactory::canRepresentProblem for details on the selection process.
518  // In short, it returns a PoseModelStateSpace if there is an IK solver and a path constraint.
519  //
520  // enforce_constrained_state_space
521  // ****************************************
522  // Check if the user wants to use an OMPL ConstrainedStateSpace for planning.
523  // This is done by setting 'enforce_constrained_state_space' to 'true' for the desired group in ompl_planing.yaml.
524  // If there are no path constraints in the planning request, this option is ignored, as the constrained state space is
525  // only usefull for paths constraints. (And at the moment only a single position constraint is supported, hence:
526  // req.path_constraints.position_constraints.size() == 1
527  // is used in the selection process below.)
528  //
529  // enforce_joint_model_state_space
530  // *******************************
531  // Check if sampling in JointModelStateSpace is enforced for this group by user.
532  // This is done by setting 'enforce_joint_model_state_space' to 'true' for the desired group in ompl_planning.yaml.
533  //
534  // Some planning problems like orientation path constraints are represented in PoseModelStateSpace and sampled via IK.
535  // However consecutive IK solutions are not checked for proximity at the moment and sometimes happen to be flipped,
536  // leading to invalid trajectories. This workaround lets the user prevent this problem by forcing rejection sampling
537  // in JointModelStateSpace.
538  StateSpaceFactoryTypeSelector factory_selector;
539  auto constrained_planning_iterator = pc->second.config.find("enforce_constrained_state_space");
540  auto joint_space_planning_iterator = pc->second.config.find("enforce_joint_model_state_space");
541 
542  if (constrained_planning_iterator != pc->second.config.end() &&
543  boost::lexical_cast<bool>(constrained_planning_iterator->second) &&
544  req.path_constraints.position_constraints.size() == 1)
545  {
546  factory_selector = std::bind(&PlanningContextManager::getStateSpaceFactory1, this, std::placeholders::_1,
547  ConstrainedPlanningStateSpace::PARAMETERIZATION_TYPE);
548  }
549  else if (joint_space_planning_iterator != pc->second.config.end() &&
550  boost::lexical_cast<bool>(joint_space_planning_iterator->second))
551  {
552  factory_selector = std::bind(&PlanningContextManager::getStateSpaceFactory1, this, std::placeholders::_1,
