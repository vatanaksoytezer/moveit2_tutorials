51 planning_pipeline::PlanningPipeline::PlanningPipeline(const moveit::core::RobotModelConstPtr& model,
52  const std::shared_ptr<rclcpp::Node>& node,
53  const std::string& parameter_namespace,
54  const std::string& planner_plugin_param_name,
55  const std::string& adapter_plugins_param_name)
56  : node_(node), parameter_namespace_(parameter_namespace), robot_model_(model)
57 {
58  std::string planner_plugin_fullname = parameter_namespace_ + "." + planner_plugin_param_name;
59  if (parameter_namespace_.empty())
60  planner_plugin_fullname = planner_plugin_param_name;
61  if (node_->has_parameter(planner_plugin_fullname))
62  {
63  node_->get_parameter(planner_plugin_fullname, planner_plugin_name_);
64  }
65 
66  std::string adapter_plugins_fullname = parameter_namespace_ + "." + adapter_plugins_param_name;
67  if (parameter_namespace_.empty())
68  adapter_plugins_fullname = adapter_plugins_param_name;
69 
70  std::string adapters;
71  if (node_->has_parameter(adapter_plugins_fullname))
72  {
73  node_->get_parameter(adapter_plugins_fullname, adapters);
74  boost::char_separator<char> sep(" ");
75  boost::tokenizer<boost::char_separator<char> > tok(adapters, sep);
76  for (boost::tokenizer<boost::char_separator<char> >::iterator beg = tok.begin(); beg != tok.end(); ++beg)
77  adapter_plugin_names_.push_back(*beg);
78  }
79 
80  configure();
81 }

97 void planning_pipeline::PlanningPipeline::configure()
98 {
99  check_solution_paths_ = false; // this is set to true below
100  publish_received_requests_ = false;
101  display_computed_motion_plans_ = false; // this is set to true below
102 
103  // load the planning plugin
104  try
105  {
106  planner_plugin_loader_.reset(new pluginlib::ClassLoader<planning_interface::PlannerManager>(
107  "moveit_core", "planning_interface::PlannerManager"));
108  }
109  catch (pluginlib::PluginlibException& ex)
110  {
111  RCLCPP_FATAL(LOGGER, "Exception while creating planning plugin loader %s", ex.what());
112  }
113 
114  std::vector<std::string> classes;
115  if (planner_plugin_loader_)
116  classes = planner_plugin_loader_->getDeclaredClasses();
117  if (planner_plugin_name_.empty() && classes.size() == 1)
118  {
119  planner_plugin_name_ = classes[0];
120  RCLCPP_INFO(
121  LOGGER,
122  "No '~planning_plugin' parameter specified, but only '%s' planning plugin is available. Using that one.",
123  planner_plugin_name_.c_str());
124  }
125  if (planner_plugin_name_.empty() && classes.size() > 1)
126  {
127  planner_plugin_name_ = classes[0];
128  RCLCPP_INFO(
129  LOGGER,
130  "Multiple planning plugins available. You should specify the '~planning_plugin' parameter. Using '%s' for "
131  "now.",
132  planner_plugin_name_.c_str());
133  }
134  try
135  {
136  planner_instance_ = planner_plugin_loader_->createUniqueInstance(planner_plugin_name_);
137  if (!planner_instance_->initialize(robot_model_, node_, parameter_namespace_))
138  throw std::runtime_error("Unable to initialize planning plugin");
139  RCLCPP_INFO(LOGGER, "Using planning interface '%s'", planner_instance_->getDescription().c_str());
140  }
141  catch (pluginlib::PluginlibException& ex)
142  {
143  std::string classes_str = boost::algorithm::join(classes, ", ");
144  RCLCPP_ERROR(LOGGER,
145  "Exception while loading planner '%s': %s"
146  "Available plugins: %s",
147  planner_plugin_name_.c_str(), ex.what(), classes_str.c_str());
148  }
149 
150  // load the planner request adapters
151  if (!adapter_plugin_names_.empty())
152  {
153  std::vector<planning_request_adapter::PlanningRequestAdapterConstPtr> ads;
154  try
155  {
156  adapter_plugin_loader_.reset(new pluginlib::ClassLoader<planning_request_adapter::PlanningRequestAdapter>(
157  "moveit_core", "planning_request_adapter::PlanningRequestAdapter"));
158  }
159  catch (pluginlib::PluginlibException& ex)
160  {
161  RCLCPP_ERROR(LOGGER, "Exception while creating planning plugin loader %s", ex.what());
162  }
163 
164  if (adapter_plugin_loader_)
165  for (const std::string& adapter_plugin_name : adapter_plugin_names_)
166  {
167  planning_request_adapter::PlanningRequestAdapterPtr ad;
168  try
169  {
170  ad = adapter_plugin_loader_->createUniqueInstance(adapter_plugin_name);
171  }
172  catch (pluginlib::PluginlibException& ex)
173  {
174  RCLCPP_ERROR(LOGGER, "Exception while loading planning adapter plugin '%s': %s", adapter_plugin_name.c_str(),
175  ex.what());
176  }
177  if (ad)
178  {
179  ad->initialize(node_, parameter_namespace_);
180  ads.push_back(std::move(ad));
181  }
182  }
183  if (!ads.empty())
184  {
185  adapter_chain_.reset(new planning_request_adapter::PlanningRequestAdapterChain());
186  for (planning_request_adapter::PlanningRequestAdapterConstPtr& ad : ads)
187  {
188  RCLCPP_INFO(LOGGER, "Using planning request adapter '%s'", ad->getDescription().c_str());
189  adapter_chain_->addAdapter(ad);
190  }
191  }
192  }
193  displayComputedMotionPlans(true);
194  checkSolutionPaths(true);
195 }

239 bool planning_pipeline::PlanningPipeline::generatePlan(const planning_scene::PlanningSceneConstPtr& planning_scene,
240  const planning_interface::MotionPlanRequest& req,
241  planning_interface::MotionPlanResponse& res,
242  std::vector<std::size_t>& adapter_added_state_index) const
243 {
244  // broadcast the request we are about to work on, if needed
245  if (publish_received_requests_)
246  received_request_publisher_->publish(req);
247  adapter_added_state_index.clear();
248 
249  if (!planner_instance_)
250  {
251  RCLCPP_ERROR(LOGGER, "No planning plugin loaded. Cannot plan.");
252  return false;
253  }
254 
255  bool solved = false;
256  try
257  {
258  if (adapter_chain_)
259  {
260  solved = adapter_chain_->adaptAndPlan(planner_instance_, planning_scene, req, res, adapter_added_state_index);
261  if (!adapter_added_state_index.empty())
262  {
263  std::stringstream ss;
264  for (std::size_t added_index : adapter_added_state_index)
265  ss << added_index << " ";
266  RCLCPP_INFO(LOGGER, "Planning adapters have added states at index positions: [ %s]", ss.str().c_str());
267  }
268  }
269  else
270  {
271  planning_interface::PlanningContextPtr context =
272  planner_instance_->getPlanningContext(planning_scene, req, res.error_code_);
273  solved = context ? context->solve(res) : false;
274  }
275  }
276  catch (std::exception& ex)
277  {
278  RCLCPP_ERROR(LOGGER, "Exception caught: '%s'", ex.what());
279  return false;
280  }
281  bool valid = true;
282 
283  if (solved && res.trajectory_)
284  {
285  std::size_t state_count = res.trajectory_->getWayPointCount();
286  RCLCPP_DEBUG(LOGGER, "Motion planner reported a solution path with %ld states", state_count);
287  if (check_solution_paths_)
288  {
289  std::vector<std::size_t> index;
290  if (!planning_scene->isPathValid(*res.trajectory_, req.path_constraints, req.group_name, false, &index))
291  {
292  // check to see if there is any problem with the states that are found to be invalid
293  // they are considered ok if they were added by a planning request adapter
294  bool problem = false;
295  for (std::size_t i = 0; i < index.size() && !problem; ++i)
296  {
297  bool found = false;
298  for (std::size_t added_index : adapter_added_state_index)
299  if (index[i] == added_index)
300  {
301  found = true;
302  break;
303  }
304  if (!found)
305  problem = true;
306  }
307  if (problem)
308  {
309  if (index.size() == 1 && index[0] == 0)
310  { // ignore cases when the robot starts at invalid location
311  RCLCPP_DEBUG(LOGGER, "It appears the robot is starting at an invalid state, but that is ok.");
312  }
313  else
314  {
315  valid = false;
316  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_MOTION_PLAN;
317 
318  // display error messages
319  std::stringstream ss;
320  for (std::size_t it : index)
321  ss << it << " ";
322 
323  RCLCPP_ERROR_STREAM(LOGGER, "Computed path is not valid. Invalid states at index locations: [ "
324  << ss.str() << "] out of " << state_count
325  << ". Explanations follow in command line. Contacts are published on "
326  << contacts_publisher_->get_topic_name());
327 
328  // call validity checks in verbose mode for the problematic states
329  visualization_msgs::msg::MarkerArray arr;
330  for (std::size_t it : index)
331  {
332  // check validity with verbose on
333  const moveit::core::RobotState& robot_state = res.trajectory_->getWayPoint(it);
334  planning_scene->isStateValid(robot_state, req.path_constraints, req.group_name, true);
335 
336  // compute the contacts if any
337  collision_detection::CollisionRequest c_req;
338  collision_detection::CollisionResult c_res;
339  c_req.contacts = true;
340  c_req.max_contacts = 10;
341  c_req.max_contacts_per_pair = 3;
342  c_req.verbose = false;
343  planning_scene->checkCollision(c_req, c_res, robot_state);
344  if (c_res.contact_count > 0)
345  {
346  visualization_msgs::msg::MarkerArray arr_i;
347  collision_detection::getCollisionMarkersFromContacts(arr_i, planning_scene->getPlanningFrame(),
348  c_res.contacts);
349  arr.markers.insert(arr.markers.end(), arr_i.markers.begin(), arr_i.markers.end());
350  }
351  }
352  RCLCPP_ERROR(LOGGER, "Completed listing of explanations for invalid states.");
353  if (!arr.markers.empty())
354  {
355  contacts_publisher_->publish(arr);
356  }
357  }
358  }
359  else
360  {
361  RCLCPP_DEBUG(LOGGER,
362  "Planned path was found to be valid, except for states that were added by planning request "
363  "adapters, but that is ok.");
364  }
365  }
366  else
367  RCLCPP_DEBUG(LOGGER, "Planned path was found to be valid when rechecked");
368  }
369  }
370 
371  // display solution path if needed
372  if (display_computed_motion_plans_ && solved)
373  {
374  moveit_msgs::msg::DisplayTrajectory disp;
375  disp.model_id = robot_model_->getName();
376  disp.trajectory.resize(1);
377  res.trajectory_->getRobotTrajectoryMsg(disp.trajectory[0]);
378  moveit::core::robotStateToRobotStateMsg(res.trajectory_->getFirstWayPoint(), disp.trajectory_start);
379  display_path_publisher_->publish(disp);
380  }
381 
382  if (!solved)
383  {
384  // This should alert the user if planning failed because of contradicting constraints.
385  // Could be checked more thoroughly, but it is probably not worth going to that length.
386  bool stacked_constraints = false;
387  if (req.path_constraints.position_constraints.size() > 1 || req.path_constraints.orientation_constraints.size() > 1)
388  stacked_constraints = true;
389  for (const auto& constraint : req.goal_constraints)
390  {
391  if (constraint.position_constraints.size() > 1 || constraint.orientation_constraints.size() > 1)
392  stacked_constraints = true;
393  }
394  if (stacked_constraints)
395  RCLCPP_WARN(LOGGER, "More than one constraint is set. If your move_group does not have multiple end "
396  "effectors/arms, this is "
397  "unusual. Are you using a move_group_interface and forgetting to call clearPoseTargets() or "
398  "equivalent?");
399  }
400 
401  return solved && valid;
402 }
