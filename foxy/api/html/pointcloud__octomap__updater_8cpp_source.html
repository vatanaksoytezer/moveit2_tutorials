81 bool PointCloudOctomapUpdater::initialize(const rclcpp::Node::SharedPtr& node)
82 {
83  node_ = node;
84  tf_buffer_.reset(new tf2_ros::Buffer(node_->get_clock()));
85  auto create_timer_interface =
86  std::make_shared<tf2_ros::CreateTimerROS>(node->get_node_base_interface(), node->get_node_timers_interface());
87  tf_buffer_->setCreateTimerInterface(create_timer_interface);
88  tf_listener_.reset(new tf2_ros::TransformListener(*tf_buffer_));
89  shape_mask_.reset(new point_containment_filter::ShapeMask());
90  shape_mask_->setTransformCallback(boost::bind(&PointCloudOctomapUpdater::getShapeTransform, this,
91  boost::placeholders::_1, boost::placeholders::_2));
92  last_update_time_ = node_->now();
93  return true;
94 }

96 void PointCloudOctomapUpdater::start()
97 {
98  if (!filtered_cloud_topic_.empty())
99  filtered_cloud_publisher_ = node_->create_publisher<sensor_msgs::msg::PointCloud2>(filtered_cloud_topic_, 10);
100 
101  if (point_cloud_subscriber_)
102  return;
103  /* subscribe to point cloud topic using tf filter*/
104  point_cloud_subscriber_ = new message_filters::Subscriber<sensor_msgs::msg::PointCloud2>(node_, point_cloud_topic_);
105  if (tf_listener_ && tf_buffer_ && !monitor_->getMapFrame().empty())
106  {
107  point_cloud_filter_ = new tf2_ros::MessageFilter<sensor_msgs::msg::PointCloud2>(
108  *point_cloud_subscriber_, *tf_buffer_, monitor_->getMapFrame(), 5, node_);
109  point_cloud_filter_->registerCallback(
110  boost::bind(&PointCloudOctomapUpdater::cloudMsgCallback, this, boost::placeholders::_1));
111  RCLCPP_INFO(LOGGER, "Listening to '%s' using message filter with target frame '%s'", point_cloud_topic_.c_str(),
112  point_cloud_filter_->getTargetFramesString().c_str());
113  }
114  else
115  {
116  point_cloud_subscriber_->registerCallback(
117  boost::bind(&PointCloudOctomapUpdater::cloudMsgCallback, this, boost::placeholders::_1));
118  RCLCPP_INFO(LOGGER, "Listening to '%s'", point_cloud_topic_.c_str());
119  }
120 }

166 void PointCloudOctomapUpdater::cloudMsgCallback(const sensor_msgs::msg::PointCloud2::ConstSharedPtr& cloud_msg)
167 {
168  RCLCPP_DEBUG(LOGGER, "Received a new point cloud message");
169  rclcpp::Time start = rclcpp::Clock(RCL_ROS_TIME).now();
170 
171  if (max_update_rate_ > 0)
172  {
173  // ensure we are not updating the octomap representation too often
174  if ((rclcpp::Clock(RCL_ROS_TIME).now() - last_update_time_) <=
175  rclcpp::Duration(std::chrono::duration<double>(1.0 / max_update_rate_)))
176  return;
177  last_update_time_ = rclcpp::Clock(RCL_ROS_TIME).now();
178  }
179 
180  if (monitor_->getMapFrame().empty())
181  monitor_->setMapFrame(cloud_msg->header.frame_id);
182 
183  /* get transform for cloud into map frame */
184  tf2::Stamped<tf2::Transform> map_h_sensor;
185  if (monitor_->getMapFrame() == cloud_msg->header.frame_id)
186  map_h_sensor.setIdentity();
187  else
188  {
189  if (tf_buffer_)
190  {
191  try
192  {
193  tf2::fromMsg(tf_buffer_->lookupTransform(monitor_->getMapFrame(), cloud_msg->header.frame_id,
194  cloud_msg->header.stamp),
195  map_h_sensor);
196  }
197  catch (tf2::TransformException& ex)
198  {
199  RCLCPP_ERROR_STREAM(LOGGER, "Transform error of sensor data: " << ex.what() << "; quitting callback");
200  return;
201  }
202  }
203  else
204  return;
205  }
206 
207  /* compute sensor origin in map frame */
208  const tf2::Vector3& sensor_origin_tf = map_h_sensor.getOrigin();
209  octomap::point3d sensor_origin(sensor_origin_tf.getX(), sensor_origin_tf.getY(), sensor_origin_tf.getZ());
210  Eigen::Vector3d sensor_origin_eigen(sensor_origin_tf.getX(), sensor_origin_tf.getY(), sensor_origin_tf.getZ());
211 
212  if (!updateTransformCache(cloud_msg->header.frame_id, cloud_msg->header.stamp))
213  return;
214 
215  /* mask out points on the robot */
216  shape_mask_->maskContainment(*cloud_msg, sensor_origin_eigen, 0.0, max_range_, mask_);
217  updateMask(*cloud_msg, sensor_origin_eigen, mask_);
218 
219  octomap::KeySet free_cells, occupied_cells, model_cells, clip_cells;
220  std::unique_ptr<sensor_msgs::msg::PointCloud2> filtered_cloud;
221 
222  // We only use these iterators if we are creating a filtered_cloud for
223  // publishing. We cannot default construct these, so we use unique_ptr's
224  // to defer construction
225  std::unique_ptr<sensor_msgs::PointCloud2Iterator<float> > iter_filtered_x;
226  std::unique_ptr<sensor_msgs::PointCloud2Iterator<float> > iter_filtered_y;
227  std::unique_ptr<sensor_msgs::PointCloud2Iterator<float> > iter_filtered_z;
228 
229  if (!filtered_cloud_topic_.empty())
230  {
231  filtered_cloud.reset(new sensor_msgs::msg::PointCloud2());
232  filtered_cloud->header = cloud_msg->header;
233  sensor_msgs::PointCloud2Modifier pcd_modifier(*filtered_cloud);
234  pcd_modifier.setPointCloud2FieldsByString(1, "xyz");
235  pcd_modifier.resize(cloud_msg->width * cloud_msg->height);
236 
237  // we have created a filtered_out, so we can create the iterators now
238  iter_filtered_x.reset(new sensor_msgs::PointCloud2Iterator<float>(*filtered_cloud, "x"));
239  iter_filtered_y.reset(new sensor_msgs::PointCloud2Iterator<float>(*filtered_cloud, "y"));
240  iter_filtered_z.reset(new sensor_msgs::PointCloud2Iterator<float>(*filtered_cloud, "z"));
241  }
242  size_t filtered_cloud_size = 0;
243 
244  tree_->lockRead();
245 
246  try
247  {
248  /* do ray tracing to find which cells this point cloud indicates should be free, and which it indicates
249  * should be occupied */
250  for (unsigned int row = 0; row < cloud_msg->height; row += point_subsample_)
251  {
252  unsigned int row_c = row * cloud_msg->width;
253  sensor_msgs::PointCloud2ConstIterator<float> pt_iter(*cloud_msg, "x");
254  // set iterator to point at start of the current row
255  pt_iter += row_c;
256 
257  for (unsigned int col = 0; col < cloud_msg->width; col += point_subsample_, pt_iter += point_subsample_)
258  {
259  // if (mask_[row_c + col] == point_containment_filter::ShapeMask::CLIP)
260  // continue;
261 
262  /* check for NaN */
263  if (!std::isnan(pt_iter[0]) && !std::isnan(pt_iter[1]) && !std::isnan(pt_iter[2]))
264  {
265  /* transform to map frame */
266  tf2::Vector3 point_tf = map_h_sensor * tf2::Vector3(pt_iter[0], pt_iter[1], pt_iter[2]);
267 
268  /* occupied cell at ray endpoint if ray is shorter than max range and this point
269  isn't on a part of the robot*/
270  if (mask_[row_c + col] == point_containment_filter::ShapeMask::INSIDE)
271  model_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
272  else if (mask_[row_c + col] == point_containment_filter::ShapeMask::CLIP)
273  clip_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
274  else
275  {
276  occupied_cells.insert(tree_->coordToKey(point_tf.getX(), point_tf.getY(), point_tf.getZ()));
277  // build list of valid points if we want to publish them
278  if (filtered_cloud)
279  {
280  **iter_filtered_x = pt_iter[0];
281  **iter_filtered_y = pt_iter[1];
282  **iter_filtered_z = pt_iter[2];
283  ++filtered_cloud_size;
284  ++*iter_filtered_x;
285  ++*iter_filtered_y;
286  ++*iter_filtered_z;
287  }
288  }
289  }
290  }
291  }
292 
293  /* compute the free cells along each ray that ends at an occupied cell */
294  for (const octomap::OcTreeKey& occupied_cell : occupied_cells)
295  if (tree_->computeRayKeys(sensor_origin, tree_->keyToCoord(occupied_cell), key_ray_))
296  free_cells.insert(key_ray_.begin(), key_ray_.end());
297 
298  /* compute the free cells along each ray that ends at a model cell */
299  for (const octomap::OcTreeKey& model_cell : model_cells)
300  if (tree_->computeRayKeys(sensor_origin, tree_->keyToCoord(model_cell), key_ray_))
301  free_cells.insert(key_ray_.begin(), key_ray_.end());
302 
303  /* compute the free cells along each ray that ends at a clipped cell */
304  for (const octomap::OcTreeKey& clip_cell : clip_cells)
305  if (tree_->computeRayKeys(sensor_origin, tree_->keyToCoord(clip_cell), key_ray_))
306  free_cells.insert(key_ray_.begin(), key_ray_.end());
307  }
308  catch (...)
309  {
310  tree_->unlockRead();
311  return;
312  }
313 
314  tree_->unlockRead();
315 
316  /* cells that overlap with the model are not occupied */
317  for (const octomap::OcTreeKey& model_cell : model_cells)
318  occupied_cells.erase(model_cell);
319 
320  /* occupied cells are not free */
321  for (const octomap::OcTreeKey& occupied_cell : occupied_cells)
322  free_cells.erase(occupied_cell);
323 
324  tree_->lockWrite();
325 
326  try
327  {
328  /* mark free cells only if not seen occupied in this cloud */
329  for (const octomap::OcTreeKey& free_cell : free_cells)
330  tree_->updateNode(free_cell, false);
331 
332  /* now mark all occupied cells */
333  for (const octomap::OcTreeKey& occupied_cell : occupied_cells)
334  tree_->updateNode(occupied_cell, true);
335 
336  // set the logodds to the minimum for the cells that are part of the model
337  const float lg = tree_->getClampingThresMinLog() - tree_->getClampingThresMaxLog();
338  for (const octomap::OcTreeKey& model_cell : model_cells)
339  tree_->updateNode(model_cell, lg);
340  }
341  catch (...)
342  {
343  RCLCPP_ERROR(LOGGER, "Internal error while updating octree");
344  }
345  tree_->unlockWrite();
346  RCLCPP_DEBUG(LOGGER, "Processed point cloud in %lf ms", (node_->now() - start).seconds() * 1000.0);
347  tree_->triggerUpdateCallback();
348 
349  if (filtered_cloud)
350  {
351  sensor_msgs::PointCloud2Modifier pcd_modifier(*filtered_cloud);
352  pcd_modifier.resize(filtered_cloud_size);
353  filtered_cloud_publisher_->publish(*filtered_cloud);
354  }
355 }
