119 static const double DEFAULT_SCALE = 0.25;
120 double RobotInteraction::computeLinkMarkerSize(const std::string& link)
121 {
122  const moveit::core::LinkModel* lm = robot_model_->getLinkModel(link);
123  double size = 0;
124 
125  while (lm)
126  {
127  Eigen::Vector3d ext = lm->getShapeExtentsAtOrigin();
128  // drop largest extension and take norm of two remaining
129  Eigen::MatrixXd::Index max_index;
130  ext.maxCoeff(&max_index);
131  ext[max_index] = 0;
132  size = 1.01 * ext.norm();
133  if (size > 0)
134  break; // break, if a non-empty shape was found
135 
136  // process kinematic chain upwards (but only following fixed joints)
137  // to find a link with some non-empty shape (to ignore virtual links)
138  if (lm->getParentJointModel()->getType() == moveit::core::JointModel::FIXED)
139  lm = lm->getParentLinkModel();
140  else
141  lm = nullptr;
142  }
143  if (!lm)
144  return DEFAULT_SCALE; // no link with non-zero shape extends found
145 
146  // the marker sphere will be half the size, so double the size here
147  return 2. * size;
148 }

150 double RobotInteraction::computeGroupMarkerSize(const std::string& group)
151 {
152  if (group.empty())
153  return DEFAULT_SCALE;
154  const moveit::core::JointModelGroup* jmg = robot_model_->getJointModelGroup(group);
155  if (!jmg)
156  return 0.0;
157 
158  const std::vector<std::string>& links = jmg->getLinkModelNames();
159  if (links.empty())
160  return DEFAULT_SCALE;
161 
162  // compute the aabb of the links that make up the group
163  double size = 0;
164  for (const std::string& link : links)
165  {
166  const moveit::core::LinkModel* lm = robot_model_->getLinkModel(link);
167  if (!lm)
168  continue;
169  Eigen::Vector3d ext = lm->getShapeExtentsAtOrigin();
170 
171  // drop largest extension and take norm of two remaining
172  Eigen::MatrixXd::Index max_index;
173  ext.maxCoeff(&max_index);
174  ext[max_index] = 0;
175  size = std::max(size, 1.01 * ext.norm());
176  }
177 
178  // if size is zero, all links have empty shapes and are placed at same position
179  // in this case, fall back to link marker size
180  if (size == 0)
181  return computeLinkMarkerSize(links[0]);
182 
183  // the marker sphere will be half the size, so double the size here
184  return 2. * size;
185 }

187 void RobotInteraction::decideActiveJoints(const std::string& group)
188 {
189  boost::unique_lock<boost::mutex> ulock(marker_access_lock_);
190  active_vj_.clear();
191 
192  if (group.empty())
193  return;
194 
195  RCLCPP_DEBUG(LOGGER, "Deciding active joints for group '%s'", group.c_str());
196 
197  const srdf::ModelConstSharedPtr& srdf = robot_model_->getSRDF();
198  const moveit::core::JointModelGroup* jmg = robot_model_->getJointModelGroup(group);
199 
200  if (!jmg || !srdf)
201  return;
202 
203  std::set<std::string> used;
204  if (jmg->hasJointModel(robot_model_->getRootJointName()))
205  {
206  moveit::core::RobotState default_state(robot_model_);
207  default_state.setToDefaultValues();
208  std::vector<double> aabb;
209  default_state.computeAABB(aabb);
210 
211  const std::vector<srdf::Model::VirtualJoint>& vj = srdf->getVirtualJoints();
212  for (const srdf::Model::VirtualJoint& joint : vj)
213  if (joint.name_ == robot_model_->getRootJointName())
214  {
215  if (joint.type_ == "planar" || joint.type_ == "floating")
216  {
217  JointInteraction v;
218  v.connecting_link = joint.child_link_;
219  v.parent_frame = joint.parent_frame_;
220  if (!v.parent_frame.empty() && v.parent_frame[0] == '/')
221  v.parent_frame = v.parent_frame.substr(1);
222  v.joint_name = joint.name_;
223  if (joint.type_ == "planar")
224  v.dof = 3;
225  else
226  v.dof = 6;
227  // take the max of the X, Y, Z extent
228  v.size = std::max(std::max(aabb[1] - aabb[0], aabb[3] - aabb[2]), aabb[5] - aabb[4]);
229  active_vj_.push_back(v);
230  used.insert(v.joint_name);
231  }
232  }
233  }
234 
235  const std::vector<const moveit::core::JointModel*>& joints = jmg->getJointModels();
236  for (const moveit::core::JointModel* joint : joints)
237  {
238  if ((joint->getType() == moveit::core::JointModel::PLANAR ||
239  joint->getType() == moveit::core::JointModel::FLOATING) &&
240  used.find(joint->getName()) == used.end())
241  {
242  JointInteraction v;
243  v.connecting_link = joint->getChildLinkModel()->getName();
244  if (joint->getParentLinkModel())
245  v.parent_frame = joint->getParentLinkModel()->getName();
246  v.joint_name = joint->getName();
247  if (joint->getType() == moveit::core::JointModel::PLANAR)
248  v.dof = 3;
249  else
250  v.dof = 6;
251  // take the max of the X, Y, Z extent
252  v.size = computeGroupMarkerSize(group);
253  active_vj_.push_back(v);
254  }
255  }
256 }

263 void RobotInteraction::decideActiveEndEffectors(const std::string& group, InteractionStyle::InteractionStyle style)
264 {
265  boost::unique_lock<boost::mutex> ulock(marker_access_lock_);
266  active_eef_.clear();
267 
268  if (group.empty())
269  return;
270 
271  RCLCPP_DEBUG(LOGGER, "Deciding active end-effectors for group '%s'", group.c_str());
272 
273  const srdf::ModelConstSharedPtr& srdf = robot_model_->getSRDF();
274  const moveit::core::JointModelGroup* jmg = robot_model_->getJointModelGroup(group);
275 
276  if (!jmg || !srdf)
277  {
278  RCLCPP_WARN(LOGGER, "Unable to decide active end effector: no joint model group or no srdf model");
279  return;
280  }
281 
282  const std::vector<srdf::Model::EndEffector>& eefs = srdf->getEndEffectors();
283  const std::pair<moveit::core::JointModelGroup::KinematicsSolver, moveit::core::JointModelGroup::KinematicsSolverMap>&
284  smap = jmg->getGroupKinematics();
285 
286  // if we have an IK solver for the selected group, we check if there are any end effectors attached to this group
287  if (smap.first)
288  {
289  for (const srdf::Model::EndEffector& eef : eefs)
290  if ((jmg->hasLinkModel(eef.parent_link_) || jmg->getName() == eef.parent_group_) &&
291  jmg->canSetStateFromIK(eef.parent_link_))
292  {
293  // We found an end-effector whose parent is the group.
294  EndEffectorInteraction ee;
295  ee.parent_group = group;
296  ee.parent_link = eef.parent_link_;
297  ee.eef_group = eef.component_group_;
298  ee.interaction = style;
299  active_eef_.push_back(ee);
300  }
301 
302  // No end effectors found. Use last link in group as the "end effector".
303  if (active_eef_.empty() && !jmg->getLinkModelNames().empty())
304  {
305  EndEffectorInteraction ee;
306  ee.parent_group = group;
307  ee.parent_link = jmg->getLinkModelNames().back();
308  ee.eef_group = group;
309  ee.interaction = style;
310  active_eef_.push_back(ee);
311  }
312  }
313  else if (!smap.second.empty())
314  {
315  for (const std::pair<const moveit::core::JointModelGroup* const, moveit::core::JointModelGroup::KinematicsSolver>&
316  it : smap.second)
317  {
318  for (const srdf::Model::EndEffector& eef : eefs)
319  {
320  if ((it.first->hasLinkModel(eef.parent_link_) || jmg->getName() == eef.parent_group_) &&
321  it.first->canSetStateFromIK(eef.parent_link_))
322  {
323  // We found an end-effector whose parent is a subgroup of the group. (May be more than one)
324  EndEffectorInteraction ee;
325  ee.parent_group = it.first->getName();
326  ee.parent_link = eef.parent_link_;
327  ee.eef_group = eef.component_group_;
328  ee.interaction = style;
329  active_eef_.push_back(ee);
330  break;
331  }
332  }
333  }
334  }
335 
336  for (EndEffectorInteraction& eef : active_eef_)
337  {
338  // if we have a separate group for the eef, we compute the scale based on it;
339  // otherwise, we use the size of the parent_link
340  eef.size = (eef.eef_group == eef.parent_group) ? computeLinkMarkerSize(eef.parent_link) :
341  computeGroupMarkerSize(eef.eef_group);
342  RCLCPP_DEBUG(LOGGER, "Found active end-effector '%s', of scale %lf", eef.eef_group.c_str(), eef.size);
343  }
344  // if there is only a single end effector marker, we can safely use a larger marker
345  if (active_eef_.size() == 1)
346  active_eef_[0].size *= 1.5;
347 }

384 void RobotInteraction::addEndEffectorMarkers(const InteractionHandlerPtr& handler, const EndEffectorInteraction& eef,
385  const geometry_msgs::msg::Pose& im_to_eef,
386  visualization_msgs::msg::InteractiveMarker& im, bool position,
387  bool orientation)
388 {
389  if (eef.parent_group == eef.eef_group || !robot_model_->hasLinkModel(eef.parent_link))
390  return;
391 
392  visualization_msgs::msg::InteractiveMarkerControl m_control;
393  m_control.always_visible = false;
394  if (position && orientation)
395  m_control.interaction_mode = m_control.MOVE_ROTATE_3D;
396  else if (orientation)
397  m_control.interaction_mode = m_control.ROTATE_3D;
398  else
399  m_control.interaction_mode = m_control.MOVE_3D;
400 
401  std_msgs::msg::ColorRGBA marker_color;
402  const float* color = handler->inError(eef) ? END_EFFECTOR_UNREACHABLE_COLOR : END_EFFECTOR_REACHABLE_COLOR;
403  marker_color.r = color[0];
404  marker_color.g = color[1];
405  marker_color.b = color[2];
406  marker_color.a = color[3];
407 
408  moveit::core::RobotStateConstPtr rstate = handler->getState();
409  const std::vector<std::string>& link_names = rstate->getJointModelGroup(eef.eef_group)->getLinkModelNames();
410  visualization_msgs::msg::MarkerArray marker_array;
411  rstate->getRobotMarkers(marker_array, link_names, marker_color, eef.eef_group, rclcpp::Duration(0.0));
412  tf2::Transform tf_root_to_link;
413  tf2::fromMsg(tf2::toMsg(rstate->getGlobalLinkTransform(eef.parent_link)), tf_root_to_link);
414  // Release the ptr count on the kinematic state
415  rstate.reset();
416 
417  for (visualization_msgs::msg::Marker& marker : marker_array.markers)
418  {
419  marker.header = im.header;
420  marker.mesh_use_embedded_materials = !marker.mesh_resource.empty();
421  // - - - - - - Do some math for the offset - - - - - -
422  tf2::Transform tf_root_to_im, tf_root_to_mesh, tf_im_to_eef;
423  tf2::fromMsg(im.pose, tf_root_to_im);
424  tf2::fromMsg(marker.pose, tf_root_to_mesh);
425  tf2::fromMsg(im_to_eef, tf_im_to_eef);
426  tf2::Transform tf_eef_to_mesh = tf_root_to_link.inverse() * tf_root_to_mesh;
427  tf2::Transform tf_im_to_mesh = tf_im_to_eef * tf_eef_to_mesh;
428  tf2::Transform tf_root_to_mesh_new = tf_root_to_im * tf_im_to_mesh;
429  tf2::toMsg(tf_root_to_mesh_new, marker.pose);
430  // - - - - - - - - - - - - - - - - - - - - - - - - - -
431  m_control.markers.push_back(marker);
432  }
433 
434  im.controls.push_back(m_control);
435 }

452 void RobotInteraction::addInteractiveMarkers(const InteractionHandlerPtr& handler, const double marker_scale)
453 {
454  // If scale is left at default size of 0, scale will be based on end effector link size. a good value is between 0-1
455  std::vector<visualization_msgs::msg::InteractiveMarker> ims;
456  {
457  boost::unique_lock<boost::mutex> ulock(marker_access_lock_);
458  moveit::core::RobotStateConstPtr s = handler->getState();
459 
460  for (std::size_t i = 0; i < active_generic_.size(); ++i)
461  {
462  visualization_msgs::msg::InteractiveMarker im;
463  if (active_generic_[i].construct_marker(*s, im))
464  {
465  im.name = getMarkerName(handler, active_generic_[i]);
466  shown_markers_[im.name] = i;
467  ims.push_back(im);
468  RCLCPP_DEBUG(LOGGER, "Publishing interactive marker %s (size = %lf)", im.name.c_str(), im.scale);
469  }
470  }
471 
472  for (std::size_t i = 0; i < active_eef_.size(); ++i)
473  {
474  geometry_msgs::msg::PoseStamped pose;
475  geometry_msgs::msg::Pose control_to_eef_tf;
476  pose.header.frame_id = robot_model_->getModelFrame();
477  pose.header.stamp = node_->now();
478  computeMarkerPose(handler, active_eef_[i], *s, pose.pose, control_to_eef_tf);
479 
480  std::string marker_name = getMarkerName(handler, active_eef_[i]);
481  shown_markers_[marker_name] = i;
482 
483  // Determine interactive maker size
484  double mscale = marker_scale < std::numeric_limits<double>::epsilon() ? active_eef_[i].size : marker_scale;
485 
486  visualization_msgs::msg::InteractiveMarker im = makeEmptyInteractiveMarker(marker_name, pose, mscale);
487  if (handler && handler->getControlsVisible())
488  {
489  if (active_eef_[i].interaction & InteractionStyle::POSITION_ARROWS)
490  addPositionControl(im, active_eef_[i].interaction & InteractionStyle::FIXED);
491  if (active_eef_[i].interaction & InteractionStyle::ORIENTATION_CIRCLES)
492  addOrientationControl(im, active_eef_[i].interaction & InteractionStyle::FIXED);
493  if (active_eef_[i].interaction & (InteractionStyle::POSITION_SPHERE | InteractionStyle::ORIENTATION_SPHERE))
494  {
495  std_msgs::msg::ColorRGBA color;
496  color.r = 0;
497  color.g = 1;
498  color.b = 1;
499  color.a = 0.5;
500  addViewPlaneControl(im, mscale * 0.25, color, active_eef_[i].interaction & InteractionStyle::POSITION_SPHERE,
501  active_eef_[i].interaction & InteractionStyle::ORIENTATION_SPHERE);
502  }
503  }
504  if (handler && handler->getMeshesVisible() &&
505  (active_eef_[i].interaction & (InteractionStyle::POSITION_EEF | InteractionStyle::ORIENTATION_EEF)))
506  addEndEffectorMarkers(handler, active_eef_[i], control_to_eef_tf, im,
507  active_eef_[i].interaction & InteractionStyle::POSITION_EEF,
508  active_eef_[i].interaction & InteractionStyle::ORIENTATION_EEF);
509  ims.push_back(im);
510  registerMoveInteractiveMarkerTopic(marker_name, handler->getName() + "_" + active_eef_[i].parent_link);
511  RCLCPP_DEBUG(LOGGER, "Publishing interactive marker %s (size = %lf)", marker_name.c_str(), mscale);
512  }
513  for (std::size_t i = 0; i < active_vj_.size(); ++i)
514  {
515  geometry_msgs::msg::PoseStamped pose;
516  pose.header.frame_id = robot_model_->getModelFrame();
517  pose.header.stamp = node_->now();
518  pose.pose = tf2::toMsg(s->getGlobalLinkTransform(active_vj_[i].connecting_link));
519  std::string marker_name = getMarkerName(handler, active_vj_[i]);
520  shown_markers_[marker_name] = i;
521 
522  // Determine interactive maker size
523  double mscale = marker_scale < std::numeric_limits<double>::epsilon() ? active_vj_[i].size : marker_scale;
524 
525  visualization_msgs::msg::InteractiveMarker im = makeEmptyInteractiveMarker(marker_name, pose, mscale);
526  if (handler && handler->getControlsVisible())
527  {
528  if (active_vj_[i].dof == 3) // planar joint
529  addPlanarXYControl(im, false);
530  else
531  add6DOFControl(im, false);
532  }
533  ims.push_back(im);
534  registerMoveInteractiveMarkerTopic(marker_name, handler->getName() + "_" + active_vj_[i].connecting_link);
535  RCLCPP_DEBUG(LOGGER, "Publishing interactive marker %s (size = %lf)", marker_name.c_str(), mscale);
536  }
537  handlers_[handler->getName()] = handler;
538  }
539 
540  // we do this while marker_access_lock_ is unlocked because the interactive marker server locks
541  // for most function calls, and maintains that lock while the feedback callback is running
542  // that can cause a deadlock if we were to run the loop below while marker_access_lock_ is locked
543  for (const visualization_msgs::msg::InteractiveMarker& im : ims)
544  {

592 void RobotInteraction::computeMarkerPose(const InteractionHandlerPtr& handler, const EndEffectorInteraction& eef,
593  const moveit::core::RobotState& robot_state, geometry_msgs::msg::Pose& pose,
594  geometry_msgs::msg::Pose& control_to_eef_tf) const
595 {
596  // Need to allow for control pose offsets
597  tf2::Transform tf_root_to_link, tf_root_to_control;
598  tf2::fromMsg(tf2::toMsg(robot_state.getGlobalLinkTransform(eef.parent_link)), tf_root_to_link);
599 
600  geometry_msgs::msg::Pose msg_link_to_control;
601  if (handler->getPoseOffset(eef, msg_link_to_control))
602  {
603  tf2::Transform tf_link_to_control;
604  tf2::fromMsg(msg_link_to_control, tf_link_to_control);
605 
606  tf_root_to_control = tf_root_to_link * tf_link_to_control;
607  tf2::toMsg(tf_link_to_control.inverse(), control_to_eef_tf);
608  }
609  else
610  {
611  tf_root_to_control = tf_root_to_link;
612  control_to_eef_tf.orientation.x = 0.0;
613  control_to_eef_tf.orientation.y = 0.0;
614  control_to_eef_tf.orientation.z = 0.0;
615  control_to_eef_tf.orientation.w = 1.0;
616  }
617 
618  tf2::toMsg(tf_root_to_control, pose);
619 }

621 void RobotInteraction::updateInteractiveMarkers(const InteractionHandlerPtr& handler)
622 {
623  std::string root_link;
624  std::map<std::string, geometry_msgs::msg::Pose> pose_updates;
625  {
626  boost::unique_lock<boost::mutex> ulock(marker_access_lock_);
627 
628  moveit::core::RobotStateConstPtr s = handler->getState();
629  root_link = s->getRobotModel()->getModelFrame();
630 
631  for (const EndEffectorInteraction& eef : active_eef_)
632  {
633  std::string marker_name = getMarkerName(handler, eef);
634  geometry_msgs::msg::Pose control_to_eef_tf;
635  computeMarkerPose(handler, eef, *s, pose_updates[marker_name], control_to_eef_tf);
636  }
637 
638  for (JointInteraction& vj : active_vj_)
639  {
640  std::string marker_name = getMarkerName(handler, vj);
641  pose_updates[marker_name] = tf2::toMsg(s->getGlobalLinkTransform(vj.connecting_link));
642  }
643 
644  for (GenericInteraction& gi : active_generic_)
645  {
646  std::string marker_name = getMarkerName(handler, gi);
647  geometry_msgs::msg::Pose pose;
648  if (gi.update_pose && gi.update_pose(*s, pose))
649  pose_updates[marker_name] = pose;
650  }
651  }
652 
653  std_msgs::msg::Header header;
654  header.frame_id = root_link; // marker poses are give w.r.t. root frame
655  for (std::map<std::string, geometry_msgs::msg::Pose>::const_iterator it = pose_updates.begin();
656  it != pose_updates.end(); ++it)
657  int_marker_server_->setPose(it->first, it->second, header);
658  int_marker_server_->applyChanges();
659 }

703 void RobotInteraction::processInteractiveMarkerFeedback(
704  const visualization_msgs::msg::InteractiveMarkerFeedback::ConstSharedPtr& feedback)
705 {
706  // perform some validity checks
707  boost::unique_lock<boost::mutex> ulock(marker_access_lock_);
708  std::map<std::string, std::size_t>::const_iterator it = shown_markers_.find(feedback->marker_name);
709  if (it == shown_markers_.end())
710  {
711  RCLCPP_ERROR(LOGGER, "Unknown marker name: '%s' (not published by RobotInteraction class)",
712  feedback->marker_name.c_str());
713  return;
714  }
715 
716  std::size_t u = feedback->marker_name.find_first_of('_');
717  if (u == std::string::npos || u < 4)
718  {
719  RCLCPP_ERROR(LOGGER, "Invalid marker name: '%s'", feedback->marker_name.c_str());
720  return;
721  }
722 
723  feedback_map_[feedback->marker_name] = feedback;
724  new_feedback_condition_.notify_all();

727 void RobotInteraction::processingThread()
728 {
729  boost::unique_lock<boost::mutex> ulock(marker_access_lock_);
730 
731  while (run_processing_thread_ && rclcpp::ok())
732  {
733  while (feedback_map_.empty() && run_processing_thread_ && rclcpp::ok())
734  new_feedback_condition_.wait(ulock);
735 
736  while (!feedback_map_.empty() && rclcpp::ok())
737  {
738  auto feedback = feedback_map_.begin()->second;
739  feedback_map_.erase(feedback_map_.begin());
740  RCLCPP_DEBUG(LOGGER, "Processing feedback from map for marker [%s]", feedback->marker_name.c_str());
741 
742  std::map<std::string, std::size_t>::const_iterator it = shown_markers_.find(feedback->marker_name);
743  if (it == shown_markers_.end())
744  {
745  RCLCPP_ERROR(LOGGER,
746  "Unknown marker name: '%s' (not published by RobotInteraction class) "
747  "(should never have ended up in the feedback_map!)",
748  feedback->marker_name.c_str());
749  continue;
750  }
751  std::size_t u = feedback->marker_name.find_first_of('_');
752  if (u == std::string::npos || u < 4)
753  {
754  RCLCPP_ERROR(LOGGER, "Invalid marker name: '%s' (should never have ended up in the feedback_map!)",
755  feedback->marker_name.c_str());
756  continue;
757  }
758  std::string marker_class = feedback->marker_name.substr(0, 2);
759  std::string handler_name = feedback->marker_name.substr(3, u - 3); // skip the ":"
760  std::map<std::string, InteractionHandlerPtr>::const_iterator jt = handlers_.find(handler_name);
761  if (jt == handlers_.end())
762  {
763  RCLCPP_ERROR(LOGGER, "Interactive Marker Handler '%s' is not known.", handler_name.c_str());
764  continue;
765  }
766 
767  // we put this in a try-catch because user specified callbacks may be triggered
768  try
769  {
770  if (marker_class == "EE")
771  {
772  // make a copy of the data, so we do not lose it while we are unlocked
773  EndEffectorInteraction eef = active_eef_[it->second];
774  InteractionHandlerPtr ih = jt->second;
775  marker_access_lock_.unlock();
776  try
777  {
778  ih->handleEndEffector(eef, feedback);
779  }
780  catch (std::exception& ex)
781  {
782  RCLCPP_ERROR(LOGGER, "Exception caught while handling end-effector update: %s", ex.what());
783  }
784  marker_access_lock_.lock();
785  }
786  else if (marker_class == "JJ")
787  {
788  // make a copy of the data, so we do not lose it while we are unlocked
789  JointInteraction vj = active_vj_[it->second];
790  InteractionHandlerPtr ih = jt->second;
791  marker_access_lock_.unlock();
792  try
793  {
794  ih->handleJoint(vj, feedback);
795  }
796  catch (std::exception& ex)
797  {
798  RCLCPP_ERROR(LOGGER, "Exception caught while handling joint update: %s", ex.what());
799  }
800  marker_access_lock_.lock();
801  }
802  else if (marker_class == "GG")
803  {
804  InteractionHandlerPtr ih = jt->second;
805  GenericInteraction g = active_generic_[it->second];
806  marker_access_lock_.unlock();
807  try
808  {
809  ih->handleGeneric(g, feedback);
810  }
811  catch (std::exception& ex)
812  {
813  RCLCPP_ERROR(LOGGER, "Exception caught while handling joint update: %s", ex.what());
814  }
815  marker_access_lock_.lock();
816  }
817  else
818  RCLCPP_ERROR(LOGGER, "Unknown marker class ('%s') for marker '%s'", marker_class.c_str(),
819  feedback->marker_name.c_str());
820  }
821  catch (std::exception& ex)
822  {
823  RCLCPP_ERROR(LOGGER, "Exception caught while processing event: %s", ex.what());
824  }
825  }
826  }
827 }
