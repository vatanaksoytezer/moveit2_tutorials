138 void computeDescendantsHelper(const JointModel* joint, std::vector<const JointModel*>& parents,
139  std::set<const JointModel*>& seen, DescMap& descendants)
140 {
141  if (!joint)
142  return;
143  if (seen.find(joint) != seen.end())
144  return;
145  seen.insert(joint);
146 
147  for (const JointModel* parent : parents)
148  descendants[parent].second.insert(joint);
149 
150  const LinkModel* lm = joint->getChildLinkModel();
151  if (!lm)
152  return;
153 
154  for (const JointModel* parent : parents)
155  descendants[parent].first.insert(lm);
156  descendants[joint].first.insert(lm);
157 
158  parents.push_back(joint);
159  const std::vector<const JointModel*>& ch = lm->getChildJointModels();
160  for (const JointModel* child_joint_model : ch)
161  computeDescendantsHelper(child_joint_model, parents, seen, descendants);
162  const std::vector<const JointModel*>& mim = joint->getMimicRequests();
163  for (const JointModel* mimic_joint_model : mim)
164  computeDescendantsHelper(mimic_joint_model, parents, seen, descendants);
165  parents.pop_back();
166 }

168 void computeCommonRootsHelper(const JointModel* joint, std::vector<int>& common_roots, int size)
169 {
170  if (!joint)
171  return;
172  const LinkModel* lm = joint->getChildLinkModel();
173  if (!lm)
174  return;
175 
176  const std::vector<const JointModel*>& ch = lm->getChildJointModels();
177  for (std::size_t i = 0; i < ch.size(); ++i)
178  {
179  const std::vector<const JointModel*>& a = ch[i]->getDescendantJointModels();
180  for (std::size_t j = i + 1; j < ch.size(); ++j)
181  {
182  const std::vector<const JointModel*>& b = ch[j]->getDescendantJointModels();
183  for (const JointModel* m : b)
184  common_roots[ch[i]->getJointIndex() * size + m->getJointIndex()] =
185  common_roots[ch[i]->getJointIndex() + m->getJointIndex() * size] = joint->getJointIndex();
186  for (const JointModel* k : a)
187  {
188  common_roots[k->getJointIndex() * size + ch[j]->getJointIndex()] =
189  common_roots[k->getJointIndex() + ch[j]->getJointIndex() * size] = joint->getJointIndex();
190  for (const JointModel* m : b)
191  common_roots[k->getJointIndex() * size + m->getJointIndex()] =
192  common_roots[k->getJointIndex() + m->getJointIndex() * size] = joint->getJointIndex();
193  }
194  }
195  computeCommonRootsHelper(ch[i], common_roots, size);
196  }

200 void RobotModel::computeCommonRoots()
201 {
202  // compute common roots for all pairs of joints;
203  // there are 3 cases of pairs (X, Y):
204  // X != Y && X and Y are not descendants of one another
205  // X == Y
206  // X != Y && X and Y are descendants of one another
207 
208  // by default, the common root is always the global root;
209  common_joint_roots_.resize(joint_model_vector_.size() * joint_model_vector_.size(), 0);
210 
211  // look at all descendants recursively; for two sibling nodes A, B, both children of X, all the pairs of respective
212  // descendants of A and B
213  // have X as the common root.
214  computeCommonRootsHelper(root_joint_, common_joint_roots_, joint_model_vector_.size());
215 
216  for (const JointModel* joint_model : joint_model_vector_)
217  {
218  // the common root of a joint and itself is the same joint:
219  common_joint_roots_[joint_model->getJointIndex() * (1 + joint_model_vector_.size())] = joint_model->getJointIndex();
220 
221  // a node N and one of its descendants have as common root the node N itself:
222  const std::vector<const JointModel*>& d = joint_model->getDescendantJointModels();
223  for (const JointModel* descendant_joint_model : d)
224  common_joint_roots_[descendant_joint_model->getJointIndex() * joint_model_vector_.size() +
225  joint_model->getJointIndex()] =
226  common_joint_roots_[descendant_joint_model->getJointIndex() +
227  joint_model->getJointIndex() * joint_model_vector_.size()] = joint_model->getJointIndex();
228  }
229 }

231 void RobotModel::computeDescendants()
232 {
233  // compute the list of descendants for all joints
234  std::vector<const JointModel*> parents;
235  std::set<const JointModel*> seen;
236 
237  DescMap descendants;
238  computeDescendantsHelper(root_joint_, parents, seen, descendants);
239  for (std::pair<const JointModel* const, std::pair<std::set<const LinkModel*, OrderLinksByIndex>,
240  std::set<const JointModel*, OrderJointsByIndex>>>& descendant :
241  descendants)
242  {
243  JointModel* jm = const_cast<JointModel*>(descendant.first);
244  for (const JointModel* jt : descendant.second.second)
245  jm->addDescendantJointModel(jt);
246  for (const LinkModel* jt : descendant.second.first)
247  jm->addDescendantLinkModel(jt);
248  }

251 void RobotModel::buildJointInfo()
252 {
253  moveit::tools::Profiler::ScopedStart prof_start;
254  moveit::tools::Profiler::ScopedBlock prof_block("RobotModel::buildJointInfo");
255 
256  // construct additional maps for easy access by name
257  variable_count_ = 0;
258  active_joint_model_start_index_.reserve(joint_model_vector_.size());
259  variable_names_.reserve(joint_model_vector_.size());
260  joints_of_variable_.reserve(joint_model_vector_.size());
261 
262  for (std::size_t i = 0; i < joint_model_vector_.size(); ++i)
263  {
264  joint_model_vector_[i]->setJointIndex(i);
265  const std::vector<std::string>& name_order = joint_model_vector_[i]->getVariableNames();
266 
267  // compute index map
268  if (!name_order.empty())
269  {
270  for (std::size_t j = 0; j < name_order.size(); ++j)
271  {
272  joint_variables_index_map_[name_order[j]] = variable_count_ + j;
273  variable_names_.push_back(name_order[j]);
274  joints_of_variable_.push_back(joint_model_vector_[i]);
275  }
276  if (joint_model_vector_[i]->getMimic() == nullptr)
277  {
278  active_joint_model_start_index_.push_back(variable_count_);
279  active_joint_model_vector_.push_back(joint_model_vector_[i]);
280  active_joint_model_names_vector_.push_back(joint_model_vector_[i]->getName());
281  active_joint_model_vector_const_.push_back(joint_model_vector_[i]);
282  active_joint_models_bounds_.push_back(&joint_model_vector_[i]->getVariableBounds());
283  }
284 
285  if (joint_model_vector_[i]->getType() == JointModel::REVOLUTE &&
286  static_cast<const RevoluteJointModel*>(joint_model_vector_[i])->isContinuous())
287  continuous_joint_model_vector_.push_back(joint_model_vector_[i]);
288 
289  joint_model_vector_[i]->setFirstVariableIndex(variable_count_);
290  joint_variables_index_map_[joint_model_vector_[i]->getName()] = variable_count_;
291 
292  // compute variable count
293  std::size_t vc = joint_model_vector_[i]->getVariableCount();
294  variable_count_ += vc;
295  if (vc == 1)
296  single_dof_joints_.push_back(joint_model_vector_[i]);
297  else
298  multi_dof_joints_.push_back(joint_model_vector_[i]);
299  }
300  }
301 
302  std::vector<bool> link_considered(link_model_vector_.size(), false);
303  for (const LinkModel* link : link_model_vector_)
304  {
305  if (link_considered[link->getLinkIndex()])
306  continue;
307 
308  LinkTransformMap associated_transforms;
309  computeFixedTransforms(link, link->getJointOriginTransform().inverse(), associated_transforms);
310  for (auto& tf_base : associated_transforms)
311  {
312  link_considered[tf_base.first->getLinkIndex()] = true;
313  for (auto& tf_target : associated_transforms)
314  {
315  if (&tf_base != &tf_target)
316  const_cast<LinkModel*>(tf_base.first) // regain write access to base LinkModel*
317  ->addAssociatedFixedTransform(tf_target.first, tf_base.second.inverse() * tf_target.second);
318  }
319  }
320  }
321 
322  computeDescendants();
323  computeCommonRoots(); // must be called _after_ list of descendants was computed
324 }

326 void RobotModel::buildGroupStates(const srdf::Model& srdf_model)
327 {
328  // copy the default states to the groups
329  const std::vector<srdf::Model::GroupState>& ds = srdf_model.getGroupStates();
330  for (const srdf::Model::GroupState& group_state : ds)
331  {
332  if (hasJointModelGroup(group_state.group_))
333  {
334  JointModelGroup* jmg = getJointModelGroup(group_state.group_);
335  std::vector<const JointModel*> remaining_joints = jmg->getActiveJointModels();
336  std::map<std::string, double> state;
337  for (std::map<std::string, std::vector<double>>::const_iterator jt = group_state.joint_values_.begin();
338  jt != group_state.joint_values_.end(); ++jt)
339  {
340  if (jmg->hasJointModel(jt->first))
341  {
342  const JointModel* jm = jmg->getJointModel(jt->first);
343  const std::vector<std::string>& vn = jm->getVariableNames();
344  // Remove current joint name from remaining list.
345  auto it_found = std::find(remaining_joints.begin(), remaining_joints.end(), jm);
346  if (it_found != remaining_joints.end())
347  remaining_joints.erase(it_found);
348  if (vn.size() == jt->second.size())
349  for (std::size_t j = 0; j < vn.size(); ++j)
350  state[vn[j]] = jt->second[j];
351  else
352  RCLCPP_ERROR(LOGGER,
353  "The model for joint '%s' requires %d variable values, "
354  "but only %d variable values were supplied in default state '%s' for group '%s'",
355  jt->first.c_str(), (int)vn.size(), (int)jt->second.size(), group_state.name_.c_str(),
356  jmg->getName().c_str());
357  }
358  else
359  RCLCPP_ERROR(LOGGER,
360  "Group state '%s' specifies value for joint '%s', "
361  "but that joint is not part of group '%s'",
362  group_state.name_.c_str(), jt->first.c_str(), jmg->getName().c_str());
363  }
364  if (!remaining_joints.empty())
365  {
366  std::stringstream missing;
367  missing << (*remaining_joints.begin())->getName();
368  for (auto j = ++remaining_joints.begin(); j != remaining_joints.end(); j++)
369  {
370  missing << ", " << (*j)->getName();
371  }
372  RCLCPP_WARN_STREAM(LOGGER, "Group state '" << group_state.name_
373  << "' doesn't specify all group joints in group '"
374  << group_state.group_ << "'. " << missing.str() << " "
375  << (remaining_joints.size() > 1 ? "are" : "is") << " missing.");
376  }
377  if (!state.empty())
378  jmg->addDefaultState(group_state.name_, state);
379  }
380  else
381  RCLCPP_ERROR(LOGGER, "Group state '%s' specified for group '%s', but that group does not exist",
382  group_state.name_.c_str(), group_state.group_.c_str());
383  }
384 }

386 void RobotModel::buildMimic(const urdf::ModelInterface& urdf_model)
387 {
388  // compute mimic joints
389  for (JointModel* joint_model : joint_model_vector_)
390  {
391  const urdf::Joint* jm = urdf_model.getJoint(joint_model->getName()).get();
392  if (jm)
393  if (jm->mimic)
394  {
395  JointModelMap::const_iterator jit = joint_model_map_.find(jm->mimic->joint_name);
396  if (jit != joint_model_map_.end())
397  {
398  if (joint_model->getVariableCount() == jit->second->getVariableCount())
399  joint_model->setMimic(jit->second, jm->mimic->multiplier, jm->mimic->offset);
400  else
401  RCLCPP_ERROR(LOGGER, "Join '%s' cannot mimic joint '%s' because they have different number of DOF",
402  joint_model->getName().c_str(), jm->mimic->joint_name.c_str());
403  }
404  else
405  RCLCPP_ERROR(LOGGER, "Joint '%s' cannot mimic unknown joint '%s'", joint_model->getName().c_str(),
406  jm->mimic->joint_name.c_str());
407  }
408  }
409 
410  // in case we have a joint that mimics a joint that already mimics another joint, we can simplify things:
411  bool change = true;
412  while (change)
413  {
414  change = false;
415  for (JointModel* joint_model : joint_model_vector_)
416  if (joint_model->getMimic())
417  {
418  if (joint_model->getMimic()->getMimic())
419  {
420  joint_model->setMimic(joint_model->getMimic()->getMimic(),
421  joint_model->getMimicFactor() * joint_model->getMimic()->getMimicFactor(),
422  joint_model->getMimicOffset() +
423  joint_model->getMimicFactor() * joint_model->getMimic()->getMimicOffset());
424  change = true;
425  }
426  if (joint_model == joint_model->getMimic())
427  {
428  RCLCPP_ERROR(LOGGER, "Cycle found in joint that mimic each other. Ignoring all mimic joints.");
429  for (JointModel* joint_model_recal : joint_model_vector_)
430  joint_model_recal->setMimic(nullptr, 0.0, 0.0);
431  change = false;
432  break;
433  }
434  }
435  }
436  // build mimic requests
437  for (JointModel* joint_model : joint_model_vector_)
438  if (joint_model->getMimic())
439  {
440  const_cast<JointModel*>(joint_model->getMimic())->addMimicRequest(joint_model);
441  mimic_joints_.push_back(joint_model);
442  }
443 }

505 void RobotModel::buildGroups(const srdf::Model& srdf_model)
506 {
507  const std::vector<srdf::Model::Group>& group_configs = srdf_model.getGroups();
508 
509  // the only thing tricky is dealing with subgroups
510  std::vector<bool> processed(group_configs.size(), false);
511 
512  bool added = true;
513  while (added)
514  {
515  added = false;
516 
517  // going to make passes until we can't do anything else
518  for (std::size_t i = 0; i < group_configs.size(); ++i)
519  if (!processed[i])
520  {
521  // if we haven't processed, check and see if the dependencies are met yet
522  bool all_subgroups_added = true;
523  for (const std::string& subgroup : group_configs[i].subgroups_)
524  if (joint_model_group_map_.find(subgroup) == joint_model_group_map_.end())
525  {
526  all_subgroups_added = false;
527  break;
528  }
529  if (all_subgroups_added)
530  {
531  added = true;
532  processed[i] = true;
533  if (!addJointModelGroup(group_configs[i]))
534  {
535  RCLCPP_WARN(LOGGER, "Failed to add group '%s'", group_configs[i].name_.c_str());
536  }
537  }
538  }
539  }
540 
541  for (std::size_t i = 0; i < processed.size(); ++i)
542  if (!processed[i])
543  {
544  RCLCPP_WARN(LOGGER, "Could not process group '%s' due to unmet subgroup dependencies",
545  group_configs[i].name_.c_str());
546  }
547 
548  for (JointModelGroupMap::const_iterator it = joint_model_group_map_.begin(); it != joint_model_group_map_.end(); ++it)
549  joint_model_groups_.push_back(it->second);
550  std::sort(joint_model_groups_.begin(), joint_model_groups_.end(), OrderGroupsByName());
551  for (JointModelGroup* joint_model_group : joint_model_groups_)
552  {
553  joint_model_groups_const_.push_back(joint_model_group);

555  }

561 void RobotModel::buildGroupsInfoSubgroups(const srdf::Model& srdf_model)
562 {
563  // compute subgroups
564  for (JointModelGroupMap::const_iterator it = joint_model_group_map_.begin(); it != joint_model_group_map_.end(); ++it)
565  {
566  JointModelGroup* jmg = it->second;
567  std::vector<std::string> subgroup_names;
568  std::set<const JointModel*> joints(jmg->getJointModels().begin(), jmg->getJointModels().end());
569  for (JointModelGroupMap::const_iterator jt = joint_model_group_map_.begin(); jt != joint_model_group_map_.end();
570  ++jt)
571  if (jt->first != it->first)
572  {
573  bool ok = true;
574  JointModelGroup* sub_jmg = jt->second;
575  const std::vector<const JointModel*>& sub_joints = sub_jmg->getJointModels();
576  for (const JointModel* sub_joint : sub_joints)
577  if (joints.find(sub_joint) == joints.end())
578  {
579  ok = false;
580  break;

590 void RobotModel::buildGroupsInfoEndEffectors(const srdf::Model& srdf_model)
591 {
592  // set the end-effector flags
593  const std::vector<srdf::Model::EndEffector>& eefs = srdf_model.getEndEffectors();
594  for (JointModelGroupMap::const_iterator it = joint_model_group_map_.begin(); it != joint_model_group_map_.end(); ++it)
595  {
596  // check if this group is a known end effector
597  for (const srdf::Model::EndEffector& eef : eefs)
598  if (eef.component_group_ == it->first)
599  {
600  // if it is, mark it as such
601  it->second->setEndEffectorName(eef.name_);
602  end_effectors_map_[eef.name_] = it->second;
603  end_effectors_.push_back(it->second);
604 
605  // check to see if there are groups that contain the parent link of this end effector.
606  // record this information if found;
607  std::vector<JointModelGroup*> possible_parent_groups;
608  for (JointModelGroupMap::const_iterator jt = joint_model_group_map_.begin(); jt != joint_model_group_map_.end();
609  ++jt)
610  if (jt->first != it->first)
611  {
612  if (jt->second->hasLinkModel(eef.parent_link_))
613  {
614  jt->second->attachEndEffector(eef.name_);
615  possible_parent_groups.push_back(jt->second);
616  }
617  }
618 
619  JointModelGroup* eef_parent_group = nullptr;
620  // if a parent group is specified in SRDF, try to use it
621  if (!eef.parent_group_.empty())
622  {
623  JointModelGroupMap::const_iterator jt = joint_model_group_map_.find(eef.parent_group_);
624  if (jt != joint_model_group_map_.end())
625  {
626  if (jt->second->hasLinkModel(eef.parent_link_))
627  {
628  if (jt->second != it->second)
629  eef_parent_group = jt->second;
630  else
631  RCLCPP_ERROR(LOGGER, "Group '%s' for end-effector '%s' cannot be its own parent",
632  eef.parent_group_.c_str(), eef.name_.c_str());
633  }
634  else
635  RCLCPP_ERROR(LOGGER,
636  "Group '%s' was specified as parent group for end-effector '%s' "
637  "but it does not include the parent link '%s'",
638  eef.parent_group_.c_str(), eef.name_.c_str(), eef.parent_link_.c_str());
639  }
640  else
641  RCLCPP_ERROR(LOGGER, "Group name '%s' not found (specified as parent group for end-effector '%s')",
642  eef.parent_group_.c_str(), eef.name_.c_str());
643  }
644 
645  // if no parent group was specified, use a default one
646  if (eef_parent_group == nullptr)
647  if (!possible_parent_groups.empty())
648  {
649  // if there are multiple options for the group that contains this end-effector,
650  // we pick the group with fewest joints.
651  std::size_t best = 0;
652  for (std::size_t g = 1; g < possible_parent_groups.size(); ++g)
653  if (possible_parent_groups[g]->getJointModels().size() <
654  possible_parent_groups[best]->getJointModels().size())
655  best = g;
656  eef_parent_group = possible_parent_groups[best];
657  }
658 
659  if (eef_parent_group)
660  {
661  it->second->setEndEffectorParent(eef_parent_group->getName(), eef.parent_link_);
662  }
663  else
664  {
665  RCLCPP_WARN(LOGGER, "Could not identify parent group for end-effector '%s'", eef.name_.c_str());
666  it->second->setEndEffectorParent("", eef.parent_link_);
667  }
668  }
669  }
670  std::sort(end_effectors_.begin(), end_effectors_.end(), OrderGroupsByName());

673 bool RobotModel::addJointModelGroup(const srdf::Model::Group& gc)
674 {
675  if (joint_model_group_map_.find(gc.name_) != joint_model_group_map_.end())
676  {
677  RCLCPP_WARN(LOGGER, "A group named '%s' already exists. Not adding.", gc.name_.c_str());
678  return false;
679  }
680 
681  std::set<const JointModel*> jset;
682 
683  // add joints from chains
684  for (const std::pair<std::string, std::string>& chain : gc.chains_)
685  {
686  const LinkModel* base_link = getLinkModel(chain.first);
687  const LinkModel* tip_link = getLinkModel(chain.second);
688  if (base_link && tip_link)
689  {
690  // go from tip, up the chain, until we hit the root or we find the base_link
691  const LinkModel* lm = tip_link;
692  std::vector<const JointModel*> cj;
693  while (lm)
694  {
695  if (lm == base_link)
696  break;
697  cj.push_back(lm->getParentJointModel());
698  lm = lm->getParentJointModel()->getParentLinkModel();
699  }
700  // if we did not find the base_link, we could have a chain like e.g.,
701  // from one end-effector to another end-effector, so the root is in between
702  if (lm != base_link)
703  {
704  // we go up the chain from the base this time, and see where we intersect the other chain
705  lm = base_link;
706  std::size_t index = 0;
707  std::vector<const JointModel*> cj2;
708  while (lm)
709  {
710  for (std::size_t j = 0; j < cj.size(); ++j)
711  if (cj[j] == lm->getParentJointModel())
712  {
713  index = j + 1;
714  break;
715  }
716  if (index > 0)
717  break;
718  cj2.push_back(lm->getParentJointModel());
719  lm = lm->getParentJointModel()->getParentLinkModel();
720  }
721  if (index > 0)
722  {
723  jset.insert(cj.begin(), cj.begin() + index);
724  jset.insert(cj2.begin(), cj2.end());
725  }
726  }
727  else
728  // if we have a simple chain, just add the joints
729  jset.insert(cj.begin(), cj.end());
730  }
731  }
732 
733  // add joints
734  for (const std::string& joint : gc.joints_)
735  {
736  const JointModel* j = getJointModel(joint);
737  if (j)
738  jset.insert(j);
739  }
740 
741  // add joints that are parents of included links
742  for (const std::string& link : gc.links_)
743  {
744  const LinkModel* l = getLinkModel(link);
745  if (l)
746  jset.insert(l->getParentJointModel());
747  }
748 
749  // add joints from subgroups
750  for (const std::string& subgroup : gc.subgroups_)
751  {
752  const JointModelGroup* sg = getJointModelGroup(subgroup);
753  if (sg)
754  {
755  // active joints
756  const std::vector<const JointModel*>& js = sg->getJointModels();
757  for (const JointModel* j : js)
758  jset.insert(j);
759 
760  // fixed joints
761  const std::vector<const JointModel*>& fs = sg->getFixedJointModels();
762  for (const JointModel* f : fs)
763  jset.insert(f);
764 
765  // mimic joints
766  const std::vector<const JointModel*>& ms = sg->getMimicJointModels();
767  for (const JointModel* m : ms)
768  jset.insert(m);
769  }
770  }
771 
772  if (jset.empty())
773  {
774  RCLCPP_WARN(LOGGER, "Group '%s' must have at least one valid joint", gc.name_.c_str());
775  return false;
776  }
777 
778  std::vector<const JointModel*> joints;
779  joints.reserve(jset.size());
780  for (const JointModel* it : jset)
781  joints.push_back(it);
782 
783  JointModelGroup* jmg = new JointModelGroup(gc.name_, gc, joints, this);
784  joint_model_group_map_[gc.name_] = jmg;
785 
786  return true;
787 }

874 JointModel* RobotModel::constructJointModel(const urdf::Joint* urdf_joint, const urdf::Link* child_link,
875  const srdf::Model& srdf_model)
876 {
877  JointModel* new_joint_model = nullptr;
878 
879  // if urdf_joint exists, must be the root link transform
880  if (urdf_joint)
881  {
882  switch (urdf_joint->type)
883  {
884  case urdf::Joint::REVOLUTE:
885  {
886  RevoluteJointModel* j = new RevoluteJointModel(urdf_joint->name);
887  j->setVariableBounds(j->getName(), jointBoundsFromURDF(urdf_joint));
888  j->setContinuous(false);
889  j->setAxis(Eigen::Vector3d(urdf_joint->axis.x, urdf_joint->axis.y, urdf_joint->axis.z));
890  new_joint_model = j;
891  }
892  break;
893  case urdf::Joint::CONTINUOUS:
894  {
895  RevoluteJointModel* j = new RevoluteJointModel(urdf_joint->name);
896  j->setVariableBounds(j->getName(), jointBoundsFromURDF(urdf_joint));
897  j->setContinuous(true);
898  j->setAxis(Eigen::Vector3d(urdf_joint->axis.x, urdf_joint->axis.y, urdf_joint->axis.z));
899  new_joint_model = j;
900  }
901  break;
902  case urdf::Joint::PRISMATIC:
903  {
904  PrismaticJointModel* j = new PrismaticJointModel(urdf_joint->name);
905  j->setVariableBounds(j->getName(), jointBoundsFromURDF(urdf_joint));
906  j->setAxis(Eigen::Vector3d(urdf_joint->axis.x, urdf_joint->axis.y, urdf_joint->axis.z));
907  new_joint_model = j;
908  }
909  break;
910  case urdf::Joint::FLOATING:
911  new_joint_model = new FloatingJointModel(urdf_joint->name);
912  break;
913  case urdf::Joint::PLANAR:
914  new_joint_model = new PlanarJointModel(urdf_joint->name);
915  break;
916  case urdf::Joint::FIXED:
917  new_joint_model = new FixedJointModel(urdf_joint->name);
918  break;
919  default:
920  RCLCPP_ERROR(LOGGER, "Unknown joint type: %d", (int)urdf_joint->type);
921  break;
922  }
923  }
924  else // if urdf_joint passed in as null, then we're at root of URDF model
925  {
926  const std::vector<srdf::Model::VirtualJoint>& virtual_joints = srdf_model.getVirtualJoints();
927  for (const srdf::Model::VirtualJoint& virtual_joint : virtual_joints)
928  {
929  if (virtual_joint.child_link_ != child_link->name)
930  {
931  RCLCPP_WARN(LOGGER,
932  "Skipping virtual joint '%s' because its child frame '%s' "
933  "does not match the URDF frame '%s'",
934  virtual_joint.name_.c_str(), virtual_joint.child_link_.c_str(), child_link->name.c_str());
935  }
936  else if (virtual_joint.parent_frame_.empty())
937  {
938  RCLCPP_WARN(LOGGER, "Skipping virtual joint '%s' because its parent frame is empty",
939  virtual_joint.name_.c_str());
940  }
941  else
942  {
943  if (virtual_joint.type_ == "fixed")
944  new_joint_model = new FixedJointModel(virtual_joint.name_);
945  else if (virtual_joint.type_ == "planar")
946  new_joint_model = new PlanarJointModel(virtual_joint.name_);
947  else if (virtual_joint.type_ == "floating")
948  new_joint_model = new FloatingJointModel(virtual_joint.name_);
949  if (new_joint_model)
950  {
951  // for fixed frames we still use the robot root link
952  if (virtual_joint.type_ != "fixed")
953  {
954  model_frame_ = virtual_joint.parent_frame_;
955  }
956  break;
957  }
958  }
959  }
960  if (!new_joint_model)
961  {
962  RCLCPP_INFO(LOGGER, "No root/virtual joint specified in SRDF. Assuming fixed joint");
963  new_joint_model = new FixedJointModel("ASSUMED_FIXED_ROOT_JOINT");
964  }
965  }
966 
967  if (new_joint_model)
968  {
969  new_joint_model->setDistanceFactor(new_joint_model->getStateSpaceDimension());
970  const std::vector<srdf::Model::PassiveJoint>& pjoints = srdf_model.getPassiveJoints();
971  for (const srdf::Model::PassiveJoint& pjoint : pjoints)
972  {
973  if (new_joint_model->getName() == pjoint.name_)
974  {
975  new_joint_model->setPassive(true);
976  break;
977  }
978  }
979 
980  for (const srdf::Model::JointProperty& property : srdf_model.getJointProperties(new_joint_model->getName()))
981  {
982  if (property.property_name_ == "angular_distance_weight")
983  {
984  double angular_distance_weight;
985  try
986  {
987  std::string::size_type sz;
988  angular_distance_weight = std::stod(property.value_, &sz);
989  if (sz != property.value_.size())
990  {
991  RCLCPP_WARN_STREAM(LOGGER, "Extra characters after property " << property.property_name_ << " for joint "
992  << property.joint_name_ << " as double: '"
993  << property.value_.substr(sz) << "'");
994  }
995  }
996  catch (const std::invalid_argument& e)
997  {
998  RCLCPP_ERROR_STREAM(LOGGER, "Unable to parse property " << property.property_name_ << " for joint "
999  << property.joint_name_ << " as double: '"
1000  << property.value_ << "'");
1001  continue;
1002  }
1003 
1004  if (new_joint_model->getType() == JointModel::JointType::PLANAR)
1005  {
1006  ((PlanarJointModel*)new_joint_model)->setAngularDistanceWeight(angular_distance_weight);
1007  }
1008  else if (new_joint_model->getType() == JointModel::JointType::FLOATING)
1009  {
1010  ((FloatingJointModel*)new_joint_model)->setAngularDistanceWeight(angular_distance_weight);
1011  }
1012  else
1013  {
1014  RCLCPP_ERROR_STREAM(LOGGER, "Cannot apply property " << property.property_name_
1015  << " to joint type: " << new_joint_model->getTypeName());
1016  }
1017  }
1018  else if (property.property_name_ == "motion_model")
1019  {
1020  if (new_joint_model->getType() != JointModel::JointType::PLANAR)
1021  {
1022  RCLCPP_ERROR(LOGGER, "Cannot apply property %s to joint type: %s", property.property_name_.c_str(),
1023  new_joint_model->getTypeName().c_str());
1024  continue;
1025  }
1026 
1027  PlanarJointModel::MotionModel motion_model;
1028  if (property.value_ == "holonomic")
1029  {
1030  motion_model = PlanarJointModel::MotionModel::HOLONOMIC;
1031  }
1032  else if (property.value_ == "diff_drive")
1033  {
1034  motion_model = PlanarJointModel::MotionModel::DIFF_DRIVE;
1035  }
1036  else
1037  {
1038  RCLCPP_ERROR_STREAM(LOGGER, "Unknown value for property " << property.property_name_ << " ("
1039  << property.joint_name_ << "): '" << property.value_
1040  << "'");
1041  RCLCPP_ERROR(LOGGER, "Valid values are 'holonomic' and 'diff_drive'");
1042  continue;
1043  }
1044 
1045  ((PlanarJointModel*)new_joint_model)->setMotionModel(motion_model);
1046  }
1047  else if (property.property_name_ == "min_translational_distance")
1048  {
1049  if (new_joint_model->getType() != JointModel::JointType::PLANAR)
1050  {
1051  RCLCPP_ERROR(LOGGER, "Cannot apply property %s to joint type: %s", property.property_name_.c_str(),
1052  new_joint_model->getTypeName().c_str());
1053  continue;
1054  }
1055  double min_translational_distance;
1056  try
1057  {
1058  std::string::size_type sz;
1059  min_translational_distance = std::stod(property.value_, &sz);
1060  if (sz != property.value_.size())
1061  {
1062  RCLCPP_WARN_STREAM(LOGGER, "Extra characters after property " << property.property_name_ << " for joint "
1063  << property.joint_name_ << " as double: '"
1064  << property.value_.substr(sz) << "'");
1065  }
1066  }
1067  catch (const std::invalid_argument& e)
1068  {
1069  RCLCPP_ERROR_STREAM(LOGGER, "Unable to parse property " << property.property_name_ << " for joint "
1070  << property.joint_name_ << " as double: '"
1071  << property.value_ << "'");
1072  continue;
1073  }
1074 
1075  ((PlanarJointModel*)new_joint_model)->setMinTranslationalDistance(min_translational_distance);
1076  }
1077  else
1078  {
1079  RCLCPP_ERROR(LOGGER, "Unknown joint property: %s", property.property_name_.c_str());
1080  }
1081  }
1082  }
1083 
1084  return new_joint_model;
1085 }

1433  std::pair<SolverAllocatorFn, SolverAllocatorMapFn> solver_allocator_pair;

1438  std::set<const JointModel*> joints;

1440

1442

1447  if (!sub) // this should actually not happen, all groups should be well defined

1453  sub_joints.insert(sub->getJointModels().begin(), sub->getJointModels().end());
1454

1456  { // sub_joints included in joints: add sub, remove sub_joints from joints set
1457  std::set<const JointModel*> joint_model_set;
1458  std::set_difference(joints.begin(), joints.end(), sub_joints.begin(), sub_joints.end(),
1459  std::inserter(joint_model_set, joint_model_set.end()));

1467  }

1471  if (!subs.empty())
1472  {

1474  for (const JointModelGroup* sub : subs)
1475  {
1476  ss << sub->getName() << " ";
1477  solver_allocator_pair.second[sub] = allocators.find(sub->getName())->second;
