60 RobotModelLoader::~RobotModelLoader()
61 {
62  // Make sure we destroy the robot model first. It contains the loaded
63  // kinematics plugins, and those must be destroyed before the pluginlib class
64  // that implements them is destroyed (that happens when kinematics_loader_ is

74 bool canSpecifyPosition(const moveit::core::JointModel* jmodel, const unsigned int index)
75 {
76  bool ok = false;
77  if (jmodel->getType() == moveit::core::JointModel::PLANAR && index == 2)
78  {
79  RCLCPP_ERROR(LOGGER, "Cannot specify position limits for orientation of planar joint '%s'",
80  jmodel->getName().c_str());
81  }
82  else if (jmodel->getType() == moveit::core::JointModel::FLOATING && index > 2)
83  {
84  RCLCPP_ERROR(LOGGER, "Cannot specify position limits for orientation of floating joint '%s'",
85  jmodel->getName().c_str());
86  }
87  else if (jmodel->getType() == moveit::core::JointModel::REVOLUTE &&
88  static_cast<const moveit::core::RevoluteJointModel*>(jmodel)->isContinuous())
89  {
90  RCLCPP_ERROR(LOGGER, "Cannot specify position limits for continuous joint '%s'", jmodel->getName().c_str());
91  }
92  else
93  {
94  ok = true;
95  }
96  return ok;
97 }

100 void RobotModelLoader::configure(const Options& opt)
101 {
102  moveit::tools::Profiler::ScopedStart prof_start;
103  moveit::tools::Profiler::ScopedBlock prof_block("RobotModelLoader::configure");
104 
105  rclcpp::Clock clock;
106  rclcpp::Time start = clock.now();
107  if (!opt.urdf_string_.empty() && !opt.srdf_string_.empty())
108  rdf_loader_.reset(new rdf_loader::RDFLoader(opt.urdf_string_, opt.srdf_string_));
109  else
110  rdf_loader_.reset(new rdf_loader::RDFLoader(node_, opt.robot_description_));
111  if (rdf_loader_->getURDF())
112  {
113  const srdf::ModelSharedPtr& srdf =
114  rdf_loader_->getSRDF() ? rdf_loader_->getSRDF() : srdf::ModelSharedPtr(new srdf::Model());
115  model_.reset(new moveit::core::RobotModel(rdf_loader_->getURDF(), srdf));
116  }
117 
118  if (model_ && !rdf_loader_->getRobotDescription().empty())
119  {
120  moveit::tools::Profiler::ScopedBlock prof_block2("RobotModelLoader::configure joint limits");
121 
122  // if there are additional joint limits specified in some .yaml file, read those in
123  for (moveit::core::JointModel* joint_model : model_->getJointModels())
124  {
125  std::vector<moveit_msgs::msg::JointLimits> joint_limit = joint_model->getVariableBoundsMsg();
126  for (std::size_t joint_id = 0; joint_id < joint_limit.size(); ++joint_id)
127  {
128  std::string prefix =
129  rdf_loader_->getRobotDescription() + "_planning.joint_limits." + joint_limit[joint_id].joint_name + ".";
130 
131  std::string param_name;
132  try
133  {
134  const auto max_pos_name = prefix + "max_position";
135  if (!node_->has_parameter(max_pos_name))
136  {
137  node_->declare_parameter(max_pos_name);
138  }
139  double max_position;
140  if (node_->get_parameter(max_pos_name, max_position))
141  {
142  if (canSpecifyPosition(joint_model, joint_id))
143  {
144  joint_limit[joint_id].has_position_limits = true;
145  joint_limit[joint_id].max_position = max_position;
146  }
147  }
148 
149  const auto min_pos_name = prefix + "min_position";
150  if (!node_->has_parameter(min_pos_name))
151  {
152  node_->declare_parameter(min_pos_name);
153  }
154  double min_position;
155  if (node_->get_parameter(min_pos_name, min_position))
156  {
157  if (canSpecifyPosition(joint_model, joint_id))
158  {
159  joint_limit[joint_id].has_position_limits = true;
160  joint_limit[joint_id].min_position = min_position;
161  }
162  }
163 
164  // Check if parameter has been declared to avoid exception
165  param_name = prefix + "has_velocity_limits";
166  if (!node_->has_parameter(param_name))
167  {
168  node_->declare_parameter(param_name);
169  }
170  bool has_vel_limits = false;
171  if (node_->get_parameter(param_name, has_vel_limits))
172  joint_limit[joint_id].has_velocity_limits = has_vel_limits;
173 
174  param_name = prefix + "has_acceleration_limits";
175  if (!node_->has_parameter(param_name))
176  {
177  node_->declare_parameter(param_name);
178  }
179  bool has_acc_limits = false;
180  if (node_->get_parameter(param_name, has_acc_limits))
181  joint_limit[joint_id].has_acceleration_limits = has_acc_limits;
182 
183  if (has_vel_limits)
184  {
185  param_name = prefix + "max_velocity";
186  if (!node_->has_parameter(param_name))
187  {
188  node_->declare_parameter(param_name);
189  }
190 
191  if (!node_->get_parameter(param_name, joint_limit[joint_id].max_velocity))
192  {
193  RCLCPP_ERROR(LOGGER, "Specified a velocity limit for joint: %s but did not set a max velocity",
194  joint_limit[joint_id].joint_name.c_str());
195  }
196  }
197 
198  if (has_acc_limits)
199  {
200  param_name = prefix + "max_acceleration";
201  if (!node_->has_parameter(param_name))
202  {
203  node_->declare_parameter(param_name);
204  }
205 
206  if (!node_->get_parameter(param_name, joint_limit[joint_id].max_acceleration))
207  {
208  RCLCPP_ERROR(LOGGER, "Specified an acceleration limit for joint: %s but did not set a max acceleration",
209  joint_limit[joint_id].joint_name.c_str());
210  }
211  }
212  }
213  catch (const rclcpp::ParameterTypeException& e)
214  {
215  RCLCPP_ERROR(LOGGER, "When getting the parameter %s: %s", param_name.c_str(), e.what());
216  }
217  }
218  joint_model->setVariableBounds(joint_limit);
219  }
220  }
221 
222  if (model_ && opt.load_kinematics_solvers_)
223  loadKinematicsSolvers();
224 
225  RCLCPP_DEBUG(node_->get_logger(), "Loaded kinematic model in %d seconds", (clock.now() - start).seconds());
226 }

228 void RobotModelLoader::loadKinematicsSolvers(const kinematics_plugin_loader::KinematicsPluginLoaderPtr& kloader)
229 {
230  moveit::tools::Profiler::ScopedStart prof_start;
231  moveit::tools::Profiler::ScopedBlock prof_block("RobotModelLoader::loadKinematicsSolvers");
232 
233  if (rdf_loader_ && model_)
234  {
235  // load the kinematics solvers
236  if (kloader)
237  kinematics_loader_ = kloader;
238  else
239  kinematics_loader_.reset(
240  new kinematics_plugin_loader::KinematicsPluginLoader(node_, rdf_loader_->getRobotDescription()));
241  moveit::core::SolverAllocatorFn kinematics_allocator =
242  kinematics_loader_->getLoaderFunction(rdf_loader_->getSRDF());
243  const std::vector<std::string>& groups = kinematics_loader_->getKnownGroups();
244  std::stringstream ss;
245  std::copy(groups.begin(), groups.end(), std::ostream_iterator<std::string>(ss, " "));
246  RCLCPP_DEBUG(LOGGER, "Loaded information about the following groups: '%s' ", ss.str().c_str());
247  if (groups.empty() && !model_->getJointModelGroups().empty())
248  RCLCPP_WARN(LOGGER, "No kinematics plugins defined. Fill and load kinematics.yaml!");
249 
250  std::map<std::string, moveit::core::SolverAllocatorFn> imap;
251  for (const std::string& group : groups)
252  {
253  // Check if a group in kinematics.yaml exists in the srdf
254  if (!model_->hasJointModelGroup(group))
255  continue;
256 
257  const moveit::core::JointModelGroup* jmg = model_->getJointModelGroup(group);
258 
259  kinematics::KinematicsBasePtr solver = kinematics_allocator(jmg);
260  if (solver)
261  {
262  std::string error_msg;
263  if (solver->supportsGroup(jmg, &error_msg))
264  {
265  imap[group] = kinematics_allocator;
266  }
267  else
268  {
269  RCLCPP_ERROR(LOGGER, "Kinematics solver %s does not support joint group %s. Error: %s",
270  typeid(*solver).name(), group.c_str(), error_msg.c_str());
271  }
272  }
273  else
274  {
275  RCLCPP_ERROR(LOGGER, "Kinematics solver could not be instantiated for joint group %s.", group.c_str());
276  }
277  }
278  model_->setKinematicsAllocators(imap);
279 
280  // set the default IK timeouts
281  const std::map<std::string, double>& timeout = kinematics_loader_->getIKTimeout();
282  for (const std::pair<const std::string, double>& it : timeout)
283  {
284  if (!model_->hasJointModelGroup(it.first))
285  continue;
286  moveit::core::JointModelGroup* jmg = model_->getJointModelGroup(it.first);
287  jmg->setDefaultIKTimeout(it.second);
288  }
289  }
290 }
