61 urdf::ModelInterfaceSharedPtr loadModelInterface(const std::string& robot_name)
62 {
63  const std::string package_name = "moveit_resources_" + robot_name + "_description";
64  boost::filesystem::path res_path(ament_index_cpp::get_package_share_directory(package_name));
65  std::string urdf_path;
66  if (robot_name == "pr2")
67  {
68  urdf_path = (res_path / "urdf/robot.xml").string();
69  }
70  else
71  {
72  urdf_path = (res_path / "urdf" / (robot_name + ".urdf")).string();
73  }
74  urdf::ModelInterfaceSharedPtr urdf_model = urdf::parseURDFFile(urdf_path);
75  if (urdf_model == nullptr)
76  {
77  RCLCPP_ERROR(LOGGER, "Cannot find URDF for %s. Make sure moveit_resources_<your_robot_description> is installed",
78  robot_name.c_str());
79  }
80  return urdf_model;
81 }

91  boost::filesystem::path res_path(ament_index_cpp::get_package_share_directory(package_name));
92  srdf_path = (res_path / "srdf/robot.xml").string();
93  }
94  else
95  {
96  const std::string package_name = "moveit_resources_" + robot_name + "_moveit_config";
97  boost::filesystem::path res_path(ament_index_cpp::get_package_share_directory(package_name));
98  srdf_path = (res_path / "config" / (robot_name + ".srdf")).string();

117 void RobotModelBuilder::addChain(const std::string& section, const std::string& type,
118  const std::vector<geometry_msgs::msg::Pose>& joint_origins, urdf::Vector3 joint_axis)
119 {
120  std::vector<std::string> link_names;
121  boost::split_regex(link_names, section, boost::regex("->"));
122  if (link_names.empty())
123  {
124  RCLCPP_ERROR(LOGGER, "No links specified (empty section?)");
125  is_valid_ = false;
126  return;
127  }
128  // First link should already be added.
129  if (!urdf_model_->getLink(link_names[0]))
130  {
131  RCLCPP_ERROR(LOGGER, "Link %s not present in builder yet!", link_names[0].c_str());
132  is_valid_ = false;
133  return;
134  }
135 
136  if (!joint_origins.empty() && link_names.size() - 1 != joint_origins.size())
137  {
138  RCLCPP_ERROR(LOGGER, "There should be one more link (%zu) than there are joint origins (%zu)", link_names.size(),
139  joint_origins.size());
140  is_valid_ = false;
141  return;
142  }
143 
144  // Iterate through each link.
145  for (size_t i = 1; i < link_names.size(); ++i)
146  {
147  // These links shouldn't be present already.
148  if (urdf_model_->getLink(link_names[i]))
149  {
150  RCLCPP_ERROR(LOGGER, "Link %s is already specified", link_names[i].c_str());
151  is_valid_ = false;
152  return;
153  }
154  urdf::LinkSharedPtr link(new urdf::Link);
155  link->name = link_names[i];
156  urdf_model_->links_.insert(std::make_pair(link_names[i], link));
157  urdf::JointSharedPtr joint(new urdf::Joint);
158  joint->name = link_names[i - 1] + "-" + link_names[i] + "-joint";
159  // Default to Identity transform for origins.
160  joint->parent_to_joint_origin_transform.clear();
161  if (!joint_origins.empty())
162  {
163  geometry_msgs::msg::Pose o = joint_origins[i - 1];
164  joint->parent_to_joint_origin_transform.position = urdf::Vector3(o.position.x, o.position.y, o.position.z);
165  joint->parent_to_joint_origin_transform.rotation =
166  urdf::Rotation(o.orientation.x, o.orientation.y, o.orientation.z, o.orientation.w);
167  }
168 
169  joint->parent_link_name = link_names[i - 1];
170  joint->child_link_name = link_names[i];
171  if (type == "planar")
172  joint->type = urdf::Joint::PLANAR;
173  else if (type == "floating")
174  joint->type = urdf::Joint::FLOATING;
175  else if (type == "revolute")
176  joint->type = urdf::Joint::REVOLUTE;
177  else if (type == "continuous")
178  joint->type = urdf::Joint::CONTINUOUS;
179  else if (type == "prismatic")
180  joint->type = urdf::Joint::PRISMATIC;
181  else if (type == "fixed")
182  joint->type = urdf::Joint::FIXED;
183  else
184  {
185  RCLCPP_ERROR(LOGGER, "No such joint type as %s", type.c_str());
186  is_valid_ = false;
187  return;
188  }
189 
190  joint->axis = joint_axis;
191  if (joint->type == urdf::Joint::REVOLUTE || joint->type == urdf::Joint::PRISMATIC)
192  {
193  urdf::JointLimitsSharedPtr limits(new urdf::JointLimits);
194  limits->lower = -boost::math::constants::pi<double>();
195  limits->upper = boost::math::constants::pi<double>();
196 
197  joint->limits = limits;
198  }
199  urdf_model_->joints_.insert(std::make_pair(joint->name, joint));
200  }
201 }

372 moveit::core::RobotModelPtr RobotModelBuilder::build()
373 {
374  moveit::core::RobotModelPtr robot_model;
375  std::map<std::string, std::string> parent_link_tree;
376  parent_link_tree.clear();
377 
378  try
379  {
380  urdf_model_->initTree(parent_link_tree);
381  }
382  catch (urdf::ParseError& e)
383  {
384  RCLCPP_ERROR(LOGGER, "Failed to build tree: %s", e.what());
385  return robot_model;
386  }
387 
388  // find the root link
389  try
390  {
391  urdf_model_->initRoot(parent_link_tree);
392  }
393  catch (urdf::ParseError& e)
394  {
395  RCLCPP_ERROR(LOGGER, "Failed to find root link: %s", e.what());
396  return robot_model;
397  }
398  srdf_writer_->updateSRDFModel(*urdf_model_);
399  robot_model.reset(new moveit::core::RobotModel(urdf_model_, srdf_writer_->srdf_model_));
400  return robot_model;
401 }
