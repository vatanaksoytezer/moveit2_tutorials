92 void RobotState::allocMemory()
93 {
94  static_assert((sizeof(Eigen::Isometry3d) / EIGEN_MAX_ALIGN_BYTES) * EIGEN_MAX_ALIGN_BYTES == sizeof(Eigen::Isometry3d),
95  "sizeof(Eigen::Isometry3d) should be a multiple of EIGEN_MAX_ALIGN_BYTES");
96 
97  constexpr unsigned int extra_alignment_bytes = EIGEN_MAX_ALIGN_BYTES - 1;
98  // memory for the dirty joint transforms
99  const int nr_doubles_for_dirty_joint_transforms =
100  1 + robot_model_->getJointModelCount() / (sizeof(double) / sizeof(unsigned char));
101  const size_t bytes =
102  sizeof(Eigen::Isometry3d) * (robot_model_->getJointModelCount() + robot_model_->getLinkModelCount() +
103  robot_model_->getLinkGeometryCount()) +
104  sizeof(double) * (robot_model_->getVariableCount() * 3 + nr_doubles_for_dirty_joint_transforms) +
105  extra_alignment_bytes;
106  memory_ = malloc(bytes);
107 
108  // make the memory for transforms align at EIGEN_MAX_ALIGN_BYTES
109  // https://eigen.tuxfamily.org/dox/classEigen_1_1aligned__allocator.html
110  variable_joint_transforms_ = reinterpret_cast<Eigen::Isometry3d*>(((uintptr_t)memory_ + extra_alignment_bytes) &
111  ~(uintptr_t)extra_alignment_bytes);
112  global_link_transforms_ = variable_joint_transforms_ + robot_model_->getJointModelCount();
113  global_collision_body_transforms_ = global_link_transforms_ + robot_model_->getLinkModelCount();
114  dirty_joint_transforms_ =
115  reinterpret_cast<unsigned char*>(global_collision_body_transforms_ + robot_model_->getLinkGeometryCount());
116  position_ = reinterpret_cast<double*>(dirty_joint_transforms_) + nr_doubles_for_dirty_joint_transforms;
117  velocity_ = position_ + robot_model_->getVariableCount();
118  // acceleration and effort share the memory (not both can be specified)
119  effort_ = acceleration_ = velocity_ + robot_model_->getVariableCount();
120 }

122 void RobotState::initTransforms()
123 {
124  // mark all transforms as dirty
125  const int nr_doubles_for_dirty_joint_transforms =
126  1 + robot_model_->getJointModelCount() / (sizeof(double) / sizeof(unsigned char));
127  memset(dirty_joint_transforms_, 1, sizeof(double) * nr_doubles_for_dirty_joint_transforms);
128 
129  // initialize last row of transformation matrices, which will not be modified by transform updates anymore
130  for (size_t i = 0, end = robot_model_->getJointModelCount() + robot_model_->getLinkModelCount() +
131  robot_model_->getLinkGeometryCount();
132  i != end; ++i)
133  variable_joint_transforms_[i].makeAffine();
134 }

143 void RobotState::copyFrom(const RobotState& other)
144 {
145  has_velocity_ = other.has_velocity_;
146  has_acceleration_ = other.has_acceleration_;
147  has_effort_ = other.has_effort_;
148 
149  dirty_collision_body_transforms_ = other.dirty_collision_body_transforms_;
150  dirty_link_transforms_ = other.dirty_link_transforms_;
151 
152  if (dirty_link_transforms_ == robot_model_->getRootJoint())
153  {
154  // everything is dirty; no point in copying transforms; copy positions, potentially velocity & acceleration
155  memcpy(position_, other.position_,
156  robot_model_->getVariableCount() * sizeof(double) *
157  (1 + (has_velocity_ ? 1 : 0) + ((has_acceleration_ || has_effort_) ? 1 : 0)));
158  // and just initialize transforms
159  initTransforms();
160  }
161  else
162  {
163  // copy all the memory; maybe avoid copying velocity and acceleration if possible
164  const int nr_doubles_for_dirty_joint_transforms =
165  1 + robot_model_->getJointModelCount() / (sizeof(double) / sizeof(unsigned char));
166  const size_t bytes =
167  sizeof(Eigen::Isometry3d) * (robot_model_->getJointModelCount() + robot_model_->getLinkModelCount() +
168  robot_model_->getLinkGeometryCount()) +
169  sizeof(double) *
170  (robot_model_->getVariableCount() * (1 + ((has_velocity_ || has_acceleration_ || has_effort_) ? 1 : 0) +
171  ((has_acceleration_ || has_effort_) ? 1 : 0)) +
172  nr_doubles_for_dirty_joint_transforms);
173  memcpy((void*)variable_joint_transforms_, (void*)other.variable_joint_transforms_, bytes);
174  }
175 
176  // copy attached bodies
177  clearAttachedBodies();
178  for (const std::pair<const std::string, AttachedBody*>& it : other.attached_body_map_)
179  attachBody(it.second->getName(), it.second->getShapes(), it.second->getFixedTransforms(),
180  it.second->getTouchLinks(), it.second->getAttachedLinkName(), it.second->getDetachPosture(),
181  it.second->getSubframeTransforms());
182 }

655 void RobotState::updateCollisionBodyTransforms()
656 {
657  if (dirty_link_transforms_ != nullptr)
658  updateLinkTransforms();
659 
660  if (dirty_collision_body_transforms_ != nullptr)
661  {
662  const std::vector<const LinkModel*>& links = dirty_collision_body_transforms_->getDescendantLinkModels();
663  dirty_collision_body_transforms_ = nullptr;
664 
665  for (const LinkModel* link : links)
666  {
667  const EigenSTL::vector_Isometry3d& ot = link->getCollisionOriginTransforms();
668  const std::vector<int>& ot_id = link->areCollisionOriginTransformsIdentity();

670  const int index_l = link->getLinkIndex();
671  for (std::size_t j = 0, end = ot.size(); j != end; ++j)
672  {
673  if (ot_id[j])
674  global_collision_body_transforms_[index_co + j] = global_link_transforms_[index_l];
675  else
676  global_collision_body_transforms_[index_co + j].affine().noalias() =
677  global_link_transforms_[index_l].affine() * ot[j].matrix();
678  }
679  }
680  }

697 void RobotState::updateLinkTransformsInternal(const JointModel* start)
698 {
699  for (const LinkModel* link : start->getDescendantLinkModels())
700  {
701  int idx_link = link->getLinkIndex();
702  const LinkModel* parent = link->getParentLinkModel();
703  if (parent) // root JointModel will not have a parent
704  {
705  int idx_parent = parent->getLinkIndex();
706  if (link->parentJointIsFixed()) // fixed joint
707  global_link_transforms_[idx_link].affine().noalias() =
708  global_link_transforms_[idx_parent].affine() * link->getJointOriginTransform().matrix();
709  else // non-fixed joint
710  {
711  if (link->jointOriginTransformIsIdentity()) // Link has identity transform
712  global_link_transforms_[idx_link].affine().noalias() =
713  global_link_transforms_[idx_parent].affine() * getJointTransform(link->getParentJointModel()).matrix();
714  else // Link has non-identity transform
715  global_link_transforms_[idx_link].affine().noalias() =
716  global_link_transforms_[idx_parent].affine() * link->getJointOriginTransform().matrix() *
717  getJointTransform(link->getParentJointModel()).matrix();
718  }
719  }
720  else // is the origin / root / 'model frame'
721  {
722  if (link->jointOriginTransformIsIdentity())
723  global_link_transforms_[idx_link] = getJointTransform(link->getParentJointModel());
724  else
725  global_link_transforms_[idx_link].affine().noalias() =
726  link->getJointOriginTransform().affine() * getJointTransform(link->getParentJointModel()).matrix();
727  }
728  }
729 
730  // update attached bodies tf; these are usually very few, so we update them all
731  for (std::map<std::string, AttachedBody*>::const_iterator it = attached_body_map_.begin();
732  it != attached_body_map_.end(); ++it)
733  it->second->computeTransform(global_link_transforms_[it->second->getAttachedLink()->getLinkIndex()]);
734 }

736 void RobotState::updateStateWithLinkAt(const LinkModel* link, const Eigen::Isometry3d& transform, bool backward)
737 {
738  updateLinkTransforms(); // no link transforms must be dirty, otherwise the transform we set will be overwritten
739 
740  // update the fact that collision body transforms are out of date
741  if (dirty_collision_body_transforms_)
742  dirty_collision_body_transforms_ =
743  robot_model_->getCommonRoot(dirty_collision_body_transforms_, link->getParentJointModel());
744  else
745  dirty_collision_body_transforms_ = link->getParentJointModel();
746 
747  global_link_transforms_[link->getLinkIndex()] = transform;
748 
749  // update link transforms for descendant links only (leaving the transform for the current link untouched)
750  const std::vector<const JointModel*>& cj = link->getChildJointModels();
751  for (const JointModel* joint : cj)
752  updateLinkTransformsInternal(joint);
753 
754  // if we also need to go backward
755  if (backward)
756  {
757  const LinkModel* parent_link = link;
758  const LinkModel* child_link;
759  while (parent_link->getParentJointModel()->getParentLinkModel())
760  {
761  child_link = parent_link;
762  parent_link = parent_link->getParentJointModel()->getParentLinkModel();
763 
764  // update the transform of the parent
765  global_link_transforms_[parent_link->getLinkIndex()] =
766  global_link_transforms_[child_link->getLinkIndex()] *
767  (child_link->getJointOriginTransform() *
768  variable_joint_transforms_[child_link->getParentJointModel()->getJointIndex()])
769  .inverse();
770 
771  // update link transforms for descendant links only (leaving the transform for the current link untouched)
772  // with the exception of the child link we are coming backwards from
773  const std::vector<const JointModel*>& cj = parent_link->getChildJointModels();
774  for (const JointModel* joint : cj)
775  if (joint != child_link->getParentJointModel())
776  updateLinkTransformsInternal(joint);
777  }
778  // all collision body transforms are invalid now
779  dirty_collision_body_transforms_ = parent_link->getParentJointModel();
780  }
781 
782  // update attached bodies tf; these are usually very few, so we update them all
783  for (std::map<std::string, AttachedBody*>::const_iterator it = attached_body_map_.begin();
784  it != attached_body_map_.end(); ++it)
785  it->second->computeTransform(global_link_transforms_[it->second->getAttachedLink()->getLinkIndex()]);
786 }

847  for (const JointModel* joint : joints)

850  continue;
851  if (joint->getType() == JointModel::REVOLUTE)
852  if (static_cast<const RevoluteJointModel*>(joint)->isContinuous())

861  upper_bounds[j] = bounds[j].max_position_;

864  if (new_distance < distance)
865  {
866  index = joint;
867  distance = new_distance;
868  }
869  new_distance = joint->distance(joint_values, &upper_bounds[0]);
870  if (new_distance < distance)
871  {
872  index = joint;
873  distance = new_distance;
874  }
875  }
876  return std::make_pair(distance, index);

1085  return getFrameInfo(frame_id.substr(1), robot_link, frame_found);
1086 
1087  static const Eigen::Isometry3d IDENTITY_TRANSFORM = Eigen::Isometry3d::Identity();
1088  if (frame_id == robot_model_->getModelFrame())
1089  {
1090  robot_link = robot_model_->getRootLink();
1091  frame_found = true;
1092  return IDENTITY_TRANSFORM;
1093  }
1094  if ((robot_link = robot_model_->getLinkModel(frame_id, &frame_found)))
1095  {
1096  BOOST_VERIFY(checkLinkTransforms());
1097  return global_link_transforms_[robot_link->getLinkIndex()];
1098  }

1103  if (jt != attached_body_map_.end())
1104  {

1106  if (tf.empty())

1109  frame_id.c_str());
1110  robot_link = nullptr;
1111  frame_found = false;
1112  return IDENTITY_TRANSFORM;
1113  }
1114  if (tf.size() > 1)
1115  RCLCPP_DEBUG(LOGGER,
1116  "There are multiple geometries associated to attached body '%s'. "
1117  "Returning the transform for the first one.",
1118  frame_id.c_str());
1119  robot_link = jt->second->getAttachedLink();
1120  frame_found = true;
1121  BOOST_VERIFY(checkLinkTransforms());
1122  return tf[0];
1123  }
1124 
1125  // Check if an AttachedBody has a subframe with name frame_id
1126  for (const std::pair<const std::string, AttachedBody*>& body : attached_body_map_)
1127  {
1128  const auto& transform = body.second->getGlobalSubframeTransform(frame_id, &frame_found);
1129  if (frame_found)
1130  {
1131  robot_link = body.second->getAttachedLink();
1132  BOOST_VERIFY(checkLinkTransforms());
1133  return transform;
1134  }
1135  }
1136 
1137  robot_link = nullptr;
1138  frame_found = false;
1139  return IDENTITY_TRANSFORM;
1140 }

1257 bool RobotState::getJacobian(const JointModelGroup* group, const LinkModel* link,
1258  const Eigen::Vector3d& reference_point_position, Eigen::MatrixXd& jacobian,
1259  bool use_quaternion_representation) const
1260 {
1261  BOOST_VERIFY(checkLinkTransforms());
1262 
1263  if (!group->isChain())
1264  {
1265  RCLCPP_ERROR(LOGGER, "The group '%s' is not a chain. Cannot compute Jacobian.", group->getName().c_str());
1266  return false;
1267  }
1268 
1269  if (!group->isLinkUpdated(link->getName()))
1270  {
1271  RCLCPP_ERROR(LOGGER, "Link name '%s' does not exist in the chain '%s' or is not a child for this chain",
1272  link->getName().c_str(), group->getName().c_str());
1273  return false;
1274  }
1275 
1276  const moveit::core::JointModel* root_joint_model = group->getJointModels()[0]; // group->getJointRoots()[0];
1277  const moveit::core::LinkModel* root_link_model = root_joint_model->getParentLinkModel();
1278  // getGlobalLinkTransform() returns a valid isometry by contract
1279  Eigen::Isometry3d reference_transform =
1280  root_link_model ? getGlobalLinkTransform(root_link_model).inverse() : Eigen::Isometry3d::Identity();
1281  int rows = use_quaternion_representation ? 7 : 6;
1282  int columns = group->getVariableCount();
1283  jacobian = Eigen::MatrixXd::Zero(rows, columns);
1284 
1285  // getGlobalLinkTransform() returns a valid isometry by contract
1286  Eigen::Isometry3d link_transform = reference_transform * getGlobalLinkTransform(link); // valid isometry
1287  Eigen::Vector3d point_transform = link_transform * reference_point_position;
1288 
1289  /*
1290  RCLCPP_DEBUG(LOGGER, "Point from reference origin expressed in world coordinates: %f %f %f",
1291  point_transform.x(),
1292  point_transform.y(),
1293  point_transform.z());
1294  */
1295 
1296  Eigen::Vector3d joint_axis;
1297  Eigen::Isometry3d joint_transform;
1298 
1299  while (link)
1300  {
1301  /*
1302  RCLCPP_DEBUG(LOGGER, "Link: %s, %f %f %f",link_state->getName().c_str(),
1303  link_state->getGlobalLinkTransform().translation().x(),
1304  link_state->getGlobalLinkTransform().translation().y(),
1305  link_state->getGlobalLinkTransform().translation().z());
1306  RCLCPP_DEBUG(LOGGER, "Joint: %s",link_state->getParentJointState()->getName().c_str());
1307  */
1308  const JointModel* pjm = link->getParentJointModel();
1309  if (pjm->getVariableCount() > 0)
1310  {
1311  if (!group->hasJointModel(pjm->getName()))
1312  {
1313  link = pjm->getParentLinkModel();
1314  continue;
1315  }
1316  unsigned int joint_index = group->getVariableGroupIndex(pjm->getName());
1317  // getGlobalLinkTransform() returns a valid isometry by contract
1318  joint_transform = reference_transform * getGlobalLinkTransform(link); // valid isometry
1319  if (pjm->getType() == moveit::core::JointModel::REVOLUTE)
1320  {
1321  joint_axis = joint_transform.linear() * static_cast<const moveit::core::RevoluteJointModel*>(pjm)->getAxis();
1322  jacobian.block<3, 1>(0, joint_index) =
1323  jacobian.block<3, 1>(0, joint_index) + joint_axis.cross(point_transform - joint_transform.translation());
1324  jacobian.block<3, 1>(3, joint_index) = jacobian.block<3, 1>(3, joint_index) + joint_axis;
1325  }
1326  else if (pjm->getType() == moveit::core::JointModel::PRISMATIC)
1327  {
1328  joint_axis = joint_transform.linear() * static_cast<const moveit::core::PrismaticJointModel*>(pjm)->getAxis();
1329  jacobian.block<3, 1>(0, joint_index) = jacobian.block<3, 1>(0, joint_index) + joint_axis;
1330  }
1331  else if (pjm->getType() == moveit::core::JointModel::PLANAR)
1332  {
1333  joint_axis = joint_transform * Eigen::Vector3d(1.0, 0.0, 0.0);
1334  jacobian.block<3, 1>(0, joint_index) = jacobian.block<3, 1>(0, joint_index) + joint_axis;
1335  joint_axis = joint_transform * Eigen::Vector3d(0.0, 1.0, 0.0);
1336  jacobian.block<3, 1>(0, joint_index + 1) = jacobian.block<3, 1>(0, joint_index + 1) + joint_axis;
1337  joint_axis = joint_transform * Eigen::Vector3d(0.0, 0.0, 1.0);
1338  jacobian.block<3, 1>(0, joint_index + 2) = jacobian.block<3, 1>(0, joint_index + 2) +
1339  joint_axis.cross(point_transform - joint_transform.translation());
1340  jacobian.block<3, 1>(3, joint_index + 2) = jacobian.block<3, 1>(3, joint_index + 2) + joint_axis;
1341  }
1342  else
1343  RCLCPP_ERROR(LOGGER, "Unknown type of joint in Jacobian computation");
1344  }
1345  if (pjm == root_joint_model)
1346  break;
1347  link = pjm->getParentLinkModel();
1348  }
1349  if (use_quaternion_representation)
1350  { // Quaternion representation
1351  // From "Advanced Dynamics and Motion Simulation" by Paul Mitiguy
1352  // d/dt ( [w] ) = 1/2 * [ -x -y -z ] * [ omega_1 ]
1353  // [x] [ w -z y ] [ omega_2 ]
1354  // [y] [ z w -x ] [ omega_3 ]
1355  // [z] [ -y x w ]
1356  Eigen::Quaterniond q(link_transform.linear());
1357  double w = q.w(), x = q.x(), y = q.y(), z = q.z();
1358  Eigen::MatrixXd quaternion_update_matrix(4, 3);
1359  quaternion_update_matrix << -x, -y, -z, w, -z, y, z, w, -x, -y, x, w;
1360  jacobian.block(3, 0, 4, columns) = 0.5 * quaternion_update_matrix * jacobian.block(3, 0, 3, columns);
1361  }
1362  return true;
1363 }

1381 void RobotState::computeVariableVelocity(const JointModelGroup* jmg, Eigen::VectorXd& qdot,
1382  const Eigen::VectorXd& twist, const LinkModel* tip) const
1383 {
1384  // Get the Jacobian of the group at the current configuration
1385  Eigen::MatrixXd j(6, jmg->getVariableCount());
1386  Eigen::Vector3d reference_point(0.0, 0.0, 0.0);
1387  getJacobian(jmg, tip, reference_point, j, false);
1388 
1389  // Rotate the jacobian to the end-effector frame
1390  Eigen::Isometry3d e_mb = getGlobalLinkTransform(tip).inverse();
1391  Eigen::MatrixXd e_wb = Eigen::ArrayXXd::Zero(6, 6);
1392  e_wb.block(0, 0, 3, 3) = e_mb.matrix().block(0, 0, 3, 3);
1393  e_wb.block(3, 3, 3, 3) = e_mb.matrix().block(0, 0, 3, 3);
1394  j = e_wb * j;
1395 
1396  // Do the Jacobian moore-penrose pseudo-inverse
1397  Eigen::JacobiSVD<Eigen::MatrixXd> svd_of_j(j, Eigen::ComputeThinU | Eigen::ComputeThinV);
1398  const Eigen::MatrixXd& u = svd_of_j.matrixU();
1399  const Eigen::MatrixXd& v = svd_of_j.matrixV();
1400  const Eigen::VectorXd& s = svd_of_j.singularValues();
1401 
1402  Eigen::VectorXd sinv = s;
1403  static const double PINVTOLER = std::numeric_limits<float>::epsilon();
1404  double maxsv = 0.0;
1405  for (std::size_t i = 0; i < static_cast<std::size_t>(s.rows()); ++i)
1406  if (fabs(s(i)) > maxsv)
1407  maxsv = fabs(s(i));
1408  for (std::size_t i = 0; i < static_cast<std::size_t>(s.rows()); ++i)
1409  {
1410  // Those singular values smaller than a percentage of the maximum singular value are removed
1411  if (fabs(s(i)) > maxsv * PINVTOLER)
1412  sinv(i) = 1.0 / s(i);
1413  else
1414  sinv(i) = 0.0;
1415  }
1416  Eigen::MatrixXd jinv = (v * sinv.asDiagonal() * u.transpose());
1417 
1418  // Compute joint velocity
1419  qdot = jinv * twist;
1420 }

1553 bool RobotState::setFromIK(const JointModelGroup* jmg, const EigenSTL::vector_Isometry3d& poses_in,

1555  const std::vector<std::vector<double> >& consistency_limit_sets, double timeout,
1556  const GroupStateValidityCallbackFn& constraint,
1557  const kinematics::KinematicsQueryOptions& options)
1558 {

1560  if (poses_in.size() != tips_in.size())
1561  {
1562  RCLCPP_ERROR(LOGGER, "Number of poses must be the same as number of tips");
1563  return false;
1564  }
1565 
1566  // Load solver
1567  const kinematics::KinematicsBaseConstPtr& solver = jmg->getSolverInstance();
1568 
1569  // Check if this jmg has a solver
1570  bool valid_solver = true;
1571  if (!solver)
1572  {
1573  valid_solver = false;
1574  }
1575  // Check if this jmg's IK solver can handle multiple tips (non-chain solver)
1576  else if (poses_in.size() > 1)
1577  {
1578  std::string error_msg;
1579  if (!solver->supportsGroup(jmg, &error_msg))
1580  {
1581  RCLCPP_ERROR(LOGGER, "Kinematics solver %s does not support joint group %s. Error: %s", typeid(*solver).name(),
1582  jmg->getName().c_str(), error_msg.c_str());
1583  valid_solver = false;
1584  }
1585  }
1586 
1587  if (!valid_solver)
1588  {
1589  // Check if there are subgroups that can solve this for us (non-chains)
1590  if (poses_in.size() > 1)
1591  {
1592  // Forward to setFromIKSubgroups() to allow different subgroup IK solvers to work together
1593  return setFromIKSubgroups(jmg, poses_in, tips_in, consistency_limit_sets, timeout, constraint, options);
1594  }
1595  else
1596  {
1597  RCLCPP_ERROR(LOGGER, "No kinematics solver instantiated for group '%s'", jmg->getName().c_str());
1598  return false;
1599  }
1600  }
1601 
1602  // Check that no, or only one set of consistency limits have been passed in, and choose that one
1603  std::vector<double> consistency_limits;
1604  if (consistency_limit_sets.size() > 1)
1605  {
1606  RCLCPP_ERROR(LOGGER,
1607  "Invalid number (%zu) of sets of consistency limits for a setFromIK request "
1608  "that is being solved by a single IK solver",
1609  consistency_limit_sets.size());
1610  return false;
1611  }
1612  else if (consistency_limit_sets.size() == 1)
1613  consistency_limits = consistency_limit_sets[0];
1614 
1615  const std::vector<std::string>& solver_tip_frames = solver->getTipFrames();
1616 
1617  // Track which possible tips frames we have filled in so far
1618  std::vector<bool> tip_frames_used(solver_tip_frames.size(), false);
1619 
1620  // Create vector to hold the output frames in the same order as solver_tip_frames
1621  std::vector<geometry_msgs::msg::Pose> ik_queries(solver_tip_frames.size());
1622 
1623  // Bring each pose to the frame of the IK solver
1624  for (std::size_t i = 0; i < poses_in.size(); ++i)
1625  {
1626  // Make non-const
1627  Eigen::Isometry3d pose = poses_in[i];
1628  std::string pose_frame = tips_in[i];
1629 
1630  // Remove extra slash
1631  if (!pose_frame.empty() && pose_frame[0] == '/')
1632  pose_frame = pose_frame.substr(1);
1633 
1634  // bring the pose to the frame of the IK solver
1635  if (!setToIKSolverFrame(pose, solver))
1636  return false;
1637 
1638  // try all of the solver's possible tip frames to see if they uniquely align with any of our passed in pose tip
1639  // frames
1640  bool found_valid_frame = false;
1641  std::size_t solver_tip_id; // our current index
1642  for (solver_tip_id = 0; solver_tip_id < solver_tip_frames.size(); ++solver_tip_id)
1643  {
1644  // Check if this tip frame is already accounted for
1645  if (tip_frames_used[solver_tip_id])
1646  continue; // already has a pose
1647 
1648  // check if the tip frame can be transformed via fixed transforms to the frame known to the IK solver
1649  std::string solver_tip_frame = solver_tip_frames[solver_tip_id];
1650 
1651  // remove the frame '/' if there is one, so we can avoid calling Transforms::sameFrame() which may copy strings
1652  // more often that we need to
1653  if (!solver_tip_frame.empty() && solver_tip_frame[0] == '/')
1654  solver_tip_frame = solver_tip_frame.substr(1);
1655 
1656  if (pose_frame != solver_tip_frame)
1657  {
1658  if (hasAttachedBody(pose_frame))
1659  {
1660  const AttachedBody* body = getAttachedBody(pose_frame);
1661  const EigenSTL::vector_Isometry3d& ab_trans = body->getFixedTransforms();
1662  if (ab_trans.size() != 1)
1663  {
1664  RCLCPP_ERROR(LOGGER, "Cannot use an attached body "
1665  "with multiple geometries as a reference frame.");
1666  return false;
1667  }
1668  pose_frame = body->getAttachedLinkName();
1669  pose = pose * ab_trans[0].inverse();
1670  }
1671  if (pose_frame != solver_tip_frame)
1672  {
1673  const moveit::core::LinkModel* link_model = getLinkModel(pose_frame);
1674  if (!link_model)
1675  {
1676  RCLCPP_ERROR(LOGGER, "The following Pose Frame does not exist ", pose_frame.c_str());
1677  return false;
1678  }
1679  const moveit::core::LinkTransformMap& fixed_links = link_model->getAssociatedFixedTransforms();
1680  for (const std::pair<const LinkModel* const, Eigen::Isometry3d>& fixed_link : fixed_links)
1681  if (Transforms::sameFrame(fixed_link.first->getName(), solver_tip_frame))
1682  {
1683  pose_frame = solver_tip_frame;
1684  pose = pose * fixed_link.second;
1685  break;
1686  }
1687  }
1688 
1689  } // end if pose_frame
1690 
1691  // Check if this pose frame works
1692  if (pose_frame == solver_tip_frame)
1693  {
1694  found_valid_frame = true;
1695  break;
1696  }

1865  // often that we need to

1905  }

1911  if (constraint)

1913

1915  {

1922  ik_queries[i].orientation.y = quat.y();
1923  ik_queries[i].orientation.z = quat.z();
1924  ik_queries[i].orientation.w = quat.w();
1925  }
1926

1928  if (timeout < std::numeric_limits<double>::epsilon())
1929  timeout = jmg->getDefaultIKTimeout();
1930 
1931  auto start = std::chrono::system_clock::now();
1932  double elapsed = 0;
1933 
1934  bool first_seed = true;
1935  unsigned int attempts = 0;
1936  do
1937  {
1938  ++attempts;
1939  RCLCPP_DEBUG(LOGGER, "IK attempt: %d", attempts);
1940  bool found_solution = true;
1941  for (std::size_t sg = 0; sg < sub_groups.size(); ++sg)
1942  {
1943  const std::vector<unsigned int>& bij = sub_groups[sg]->getKinematicsSolverJointBijection();
1944  std::vector<double> seed(bij.size());
1945  // the first seed is the initial state
1946  if (first_seed)
1947  {
1948  std::vector<double> initial_values;
1949  copyJointGroupPositions(sub_groups[sg], initial_values);
1950  for (std::size_t i = 0; i < bij.size(); ++i)
1951  seed[i] = initial_values[bij[i]];
1952  }
1953  else
1954  {
1955  // sample a random seed
1956  random_numbers::RandomNumberGenerator& rng = getRandomNumberGenerator();
1957  std::vector<double> random_values;
1958  sub_groups[sg]->getVariableRandomPositions(rng, random_values);
1959  for (std::size_t i = 0; i < bij.size(); ++i)
1960  seed[i] = random_values[bij[i]];
1961  }
1962 
1963  // compute the IK solution
1964  std::vector<double> ik_sol;
1965  moveit_msgs::msg::MoveItErrorCodes error;
1966  const std::vector<double>& climits = consistency_limits.empty() ? std::vector<double>() : consistency_limits[sg];
1967  if (solvers[sg]->searchPositionIK(ik_queries[sg], seed, (timeout - elapsed) / sub_groups.size(), climits, ik_sol,
1968  error))
1969  {
1970  std::vector<double> solution(bij.size());
1971  for (std::size_t i = 0; i < bij.size(); ++i)
1972  solution[bij[i]] = ik_sol[i];
1973  setJointGroupPositions(sub_groups[sg], solution);
1974  }
1975  else
1976  {
1977  found_solution = false;
1978  break;
1979  }
1980  }
1981  if (found_solution)
1982  {
1983  std::vector<double> full_solution;
1984  copyJointGroupPositions(jmg, full_solution);
1985  if (constraint ? constraint(this, jmg, &full_solution[0]) : true)
1986  {
1987  RCLCPP_DEBUG(LOGGER, "Found IK solution");
1988  return true;
1989  }
1990  }
1991  elapsed = std::chrono::duration_cast<std::chrono::seconds>(std::chrono::system_clock::now() - start).count();
1992  first_seed = false;
1993  } while (elapsed < timeout);
1994  return false;
