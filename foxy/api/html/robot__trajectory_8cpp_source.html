104 void RobotTrajectory::append(const RobotTrajectory& source, double dt, size_t start_index, size_t end_index)
105 {
106  end_index = std::min(end_index, source.waypoints_.size());
107  if (start_index >= end_index)
108  return;
109  waypoints_.insert(waypoints_.end(), std::next(source.waypoints_.begin(), start_index),
110  std::next(source.waypoints_.begin(), end_index));
111  std::size_t index = duration_from_previous_.size();
112  duration_from_previous_.insert(duration_from_previous_.end(),
113  std::next(source.duration_from_previous_.begin(), start_index),
114  std::next(source.duration_from_previous_.begin(), end_index));
115  if (duration_from_previous_.size() > index)
116  duration_from_previous_[index] += dt;
117 }

120 {

135 void RobotTrajectory::unwind()
136 {
137  if (waypoints_.empty())
138  return;
139 
140  const std::vector<const moveit::core::JointModel*>& cont_joints =
141  group_ ? group_->getContinuousJointModels() : robot_model_->getContinuousJointModels();
142 
143  for (const moveit::core::JointModel* cont_joint : cont_joints)
144  {
145  // unwrap continuous joints
146  double running_offset = 0.0;
147  double last_value = waypoints_[0]->getJointPositions(cont_joint)[0];
148 
149  for (std::size_t j = 1; j < waypoints_.size(); ++j)
150  {
151  double current_value = waypoints_[j]->getJointPositions(cont_joint)[0];
152  if (last_value > current_value + boost::math::constants::pi<double>())
153  running_offset += 2.0 * boost::math::constants::pi<double>();
154  else if (current_value > last_value + boost::math::constants::pi<double>())
155  running_offset -= 2.0 * boost::math::constants::pi<double>();
156 
157  last_value = current_value;
158  if (running_offset > std::numeric_limits<double>::epsilon() ||
159  running_offset < -std::numeric_limits<double>::epsilon())
160  {
161  current_value += running_offset;
162  waypoints_[j]->setJointPositions(cont_joint, &current_value);
163  }
164  }
165  }
166  for (moveit::core::RobotStatePtr& waypoint : waypoints_)
167  waypoint->update();
168 }

170 void RobotTrajectory::unwind(const moveit::core::RobotState& state)
171 {
172  if (waypoints_.empty())
173  return;
174 
175  const std::vector<const moveit::core::JointModel*>& cont_joints =
176  group_ ? group_->getContinuousJointModels() : robot_model_->getContinuousJointModels();
177 
178  for (const moveit::core::JointModel* cont_joint : cont_joints)
179  {
180  double reference_value0 = state.getJointPositions(cont_joint)[0];
181  double reference_value = reference_value0;
182  cont_joint->enforcePositionBounds(&reference_value);
183 
184  // unwrap continuous joints
185  double running_offset = reference_value0 - reference_value;
186 
187  double last_value = waypoints_[0]->getJointPositions(cont_joint)[0];
188  if (running_offset > std::numeric_limits<double>::epsilon() ||
189  running_offset < -std::numeric_limits<double>::epsilon())
190  {
191  double current_value = last_value + running_offset;
192  waypoints_[0]->setJointPositions(cont_joint, &current_value);
193  }
194 
195  for (std::size_t j = 1; j < waypoints_.size(); ++j)
196  {
197  double current_value = waypoints_[j]->getJointPositions(cont_joint)[0];
198  if (last_value > current_value + boost::math::constants::pi<double>())
199  running_offset += 2.0 * boost::math::constants::pi<double>();
200  else if (current_value > last_value + boost::math::constants::pi<double>())
201  running_offset -= 2.0 * boost::math::constants::pi<double>();
202 
203  last_value = current_value;
204  if (running_offset > std::numeric_limits<double>::epsilon() ||
205  running_offset < -std::numeric_limits<double>::epsilon())
206  {
207  current_value += running_offset;
208  waypoints_[j]->setJointPositions(cont_joint, &current_value);
209  }
210  }
211  }
212  for (moveit::core::RobotStatePtr& waypoint : waypoints_)
213  waypoint->update();

222 void RobotTrajectory::getRobotTrajectoryMsg(moveit_msgs::msg::RobotTrajectory& trajectory,
223  const std::vector<std::string>& joint_filter) const
224 {
225  trajectory = moveit_msgs::msg::RobotTrajectory();
226  if (waypoints_.empty())
227  return;
228  const std::vector<const moveit::core::JointModel*>& jnts =
229  group_ ? group_->getActiveJointModels() : robot_model_->getActiveJointModels();
230 
231  std::vector<const moveit::core::JointModel*> onedof;
232  std::vector<const moveit::core::JointModel*> mdof;
233  trajectory.joint_trajectory.joint_names.clear();
234  trajectory.multi_dof_joint_trajectory.joint_names.clear();
235 
236  for (const moveit::core::JointModel* active_joint : jnts)
237  {
238  // only consider joints listed in joint_filter
239  if (!joint_filter.empty() &&
240  std::find(joint_filter.begin(), joint_filter.end(), active_joint->getName()) == joint_filter.end())
241  continue;
242 
243  if (active_joint->getVariableCount() == 1)
244  {
245  trajectory.joint_trajectory.joint_names.push_back(active_joint->getName());
246  onedof.push_back(active_joint);
247  }
248  else
249  {
250  trajectory.multi_dof_joint_trajectory.joint_names.push_back(active_joint->getName());
251  mdof.push_back(active_joint);
252  }
253  }
254 
255  if (!onedof.empty())
256  {
257  trajectory.joint_trajectory.header.frame_id = robot_model_->getModelFrame();
258  trajectory.joint_trajectory.header.stamp = rclcpp::Time(0, 0, RCL_ROS_TIME);
259  trajectory.joint_trajectory.points.resize(waypoints_.size());
260  }
261 
262  if (!mdof.empty())
263  {
264  trajectory.multi_dof_joint_trajectory.header.frame_id = robot_model_->getModelFrame();
265  trajectory.multi_dof_joint_trajectory.header.stamp = rclcpp::Time(0, 0, RCL_ROS_TIME);
266  trajectory.multi_dof_joint_trajectory.points.resize(waypoints_.size());
267  }
268 
269  static const rclcpp::Duration ZERO_DURATION(0.0);
270  double total_time = 0.0;
271  for (std::size_t i = 0; i < waypoints_.size(); ++i)
272  {
273  if (duration_from_previous_.size() > i)
274  total_time += duration_from_previous_[i];
275 
276  if (!onedof.empty())
277  {
278  trajectory.joint_trajectory.points[i].positions.resize(onedof.size());
279  trajectory.joint_trajectory.points[i].velocities.reserve(onedof.size());
280 
281  for (std::size_t j = 0; j < onedof.size(); ++j)
282  {
283  trajectory.joint_trajectory.points[i].positions[j] =
284  waypoints_[i]->getVariablePosition(onedof[j]->getFirstVariableIndex());
285  // if we have velocities/accelerations/effort, copy those too
286  if (waypoints_[i]->hasVelocities())
287  trajectory.joint_trajectory.points[i].velocities.push_back(
288  waypoints_[i]->getVariableVelocity(onedof[j]->getFirstVariableIndex()));
289  if (waypoints_[i]->hasAccelerations())
290  trajectory.joint_trajectory.points[i].accelerations.push_back(
291  waypoints_[i]->getVariableAcceleration(onedof[j]->getFirstVariableIndex()));
292  if (waypoints_[i]->hasEffort())
293  trajectory.joint_trajectory.points[i].effort.push_back(
294  waypoints_[i]->getVariableEffort(onedof[j]->getFirstVariableIndex()));
295  }
296  // clear velocities if we have an incomplete specification
297  if (trajectory.joint_trajectory.points[i].velocities.size() != onedof.size())
298  trajectory.joint_trajectory.points[i].velocities.clear();
299  // clear accelerations if we have an incomplete specification
300  if (trajectory.joint_trajectory.points[i].accelerations.size() != onedof.size())
301  trajectory.joint_trajectory.points[i].accelerations.clear();
302  // clear effort if we have an incomplete specification
303  if (trajectory.joint_trajectory.points[i].effort.size() != onedof.size())
304  trajectory.joint_trajectory.points[i].effort.clear();
305 
306  if (duration_from_previous_.size() > i)
307  trajectory.joint_trajectory.points[i].time_from_start = rclcpp::Duration::from_seconds(total_time);
308  else
309  trajectory.joint_trajectory.points[i].time_from_start = ZERO_DURATION;
310  }
311  if (!mdof.empty())
312  {
313  trajectory.multi_dof_joint_trajectory.points[i].transforms.resize(mdof.size());
314  for (std::size_t j = 0; j < mdof.size(); ++j)
315  {
316  geometry_msgs::msg::TransformStamped ts = tf2::eigenToTransform(waypoints_[i]->getJointTransform(mdof[j]));
317  trajectory.multi_dof_joint_trajectory.points[i].transforms[j] = ts.transform;
318  // TODO: currently only checking for planar multi DOF joints / need to add check for floating
319  if (waypoints_[i]->hasVelocities() && (mdof[j]->getType() == moveit::core::JointModel::JointType::PLANAR))
320  {
321  const std::vector<std::string> names = mdof[j]->getVariableNames();
322  const double* velocities = waypoints_[i]->getJointVelocities(mdof[j]);
323 
324  geometry_msgs::msg::Twist point_velocity;
325 
326  for (std::size_t k = 0; k < names.size(); ++k)
327  {
328  if (names[k].find("/x") != std::string::npos)
329  {
330  point_velocity.linear.x = velocities[k];
331  }
332  else if (names[k].find("/y") != std::string::npos)
333  {
334  point_velocity.linear.y = velocities[k];
335  }
336  else if (names[k].find("/z") != std::string::npos)
337  {
338  point_velocity.linear.z = velocities[k];
339  }
340  else if (names[k].find("/theta") != std::string::npos)
341  {
342  point_velocity.angular.z = velocities[k];
343  }
344  }
345  trajectory.multi_dof_joint_trajectory.points[i].velocities.push_back(point_velocity);
346  }
347  }
348  if (duration_from_previous_.size() > i)
349  trajectory.multi_dof_joint_trajectory.points[i].time_from_start = rclcpp::Duration::from_seconds(total_time);
350  else
351  trajectory.multi_dof_joint_trajectory.points[i].time_from_start = ZERO_DURATION;
352  }
353  }
354 }

356 void RobotTrajectory::setRobotTrajectoryMsg(const moveit::core::RobotState& reference_state,
357  const trajectory_msgs::msg::JointTrajectory& trajectory)
358 {
359  // make a copy just in case the next clear() removes the memory for the reference passed in
360  const moveit::core::RobotState& copy = reference_state;
361  clear();
362  std::size_t state_count = trajectory.points.size();
363  rclcpp::Time last_time_stamp = trajectory.header.stamp;
364  rclcpp::Time this_time_stamp = last_time_stamp;
365 
366  for (std::size_t i = 0; i < state_count; ++i)
367  {
368  this_time_stamp = rclcpp::Time(trajectory.header.stamp) + trajectory.points[i].time_from_start;
369  moveit::core::RobotStatePtr st(new moveit::core::RobotState(copy));
370  st->setVariablePositions(trajectory.joint_names, trajectory.points[i].positions);
371  if (!trajectory.points[i].velocities.empty())
372  st->setVariableVelocities(trajectory.joint_names, trajectory.points[i].velocities);
373  if (!trajectory.points[i].accelerations.empty())
374  st->setVariableAccelerations(trajectory.joint_names, trajectory.points[i].accelerations);
375  if (!trajectory.points[i].effort.empty())
376  st->setVariableEffort(trajectory.joint_names, trajectory.points[i].effort);
377  addSuffixWayPoint(st, (this_time_stamp - last_time_stamp).seconds());
378  last_time_stamp = this_time_stamp;
379  }
380 }

382 void RobotTrajectory::setRobotTrajectoryMsg(const moveit::core::RobotState& reference_state,
383  const moveit_msgs::msg::RobotTrajectory& trajectory)
384 {
385  // make a copy just in case the next clear() removes the memory for the reference passed in
386  const moveit::core::RobotState& copy = reference_state;
387  clear();
388 
389  std::size_t state_count =
390  std::max(trajectory.joint_trajectory.points.size(), trajectory.multi_dof_joint_trajectory.points.size());
391  rclcpp::Time last_time_stamp = trajectory.joint_trajectory.points.empty() ?
392  trajectory.multi_dof_joint_trajectory.header.stamp :
393  trajectory.joint_trajectory.header.stamp;
394  rclcpp::Time this_time_stamp = last_time_stamp;
395 
396  for (std::size_t i = 0; i < state_count; ++i)
397  {
398  moveit::core::RobotStatePtr st(new moveit::core::RobotState(copy));
399  if (trajectory.joint_trajectory.points.size() > i)
400  {
401  st->setVariablePositions(trajectory.joint_trajectory.joint_names, trajectory.joint_trajectory.points[i].positions);
402  if (!trajectory.joint_trajectory.points[i].velocities.empty())
403  st->setVariableVelocities(trajectory.joint_trajectory.joint_names,
404  trajectory.joint_trajectory.points[i].velocities);
405  if (!trajectory.joint_trajectory.points[i].accelerations.empty())
406  st->setVariableAccelerations(trajectory.joint_trajectory.joint_names,
407  trajectory.joint_trajectory.points[i].accelerations);
408  if (!trajectory.joint_trajectory.points[i].effort.empty())
409  st->setVariableEffort(trajectory.joint_trajectory.joint_names, trajectory.joint_trajectory.points[i].effort);
410  this_time_stamp = rclcpp::Time(trajectory.joint_trajectory.header.stamp) +
411  trajectory.joint_trajectory.points[i].time_from_start;
412  }
413  if (trajectory.multi_dof_joint_trajectory.points.size() > i)
414  {
415  for (std::size_t j = 0; j < trajectory.multi_dof_joint_trajectory.joint_names.size(); ++j)
416  {
417  Eigen::Isometry3d t = tf2::transformToEigen(trajectory.multi_dof_joint_trajectory.points[i].transforms[j]);
418  st->setJointPositions(trajectory.multi_dof_joint_trajectory.joint_names[j], t);
419  }
420  this_time_stamp = rclcpp::Time(trajectory.multi_dof_joint_trajectory.header.stamp) +
421  trajectory.multi_dof_joint_trajectory.points[i].time_from_start;
422  }
423 
424  addSuffixWayPoint(st, (this_time_stamp - last_time_stamp).seconds());
425  last_time_stamp = this_time_stamp;
426  }
427 }

438 void RobotTrajectory::findWayPointIndicesForDurationAfterStart(const double& duration, int& before, int& after,
439  double& blend) const
440 {
441  if (duration < 0.0)
442  {
443  before = 0;
444  after = 0;
445  blend = 0;
446  return;
447  }
448 
449  // Find indicies
450  std::size_t index = 0, num_points = waypoints_.size();
451  double running_duration = 0.0;
452  for (; index < num_points; ++index)
453  {
454  running_duration += duration_from_previous_[index];
455  if (running_duration >= duration)
456  break;
457  }
458  before = std::max<int>(index - 1, 0);
459  after = std::min<int>(index, num_points - 1);
460 
461  // Compute duration blend
462  double before_time = running_duration - duration_from_previous_[index];
463  if (after == before)
464  blend = 1.0;
465  else
466  blend = (duration - before_time) / duration_from_previous_[index];
467 }
