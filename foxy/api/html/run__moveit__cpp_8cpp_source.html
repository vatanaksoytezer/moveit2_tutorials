49 class MoveItCppDemo
50 {
51 public:
52  MoveItCppDemo(const rclcpp::Node::SharedPtr& node)
53  : node_(node)
54  , robot_state_publisher_(node_->create_publisher<moveit_msgs::msg::DisplayRobotState>("display_robot_state", 1))
55  {
56  }
57 
58  void run()
59  {
60  RCLCPP_INFO(LOGGER, "Initialize MoveItCpp");
61  moveit_cpp_ = std::make_shared<moveit_cpp::MoveItCpp>(node_);
62  moveit_cpp_->getPlanningSceneMonitor()->providePlanningSceneService(); // let RViz display query PlanningScene
63  moveit_cpp_->getPlanningSceneMonitor()->setPlanningScenePublishingFrequency(100);
64 
65  RCLCPP_INFO(LOGGER, "Initialize PlanningComponent");
66  moveit_cpp::PlanningComponent arm("panda_arm", moveit_cpp_);
67 
68  // A little delay before running the plan
69  rclcpp::sleep_for(std::chrono::seconds(3));
70 
71  // Create collision object, planning shouldn't be too easy
72  moveit_msgs::msg::CollisionObject collision_object;
73  collision_object.header.frame_id = "panda_link0";
74  collision_object.id = "box";
75 
76  shape_msgs::msg::SolidPrimitive box;
77  box.type = box.BOX;
78  box.dimensions = { 0.1, 0.4, 0.1 };
79 
80  geometry_msgs::msg::Pose box_pose;
81  box_pose.position.x = 0.4;
82  box_pose.position.y = 0.0;
83  box_pose.position.z = 1.0;
84 
85  collision_object.primitives.push_back(box);
86  collision_object.primitive_poses.push_back(box_pose);
87  collision_object.operation = collision_object.ADD;
88 
89  // Add object to planning scene
90  { // Lock PlanningScene
91  planning_scene_monitor::LockedPlanningSceneRW scene(moveit_cpp_->getPlanningSceneMonitor());
92  scene->processCollisionObjectMsg(collision_object);
93  } // Unlock PlanningScene
94 
95  // Set joint state goal
96  RCLCPP_INFO(LOGGER, "Set goal");
97  arm.setGoal("extended");
98 
99  // Run actual plan
100  RCLCPP_INFO(LOGGER, "Plan to goal");
101  const auto plan_solution = arm.plan();
102  if (plan_solution)
103  {
104  RCLCPP_INFO(LOGGER, "arm.execute()");
105  arm.execute();
106  }
107  }
108 
109 private:
110  rclcpp::Node::SharedPtr node_;
111  rclcpp::Publisher<moveit_msgs::msg::DisplayRobotState>::SharedPtr robot_state_publisher_;
112  moveit_cpp::MoveItCppPtr moveit_cpp_;
113 };

115 int main(int argc, char** argv)
116 {
117  RCLCPP_INFO(LOGGER, "Initialize node");
118  rclcpp::init(argc, argv);
119  rclcpp::NodeOptions node_options;
120  // This enables loading undeclared parameters
121  // best practice would be to declare parameters in the corresponding classes
122  // and provide descriptions about expected use
123  node_options.automatically_declare_parameters_from_overrides(true);
124  rclcpp::Node::SharedPtr node = rclcpp::Node::make_shared("run_moveit_cpp", "", node_options);
125 
126  MoveItCppDemo demo(node);
127  std::thread run_demo([&demo]() {
128  // Let RViz initialize before running demo
129  // TODO(henningkayser): use lifecycle events to launch node
130  rclcpp::sleep_for(std::chrono::seconds(5));
131  demo.run();
132  });
133 
134  rclcpp::spin(node);
135  run_demo.join();
136 
137  return 0;
