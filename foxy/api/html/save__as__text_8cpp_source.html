55 void collectLinkConstraints(const moveit_msgs::msg::Constraints& constraints, LinkConstraintMap& lcmap)
56 {
57  for (const moveit_msgs::msg::PositionConstraint& position_constraint : constraints.position_constraints)
58  {
59  LinkConstraintPair lcp;
60  const moveit_msgs::msg::PositionConstraint& pc = position_constraint;
61  lcp.first = pc.constraint_region.primitive_poses[0].position;
62  lcmap[position_constraint.link_name] = lcp;
63  }
64 
65  for (const moveit_msgs::msg::OrientationConstraint& orientation_constraint : constraints.orientation_constraints)
66  {
67  if (lcmap.count(orientation_constraint.link_name))
68  {
69  lcmap[orientation_constraint.link_name].second = orientation_constraint.orientation;
70  }
71  else
72  {
73  RCLCPP_WARN(LOGGER, "Orientation constraint for %s has no matching position constraint",
74  orientation_constraint.link_name.c_str());
75  }
76  }
77 }

111  moveit_warehouse::PlanningSceneStorage pss(conn);

113  moveit_warehouse::ConstraintsStorage cs(conn);

115  std::vector<std::string> scene_names;

118  for (const std::string& scene_name : scene_names)

121  if (pss.getPlanningScene(pswm, scene_name))
122  {

128

130  moveit::core::RobotModelConstPtr km = psm.getRobotModel();
131  // Get start states for scene

133  rsregex << ".*" << scene_name << ".*";
134  rss.getKnownRobotStates(rsregex.str(), robot_state_names);
135 
136  // Get goal constraints for scene
137  std::vector<std::string> constraint_names;
138 
139  std::stringstream csregex;
140  csregex << ".*" << scene_name << ".*";
141  cs.getKnownConstraints(csregex.str(), constraint_names);
142 
143  if (!(robot_state_names.empty() && constraint_names.empty()))
144  {
145  std::ofstream qfout((scene_name + ".queries").c_str());
146  qfout << scene_name << std::endl;
147  if (!robot_state_names.empty())
148  {
149  qfout << "start" << std::endl;

151  for (const std::string& robot_state_name : robot_state_names)
152  {

154  qfout << robot_state_name << std::endl;
155  moveit_warehouse::RobotStateWithMetadata robot_state;
156  rss.getRobotState(robot_state, robot_state_name);
157  moveit::core::RobotState ks(km);
158  moveit::core::robotStateMsgToRobotState(*robot_state, ks, false);
159  ks.printStateInfo(qfout);
160  qfout << "." << std::endl;
161  }
162  }
163 
164  if (!constraint_names.empty())
165  {
166  qfout << "goal" << std::endl;
167  qfout << constraint_names.size() << std::endl;
168  for (const std::string& constraint_name : constraint_names)
169  {
170  RCLCPP_INFO(LOGGER, "Saving goal %s for scene %s", constraint_name.c_str(), scene_name.c_str());
171  qfout << "link_constraint" << std::endl;
172  qfout << constraint_name << std::endl;
173  moveit_warehouse::ConstraintsWithMetadata constraints;
174  cs.getConstraints(constraints, constraint_name);
175 
176  LinkConstraintMap lcmap;
177  collectLinkConstraints(*constraints, lcmap);
178  for (std::pair<const std::string, LinkConstraintPair>& iter : lcmap)
179  {
180  std::string link_name = iter.first;
181  LinkConstraintPair lcp = iter.second;
182  qfout << link_name << std::endl;
183  qfout << "xyz " << lcp.first.x << " " << lcp.first.y << " " << lcp.first.z << std::endl;
184  Eigen::Quaterniond orientation(lcp.second.w, lcp.second.x, lcp.second.y, lcp.second.z);
185  Eigen::Vector3d rpy = orientation.matrix().eulerAngles(0, 1, 2);
186  qfout << "rpy " << rpy[0] << " " << rpy[1] << " " << rpy[2] << std::endl;
187  }
188  qfout << "." << std::endl;
189  }
190  }
191  qfout.close();
192  }
193  }
194  }
195 
196  RCLCPP_INFO(LOGGER, "Done.");
197  rclcpp::spin(node);
198  return 0;
