182 void ServoCalcs::start()
183 {
184  // Stop the thread if we are currently running
185  stop();
186 
187  // Set up the "last" published message, in case we need to send it first
188  auto initial_joint_trajectory = std::make_unique<trajectory_msgs::msg::JointTrajectory>();
189  initial_joint_trajectory->header.stamp = node_->now();
190  initial_joint_trajectory->header.frame_id = parameters_->planning_frame;
191  initial_joint_trajectory->joint_names = internal_joint_state_.name;
192  trajectory_msgs::msg::JointTrajectoryPoint point;
193  point.time_from_start = rclcpp::Duration::from_seconds(parameters_->publish_period);
194  if (parameters_->publish_joint_positions)
195  planning_scene_monitor_->getStateMonitor()->getCurrentState()->copyJointGroupPositions(joint_model_group_,
196  point.positions);
197  if (parameters_->publish_joint_velocities)
198  {
199  std::vector<double> velocity(num_joints_);
200  point.velocities = velocity;
201  }
202  if (parameters_->publish_joint_accelerations)
203  {
204  // I do not know of a robot that takes acceleration commands.
205  // However, some controllers check that this data is non-empty.
206  // Send all zeros, for now.
207  point.accelerations.resize(num_joints_);
208  }
209  initial_joint_trajectory->points.push_back(point);
210  last_sent_command_ = std::move(initial_joint_trajectory);
211 
212  current_state_ = planning_scene_monitor_->getStateMonitor()->getCurrentState();
213  tf_moveit_to_ee_frame_ = current_state_->getGlobalLinkTransform(parameters_->planning_frame).inverse() *
214  current_state_->getGlobalLinkTransform(parameters_->ee_frame_name);
215  tf_moveit_to_robot_cmd_frame_ = current_state_->getGlobalLinkTransform(parameters_->planning_frame).inverse() *
216  current_state_->getGlobalLinkTransform(robot_link_command_frame_);
217 
218  stop_requested_ = false;
219  thread_ = std::thread([this] { mainCalcLoop(); });
220  new_input_cmd_ = false;
221 }

244 void ServoCalcs::mainCalcLoop()
245 {
246  rclcpp::Rate rate(1.0 / parameters_->publish_period);
247 
248  while (rclcpp::ok() && !stop_requested_)
249  {
250  // lock the input state mutex
251  std::unique_lock<std::mutex> main_loop_lock(main_loop_mutex_);
252 
253  // low latency mode -- begin calculations as soon as a new command is received.
254  if (parameters_->low_latency_mode)
255  {
256  input_cv_.wait(main_loop_lock, [this] { return (new_input_cmd_ || stop_requested_); });
257  }
258 
259  // reset new_input_cmd_ flag
260  new_input_cmd_ = false;
261 
262  // run servo calcs
263  const auto start_time = node_->now();
264  calculateSingleIteration();
265  const auto run_duration = node_->now() - start_time;
266 
267  // Log warning when the run duration was longer than the period
268  if (run_duration.seconds() > parameters_->publish_period)
269  {
270  rclcpp::Clock& clock = *node_->get_clock();
271  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD,
272  "run_duration: " << run_duration.seconds() << " (" << parameters_->publish_period
273  << ")");
274  }
275 
276  // normal mode, unlock input mutex and wait for the period of the loop
277  if (!parameters_->low_latency_mode)
278  {
279  main_loop_lock.unlock();
280  rate.sleep();
281  }
282  }

285 void ServoCalcs::calculateSingleIteration()
286 {
287  // Publish status each loop iteration
288  auto status_msg = std::make_unique<std_msgs::msg::Int8>();
289  status_msg->data = static_cast<int8_t>(status_);
290  status_pub_->publish(std::move(status_msg));
291 
292  // After we publish, status, reset it back to no warnings
293  status_ = StatusCode::NO_WARNING;
294 
295  // Always update the joints and end-effector transform for 2 reasons:
296  // 1) in case the getCommandFrameTransform() method is being used
297  // 2) so the low-pass filters are up to date and don't cause a jump
298  updateJoints();
299 
300  // Calculate and publish worst stop time for collision checker
301  if (parameters_->check_collisions && parameters_->collision_check_type == "stop_distance")
302  calculateWorstCaseStopTime();
303 
304  // Update from latest state
305  current_state_ = planning_scene_monitor_->getStateMonitor()->getCurrentState();
306 
307  if (latest_twist_stamped_)
308  twist_stamped_cmd_ = *latest_twist_stamped_;
309  if (latest_joint_cmd_)
310  joint_servo_cmd_ = *latest_joint_cmd_;
311 
312  // Check for stale cmds
313  twist_command_is_stale_ = ((node_->now() - latest_twist_command_stamp_) >=
314  rclcpp::Duration::from_seconds(parameters_->incoming_command_timeout));
315  joint_command_is_stale_ = ((node_->now() - latest_joint_command_stamp_) >=
316  rclcpp::Duration::from_seconds(parameters_->incoming_command_timeout));
317 
318  have_nonzero_twist_stamped_ = latest_nonzero_twist_stamped_;
319  have_nonzero_joint_command_ = latest_nonzero_joint_cmd_;
320 
321  // Get the transform from MoveIt planning frame to servoing command frame
322  // Calculate this transform to ensure it is available via C++ API
323  // We solve (planning_frame -> base -> robot_link_command_frame)
324  // by computing (base->planning_frame)^-1 * (base->robot_link_command_frame)
325  tf_moveit_to_robot_cmd_frame_ = current_state_->getGlobalLinkTransform(parameters_->planning_frame).inverse() *
326  current_state_->getGlobalLinkTransform(robot_link_command_frame_);
327 
328  // Calculate the transform from MoveIt planning frame to End Effector frame
329  // Calculate this transform to ensure it is available via C++ API
330  tf_moveit_to_ee_frame_ = current_state_->getGlobalLinkTransform(parameters_->planning_frame).inverse() *
331  current_state_->getGlobalLinkTransform(parameters_->ee_frame_name);
332 
333  have_nonzero_command_ = have_nonzero_twist_stamped_ || have_nonzero_joint_command_;
334 
335  // Don't end this function without updating the filters
336  updated_filters_ = false;
337 
338  // If paused or while waiting for initial servo commands, just keep the low-pass filters up to date with current
339  // joints so a jump doesn't occur when restarting
340  if (wait_for_servo_commands_ || paused_)
341  {
342  resetLowPassFilters(original_joint_state_);
343 
344  // Check if there are any new commands with valid timestamp
345  wait_for_servo_commands_ =
346  twist_stamped_cmd_.header.stamp == rclcpp::Time(0.) && joint_servo_cmd_.header.stamp == rclcpp::Time(0.);
347 
348  // Early exit
349  return;
350  }
351 
352  // If not waiting for initial command, and not paused.
353  // Do servoing calculations only if the robot should move, for efficiency
354  // Create new outgoing joint trajectory command message
355  auto joint_trajectory = std::make_unique<trajectory_msgs::msg::JointTrajectory>();
356 
357  // Prioritize cartesian servoing above joint servoing
358  // Only run commands if not stale and nonzero
359  if (have_nonzero_twist_stamped_ && !twist_command_is_stale_)
360  {
361  if (!cartesianServoCalcs(twist_stamped_cmd_, *joint_trajectory))
362  {
363  resetLowPassFilters(original_joint_state_);
364  return;
365  }
366  }
367  else if (have_nonzero_joint_command_ && !joint_command_is_stale_)
368  {
369  if (!jointServoCalcs(joint_servo_cmd_, *joint_trajectory))
370  {
371  resetLowPassFilters(original_joint_state_);
372  return;
373  }
374  }
375  else
376  {
377  // Joint trajectory is not populated with anything, so set it to the last positions and 0 velocity
378  *joint_trajectory = *last_sent_command_;
379  for (auto& point : joint_trajectory->points)
380  {
381  point.velocities.assign(point.velocities.size(), 0);
382  }
383  }
384 
385  // Print a warning to the user if both are stale
386  if (twist_command_is_stale_ && joint_command_is_stale_)
387  {
388  rclcpp::Clock& clock = *node_->get_clock();
389  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD,
390  "Stale command. Try a larger 'incoming_command_timeout' parameter?");
391  }
392 
393  // If we should halt
394  if (!have_nonzero_command_)
395  {
396  suddenHalt(*joint_trajectory);
397  have_nonzero_twist_stamped_ = false;
398  have_nonzero_joint_command_ = false;
399  }
400 
401  // Skip the servoing publication if all inputs have been zero for several cycles in a row.
402  // num_outgoing_halt_msgs_to_publish == 0 signifies that we should keep republishing forever.
403  if (!have_nonzero_command_ && (parameters_->num_outgoing_halt_msgs_to_publish != 0) &&
404  (zero_velocity_count_ > parameters_->num_outgoing_halt_msgs_to_publish))
405  {
406  ok_to_publish_ = false;
407  rclcpp::Clock& clock = *node_->get_clock();
408  RCLCPP_DEBUG_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD, "All-zero command. Doing nothing.");
409  }
410  else
411  {
412  ok_to_publish_ = true;
413  }
414 
415  // Store last zero-velocity message flag to prevent superfluous warnings.
416  // Cartesian and joint commands must both be zero.
417  if (!have_nonzero_command_)
418  {
419  // Avoid overflow
420  if (zero_velocity_count_ < std::numeric_limits<int>::max())
421  ++zero_velocity_count_;
422  }
423  else
424  {
425  zero_velocity_count_ = 0;
426  }
427 
428  if (ok_to_publish_ && !paused_)
429  {
430  // Put the outgoing msg in the right format
431  // (trajectory_msgs/JointTrajectory or std_msgs/Float64MultiArray).
432  if (parameters_->command_out_type == "trajectory_msgs/JointTrajectory")
433  {
434  // When a joint_trajectory_controller receives a new command, a stamp of 0 indicates "begin immediately"
435  // See http://wiki.ros.org/joint_trajectory_controller#Trajectory_replacement
436  joint_trajectory->header.stamp = rclcpp::Time(0);
437  *last_sent_command_ = *joint_trajectory;
438  trajectory_outgoing_cmd_pub_->publish(std::move(joint_trajectory));
439  }
440  else if (parameters_->command_out_type == "std_msgs/Float64MultiArray")
441  {
442  auto joints = std::make_unique<std_msgs::msg::Float64MultiArray>();
443  if (parameters_->publish_joint_positions && !joint_trajectory->points.empty())
444  joints->data = joint_trajectory->points[0].positions;
445  else if (parameters_->publish_joint_velocities && !joint_trajectory->points.empty())
446  joints->data = joint_trajectory->points[0].velocities;
447  *last_sent_command_ = *joint_trajectory;
448  multiarray_outgoing_cmd_pub_->publish(std::move(joints));
449  }
450  }
451 
452  // Update the filters if we haven't yet
453  if (!updated_filters_)
454  resetLowPassFilters(original_joint_state_);
455 }

467 // Perform the servoing calculations
468 bool ServoCalcs::cartesianServoCalcs(geometry_msgs::msg::TwistStamped& cmd,
469  trajectory_msgs::msg::JointTrajectory& joint_trajectory)
470 {
471  // Check for nan's in the incoming command
472  if (!checkValidCommand(cmd))
473  return false;
474 
475  // Set uncontrolled dimensions to 0 in command frame
476  enforceControlDimensions(cmd);
477 
478  // Transform the command to the MoveGroup planning frame
479  if (cmd.header.frame_id != parameters_->planning_frame)
480  {
481  Eigen::Vector3d translation_vector(cmd.twist.linear.x, cmd.twist.linear.y, cmd.twist.linear.z);
482  Eigen::Vector3d angular_vector(cmd.twist.angular.x, cmd.twist.angular.y, cmd.twist.angular.z);
483 
484  // If the incoming frame is empty or is the command frame, we use the previously calculated tf
485  if (cmd.header.frame_id.empty() || cmd.header.frame_id == robot_link_command_frame_)
486  {
487  translation_vector = tf_moveit_to_robot_cmd_frame_.linear() * translation_vector;
488  angular_vector = tf_moveit_to_robot_cmd_frame_.linear() * angular_vector;
489  }
490  else if (cmd.header.frame_id == parameters_->ee_frame_name)
491  {
492  // If the frame is the EE frame, we already have that transform as well
493  translation_vector = tf_moveit_to_ee_frame_.linear() * translation_vector;
494  angular_vector = tf_moveit_to_ee_frame_.linear() * angular_vector;
495  }
496  else
497  {
498  // We solve (planning_frame -> base -> cmd.header.frame_id)
499  // by computing (base->planning_frame)^-1 * (base->cmd.header.frame_id)
500  const auto tf_moveit_to_incoming_cmd_frame =
501  current_state_->getGlobalLinkTransform(parameters_->planning_frame).inverse() *
502  current_state_->getGlobalLinkTransform(cmd.header.frame_id);
503 
504  translation_vector = tf_moveit_to_incoming_cmd_frame.linear() * translation_vector;
505  angular_vector = tf_moveit_to_incoming_cmd_frame.linear() * angular_vector;
506  }
507 
508  // Put these components back into a TwistStamped
509  cmd.header.frame_id = parameters_->planning_frame;
510  cmd.twist.linear.x = translation_vector(0);
511  cmd.twist.linear.y = translation_vector(1);
512  cmd.twist.linear.z = translation_vector(2);
513  cmd.twist.angular.x = angular_vector(0);
514  cmd.twist.angular.y = angular_vector(1);
515  cmd.twist.angular.z = angular_vector(2);
516  }
517 
518  Eigen::VectorXd delta_x = scaleCartesianCommand(cmd);
519 
520  // Convert from cartesian commands to joint commands
521  Eigen::MatrixXd jacobian = current_state_->getJacobian(joint_model_group_);
522 
523  removeDriftDimensions(jacobian, delta_x);
524 
525  Eigen::JacobiSVD<Eigen::MatrixXd> svd =
526  Eigen::JacobiSVD<Eigen::MatrixXd>(jacobian, Eigen::ComputeThinU | Eigen::ComputeThinV);
527  Eigen::MatrixXd matrix_s = svd.singularValues().asDiagonal();
528  Eigen::MatrixXd pseudo_inverse = svd.matrixV() * matrix_s.inverse() * svd.matrixU().transpose();
529 
530  delta_theta_ = pseudo_inverse * delta_x;
531  delta_theta_ *= velocityScalingFactorForSingularity(delta_x, svd, pseudo_inverse);
532 
533  return internalServoUpdate(delta_theta_, joint_trajectory, ServoType::CARTESIAN_SPACE);
534 }

550 bool ServoCalcs::internalServoUpdate(Eigen::ArrayXd& delta_theta,
551  trajectory_msgs::msg::JointTrajectory& joint_trajectory,
552  const ServoType servo_type)
553 {
554  // Set internal joint state from original
555  internal_joint_state_ = original_joint_state_;
556 
557  // Enforce SRDF Velocity, Acceleration limits
558  delta_theta = enforceVelocityLimits(joint_model_group_, parameters_->publish_period, delta_theta);
559 
560  // Apply collision scaling
561  double collision_scale = collision_velocity_scale_;
562  if (collision_scale > 0 && collision_scale < 1)
563  {
564  status_ = StatusCode::DECELERATE_FOR_COLLISION;
565  rclcpp::Clock& clock = *node_->get_clock();
566  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD, SERVO_STATUS_CODE_MAP.at(status_));
567  }
568  else if (collision_scale == 0)
569  {
570  status_ = StatusCode::HALT_FOR_COLLISION;
571  rclcpp::Clock& clock = *node_->get_clock();
572  RCLCPP_ERROR_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD, "Halting for collision!");
573  }
574  delta_theta *= collision_scale;
575 
576  // Loop thru joints and update them, calculate velocities, and filter
577  if (!applyJointUpdate(delta_theta, internal_joint_state_, prev_joint_velocity_))
578  return false;
579 
580  // Mark the lowpass filters as updated for this cycle
581  updated_filters_ = true;
582 
583  // Enforce SRDF position limits, might halt if needed, set prev_vel to 0
584  const auto joints_to_halt = enforcePositionLimits(internal_joint_state_);
585  if (!joints_to_halt.empty())
586  {
587  status_ = StatusCode::JOINT_BOUND;
588  if ((servo_type == ServoType::JOINT_SPACE && !parameters_->halt_all_joints_in_joint_mode) ||
589  (servo_type == ServoType::CARTESIAN_SPACE && !parameters_->halt_all_joints_in_cartesian_mode))
590  {
591  suddenHalt(internal_joint_state_, joints_to_halt);
592  prev_joint_velocity_ =
593  Eigen::ArrayXd::Map(internal_joint_state_.velocity.data(), internal_joint_state_.velocity.size());
594  }
595  else
596  {
597  suddenHalt(internal_joint_state_, joint_model_group_->getActiveJointModels());
598  prev_joint_velocity_.setZero();
599  }
600  }
601 
602  // compose outgoing message
603  composeJointTrajMessage(internal_joint_state_, joint_trajectory);
604 
605  // Modify the output message if we are using gazebo
606  if (parameters_->use_gazebo)
607  {
608  insertRedundantPointsIntoTrajectory(joint_trajectory, gazebo_redundant_message_count_);
609  }
610 
611  return true;
612 }

614 bool ServoCalcs::applyJointUpdate(const Eigen::ArrayXd& delta_theta, sensor_msgs::msg::JointState& joint_state,
615  Eigen::ArrayXd& previous_vel)
616 {
617  // All the sizes must match
618  if (joint_state.position.size() != static_cast<std::size_t>(delta_theta.size()) ||
619  joint_state.velocity.size() != joint_state.position.size() ||
620  static_cast<std::size_t>(previous_vel.size()) != joint_state.position.size())
621  {
622  rclcpp::Clock& clock = *node_->get_clock();
623  RCLCPP_ERROR_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD,
624  "Lengths of output and increments do not match.");
625  return false;
626  }
627 
628  for (std::size_t i = 0; i < joint_state.position.size(); ++i)
629  {
630  // Increment joint
631  joint_state.position[i] += delta_theta[i];
632 
633  // Lowpass filter position
634  joint_state.position[i] = position_filters_[i].filter(joint_state.position[i]);
635 
636  // Calculate joint velocity
637  joint_state.velocity[i] = delta_theta[i] / parameters_->publish_period;
638 
639  // Save this velocity for future accel calculations
640  previous_vel[i] = joint_state.velocity[i];
641  }
642  return true;
643 }

646 // time stamp is in the past when it reaches the client. Needed for gazebo simulation.

657  {
658  point.time_from_start = rclcpp::Duration::from_seconds((i + 1) * parameters_->publish_period);
659  joint_trajectory.points[i] = point;
660  }

699 // Possibly calculate a velocity scaling factor, due to proximity of singularity and direction of motion
700 double ServoCalcs::velocityScalingFactorForSingularity(const Eigen::VectorXd& commanded_velocity,
701  const Eigen::JacobiSVD<Eigen::MatrixXd>& svd,
702  const Eigen::MatrixXd& pseudo_inverse)
703 {
704  double velocity_scale = 1;
705  std::size_t num_dimensions = commanded_velocity.size();
706 
707  // Find the direction away from nearest singularity.
708  // The last column of U from the SVD of the Jacobian points directly toward or away from the singularity.
709  // The sign can flip at any time, so we have to do some extra checking.
710  // Look ahead to see if the Jacobian's condition will decrease.
711  Eigen::VectorXd vector_toward_singularity = svd.matrixU().col(num_dimensions - 1);
712 
713  double ini_condition = svd.singularValues()(0) / svd.singularValues()(svd.singularValues().size() - 1);
714 
715  // This singular vector tends to flip direction unpredictably. See R. Bro,
716  // "Resolving the Sign Ambiguity in the Singular Value Decomposition".
717  // Look ahead to see if the Jacobian's condition will decrease in this
718  // direction. Start with a scaled version of the singular vector
719  Eigen::VectorXd delta_x(num_dimensions);
720  double scale = 100;
721  delta_x = vector_toward_singularity / scale;
722 
723  // Calculate a small change in joints
724  Eigen::VectorXd new_theta;
725  current_state_->copyJointGroupPositions(joint_model_group_, new_theta);
726  new_theta += pseudo_inverse * delta_x;
727  current_state_->setJointGroupPositions(joint_model_group_, new_theta);
728  Eigen::MatrixXd new_jacobian = current_state_->getJacobian(joint_model_group_);
729 
730  Eigen::JacobiSVD<Eigen::MatrixXd> new_svd(new_jacobian);
731  double new_condition = new_svd.singularValues()(0) / new_svd.singularValues()(new_svd.singularValues().size() - 1);
732  // If new_condition < ini_condition, the singular vector does point towards a
733  // singularity. Otherwise, flip its direction.
734  if (ini_condition >= new_condition)
735  {
736  vector_toward_singularity *= -1;
737  }
738 
739  // If this dot product is positive, we're moving toward singularity ==> decelerate
740  double dot = vector_toward_singularity.dot(commanded_velocity);
741  if (dot > 0)
742  {
743  // Ramp velocity down linearly when the Jacobian condition is between lower_singularity_threshold and
744  // hard_stop_singularity_threshold, and we're moving towards the singularity
745  if ((ini_condition > parameters_->lower_singularity_threshold) &&
746  (ini_condition < parameters_->hard_stop_singularity_threshold))
747  {
748  velocity_scale =
749  1. - (ini_condition - parameters_->lower_singularity_threshold) /
750  (parameters_->hard_stop_singularity_threshold - parameters_->lower_singularity_threshold);
751  status_ = StatusCode::DECELERATE_FOR_SINGULARITY;
752  rclcpp::Clock& clock = *node_->get_clock();
753  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD, SERVO_STATUS_CODE_MAP.at(status_));
754  }
755 
756  // Very close to singularity, so halt.
757  else if (ini_condition > parameters_->hard_stop_singularity_threshold)
758  {
759  velocity_scale = 0;
760  status_ = StatusCode::HALT_FOR_SINGULARITY;
761  rclcpp::Clock& clock = *node_->get_clock();
762  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD, SERVO_STATUS_CODE_MAP.at(status_));
763  }
764  }
765 
766  return velocity_scale;
767 }

769 std::vector<const moveit::core::JointModel*>
770 ServoCalcs::enforcePositionLimits(sensor_msgs::msg::JointState& joint_state) const
771 {
772  // Halt if we're past a joint margin and joint velocity is moving even farther past
773  double joint_angle = 0;
774  std::vector<const moveit::core::JointModel*> joints_to_halt;
775  for (auto joint : joint_model_group_->getActiveJointModels())
776  {
777  for (std::size_t c = 0; c < joint_state.name.size(); ++c)
778  {
779  // Use the most recent robot joint state
780  if (joint_state.name[c] == joint->getName())
781  {
782  joint_angle = joint_state.position.at(c);
783  break;
784  }
785  }
786 
787  if (!joint->satisfiesPositionBounds(&joint_angle, -parameters_->joint_limit_margin))
788  {
789  const std::vector<moveit_msgs::msg::JointLimits> limits = joint->getVariableBoundsMsg();
790 
791  // Joint limits are not defined for some joints. Skip them.
792  if (!limits.empty())
793  {
794  // Check if pending velocity command is moving in the right direction
795  auto joint_itr = std::find(joint_state.name.begin(), joint_state.name.end(), joint->getName());
796  auto joint_idx = std::distance(joint_state.name.begin(), joint_itr);
797 
798  if ((joint_state.velocity.at(joint_idx) < 0 &&
799  (joint_angle < (limits[0].min_position + parameters_->joint_limit_margin))) ||
800  (joint_state.velocity.at(joint_idx) > 0 &&
801  (joint_angle > (limits[0].max_position - parameters_->joint_limit_margin))))
802  {
803  joints_to_halt.push_back(joint);
804  }
805  }
806  }
807  }
808  if (!joints_to_halt.empty())
809  {
810  std::ostringstream joints_names;
811  std::transform(joints_to_halt.cbegin(), std::prev(joints_to_halt.cend()),
812  std::ostream_iterator<std::string>(joints_names, ", "),
813  [](const auto& joint) { return joint->getName(); });
814  joints_names << joints_to_halt.back()->getName();
815  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, *node_->get_clock(), ROS_LOG_THROTTLE_PERIOD,
816  node_->get_name()
817  << " " << joints_names.str() << " close to a position limit. Halting.");
818  }
819  return joints_to_halt;
820 }

823 // Is handled differently for position vs. velocity control.
824 void ServoCalcs::suddenHalt(trajectory_msgs::msg::JointTrajectory& joint_trajectory) const
825 {
826  // Prepare the joint trajectory message to stop the robot
827  joint_trajectory.points.clear();

833  // arm has to reach the first trajectory point the moment execution begins. To prevent errors about points

849  if (parameters_->publish_joint_positions)
850  point.positions[i] = original_joint_state_.position[i];
851 
852  // For velocity-controlled robots, stop
853  if (parameters_->publish_joint_velocities)
854  point.velocities[i] = 0;
855  }

886 // Calculate worst case joint stop time, for collision checking
887 void ServoCalcs::calculateWorstCaseStopTime()
888 {
889  std::string joint_name = "";
890  moveit::core::JointModel::Bounds kinematic_bounds;
891  double accel_limit = 0;
892  double joint_velocity = 0;
893  double worst_case_stop_time = 0;
894  for (size_t jt_state_idx = 0; jt_state_idx < internal_joint_state_.velocity.size(); ++jt_state_idx)
895  {
896  joint_name = internal_joint_state_.name[jt_state_idx];
897 
898  // Get acceleration limit for this joint
899  for (auto joint_model : joint_model_group_->getActiveJointModels())
900  {
901  if (joint_model->getName() == joint_name)
902  {
903  kinematic_bounds = joint_model->getVariableBounds();
904  // Some joints do not have acceleration limits
905  if (kinematic_bounds[0].acceleration_bounded_)
906  {
907  // Be conservative when calculating overall acceleration limit from min and max limits
908  accel_limit =
909  std::min(fabs(kinematic_bounds[0].min_acceleration_), fabs(kinematic_bounds[0].max_acceleration_));
910  }
911  else
912  {
913  rclcpp::Clock& clock = *node_->get_clock();
914  RCLCPP_ERROR_STREAM_THROTTLE(LOGGER, clock, ROS_LOG_THROTTLE_PERIOD,
915  "An acceleration limit is not defined for this joint; minimum stop distance "
916  "should not be used for collision checking");
917 
918  // TODO(adamp): figure out what to do here. We definitely don't want to allow 'stop_distance' collision
919  // checking with no acceleration limits defined.
920  }
921  break;
922  }
923  }
924 
925  // Get the current joint velocity
926  joint_velocity = internal_joint_state_.velocity[jt_state_idx];
927 
928  // Calculate worst case stop time
929  worst_case_stop_time = std::max(worst_case_stop_time, fabs(joint_velocity / accel_limit));
930  }
931 
932  // publish message
933  {
934  auto msg = std::make_unique<std_msgs::msg::Float64>();
935  msg->data = worst_case_stop_time;
936  worst_case_stop_time_pub_->publish(std::move(msg));
937  }
