402  bool start()
403  {
404  auto time_start = node_->now();
405  auto start_result = client_servo_start_->async_send_request(std::make_shared<std_srvs::srv::Trigger::Request>());
406  if (!start_result.get()->success)
407  {
408  RCLCPP_ERROR(LOGGER, "Error returned form service call to start servo");
409  return false;
410  }
411  RCLCPP_INFO_STREAM(LOGGER, "Wait for start servo: " << (node_->now() - time_start).seconds());
412 
413  // Test that status messages start
414  rclcpp::Rate publish_loop_rate(test_parameters_->publish_hz);
415  time_start = node_->now();
416  auto num_statuses_start = getNumStatus();
417  size_t iterations = 0;
418  while (getNumStatus() == num_statuses_start && iterations++ < test_parameters_->timeout_iterations)
419  {
420  publish_loop_rate.sleep();
421  }
422 
423  RCLCPP_INFO_STREAM(LOGGER, "Wait for status num increasing: " << (node_->now() - time_start).seconds());
424 
425  if (iterations >= test_parameters_->timeout_iterations)
426  {
427  RCLCPP_ERROR(LOGGER, "Timeout waiting for status num increasing");
428  return false;
429  }
430 
431  return getNumStatus() > num_statuses_start;
432  }
433 
434  bool stop()
435  {
436  auto time_start = node_->now();
437  auto stop_result = client_servo_stop_->async_send_request(std::make_shared<std_srvs::srv::Trigger::Request>());
438  if (!stop_result.get()->success)
439  {
440  RCLCPP_ERROR(LOGGER, "Error returned form service call to stop servo");
441  return false;
442  }
443  RCLCPP_INFO_STREAM(LOGGER, "Wait for stop servo service: " << (node_->now() - time_start).seconds());
444 
445  // Test that status messages stop
446  rclcpp::Rate publish_loop_rate(test_parameters_->publish_hz);
447  time_start = node_->now();
448  size_t num_statuses_start = 0;
449  size_t iterations = 0;
450  do
451  {
452  num_statuses_start = getNumStatus();
453  // Wait 4x the loop rate
454  for (size_t i = 0; i < 4; ++i)
455  publish_loop_rate.sleep();
456  } while (getNumStatus() != num_statuses_start && ++iterations < test_parameters_->timeout_iterations);
457  RCLCPP_INFO_STREAM(LOGGER, "Wait for status to stop: " << (node_->now() - time_start).seconds());
458 
459  if (iterations >= test_parameters_->timeout_iterations)
460  {
461  RCLCPP_ERROR(LOGGER, "Timeout waiting for status num increasing");
462  return false;
463  }
464 
465  return true;
466  }
