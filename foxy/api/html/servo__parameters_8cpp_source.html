50 rcl_interfaces::msg::SetParametersResult
51 ServoParameters::setParametersCallback(std::vector<rclcpp::Parameter> parameters)
52 {
53  const std::lock_guard<std::mutex> guard(callback_mutex_);
54  auto result = rcl_interfaces::msg::SetParametersResult();
55  result.successful = true;
56 
57  for (const auto& parameter : parameters)
58  {
59  auto search = set_parameter_callbacks_.find(parameter.get_name());
60  if (search != set_parameter_callbacks_.end())
61  {
62  RCLCPP_INFO_STREAM(logger_, "setParametersCallback - "
63  << parameter.get_name() << "<" << parameter.get_type_name()
64  << ">: " << rclcpp::to_string(parameter.get_parameter_value()));
65  for (const auto& callback : search->second)
66  {
67  result = callback(parameter);
68 
69  if (!result.successful)
70  {
71  // Handle automatic parameter set failure
72  return result;
73  }
74  }
75  }
76  }
77 
78  return result;
79 }
