153 // Helper template for declaring and getting ros param
154 // Must be declared here to ensure template class is built for required templates when included.
155 // (The CPP file can't just be included because not everything there is templated, so you'd get duplicate symbols)
156 template <typename T>
157 void declareOrGetParam(T& output_value, const std::string& param_name, const rclcpp::Node::SharedPtr& node,
158  const rclcpp::Logger& logger, const T default_value = T{})
159 {
160  try
161  {
162  if (node->has_parameter(param_name))
163  {
164  node->get_parameter<T>(param_name, output_value);
165  }
166  else
167  {
168  output_value = node->declare_parameter<T>(param_name, default_value);
169  }
170  }
171  catch (const rclcpp::exceptions::InvalidParameterTypeException& e)
172  {
173  RCLCPP_WARN_STREAM(logger, "InvalidParameterTypeException(" << param_name << "): " << e.what());
174  RCLCPP_ERROR_STREAM(logger, "Error getting parameter \'" << param_name << "\', check parameter type in YAML file");
175  throw e;
176  }
177 
178  RCLCPP_INFO_STREAM(logger, "Found parameter - " << param_name << ": " << output_value);
179 }
