65 // Function copied from tutorial
66 // a small helper function to create our planning requests and move the robot.
67 bool moveToCartPose(const geometry_msgs::PoseStamped& pose, moveit::planning_interface::MoveGroupInterface& group,
68  const std::string& end_effector_link)
69 {
70  group.clearPoseTargets();
71  group.setEndEffectorLink(end_effector_link);
72  group.setStartStateToCurrentState();
73  group.setPoseTarget(pose);
74 
75  moveit::planning_interface::MoveGroupInterface::Plan myplan;
76  if (group.plan(myplan) && group.execute(myplan))
77  {
78  return true;
79  }
80 
81  ROS_WARN("Failed to perform motion.");
82  return false;
83 }

85 // Function copied from tutorial
86 // This helper function creates two objects and publishes them to the PlanningScene: a box and a cylinder.
87 // The box spawns in front of the gripper, the cylinder at the tip of the gripper, as if it had been grasped.
88 void spawnCollisionObjects(moveit::planning_interface::PlanningSceneInterface& planning_scene_interface)
89 {
90  const std::string log_name = "spawn_collision_objects";
91  double z_offset_box = .25; // The z-axis points away from the gripper
92  double z_offset_cylinder = .1;
93 
94  // First, we start defining the CollisionObject as usual.
95  moveit_msgs::CollisionObject box;
96  box.id = "box";
97  box.header.frame_id = "panda_hand";
98  box.primitives.resize(1);
99  box.primitive_poses.resize(1);
100  box.primitives[0].type = box.primitives[0].BOX;
101  box.primitives[0].dimensions.resize(3);
102  box.primitives[0].dimensions[0] = 0.05;
103  box.primitives[0].dimensions[1] = 0.1;
104  box.primitives[0].dimensions[2] = 0.02;
105  box.primitive_poses[0].position.z = z_offset_box;
106 
107  // Then, we define the subframes of the CollisionObject.
108  box.subframe_names.resize(1);
109  box.subframe_poses.resize(1);
110  box.subframe_names[0] = "bottom";
111  box.subframe_poses[0].position.y = -.05;
112  box.subframe_poses[0].position.z = 0.0 + z_offset_box;
113  tf2::Quaternion orientation;
114  orientation.setRPY(90.0 / 180.0 * M_PI, 0, 0);
115  box.subframe_poses[0].orientation = tf2::toMsg(orientation);
116 
117  // Next, define the cylinder
118  moveit_msgs::CollisionObject cylinder;
119  cylinder.id = "cylinder";
120  cylinder.header.frame_id = "panda_hand";
121  cylinder.primitives.resize(1);
122  cylinder.primitive_poses.resize(1);
123  cylinder.primitives[0].type = box.primitives[0].CYLINDER;
124  cylinder.primitives[0].dimensions.resize(2);
125  cylinder.primitives[0].dimensions[0] = 0.06; // height (along x)
126  cylinder.primitives[0].dimensions[1] = 0.005; // radius
127  cylinder.primitive_poses[0].position.x = 0.0;
128  cylinder.primitive_poses[0].position.y = 0.0;
129  cylinder.primitive_poses[0].position.z = 0.0 + z_offset_cylinder;
130  orientation.setRPY(0, 90.0 / 180.0 * M_PI, 0);
131  cylinder.primitive_poses[0].orientation = tf2::toMsg(orientation);
132 
133  cylinder.subframe_poses.resize(1);
134  cylinder.subframe_names.resize(1);
135  cylinder.subframe_names[0] = "tip";
136  cylinder.subframe_poses[0].position.x = 0.03;
137  cylinder.subframe_poses[0].position.y = 0.0;
138  cylinder.subframe_poses[0].position.z = 0.0 + z_offset_cylinder;
139  orientation.setRPY(0, 90.0 / 180.0 * M_PI, 0);
140  cylinder.subframe_poses[0].orientation = tf2::toMsg(orientation);
141 
142  // Lastly, the objects are published to the PlanningScene. In this tutorial, we publish a box and a cylinder.
143  box.operation = moveit_msgs::CollisionObject::ADD;
144  cylinder.operation = moveit_msgs::CollisionObject::ADD;
145  planning_scene_interface.applyCollisionObjects({ box, cylinder });
146 }

148 TEST(TestPlanUsingSubframes, SubframesTests)
149 {
150  const std::string log_name = "test_plan_using_subframes";
151  ros::NodeHandle nh(log_name);
152 
153  auto planning_scene_monitor = std::make_shared<planning_scene_monitor::PlanningSceneMonitor>("robot_description");
154  moveit::planning_interface::PlanningSceneInterface planning_scene_interface;
155  moveit::planning_interface::MoveGroupInterface group("panda_arm");
156  group.setPlanningTime(PLANNING_TIME_S);
157 
158  spawnCollisionObjects(planning_scene_interface);
159  moveit_msgs::AttachedCollisionObject att_coll_object;
160  att_coll_object.object.id = "cylinder";
161  att_coll_object.link_name = "panda_hand";
162  att_coll_object.object.operation = att_coll_object.object.ADD;
163  planning_scene_interface.applyAttachedCollisionObject(att_coll_object);
164 
165  tf2::Quaternion target_orientation;
166  target_orientation.setRPY(0, 180.0 / 180.0 * M_PI, 90.0 / 180.0 * M_PI);
167  geometry_msgs::PoseStamped target_pose_stamped;
168  target_pose_stamped.pose.orientation = tf2::toMsg(target_orientation);
169  target_pose_stamped.pose.position.z = Z_OFFSET;
170 
171  ROS_INFO_STREAM_NAMED(log_name, "Moving to bottom of box with cylinder tip");
172  target_pose_stamped.header.frame_id = "box/bottom";
173  ASSERT_TRUE(moveToCartPose(target_pose_stamped, group, "cylinder/tip"));
174  planning_scene_monitor->requestPlanningSceneState();
175  {
176  planning_scene_monitor::LockedPlanningSceneRO planning_scene(planning_scene_monitor);
177 
178  // get the tip and box subframe locations in world
179  Eigen::Isometry3d eef = planning_scene->getFrameTransform("cylinder/tip");
180  Eigen::Isometry3d box_subframe = planning_scene->getFrameTransform(target_pose_stamped.header.frame_id);
181  Eigen::Isometry3d target_pose;
182  tf2::fromMsg(target_pose_stamped.pose, target_pose);
183 
184  // expect that they are identical
185  std::stringstream ss;
186  ss << "target frame: \n" << (box_subframe * target_pose).matrix() << "\ncylinder frame: \n" << eef.matrix();
187  EXPECT_TRUE(eef.isApprox(box_subframe * target_pose, EPSILON)) << ss.str();
188  }
189 }
