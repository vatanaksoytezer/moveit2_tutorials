133  "<virtual_joint name=\"base_joint\" child_link=\"base_footprint\" "
134  "parent_frame=\"base_footprint\" type=\"planar\"/>"
135  "<group name=\"left_arm_base_tip\">"

150  moveit::core::RobotModelPtr robot_model2(new moveit::core::RobotModel(urdf_model_, srdf_model));

152  left_arm_base_tip_group = robot_model2->getJointModelGroup("left_arm_base_tip");
153  ASSERT_TRUE(left_arm_base_tip_group != nullptr);

155  left_arm_joints_group = robot_model2->getJointModelGroup("left_arm_joints");
156  ASSERT_TRUE(left_arm_joints_group != nullptr);

159  EXPECT_EQ(left_arm_joints_group->getJointModels().size(), 7u);

163

165 
166  bool found_shoulder_pan_link = false;

168  for (const moveit::core::LinkModel* link_model : left_arm_base_tip_group->getLinkModels())
169  {

192  moveit::core::robotStateToRobotStateMsg(ks, robot_state);
193 
194  moveit::core::RobotState ks2(robot_model2);

196 
197  const double* v1 = ks.getVariablePositions();
198  const double* v2 = ks2.getVariablePositions();
199  for (std::size_t i = 0; i < ks.getVariableCount(); ++i)
200  EXPECT_NEAR(v1[i], v2[i], 1e-5);
201 }

223 TEST_F(LoadPlanningModelsPr2, FullTest)
224 {
225  moveit::core::RobotModelPtr robot_model(new moveit::core::RobotModel(urdf_model_, srdf_model_));
226 
227  moveit::core::RobotState ks(robot_model);
228  ks.setToDefaultValues();
229 
230  moveit::core::RobotState ks2(robot_model);
231  ks2.setToDefaultValues();
232 
233  std::vector<shapes::ShapeConstPtr> shapes;
234  EigenSTL::vector_Isometry3d poses;
235  shapes::Shape* shape = new shapes::Box(.1, .1, .1);
236  shapes.push_back(shapes::ShapeConstPtr(shape));
237  poses.push_back(Eigen::Isometry3d::Identity());
238  std::set<std::string> touch_links;
239 
240  trajectory_msgs::msg::JointTrajectory empty_state;
241  moveit::core::AttachedBody* attached_body = new moveit::core::AttachedBody(
242  robot_model->getLinkModel("r_gripper_palm_link"), "box", shapes, poses, touch_links, empty_state);
243  ks.attachBody(attached_body);
244 
245  std::vector<const moveit::core::AttachedBody*> attached_bodies_1;
246  ks.getAttachedBodies(attached_bodies_1);
247  ASSERT_EQ(attached_bodies_1.size(), 1u);
248 
249  std::vector<const moveit::core::AttachedBody*> attached_bodies_2;
250  ks2 = ks;
251  ks2.getAttachedBodies(attached_bodies_2);
252  ASSERT_EQ(attached_bodies_2.size(), 1u);
253 
254  ks.clearAttachedBody("box");
255  attached_bodies_1.clear();
256  ks.getAttachedBodies(attached_bodies_1);
257  ASSERT_EQ(attached_bodies_1.size(), 0u);
258 
259  ks2 = ks;
260  attached_bodies_2.clear();
261  ks2.getAttachedBodies(attached_bodies_2);
262  ASSERT_EQ(attached_bodies_2.size(), 0u);
263 }
