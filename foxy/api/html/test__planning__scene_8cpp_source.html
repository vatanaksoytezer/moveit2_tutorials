62 TEST(PlanningScene, LoadRestoreDiff)
63 {
64  urdf::ModelInterfaceSharedPtr urdf_model = moveit::core::loadModelInterface("pr2");
65  srdf::ModelSharedPtr srdf_model(new srdf::Model());
66  auto ps = std::make_shared<planning_scene::PlanningScene>(urdf_model, srdf_model);
67 
68  collision_detection::World& world = *ps->getWorldNonConst();
69 
70  /* add one object to ps's world */
71  Eigen::Isometry3d id = Eigen::Isometry3d::Identity();
72  world.addToObject("sphere", shapes::ShapeConstPtr(new shapes::Sphere(0.4)), id);
73 
74  /* ps can be written to and set from message */
75  moveit_msgs::msg::PlanningScene ps_msg;
76  ps_msg.robot_state.is_diff = true;
77  EXPECT_TRUE(moveit::core::isEmpty(ps_msg));
78  ps->getPlanningSceneMsg(ps_msg);
79  ps->setPlanningSceneMsg(ps_msg);
80  EXPECT_EQ(ps_msg.world.collision_objects.size(), 1u);
81  EXPECT_EQ(ps_msg.world.collision_objects[0].id, "sphere");
82  EXPECT_TRUE(world.hasObject("sphere"));
83 
84  /* test diff scene on top of ps */
85  planning_scene::PlanningScenePtr next = ps->diff();
86  /* world is inherited from ps */
87  EXPECT_TRUE(next->getWorld()->hasObject("sphere"));
88 
89  /* object in overlay is only added in overlay */
90  next->getWorldNonConst()->addToObject("sphere_in_next_only", shapes::ShapeConstPtr(new shapes::Sphere(0.5)), id);
91  EXPECT_EQ(next->getWorld()->size(), 2u);
92  EXPECT_EQ(ps->getWorld()->size(), 1u);
93 
94  /* the worlds used for collision detection contain one and two objects, respectively */
95  EXPECT_EQ(ps->getCollisionEnv()->getWorld()->size(), 1u);
96  EXPECT_EQ(ps->getCollisionEnvUnpadded()->getWorld()->size(), 1u);
97 
98  EXPECT_EQ(next->getCollisionEnv()->getWorld()->size(), 2u);
99  EXPECT_EQ(next->getCollisionEnvUnpadded()->getWorld()->size(), 2u);
100 
101  /* maintained diff contains only overlay object */
102  next->getPlanningSceneDiffMsg(ps_msg);
103  EXPECT_EQ(ps_msg.world.collision_objects.size(), 1u);
104 
105  /* copy ps to next and apply diff */
106  next->decoupleParent();
107  moveit_msgs::msg::PlanningScene ps_msg2;
108 
109  /* diff is empty now */
110  next->getPlanningSceneDiffMsg(ps_msg2);
111  EXPECT_EQ(ps_msg2.world.collision_objects.size(), 0u);
112 
113  /* next's world contains both objects */
114  next->getPlanningSceneMsg(ps_msg);
115  EXPECT_EQ(ps_msg.world.collision_objects.size(), 2u);
116  ps->setPlanningSceneMsg(ps_msg);
117  EXPECT_EQ(ps->getWorld()->size(), 2u);
118  EXPECT_EQ(ps->getCollisionEnv()->getWorld()->size(), 2u);
119  EXPECT_EQ(ps->getCollisionEnvUnpadded()->getWorld()->size(), 2u);
120 }

122 TEST(PlanningScene, MakeAttachedDiff)
123 {
124  urdf::ModelInterfaceSharedPtr urdf_model = moveit::core::loadModelInterface("pr2");
125  srdf::ModelSharedPtr srdf_model(new srdf::Model());
126  auto ps = std::make_shared<planning_scene::PlanningScene>(urdf_model, srdf_model);
127 
128  /* add a single object to ps's world */
129  collision_detection::World& world = *ps->getWorldNonConst();
130  Eigen::Isometry3d id = Eigen::Isometry3d::Identity();
131  world.addToObject("sphere", shapes::ShapeConstPtr(new shapes::Sphere(0.4)), id);
132 
133  /* attach object in diff */
134  planning_scene::PlanningScenePtr attached_object_diff_scene = ps->diff();
135 
136  moveit_msgs::msg::AttachedCollisionObject att_obj;
137  att_obj.link_name = "r_wrist_roll_link";
138  att_obj.object.operation = moveit_msgs::msg::CollisionObject::ADD;
139  att_obj.object.id = "sphere";
140  attached_object_diff_scene->processAttachedCollisionObjectMsg(att_obj);
141 
142  /* object is not in world anymore */
143  EXPECT_EQ(attached_object_diff_scene->getWorld()->size(), 0u);
144  /* it became part of the robot state though */
145  EXPECT_TRUE(attached_object_diff_scene->getCurrentState().hasAttachedBody("sphere"));
146 
147  collision_detection::CollisionRequest req;
148  collision_detection::CollisionResult res;
149  attached_object_diff_scene->checkCollision(req, res);
150  ps->checkCollision(req, res);
151 }
