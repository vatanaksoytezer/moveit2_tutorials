64  void initTestTrajectory(robot_trajectory::RobotTrajectoryPtr& trajectory)
65  {
66  // Init a traj
67  ASSERT_TRUE(robot_model_->hasJointModelGroup(arm_jmg_name_))
68  << "Robot model does not have group: " << arm_jmg_name_;
69 
70  trajectory = std::make_shared<robot_trajectory::RobotTrajectory>(robot_model_, arm_jmg_name_);
71 
72  EXPECT_EQ(trajectory->getGroupName(), arm_jmg_name_) << "Generated trajectory group name does not match";
73  EXPECT_TRUE(trajectory->empty()) << "Generated trajectory not empty";
74 
75  double duration_from_previous = 0.1;
76  std::size_t waypoint_count = 5;
77  for (std::size_t ix = 0; ix < waypoint_count; ++ix)
78  trajectory->addSuffixWayPoint(robot_state_, duration_from_previous);
79  // Quick check that getDuration is working correctly
80  EXPECT_EQ(trajectory->getDuration(), duration_from_previous * waypoint_count)
81  << "Generated trajectory duration incorrect";
82  EXPECT_EQ(waypoint_count, trajectory->getWayPointDurations().size())
83  << "Generated trajectory has the wrong number of waypoints";
84  }

96  void modifyFirstWaypointPtrAndCheckTrajectory(robot_trajectory::RobotTrajectoryPtr& trajectory)
97  {
99  // Get the first waypoint by POINTER, modify it, and check that the value WAS updated in trajectory
101  // Get the first waypoint by shared pointer
102  moveit::core::RobotStatePtr trajectory_first_waypoint = trajectory->getWayPointPtr(0);
103  // Get the first waypoint joint values
104  std::vector<double> trajectory_first_state;
105  trajectory_first_waypoint->copyJointGroupPositions(arm_jmg_name_, trajectory_first_state);
106 
107  // Modify the first waypoint joint values
108  trajectory_first_state[0] += 0.01;
109  trajectory_first_waypoint->setJointGroupPositions(arm_jmg_name_, trajectory_first_state);
110 
111  // Check that the trajectory's first waypoint was updated
112  moveit::core::RobotStatePtr trajectory_first_waypoint_after_update = trajectory->getWayPointPtr(0);
113  std::vector<double> trajectory_first_state_after_update;
114  trajectory_first_waypoint_after_update->copyJointGroupPositions(arm_jmg_name_, trajectory_first_state_after_update);
115  EXPECT_EQ(trajectory_first_state[0], trajectory_first_state_after_update[0]);
116  }
