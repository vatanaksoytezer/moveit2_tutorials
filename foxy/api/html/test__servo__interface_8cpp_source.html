152  auto num_received = getNumCommands();

156  log_time_end = node_->now();

161 
162  // Now let's test the Servo input going stale

164  // to continually receive Servo output, but with 0 velocity/delta_position
165  log_time_start = node_->now();
166

176

178  RCLCPP_INFO_STREAM(LOGGER, "Wait stale command: " << (log_time_end - log_time_start).seconds());

208

216
