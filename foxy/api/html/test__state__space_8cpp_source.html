114 TEST_F(LoadPlanningModelsPr2, StateSpaceCopy)
115 {
116  ompl_interface::ModelBasedStateSpaceSpecification spec(robot_model_, "right_arm");
117  ompl_interface::JointModelStateSpace joint_model_state_space(spec);
118  joint_model_state_space.setPlanningVolume(-1, 1, -1, 1, -1, 1);
119  joint_model_state_space.setup();
120  std::ofstream fout("ompl_interface_test_state_space_diagram1.dot");
121  joint_model_state_space.diagram(fout);
122  bool passed = false;
123  try
124  {
125  joint_model_state_space.sanityChecks();
126  passed = true;
127  }
128  catch (ompl::Exception& ex)
129  {
130  RCLCPP_ERROR(LOGGER, "Sanity checks did not pass: %s", ex.what());
131  }
132  EXPECT_TRUE(passed);
133 
134  moveit::core::RobotState robot_state(robot_model_);
135  robot_state.setToRandomPositions();
136  EXPECT_TRUE(robot_state.distance(robot_state) < EPSILON);
137  ompl::base::State* state = joint_model_state_space.allocState();
138  for (int i = 0; i < 10; ++i)
139  {
140  moveit::core::RobotState robot_state2(robot_state);
141  EXPECT_TRUE(robot_state.distance(robot_state2) < EPSILON);
142  joint_model_state_space.copyToOMPLState(state, robot_state);
143  robot_state.setToRandomPositions(
144  robot_state.getRobotModel()->getJointModelGroup(joint_model_state_space.getJointModelGroupName()));
145  std::cout << (robot_state.getGlobalLinkTransform("r_wrist_roll_link").translation() -
146  robot_state2.getGlobalLinkTransform("r_wrist_roll_link").translation())
147  << std::endl;
148  EXPECT_TRUE(robot_state.distance(robot_state2) > EPSILON);
149  joint_model_state_space.copyToRobotState(robot_state, state);
150  std::cout << (robot_state.getGlobalLinkTransform("r_wrist_roll_link").translation() -
151  robot_state2.getGlobalLinkTransform("r_wrist_roll_link").translation())
152  << std::endl;
153  EXPECT_TRUE(robot_state.distance(robot_state2) < EPSILON);
154  }
155 
156  // repeat the above test with a different method to copy the state to OMPL
157  for (int i = 0; i < 10; ++i)
158  {
159  // create two equal states
160  moveit::core::RobotState robot_state2(robot_state);
161  EXPECT_LT(robot_state.distance(robot_state2), EPSILON);
162 
163  // copy the first state to OMPL as backup (this is where the 'different' method comes into play)
164  const moveit::core::JointModelGroup* joint_model_group =
165  robot_state.getRobotModel()->getJointModelGroup(joint_model_state_space.getJointModelGroupName());
166  std::vector<std::string> joint_model_names = joint_model_group->getActiveJointModelNames();
167  for (std::size_t joint_index = 0; joint_index < joint_model_group->getVariableCount(); ++joint_index)
168  {
169  const moveit::core::JointModel* joint_model = joint_model_group->getJointModel(joint_model_names[joint_index]);
170  EXPECT_NE(joint_model, nullptr);
171  joint_model_state_space.copyJointToOMPLState(state, robot_state, joint_model, joint_index);
172  }
173 
174  // and change the joint values of the moveit state, so it is different that robot_state2
175  robot_state.setToRandomPositions(
176  robot_state.getRobotModel()->getJointModelGroup(joint_model_state_space.getJointModelGroupName()));
177  EXPECT_GT(robot_state.distance(robot_state2), EPSILON);
178 
179  // copy the backup values in the OMPL state back to the first state,
180  // and check if it is still equal to the second
181  joint_model_state_space.copyToRobotState(robot_state, state);
182  EXPECT_LT(robot_state.distance(robot_state2), EPSILON);
183  }
184 
185  joint_model_state_space.freeState(state);
186 }
