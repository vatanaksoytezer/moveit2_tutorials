218 bool testutils::checkCartesianLinearity(const moveit::core::RobotModelConstPtr& robot_model,
219  const trajectory_msgs::JointTrajectory& trajectory,
220  const planning_interface::MotionPlanRequest& req,
221  const double translation_norm_tolerance, const double rot_axis_norm_tolerance,
222  const double rot_angle_tolerance)
223 {
224  std::string link_name;
225  Eigen::Isometry3d goal_pose_expect;
226  if (!testutils::getExpectedGoalPose(robot_model, req, link_name, goal_pose_expect))
227  {
228  return false;
229  }
230 
231  // compute start pose
232  robot_state::RobotState rstate(robot_model);
233  rstate.setToDefaultValues();
234  moveit::core::jointStateToRobotState(req.start_state.joint_state, rstate);
235  Eigen::Isometry3d start_pose = rstate.getFrameTransform(link_name);
236 
237  // start goal angle axis
238  Eigen::AngleAxisd start_goal_aa(start_pose.rotation().transpose() * goal_pose_expect.rotation());
239 
240  // check the linearity
241  for (trajectory_msgs::JointTrajectoryPoint way_point : trajectory.points)
242  {
243  Eigen::Isometry3d way_point_pose;
244  std::map<std::string, double> way_point_joint_state;
245 
246  // initializing all joints of the model
247  for (const auto& joint_name : robot_model->getVariableNames())
248  {
249  way_point_joint_state[joint_name] = 0;
250  }
251 
252  for (std::size_t i = 0; i < trajectory.joint_names.size(); ++i)
253  {
254  way_point_joint_state[trajectory.joint_names.at(i)] = way_point.positions.at(i);
255  }
256 
257  if (!computeLinkFK(robot_model, link_name, way_point_joint_state, way_point_pose))
258  {
259  std::cerr << "Failed to compute forward kinematics for link in goal "
260  "constraints \n";
261  return false;
262  }
263 
264  // translational linearity
265  Eigen::Vector3d goal_start_translation = goal_pose_expect.translation() - start_pose.translation();
266  // https://de.wikipedia.org/wiki/Geradengleichung
267  // Determined form of a straight line in 3D space
268  if (fabs((goal_start_translation.cross(way_point_pose.translation()) -
269  goal_start_translation.cross(start_pose.translation()))
270  .norm()) > fabs(translation_norm_tolerance))
271  {
272  std::cout << "Translational linearity is violated. \n"
273  << "goal tanslation: " << goal_pose_expect.translation() << std::endl
274  << "start translation: " << start_pose.translation() << std::endl
275  << "acutual translation " << way_point_pose.translation() << std::endl;
276  return false;
277  }
278 
279  if (!checkSLERP(start_pose, goal_pose_expect, way_point_pose, rot_axis_norm_tolerance))
280  {
281  return false;
282  }
283  }
284 
285  return true;
286 }

288 bool testutils::checkSLERP(const Eigen::Isometry3d& start_pose, const Eigen::Isometry3d& goal_pose,
289  const Eigen::Isometry3d& wp_pose, const double rot_axis_norm_tolerance,
290  const double rot_angle_tolerance)
291 {
292  // rotational linearity
293  // start way point angle axis
294  Eigen::AngleAxisd start_goal_aa(start_pose.rotation().transpose() * goal_pose.rotation());
295  Eigen::AngleAxisd start_wp_aa(start_pose.rotation().transpose() * wp_pose.rotation());
296 
297  // parallel rotation axis
298  // it is possible the axis opposite is
299  // if the angle is zero, axis is arbitrary
300  if (!(((start_goal_aa.axis() - start_wp_aa.axis()).norm() < fabs(rot_axis_norm_tolerance)) ||
301  ((start_goal_aa.axis() + start_wp_aa.axis()).norm() < fabs(rot_axis_norm_tolerance)) ||
302  (fabs(start_wp_aa.angle()) < fabs(rot_angle_tolerance))))
303  {
304  std::cout << "Rotational linearity is violated. \n"

306  << "start goal angle: " << start_goal_aa.angle() << "; axis: " << std::endl
307  << start_goal_aa.axis() << std::endl
308  << "start waypoint angle: " << start_wp_aa.angle() << "; axis: " << std::endl
309  << start_wp_aa.axis() << std::endl;
310 
311  return false;
312  }

314  return true;

317 bool testutils::computeLinkFK(const moveit::core::RobotModelConstPtr& robot_model, const std::string& link_name,
318  const std::map<std::string, double>& joint_state, Eigen::Isometry3d& pose)
319 {
320  // create robot state
321  robot_state::RobotState rstate(robot_model);
322  rstate.setToDefaultValues();
323 
324  // check the reference frame of the target pose
325  if (!rstate.knowsFrameTransform(link_name))
326  {
327  ROS_ERROR_STREAM("The target link " << link_name << " is not known by robot.");
328  return false;
329  }
330 
331  // set the joint positions
332  try
333  {
334  rstate.setVariablePositions(joint_state);
335  }
336  catch (const std::exception& e)
337  {
338  std::cerr << e.what() << '\n';
339  return false;
340  }
341 
342  // update the frame
343  rstate.update();
344  pose = rstate.getFrameTransform(link_name);
345 
346  return true;
