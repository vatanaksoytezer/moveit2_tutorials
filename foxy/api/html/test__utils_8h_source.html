101 inline sensor_msgs::JointState generateJointState(const std::vector<double>& pos, const std::vector<double>& vel,
102  const std::string& joint_prefix = testutils::JOINT_NAME_PREFIX)
103 {
104  sensor_msgs::JointState state;
105  auto posit = pos.cbegin();
106  size_t i = 0;
107 
108  while (posit != pos.cend())
109  {
110  state.name.push_back(testutils::getJointName(i + 1, joint_prefix));
111  state.position.push_back(*posit);
112 
113  i++;
114  posit++;
115  }
116  for (const auto& one_vel : vel)
117  {
118  state.velocity.push_back(one_vel);
119  }
120  return state;
121 }

129 inline moveit_msgs::Constraints generateJointConstraint(const std::vector<double>& pos_list,
130  const std::string& joint_prefix = testutils::JOINT_NAME_PREFIX)
131 {
132  moveit_msgs::Constraints gc;
133 
134  auto pos_it = pos_list.begin();
135 
136  for (size_t i = 0; i < pos_list.size(); ++i)
137  {
138  moveit_msgs::JointConstraint jc;
139  jc.joint_name = testutils::getJointName(i + 1, joint_prefix);
140  jc.position = *pos_it;
141  gc.joint_constraints.push_back(jc);
142 
143  ++pos_it;
144  }
145 
146  return gc;
147 }

377 inline geometry_msgs::Quaternion fromEuler(double a, double b, double c)
378 {
379  tf2::Vector3 qvz(0., 0., 1.);
380  tf2::Vector3 qvy(0., 1., 0.);
381  tf2::Quaternion q1(qvz, a);
382 
383  q1 = q1 * tf2::Quaternion(qvy, b);
384  q1 = q1 * tf2::Quaternion(qvz, c);
385 
386  return tf2::toMsg(q1);
387 }
