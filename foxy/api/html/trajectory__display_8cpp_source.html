73 void TrajectoryDisplay::loadRobotModel()
74 {
75  try
76  {
77  rdf_loader_.reset(new rdf_loader::RDFLoader(node_, robot_description_property_->getStdString()));
78 
79  if (!rdf_loader_->getURDF())
80  {
81  this->setStatus(rviz_common::properties::StatusProperty::Error, "Robot Model",
82  "Failed to load from parameter " + robot_description_property_->getString());
83  return;
84  }
85  this->setStatus(rviz_common::properties::StatusProperty::Ok, "Robot Model", "Successfully loaded");
86 
87  const srdf::ModelSharedPtr& srdf =
88  rdf_loader_->getSRDF() ? rdf_loader_->getSRDF() : srdf::ModelSharedPtr(new srdf::Model());
89  robot_model_.reset(new moveit::core::RobotModel(rdf_loader_->getURDF(), srdf));
90 
91  // Send to child class
92  trajectory_visual_->onRobotModelLoaded(robot_model_);
93  }
94  catch (std::exception& e)
95  {
96  setStatus(rviz_common::properties::StatusProperty::Error, "RobotModel", QString("Loading failed: %1").arg(e.what()));
97  }
98 }
