84 void TrajectoryExecutionManager::initialize()
85 {
86  verbose_ = false;
87  execution_complete_ = true;
88  stop_continuous_execution_ = false;
89  current_context_ = -1;
90  last_execution_status_ = moveit_controller_manager::ExecutionStatus::SUCCEEDED;
91  run_continuous_execution_thread_ = true;
92  execution_duration_monitoring_ = true;
93  execution_velocity_scaling_ = 1.0;
94  allowed_start_tolerance_ = 0.01;
95  wait_for_trajectory_completion_ = true;
96 
97  allowed_execution_duration_scaling_ = DEFAULT_CONTROLLER_GOAL_DURATION_SCALING;
98  allowed_goal_duration_margin_ = DEFAULT_CONTROLLER_GOAL_DURATION_MARGIN;
99 
100  // load controller-specific values for allowed_execution_duration_scaling and allowed_goal_duration_margin
101  loadControllerParams();
102 
103  // load the controller manager plugin
104  try
105  {
106  controller_manager_loader_.reset(new pluginlib::ClassLoader<moveit_controller_manager::MoveItControllerManager>(
107  "moveit_core", "moveit_controller_manager::MoveItControllerManager"));
108  }
109  catch (pluginlib::PluginlibException& ex)
110  {
111  RCLCPP_FATAL_STREAM(LOGGER, "Exception while creating controller manager plugin loader: " << ex.what());
112  return;
113  }
114 
115  if (controller_manager_loader_)
116  {
117  std::string controller;
118 
119  if (!node_->get_parameter("moveit_controller_manager", controller))
120  {
121  const std::vector<std::string>& classes = controller_manager_loader_->getDeclaredClasses();
122  if (classes.size() == 1)
123  {
124  controller = classes[0];
125  RCLCPP_WARN(LOGGER,
126  "Parameter '~moveit_controller_manager' is not specified but only one "
127  "matching plugin was found: '%s'. Using that one.",
128  controller.c_str());
129  }
130  else
131  {
132  RCLCPP_FATAL(LOGGER, "Parameter '~moveit_controller_manager' not specified. This is needed to "
133  "identify the plugin to use for interacting with controllers. No paths can "
134  "be executed.");
135  }
136  }
137 
138  if (!controller.empty())
139  try
140  {
141  // We make a node called moveit_simple_controller_manager so it's able to
142  // receive callbacks on another thread. We then copy parameters from the move_group node
143  // and then add it to the multithreadedexecutor
144  rclcpp::NodeOptions opt;
145  opt.allow_undeclared_parameters(true);
146  opt.automatically_declare_parameters_from_overrides(true);
147  controller_mgr_node_.reset(new rclcpp::Node("moveit_simple_controller_manager", opt));
148 
149  auto all_params = node_->get_node_parameters_interface()->get_parameter_overrides();
150  for (const auto& param : all_params)
151  controller_mgr_node_->set_parameter(rclcpp::Parameter(param.first, param.second));
152 
153  controller_manager_ = controller_manager_loader_->createUniqueInstance(controller);
154  controller_manager_->initialize(controller_mgr_node_);
155  private_executor_ = std::make_shared<rclcpp::executors::SingleThreadedExecutor>();
156  private_executor_->add_node(controller_mgr_node_);
157 
158  // start executor on a different thread now
159  private_executor_thread_ = std::thread([this]() { private_executor_->spin(); });
160  }
161  catch (pluginlib::PluginlibException& ex)
162  {
163  RCLCPP_FATAL_STREAM(LOGGER, "Exception while loading controller manager '" << controller << "': " << ex.what());
164  }
165  }
166 
167  // other configuration steps
168  reloadControllerInformation();
169  // The default callback group for rclcpp::Node is MutuallyExclusive which means we cannot call
170  // receiveEvent while processing a different callback. To fix this we create a new callback group (the type is not
171  // important since we only use it to process one callback) and associate event_topic_subscriber_ with this callback group
172  auto callback_group = node_->create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
173  auto options = rclcpp::SubscriptionOptions();
174  options.callback_group = callback_group;
175  event_topic_subscriber_ = node_->create_subscription<std_msgs::msg::String>(
176  EXECUTION_EVENT_TOPIC, 100, std::bind(&TrajectoryExecutionManager::receiveEvent, this, std::placeholders::_1),
177  options);
178 
179  controller_mgr_node_->get_parameter("trajectory_execution.allowed_execution_duration_scaling",
180  allowed_execution_duration_scaling_);
181  controller_mgr_node_->get_parameter("trajectory_execution.allowed_goal_duration_margin",
182  allowed_goal_duration_margin_);
183  controller_mgr_node_->get_parameter("trajectory_execution.allowed_start_tolerance", allowed_start_tolerance_);
184 
185  if (manage_controllers_)
186  RCLCPP_INFO(LOGGER, "Trajectory execution is managing controllers");
187  else
188  RCLCPP_INFO(LOGGER, "Trajectory execution is not managing controllers");
189 
190  auto controller_mgr_parameter_set_callback = [this](std::vector<rclcpp::Parameter> parameters) {
191  auto result = rcl_interfaces::msg::SetParametersResult();
192  result.successful = true;
193  for (const auto& parameter : parameters)
194  {
195  const std::string& name = parameter.get_name();
196  if (name == "trajectory_execution.execution_duration_monitoring")
197  enableExecutionDurationMonitoring(parameter.as_bool());
198  else if (name == "trajectory_execution.allowed_execution_duration_scaling")
199  setAllowedExecutionDurationScaling(parameter.as_double());
200  else if (name == "trajectory_execution.allowed_goal_duration_margin")
201  setAllowedGoalDurationMargin(parameter.as_double());
202  else if (name == "trajectory_execution.execution_velocity_scaling")
203  setExecutionVelocityScaling(parameter.as_double());
204  else if (name == "trajectory_execution.allowed_start_tolerance")
205  setAllowedStartTolerance(parameter.as_double());
206  else if (name == "trajectory_execution.wait_for_trajectory_completion")
207  setWaitForTrajectoryCompletion(parameter.as_bool());
208  else
209  result.successful = false;
210  }
211  return result;
212  };
213  callback_handler_ = controller_mgr_node_->add_on_set_parameters_callback(controller_mgr_parameter_set_callback);
214 }

379 bool TrajectoryExecutionManager::pushAndExecute(const moveit_msgs::msg::RobotTrajectory& trajectory,
380  const std::vector<std::string>& controllers)
381 {
382  if (!execution_complete_)
383  {
384  RCLCPP_ERROR(LOGGER, "Cannot push & execute a new trajectory while another is being executed");
385  return false;
386  }
387 
388  TrajectoryExecutionContext* context = new TrajectoryExecutionContext();
389  if (configure(*context, trajectory, controllers))
390  {
391  {
392  boost::mutex::scoped_lock slock(continuous_execution_mutex_);
393  continuous_execution_queue_.push_back(context);
394  if (!continuous_execution_thread_)
395  continuous_execution_thread_.reset(
396  new boost::thread(boost::bind(&TrajectoryExecutionManager::continuousExecutionThread, this)));
397  }
398  last_execution_status_ = moveit_controller_manager::ExecutionStatus::SUCCEEDED;
399  continuous_execution_condition_.notify_all();
400  return true;
401  }
402  else
403  {
404  delete context;
405  last_execution_status_ = moveit_controller_manager::ExecutionStatus::ABORTED;
406  return false;
407  }

410 void TrajectoryExecutionManager::continuousExecutionThread()
411 {
412  std::set<moveit_controller_manager::MoveItControllerHandlePtr> used_handles;
413  while (run_continuous_execution_thread_)
414  {
415  if (!stop_continuous_execution_)
416  {
417  boost::unique_lock<boost::mutex> ulock(continuous_execution_mutex_);
418  while (continuous_execution_queue_.empty() && run_continuous_execution_thread_ && !stop_continuous_execution_)
419  continuous_execution_condition_.wait(ulock);
420  }
421 
422  if (stop_continuous_execution_ || !run_continuous_execution_thread_)
423  {
424  for (const moveit_controller_manager::MoveItControllerHandlePtr& used_handle : used_handles)
425  if (used_handle->getLastExecutionStatus() == moveit_controller_manager::ExecutionStatus::RUNNING)
426  used_handle->cancelExecution();
427  used_handles.clear();
428  while (!continuous_execution_queue_.empty())
429  {
430  TrajectoryExecutionContext* context = continuous_execution_queue_.front();
431  continuous_execution_queue_.pop_front();
432  delete context;
433  }
434  stop_continuous_execution_ = false;
435  continue;
436  }
437 
438  while (!continuous_execution_queue_.empty())
439  {
440  TrajectoryExecutionContext* context = nullptr;
441  {
442  boost::mutex::scoped_lock slock(continuous_execution_mutex_);
443  if (continuous_execution_queue_.empty())
444  break;
445  context = continuous_execution_queue_.front();
446  continuous_execution_queue_.pop_front();
447  if (continuous_execution_queue_.empty())
448  continuous_execution_condition_.notify_all();
449  }
450 
451  // remove handles we no longer need
452  std::set<moveit_controller_manager::MoveItControllerHandlePtr>::iterator uit = used_handles.begin();
453  while (uit != used_handles.end())
454  if ((*uit)->getLastExecutionStatus() != moveit_controller_manager::ExecutionStatus::RUNNING)
455  {
456  std::set<moveit_controller_manager::MoveItControllerHandlePtr>::iterator to_erase = uit;
457  ++uit;
458  used_handles.erase(to_erase);
459  }
460  else
461  ++uit;
462 
463  // now send stuff to controllers
464 
465  // first make sure desired controllers are active
466  if (areControllersActive(context->controllers_))
467  {
468  // get the controller handles needed to execute the new trajectory
469  std::vector<moveit_controller_manager::MoveItControllerHandlePtr> handles(context->controllers_.size());
470  for (std::size_t i = 0; i < context->controllers_.size(); ++i)
471  {
472  moveit_controller_manager::MoveItControllerHandlePtr h;
473  try
474  {
475  h = controller_manager_->getControllerHandle(context->controllers_[i]);
476  }
477  catch (std::exception& ex)
478  {
479  RCLCPP_ERROR(LOGGER, "%s caught when retrieving controller handle", ex.what());
480  }
481  if (!h)
482  {
483  last_execution_status_ = moveit_controller_manager::ExecutionStatus::ABORTED;
484  RCLCPP_ERROR(LOGGER, "No controller handle for controller '%s'. Aborting.",
485  context->controllers_[i].c_str());
486  handles.clear();
487  break;
488  }
489  handles[i] = h;
490  }
491 
492  if (stop_continuous_execution_ || !run_continuous_execution_thread_)
493  {
494  delete context;
495  break;
496  }
497 
498  // push all trajectories to all controllers simultaneously
499  if (!handles.empty())
500  for (std::size_t i = 0; i < context->trajectory_parts_.size(); ++i)
501  {
502  bool ok = false;
503  try
504  {
505  ok = handles[i]->sendTrajectory(context->trajectory_parts_[i]);
506  }
507  catch (std::exception& ex)
508  {
509  RCLCPP_ERROR(LOGGER, "Caught %s when sending trajectory to controller", ex.what());
510  }
511  if (!ok)
512  {
513  for (std::size_t j = 0; j < i; ++j)
514  try
515  {
516  handles[j]->cancelExecution();
517  }
518  catch (std::exception& ex)
519  {
520  RCLCPP_ERROR(LOGGER, "Caught %s when canceling execution", ex.what());
521  }
522  RCLCPP_ERROR(LOGGER, "Failed to send trajectory part %zu of %zu to controller %s", i + 1,
523  context->trajectory_parts_.size(), handles[i]->getName().c_str());
524  if (i > 0)
525  RCLCPP_ERROR(LOGGER, "Cancelling previously sent trajectory parts");
526  last_execution_status_ = moveit_controller_manager::ExecutionStatus::ABORTED;
527  handles.clear();
528  break;
529  }
530  }
531  delete context;
532 
533  // remember which handles we used
534  for (const moveit_controller_manager::MoveItControllerHandlePtr& handle : handles)
535  used_handles.insert(handle);
536  }
537  else
538  {
539  RCLCPP_ERROR(LOGGER, "Not all needed controllers are active. Cannot push and execute. You can try "
540  "calling ensureActiveControllers() before pushAndExecute()");
541  last_execution_status_ = moveit_controller_manager::ExecutionStatus::ABORTED;
542  delete context;
543  }
544  }
545  }

548 void TrajectoryExecutionManager::reloadControllerInformation()
549 {
550  known_controllers_.clear();
551  if (controller_manager_)
552  {
553  std::vector<std::string> names;
554  controller_manager_->getControllersList(names);
555  for (const std::string& name : names)
556  {
557  std::vector<std::string> joints;
558  controller_manager_->getControllerJoints(name, joints);
559  ControllerInformation ci;
560  ci.name_ = name;
561  ci.joints_.insert(joints.begin(), joints.end());
562  known_controllers_[ci.name_] = ci;
563  }
564 
565  for (std::map<std::string, ControllerInformation>::iterator it = known_controllers_.begin();
566  it != known_controllers_.end(); ++it)
567  for (std::map<std::string, ControllerInformation>::iterator jt = known_controllers_.begin();
568  jt != known_controllers_.end(); ++jt)
569  if (it != jt)
570  {
571  std::vector<std::string> intersect;
572  std::set_intersection(it->second.joints_.begin(), it->second.joints_.end(), jt->second.joints_.begin(),
573  jt->second.joints_.end(), std::back_inserter(intersect));
574  if (!intersect.empty())
575  {
576  it->second.overlapping_controllers_.insert(jt->first);
577  jt->second.overlapping_controllers_.insert(it->first);
578  }
579  }
580  }

621  combined_joints.insert(ci.joints_.begin(), ci.joints_.end());

631  for (const std::string& combined_joint : combined_joints)

634  sac.str().c_str(), saj.str().c_str());

640 void TrajectoryExecutionManager::generateControllerCombination(std::size_t start_index, std::size_t controller_count,
641  const std::vector<std::string>& available_controllers,
642  std::vector<std::string>& selected_controllers,
643  std::vector<std::vector<std::string> >& selected_options,
644  const std::set<std::string>& actuated_joints)
645 {
646  if (selected_controllers.size() == controller_count)
647  {
648  if (checkControllerCombination(selected_controllers, actuated_joints))
649  selected_options.push_back(selected_controllers);
650  return;
651  }
652 
653  for (std::size_t i = start_index; i < available_controllers.size(); ++i)
654  {
655  bool overlap = false;
656  const ControllerInformation& ci = known_controllers_[available_controllers[i]];
657  for (std::size_t j = 0; j < selected_controllers.size() && !overlap; ++j)
658  {
659  if (ci.overlapping_controllers_.find(selected_controllers[j]) != ci.overlapping_controllers_.end())
660  overlap = true;
661  }
662  if (overlap)
663  continue;
664  selected_controllers.push_back(available_controllers[i]);
665  generateControllerCombination(i + 1, controller_count, available_controllers, selected_controllers,
666  selected_options, actuated_joints);
667  selected_controllers.pop_back();
668  }
669 }

673 struct OrderPotentialControllerCombination
674 {
675  bool operator()(const std::size_t a, const std::size_t b) const
676  {

678  if (nrdefault[a] > nrdefault[b])
679  return true;
680  if (nrdefault[a] < nrdefault[b])

684  if (nrjoints[a] < nrjoints[b])
685  return true;

690  if (nractive[a] < nractive[b])
691  return true;
692  if (nractive[a] > nractive[b])
693  return false;
694 
695  return false;
696  }

699  std::vector<std::size_t> nrdefault;
700  std::vector<std::size_t> nrjoints;
701  std::vector<std::size_t> nractive;
702 };

705 bool TrajectoryExecutionManager::findControllers(const std::set<std::string>& actuated_joints,
706  std::size_t controller_count,
707  const std::vector<std::string>& available_controllers,
708  std::vector<std::string>& selected_controllers)
709 {
710  // generate all combinations of controller_count controllers that operate on disjoint sets of joints
711  std::vector<std::string> work_area;
712  OrderPotentialControllerCombination order;
713  std::vector<std::vector<std::string> >& selected_options = order.selected_options;
714  generateControllerCombination(0, controller_count, available_controllers, work_area, selected_options,
715  actuated_joints);
716 
717  if (verbose_)
718  {
719  std::stringstream saj;
720  std::stringstream sac;
721  for (const std::string& available_controller : available_controllers)
722  sac << available_controller << " ";
723  for (const std::string& actuated_joint : actuated_joints)
724  saj << actuated_joint << " ";
725  RCLCPP_INFO(LOGGER, "Looking for %zu controllers among [ %s] that cover joints [ %s]. Found %zd options.",
726  controller_count, sac.str().c_str(), saj.str().c_str(), selected_options.size());
727  }
728 
729  // if none was found, this is a problem
730  if (selected_options.empty())
731  return false;
732 
733  // if only one was found, return it
734  if (selected_options.size() == 1)
735  {
736  selected_controllers.swap(selected_options[0]);
737  return true;
738  }
739 
740  // if more options were found, evaluate them all and return the best one
741 
742  // count how many default controllers are used in each reported option, and how many joints are actuated in total by
743  // the selected controllers,
744  // to use that in the ranking of the options
745  order.nrdefault.resize(selected_options.size(), 0);
746  order.nrjoints.resize(selected_options.size(), 0);
747  order.nractive.resize(selected_options.size(), 0);
748  for (std::size_t i = 0; i < selected_options.size(); ++i)
749  {
750  for (std::size_t k = 0; k < selected_options[i].size(); ++k)
751  {
752  updateControllerState(selected_options[i][k], DEFAULT_CONTROLLER_INFORMATION_VALIDITY_AGE);
753  const ControllerInformation& ci = known_controllers_[selected_options[i][k]];
754 
755  if (ci.state_.default_)
756  order.nrdefault[i]++;
757  if (ci.state_.active_)
758  order.nractive[i]++;
759  order.nrjoints[i] += ci.joints_.size();
760  }
761  }
762 
763  // define a bijection to compute the raking of the found options
764  std::vector<std::size_t> bijection(selected_options.size(), 0);
765  for (std::size_t i = 0; i < selected_options.size(); ++i)
766  bijection[i] = i;
767 
768  // sort the options
769  std::sort(bijection.begin(), bijection.end(), order);
770 
771  // depending on whether we are allowed to load & unload controllers,
772  // we have different preference on deciding between options
773  if (!manage_controllers_)
774  {
775  // if we can't load different options at will, just choose one that is already loaded
776  for (std::size_t i = 0; i < selected_options.size(); ++i)
777  if (areControllersActive(selected_options[bijection[i]]))
778  {
779  selected_controllers.swap(selected_options[bijection[i]]);
780  return true;
781  }
782  }
783 
784  // otherwise, just use the first valid option
785  selected_controllers.swap(selected_options[bijection[0]]);
786  return true;
787 }

968 bool TrajectoryExecutionManager::validate(const TrajectoryExecutionContext& context) const
969 {
970  if (allowed_start_tolerance_ == 0) // skip validation on this magic number
971  return true;
972 
973  RCLCPP_INFO(LOGGER, "Validating trajectory with allowed_start_tolerance %g", allowed_start_tolerance_);
974 
975  moveit::core::RobotStatePtr current_state;
976  if (!csm_->waitForCurrentState(node_->now()) || !(current_state = csm_->getCurrentState()))
977  {
978  RCLCPP_WARN(LOGGER, "Failed to validate trajectory: couldn't receive full current joint state within 1s");
979  return false;
980  }
981 
982  for (const auto& trajectory : context.trajectory_parts_)
983  {
984  if (!trajectory.joint_trajectory.points.empty())
985  {
986  // Check single-dof trajectory
987  const std::vector<double>& positions = trajectory.joint_trajectory.points.front().positions;
988  const std::vector<std::string>& joint_names = trajectory.joint_trajectory.joint_names;
989  if (positions.size() != joint_names.size())
990  {
991  RCLCPP_ERROR(LOGGER, "Wrong trajectory: #joints: %zu != #positions: %zu", joint_names.size(), positions.size());
992  return false;
993  }
994 
995  for (std::size_t i = 0, end = joint_names.size(); i < end; ++i)
996  {
997  const moveit::core::JointModel* jm = current_state->getJointModel(joint_names[i]);
998  if (!jm)
999  {
1000  RCLCPP_ERROR_STREAM(LOGGER, "Unknown joint in trajectory: " << joint_names[i]);
1001  return false;
1002  }
1003 
1004  double cur_position = current_state->getJointPositions(jm)[0];
1005  double traj_position = positions[i];
1006  // normalize positions and compare
1007  jm->enforcePositionBounds(&cur_position);
1008  jm->enforcePositionBounds(&traj_position);
1009  if (jm->distance(&cur_position, &traj_position) > allowed_start_tolerance_)
1010  {
1011  RCLCPP_ERROR(LOGGER,
1012  "\nInvalid Trajectory: start point deviates from current robot state more than %g"
1013  "\njoint '%s': expected: %g, current: %g",
1014  allowed_start_tolerance_, joint_names[i].c_str(), traj_position, cur_position);
1015  return false;
1016  }
1017  }
1018  }
1019  if (!trajectory.multi_dof_joint_trajectory.points.empty())
1020  {
1021  // Check multi-dof trajectory
1022  const std::vector<geometry_msgs::msg::Transform>& transforms =
1023  trajectory.multi_dof_joint_trajectory.points.front().transforms;
1024  const std::vector<std::string>& joint_names = trajectory.multi_dof_joint_trajectory.joint_names;
1025  if (transforms.size() != joint_names.size())
1026  {
1027  RCLCPP_ERROR(LOGGER, "Wrong trajectory: #joints: %zu != #transforms: %zu", joint_names.size(),
1028  transforms.size());
1029  return false;
1030  }
1031 
1032  for (std::size_t i = 0, end = joint_names.size(); i < end; ++i)
1033  {
1034  const moveit::core::JointModel* jm = current_state->getJointModel(joint_names[i]);
1035  if (!jm)
1036  {
1037  RCLCPP_ERROR_STREAM(LOGGER, "Unknown joint in trajectory: " << joint_names[i]);
1038  return false;
1039  }
1040 
1041  // compute difference (offset vector and rotation angle) between current transform
1042  // and start transform in trajectory
1043  Eigen::Isometry3d cur_transform, start_transform;
1044  // computeTransform() computes a valid isometry by contract
1045  jm->computeTransform(current_state->getJointPositions(jm), cur_transform);
1046  start_transform = tf2::transformToEigen(transforms[i]);
1047  ASSERT_ISOMETRY(start_transform) // unsanitized input, could contain a non-isometry
1048  Eigen::Vector3d offset = cur_transform.translation() - start_transform.translation();
1049  Eigen::AngleAxisd rotation;
1050  rotation.fromRotationMatrix(cur_transform.linear().transpose() * start_transform.linear());
1051  if ((offset.array() > allowed_start_tolerance_).any() || rotation.angle() > allowed_start_tolerance_)
1052  {
1053  RCLCPP_ERROR_STREAM(LOGGER, "\nInvalid Trajectory: start point deviates from current robot state more than "
1054  << allowed_start_tolerance_ << "\nmulti-dof joint '" << joint_names[i]
1055  << "': pos delta: " << offset.transpose()
1056  << " rot delta: " << rotation.angle());
1057  return false;
1058  }
1059  }
1060  }
1061  }
1062  return true;
1063 }

1065 bool TrajectoryExecutionManager::configure(TrajectoryExecutionContext& context,
1066  const moveit_msgs::msg::RobotTrajectory& trajectory,
1067  const std::vector<std::string>& controllers)
1068 {
1069  if (trajectory.multi_dof_joint_trajectory.points.empty() && trajectory.joint_trajectory.points.empty())
1070  {
1071  // empty trajectories don't need to configure anything
1072  return true;
1073  }
1074  std::set<std::string> actuated_joints;
1075 
1076  auto is_actuated = [this](const std::string& joint_name) -> bool {
1077  const moveit::core::JointModel* jm = robot_model_->getJointModel(joint_name);
1078  return (jm && !jm->isPassive() && !jm->getMimic() && jm->getType() != moveit::core::JointModel::FIXED);
1079  };
1080  for (const std::string& joint_name : trajectory.multi_dof_joint_trajectory.joint_names)
1081  if (is_actuated(joint_name))
1082  actuated_joints.insert(joint_name);
1083  for (const std::string& joint_name : trajectory.joint_trajectory.joint_names)
1084  if (is_actuated(joint_name))
1085  actuated_joints.insert(joint_name);
1086 
1087  if (actuated_joints.empty())
1088  {
1089  RCLCPP_WARN(LOGGER, "The trajectory to execute specifies no joints");
1090  return false;
1091  }
1092 
1093  if (controllers.empty())
1094  {
1095  bool retry = true;
1096  bool reloaded = false;
1097  while (retry)
1098  {
1099  retry = false;
1100  std::vector<std::string> all_controller_names;
1101  for (std::map<std::string, ControllerInformation>::const_iterator it = known_controllers_.begin();
1102  it != known_controllers_.end(); ++it)
1103  all_controller_names.push_back(it->first);
1104  if (selectControllers(actuated_joints, all_controller_names, context.controllers_))
1105  {
1106  if (distributeTrajectory(trajectory, context.controllers_, context.trajectory_parts_))
1107  return true;
1108  }
1109  else
1110  {
1111  // maybe we failed because we did not have a complete list of controllers
1112  if (!reloaded)
1113  {
1114  reloadControllerInformation();
1115  reloaded = true;
1116  retry = true;
1117  }
1118  }
1119  }
1120  }
1121  else
1122  {
1123  // check if the specified controllers are valid names;
1124  // if they appear not to be, try to reload the controller information, just in case they are new in the system
1125  bool reloaded = false;
1126  for (const std::string& controller : controllers)
1127  if (known_controllers_.find(controller) == known_controllers_.end())
1128  {
1129  reloadControllerInformation();
1130  reloaded = true;
1131  break;
1132  }
1133  if (reloaded)
1134  for (const std::string& controller : controllers)
1135  if (known_controllers_.find(controller) == known_controllers_.end())
1136  {
1137  RCLCPP_ERROR(LOGGER, "Controller '%s' is not known", controller.c_str());
1138  return false;
1139  }
1140  if (selectControllers(actuated_joints, controllers, context.controllers_))
1141  {
1142  if (distributeTrajectory(trajectory, context.controllers_, context.trajectory_parts_))

1303  const PathSegmentCompleteCallback& part_callback, bool auto_clear)
1304 {

1306  if (execution_complete_)
1307  {
1308  last_execution_status_ = moveit_controller_manager::ExecutionStatus::ABORTED;
1309  if (callback)
1310  callback(last_execution_status_);
1311  return;
1312  }
1313

1315  // assume everything will be OK
1316  last_execution_status_ = moveit_controller_manager::ExecutionStatus::SUCCEEDED;

1342  execution_state_mutex_.unlock();

1428  }

1454  context.trajectory_parts_[i].multi_dof_joint_trajectory.points.back().time_from_start));

1528  last_execution_status_ = moveit_controller_manager::ExecutionStatus::TIMED_OUT;
1529  result = false;
1530  break;
1531  }
1532  }
1533  else
1534  handle->waitForExecution();
1535 
1536  // if something made the trajectory stop, we stop this thread too
1537  if (execution_complete_)
1538  {
1539  result = false;
1540  break;
1541  }
1542  else if (handle->getLastExecutionStatus() != moveit_controller_manager::ExecutionStatus::SUCCEEDED)
1543  {

1546  last_execution_status_ = handle->getLastExecutionStatus();

1548  }
1549  }
1550

1554 
1555  // clear the time index
1556  time_index_mutex_.lock();

1559  time_index_mutex_.unlock();

1571 bool TrajectoryExecutionManager::waitForRobotToStop(const TrajectoryExecutionContext& context, double wait_time)
1572 {
1573  // skip waiting for convergence?
1574  if (allowed_start_tolerance_ == 0 || !wait_for_trajectory_completion_)
1575  {
1576  RCLCPP_INFO(LOGGER, "Not waiting for trajectory completion");
1577  return true;
1578  }
1579 
1580  auto start = std::chrono::system_clock::now();
1581  double time_remaining = wait_time;
1582 
1583  moveit::core::RobotStatePtr prev_state, cur_state;
1584  prev_state = csm_->getCurrentState();
1585  prev_state->enforceBounds();
1586 
1587  // assume robot stopped when 3 consecutive checks yield the same robot state
1588  unsigned int no_motion_count = 0; // count iterations with no motion
1589  while (time_remaining > 0. && no_motion_count < 3)
1590  {
1591  if (!csm_->waitForCurrentState(node_->now(), time_remaining) || !(cur_state = csm_->getCurrentState()))

1598  time_remaining = wait_time - elapsed_seconds.count(); // remaining wait_time

1601  bool moved = false;
1602  for (const auto& trajectory : context.trajectory_parts_)
1603  {
1604  const std::vector<std::string>& joint_names = trajectory.joint_trajectory.joint_names;

1617  break;
1618  }
1619  }

1621

1624

1626  }
