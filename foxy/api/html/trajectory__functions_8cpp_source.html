43 bool pilz_industrial_motion_planner::computePoseIK(const moveit::core::RobotModelConstPtr& robot_model,
44  const std::string& group_name, const std::string& link_name,
45  const Eigen::Isometry3d& pose, const std::string& frame_id,
46  const std::map<std::string, double>& seed,
47  std::map<std::string, double>& solution, bool check_self_collision,
48  const double timeout)
49 {
50  if (!robot_model->hasJointModelGroup(group_name))
51  {
52  ROS_ERROR_STREAM("Robot model has no planning group named as " << group_name);
53  return false;
54  }
55 
56  if (!robot_model->getJointModelGroup(group_name)->canSetStateFromIK(link_name))
57  {
58  ROS_ERROR_STREAM("No valid IK solver exists for " << link_name << " in planning group " << group_name);
59  return false;
60  }
61 
62  if (frame_id != robot_model->getModelFrame())
63  {
64  ROS_ERROR_STREAM("Given frame (" << frame_id << ") is unequal to model frame(" << robot_model->getModelFrame()
65  << ")");
66  return false;
67  }
68 
69  robot_state::RobotState rstate(robot_model);
70  // By setting the robot state to default values, we basically allow
71  // the user of this function to supply an incomplete or even empty seed.
72  rstate.setToDefaultValues();
73  rstate.setVariablePositions(seed);
74 
75  moveit::core::GroupStateValidityCallbackFn ik_constraint_function;
76  ik_constraint_function =
77  boost::bind(&pilz_industrial_motion_planner::isStateColliding, check_self_collision, robot_model, _1, _2, _3);
78 
79  // call ik
80  if (rstate.setFromIK(robot_model->getJointModelGroup(group_name), pose, link_name, timeout, ik_constraint_function))
81  {
82  // copy the solution
83  for (const auto& joint_name : robot_model->getJointModelGroup(group_name)->getActiveJointModelNames())
84  {
85  solution[joint_name] = rstate.getVariablePosition(joint_name);
86  }
87  return true;
88  }
89  else
90  {
91  ROS_ERROR_STREAM("Inverse kinematics for pose \n" << pose.translation() << " has no solution.");
92  return false;
93  }
94 }

134 bool pilz_industrial_motion_planner::verifySampleJointLimits(
135  const std::map<std::string, double>& position_last, const std::map<std::string, double>& velocity_last,
136  const std::map<std::string, double>& position_current, double duration_last, double duration_current,
137  const pilz_industrial_motion_planner::JointLimitsContainer& joint_limits)
138 {
139  const double epsilon = 10e-6;
140  if (duration_current <= epsilon)
141  {
142  ROS_ERROR("Sample duration too small, cannot compute the velocity");
143  return false;
144  }
145 
146  double velocity_current, acceleration_current;
147 
148  for (const auto& pos : position_current)
149  {
150  velocity_current = (pos.second - position_last.at(pos.first)) / duration_current;
151 
152  if (!joint_limits.verifyVelocityLimit(pos.first, velocity_current))
153  {
154  ROS_ERROR_STREAM("Joint velocity limit of " << pos.first << " violated. Set the velocity scaling factor lower!"
155  << " Actual joint velocity is " << velocity_current
156  << ", while the limit is "
157  << joint_limits.getLimit(pos.first).max_velocity << ". ");
158  return false;
159  }
160 
161  acceleration_current = (velocity_current - velocity_last.at(pos.first)) / (duration_last + duration_current) * 2;
162  // acceleration case
163  if (fabs(velocity_last.at(pos.first)) <= fabs(velocity_current))
164  {
165  if (joint_limits.getLimit(pos.first).has_acceleration_limits &&
166  fabs(acceleration_current) > fabs(joint_limits.getLimit(pos.first).max_acceleration))
167  {
168  ROS_ERROR_STREAM("Joint acceleration limit of "
169  << pos.first << " violated. Set the acceleration scaling factor lower!"
170  << " Actual joint acceleration is " << acceleration_current << ", while the limit is "
171  << joint_limits.getLimit(pos.first).max_acceleration << ". ");
172  return false;
173  }
174  }
175  // deceleration case
176  else
177  {
178  if (joint_limits.getLimit(pos.first).has_deceleration_limits &&
179  fabs(acceleration_current) > fabs(joint_limits.getLimit(pos.first).max_deceleration))
180  {
181  ROS_ERROR_STREAM("Joint deceleration limit of "
182  << pos.first << " violated. Set the acceleration scaling factor lower!"
183  << " Actual joint deceleration is " << acceleration_current << ", while the limit is "
184  << joint_limits.getLimit(pos.first).max_deceleration << ". ");
185  return false;
186  }
187  }
188  }
189 
190  return true;
191 }

193 bool pilz_industrial_motion_planner::generateJointTrajectory(
194  const moveit::core::RobotModelConstPtr& robot_model,
195  const pilz_industrial_motion_planner::JointLimitsContainer& joint_limits, const KDL::Trajectory& trajectory,
196  const std::string& group_name, const std::string& link_name,
197  const std::map<std::string, double>& initial_joint_position, const double& sampling_time,
198  trajectory_msgs::JointTrajectory& joint_trajectory, moveit_msgs::MoveItErrorCodes& error_code,
199  bool check_self_collision)
200 {
201  ROS_DEBUG("Generate joint trajectory from a Cartesian trajectory.");
202 
203  ros::Time generation_begin = ros::Time::now();
204 
205  // generate the time samples
206  const double epsilon = 10e-06; // avoid adding the last time sample twice
207  std::vector<double> time_samples;
208  for (double t_sample = 0.0; t_sample < trajectory.Duration() - epsilon; t_sample += sampling_time)
209  {
210  time_samples.push_back(t_sample);
211  }
212  time_samples.push_back(trajectory.Duration());
213 
214  // sample the trajectory and solve the inverse kinematics
215  Eigen::Isometry3d pose_sample;
216  std::map<std::string, double> ik_solution_last, ik_solution, joint_velocity_last;
217  ik_solution_last = initial_joint_position;
218  for (const auto& item : ik_solution_last)
219  {
220  joint_velocity_last[item.first] = 0.0;
221  }
222 
223  for (std::vector<double>::const_iterator time_iter = time_samples.begin(); time_iter != time_samples.end();
224  ++time_iter)
225  {
226  tf2::fromMsg(tf2::toMsg(trajectory.Pos(*time_iter)), pose_sample);
227 
228  if (!computePoseIK(robot_model, group_name, link_name, pose_sample, robot_model->getModelFrame(), ik_solution_last,
229  ik_solution, check_self_collision))
230  {
231  ROS_ERROR("Failed to compute inverse kinematics solution for sampled "
232  "Cartesian pose.");
233  error_code.val = moveit_msgs::MoveItErrorCodes::NO_IK_SOLUTION;
234  joint_trajectory.points.clear();
235  return false;
236  }
237 
238  // check the joint limits
239  double duration_current_sample = sampling_time;
240  // last interval can be shorter than the sampling time
241  if (time_iter == (time_samples.end() - 1) && time_samples.size() > 1)
242  {
243  duration_current_sample = *time_iter - *(time_iter - 1);
244  }
245  if (time_samples.size() == 1)
246  {
247  duration_current_sample = *time_iter;
248  }
249 
250  // skip the first sample with zero time from start for limits checking
251  if (time_iter != time_samples.begin() &&
252  !verifySampleJointLimits(ik_solution_last, joint_velocity_last, ik_solution, sampling_time,
253  duration_current_sample, joint_limits))
254  {
255  ROS_ERROR_STREAM("Inverse kinematics solution at "
256  << *time_iter << "s violates the joint velocity/acceleration/deceleration limits.");
257  error_code.val = moveit_msgs::MoveItErrorCodes::PLANNING_FAILED;
258  joint_trajectory.points.clear();
259  return false;
260  }
261 
262  // fill the point with joint values
263  trajectory_msgs::JointTrajectoryPoint point;
264 
265  // set joint names
266  joint_trajectory.joint_names.clear();
267  for (const auto& start_joint : initial_joint_position)
268  {
269  joint_trajectory.joint_names.push_back(start_joint.first);
270  }
271 
272  point.time_from_start = ros::Duration(*time_iter);
273  for (const auto& joint_name : joint_trajectory.joint_names)
274  {
275  point.positions.push_back(ik_solution.at(joint_name));
276 
277  if (time_iter != time_samples.begin() && time_iter != time_samples.end() - 1)
278  {
279  double joint_velocity =
280  (ik_solution.at(joint_name) - ik_solution_last.at(joint_name)) / duration_current_sample;
281  point.velocities.push_back(joint_velocity);
282  point.accelerations.push_back((joint_velocity - joint_velocity_last.at(joint_name)) /
283  (duration_current_sample + sampling_time) * 2);
284  joint_velocity_last[joint_name] = joint_velocity;
285  }
286  else
287  {
288  point.velocities.push_back(0.);
289  point.accelerations.push_back(0.);
290  joint_velocity_last[joint_name] = 0.;
291  }
292  }
293 
294  // update joint trajectory
295  joint_trajectory.points.push_back(point);
296  ik_solution_last = ik_solution;
297  }
298 
299  error_code.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
300  double duration_ms = (ros::Time::now() - generation_begin).toSec() * 1000;
301  ROS_DEBUG_STREAM("Generate trajectory (N-Points: " << joint_trajectory.points.size() << ") took " << duration_ms
302  << " ms | " << duration_ms / joint_trajectory.points.size()
303  << " ms per Point");
304 
305  return true;
306 }

308 bool pilz_industrial_motion_planner::generateJointTrajectory(
309  const moveit::core::RobotModelConstPtr& robot_model,
310  const pilz_industrial_motion_planner::JointLimitsContainer& joint_limits,
311  const pilz_industrial_motion_planner::CartesianTrajectory& trajectory, const std::string& group_name,
312  const std::string& link_name, const std::map<std::string, double>& initial_joint_position,
313  const std::map<std::string, double>& initial_joint_velocity, trajectory_msgs::JointTrajectory& joint_trajectory,
314  moveit_msgs::MoveItErrorCodes& error_code, bool check_self_collision)
315 {
316  ROS_DEBUG("Generate joint trajectory from a Cartesian trajectory.");
317 
318  ros::Time generation_begin = ros::Time::now();
319 
320  std::map<std::string, double> ik_solution_last = initial_joint_position;
321  std::map<std::string, double> joint_velocity_last = initial_joint_velocity;
322  double duration_last = 0;
323  double duration_current = 0;
324  joint_trajectory.joint_names.clear();
325  for (const auto& joint_position : ik_solution_last)
326  {
327  joint_trajectory.joint_names.push_back(joint_position.first);
328  }
329  std::map<std::string, double> ik_solution;
330  for (size_t i = 0; i < trajectory.points.size(); ++i)
331  {
332  // compute inverse kinematics
333  if (!computePoseIK(robot_model, group_name, link_name, trajectory.points.at(i).pose, robot_model->getModelFrame(),
334  ik_solution_last, ik_solution, check_self_collision))
335  {
336  ROS_ERROR("Failed to compute inverse kinematics solution for sampled "
337  "Cartesian pose.");
338  error_code.val = moveit_msgs::MoveItErrorCodes::NO_IK_SOLUTION;
339  joint_trajectory.points.clear();
340  return false;
341  }
342 
343  // verify the joint limits
344  if (i == 0)
345  {
346  duration_current = trajectory.points.front().time_from_start.toSec();
347  duration_last = duration_current;
348  }
349  else
350  {
351  duration_current =
352  trajectory.points.at(i).time_from_start.toSec() - trajectory.points.at(i - 1).time_from_start.toSec();
353  }
354 
355  if (!verifySampleJointLimits(ik_solution_last, joint_velocity_last, ik_solution, duration_last, duration_current,
356  joint_limits))
357  {
358  // LCOV_EXCL_START since the same code was captured in a test in the other
359  // overload generateJointTrajectory(...,
360  // KDL::Trajectory, ...)
361  // TODO: refactor to avoid code duplication.
362  ROS_ERROR_STREAM("Inverse kinematics solution of the " << i
363  << "th sample violates the joint "
364  "velocity/acceleration/deceleration limits.");
365  error_code.val = moveit_msgs::MoveItErrorCodes::PLANNING_FAILED;
366  joint_trajectory.points.clear();
367  return false;
368  // LCOV_EXCL_STOP
369  }
370 
371  // compute the waypoint
372  trajectory_msgs::JointTrajectoryPoint waypoint_joint;
373  waypoint_joint.time_from_start = ros::Duration(trajectory.points.at(i).time_from_start);
374  for (const auto& joint_name : joint_trajectory.joint_names)
375  {
376  waypoint_joint.positions.push_back(ik_solution.at(joint_name));
377  double joint_velocity = (ik_solution.at(joint_name) - ik_solution_last.at(joint_name)) / duration_current;
378  waypoint_joint.velocities.push_back(joint_velocity);
379  waypoint_joint.accelerations.push_back((joint_velocity - joint_velocity_last.at(joint_name)) /
380  (duration_current + duration_last) * 2);
381  // update the joint velocity
382  joint_velocity_last[joint_name] = joint_velocity;
383  }
384 
385  // update joint trajectory
386  joint_trajectory.points.push_back(waypoint_joint);
387  ik_solution_last = ik_solution;
388  duration_last = duration_current;
389  }
390 
391  error_code.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
392 
393  double duration_ms = (ros::Time::now() - generation_begin).toSec() * 1000;
394  ROS_DEBUG_STREAM("Generate trajectory (N-Points: " << joint_trajectory.points.size() << ") took " << duration_ms
395  << " ms | " << duration_ms / joint_trajectory.points.size()
396  << " ms per Point");
397 
398  return true;
399 }

401 bool pilz_industrial_motion_planner::determineAndCheckSamplingTime(
402  const robot_trajectory::RobotTrajectoryPtr& first_trajectory,
403  const robot_trajectory::RobotTrajectoryPtr& second_trajectory, double epsilon, double& sampling_time)
404 {
405  // The last sample is ignored because it is allowed to violate the sampling
406  // time.
407  std::size_t n1 = first_trajectory->getWayPointCount() - 1;
408  std::size_t n2 = second_trajectory->getWayPointCount() - 1;
409  if ((n1 < 2) && (n2 < 2))
410  {
411  ROS_ERROR_STREAM("Both trajectories do not have enough points to determine "
412  "sampling time.");
413  return false;
414  }
415 
416  if (n1 >= 2)
417  {
418  sampling_time = first_trajectory->getWayPointDurationFromPrevious(1);
419  }
420  else
421  {
422  sampling_time = second_trajectory->getWayPointDurationFromPrevious(1);
423  }
424 
425  for (std::size_t i = 1; i < std::max(n1, n2); ++i)
426  {
427  if (i < n1)
428  {
429  if (fabs(sampling_time - first_trajectory->getWayPointDurationFromPrevious(i)) > epsilon)
430  {
431  ROS_ERROR_STREAM("First trajectory violates sampline time " << sampling_time << " between points " << (i - 1)
432  << "and " << i << " (indices).");
433  return false;
434  }
435  }
436 
437  if (i < n2)
438  {
439  if (fabs(sampling_time - second_trajectory->getWayPointDurationFromPrevious(i)) > epsilon)
440  {
441  ROS_ERROR_STREAM("Second trajectory violates sampline time " << sampling_time << " between points " << (i - 1)
442  << "and " << i << " (indices).");
443  return false;
444  }
445  }
446  }
447 
448  return true;
449 }

451 bool pilz_industrial_motion_planner::isRobotStateEqual(const moveit::core::RobotState& state1,
452  const moveit::core::RobotState& state2,
453  const std::string& joint_group_name, double epsilon)
454 {
455  Eigen::VectorXd joint_position_1, joint_position_2;
456 
457  state1.copyJointGroupPositions(joint_group_name, joint_position_1);
458  state2.copyJointGroupPositions(joint_group_name, joint_position_2);
459 
460  if ((joint_position_1 - joint_position_2).norm() > epsilon)
461  {
462  ROS_DEBUG_STREAM("Joint positions of the two states are different. state1: " << joint_position_1
463  << " state2: " << joint_position_2);
464  return false;
465  }
466 
467  Eigen::VectorXd joint_velocity_1, joint_velocity_2;
468 
469  state1.copyJointGroupVelocities(joint_group_name, joint_velocity_1);
470  state2.copyJointGroupVelocities(joint_group_name, joint_velocity_2);
471 
472  if ((joint_velocity_1 - joint_velocity_2).norm() > epsilon)
473  {
474  ROS_DEBUG_STREAM("Joint velocities of the two states are different. state1: " << joint_velocity_1
475  << " state2: " << joint_velocity_2);
476  return false;
477  }
478 
479  Eigen::VectorXd joint_acc_1, joint_acc_2;
480 
481  state1.copyJointGroupAccelerations(joint_group_name, joint_acc_1);
482  state2.copyJointGroupAccelerations(joint_group_name, joint_acc_2);
483 
484  if ((joint_acc_1 - joint_acc_2).norm() > epsilon)
485  {
486  ROS_DEBUG_STREAM("Joint accelerations of the two states are different. state1: " << joint_acc_1
487  << " state2: " << joint_acc_2);
488  return false;
489  }
490 
491  return true;
492 }

513 bool pilz_industrial_motion_planner::linearSearchIntersectionPoint(const std::string& link_name,
514  const Eigen::Vector3d& center_position,
515  const double& r,
516  const robot_trajectory::RobotTrajectoryPtr& traj,
517  bool inverseOrder, std::size_t& index)
518 {
519  ROS_DEBUG("Start linear search for intersection point.");
520 
521  const size_t waypoint_num = traj->getWayPointCount();
522 
523  if (inverseOrder)
524  {
525  for (size_t i = waypoint_num - 1; i > 0; --i)
526  {
527  if (intersectionFound(center_position, traj->getWayPointPtr(i)->getFrameTransform(link_name).translation(),
528  traj->getWayPointPtr(i - 1)->getFrameTransform(link_name).translation(), r))
529  {
530  index = i;
531  return true;
532  }
533  }
534  }
535  else
536  {
537  for (size_t i = 0; i < waypoint_num - 1; ++i)
538  {
539  if (intersectionFound(center_position, traj->getWayPointPtr(i)->getFrameTransform(link_name).translation(),
540  traj->getWayPointPtr(i + 1)->getFrameTransform(link_name).translation(), r))
541  {
542  index = i;
543  return true;
544  }
545  }
546  }
547 
548  return false;
549 }

558 bool pilz_industrial_motion_planner::isStateColliding(const bool test_for_self_collision,
559  const moveit::core::RobotModelConstPtr& robot_model,
560  robot_state::RobotState* rstate,
561  const robot_state::JointModelGroup* const group,
562  const double* const ik_solution)
563 {
564  if (!test_for_self_collision)
565  {
566  return true;
567  }
568 
569  rstate->setJointGroupPositions(group, ik_solution);
570  rstate->update();
571  collision_detection::CollisionRequest collision_req;
572  collision_req.group_name = group->getName();

575 
576  return !collision_res.collision;
