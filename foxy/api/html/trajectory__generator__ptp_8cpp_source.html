47 TrajectoryGeneratorPTP::TrajectoryGeneratorPTP(const robot_model::RobotModelConstPtr& robot_model,
48  const LimitsContainer& planner_limits)
49  : TrajectoryGenerator::TrajectoryGenerator(robot_model, planner_limits)
50 {
51  if (!planner_limits_.hasJointLimits())
52  {
53  throw TrajectoryGeneratorInvalidLimitsException("joint limit not set");
54  }
55 
56  joint_limits_ = planner_limits_.getJointLimitContainer();
57 
58  // collect most strict joint limits for each group in robot model
59  for (const auto& jmg : robot_model->getJointModelGroups())
60  {
61  auto active_joints = jmg->getActiveJointModelNames();
62 
63  // no active joints
64  if (active_joints.empty())
65  {
66  continue;
67  }
68 
69  JointLimit most_strict_limit = joint_limits_.getCommonLimit(active_joints);
70 
71  if (!most_strict_limit.has_velocity_limits)
72  {
73  ROS_ERROR_STREAM("velocity limit not set for group " << jmg->getName());
74  throw TrajectoryGeneratorInvalidLimitsException("velocity limit not set for group " + jmg->getName());
75  }
76  if (!most_strict_limit.has_acceleration_limits)
77  {
78  ROS_ERROR_STREAM("acceleration limit not set for group " << jmg->getName());
79  throw TrajectoryGeneratorInvalidLimitsException("acceleration limit not set for group " + jmg->getName());
80  }
81  if (!most_strict_limit.has_deceleration_limits)
82  {
83  ROS_ERROR_STREAM("deceleration limit not set for group " << jmg->getName());
84  throw TrajectoryGeneratorInvalidLimitsException("deceleration limit not set for group " + jmg->getName());
85  }
86 
87  most_strict_limits_.insert(std::pair<std::string, JointLimit>(jmg->getName(), most_strict_limit));
88  }
89 
90  ROS_INFO("Initialized Point-to-Point Trajectory Generator.");
91 }

93 void TrajectoryGeneratorPTP::planPTP(const std::map<std::string, double>& start_pos,
94  const std::map<std::string, double>& goal_pos,
95  trajectory_msgs::JointTrajectory& joint_trajectory, const std::string& group_name,
96  const double& velocity_scaling_factor, const double& acceleration_scaling_factor,
97  const double& sampling_time)
98 {
99  // initialize joint names
100  for (const auto& item : goal_pos)
101  {
102  joint_trajectory.joint_names.push_back(item.first);
103  }
104 
105  // check if goal already reached
106  bool goal_reached = true;
107  for (auto const& goal : goal_pos)
108  {
109  if (fabs(start_pos.at(goal.first) - goal.second) >= MIN_MOVEMENT)
110  {
111  goal_reached = false;
112  break;
113  }
114  }
115  if (goal_reached)
116  {
117  ROS_INFO_STREAM("Goal already reached, set one goal point explicitly.");
118  if (joint_trajectory.points.empty())
119  {
120  trajectory_msgs::JointTrajectoryPoint point;
121  point.time_from_start = ros::Duration(sampling_time);
122  for (const std::string& joint_name : joint_trajectory.joint_names)
123  {
124  point.positions.push_back(start_pos.at(joint_name));
125  point.velocities.push_back(0);
126  point.accelerations.push_back(0);
127  }
128  joint_trajectory.points.push_back(point);
129  }
130  return;
131  }
132 
133  // compute the fastest trajectory and choose the slowest joint as leading axis
134  std::string leading_axis = joint_trajectory.joint_names.front();
135  double max_duration = -1.0;
136 
137  std::map<std::string, VelocityProfileATrap> velocity_profile;
138  for (const auto& joint_name : joint_trajectory.joint_names)
139  {
140  // create vecocity profile if necessary
141  velocity_profile.insert(std::make_pair(
142  joint_name,
143  VelocityProfileATrap(velocity_scaling_factor * most_strict_limits_.at(group_name).max_velocity,
144  acceleration_scaling_factor * most_strict_limits_.at(group_name).max_acceleration,
145  acceleration_scaling_factor * most_strict_limits_.at(group_name).max_deceleration)));
146 
147  velocity_profile.at(joint_name).SetProfile(start_pos.at(joint_name), goal_pos.at(joint_name));
148  if (velocity_profile.at(joint_name).Duration() > max_duration)
149  {
150  max_duration = velocity_profile.at(joint_name).Duration();
151  leading_axis = joint_name;
152  }
153  }
154 
155  // Full Synchronization
156  // This should only work if all axes have same max_vel, max_acc, max_dec
157  // values
158  // reset the velocity profile for other joints
159  double acc_time = velocity_profile.at(leading_axis).firstPhaseDuration();
160  double const_time = velocity_profile.at(leading_axis).secondPhaseDuration();
161  double dec_time = velocity_profile.at(leading_axis).thirdPhaseDuration();
162 
163  for (const auto& joint_name : joint_trajectory.joint_names)
164  {
165  if (joint_name != leading_axis)
166  {
167  // make full synchronization
168  // causes the program to terminate if acc_time<=0 or dec_time<=0 (should
169  // be prevented by goal_reached block above)
170  // by using the most strict limit, the following should always return true
171  if (!velocity_profile.at(joint_name)
172  .setProfileAllDurations(start_pos.at(joint_name), goal_pos.at(joint_name), acc_time, const_time,
173  dec_time))
174  // LCOV_EXCL_START
175  {
176  std::stringstream error_str;
177  error_str << "TrajectoryGeneratorPTP::planPTP(): Can not synchronize "
178  "velocity profile of axis "
179  << joint_name << " with leading axis " << leading_axis;
180  throw PtpVelocityProfileSyncFailed(error_str.str());
181  }
182  // LCOV_EXCL_STOP
183  }
184  }
185 
186  // first generate the time samples
187  std::vector<double> time_samples;
188  for (double t_sample = 0.0; t_sample < max_duration; t_sample += sampling_time)
189  {
190  time_samples.push_back(t_sample);
191  }
192  // add last time
193  time_samples.push_back(max_duration);
194 
195  // construct joint trajectory point
196  for (double time_stamp : time_samples)
197  {
198  trajectory_msgs::JointTrajectoryPoint point;
199  point.time_from_start = ros::Duration(time_stamp);
200  for (std::string& joint_name : joint_trajectory.joint_names)
201  {
202  point.positions.push_back(velocity_profile.at(joint_name).Pos(time_stamp));
203  point.velocities.push_back(velocity_profile.at(joint_name).Vel(time_stamp));
204  point.accelerations.push_back(velocity_profile.at(joint_name).Acc(time_stamp));
205  }
206  joint_trajectory.points.push_back(point);
207  }
208 
209  // Set last point velocity and acceleration to zero
210  std::fill(joint_trajectory.points.back().velocities.begin(), joint_trajectory.points.back().velocities.end(), 0.0);
211  std::fill(joint_trajectory.points.back().accelerations.begin(), joint_trajectory.points.back().accelerations.end(),
212  0.0);
213 }

215 void TrajectoryGeneratorPTP::extractMotionPlanInfo(const planning_interface::MotionPlanRequest& req,
216  MotionPlanInfo& info) const
217 {
218  info.group_name = req.group_name;
219 
220  // extract start state information
221  info.start_joint_position.clear();
222  for (std::size_t i = 0; i < req.start_state.joint_state.name.size(); ++i)
223  {
224  info.start_joint_position[req.start_state.joint_state.name[i]] = req.start_state.joint_state.position[i];
225  }
226 
227  // extract goal
228  info.goal_joint_position.clear();
229  if (!req.goal_constraints.at(0).joint_constraints.empty())
230  {
231  for (const auto& joint_constraint : req.goal_constraints.at(0).joint_constraints)
232  {
233  info.goal_joint_position[joint_constraint.joint_name] = joint_constraint.position;
234  }
235  }
236  // slove the ik
237  else
238  {
239  geometry_msgs::Point p =
240  req.goal_constraints.at(0).position_constraints.at(0).constraint_region.primitive_poses.at(0).position;
241  p.x -= req.goal_constraints.at(0).position_constraints.at(0).target_point_offset.x;
242  p.y -= req.goal_constraints.at(0).position_constraints.at(0).target_point_offset.y;
243  p.z -= req.goal_constraints.at(0).position_constraints.at(0).target_point_offset.z;
244 
245  geometry_msgs::Pose pose;
246  pose.position = p;
247  pose.orientation = req.goal_constraints.at(0).orientation_constraints.at(0).orientation;
248  Eigen::Isometry3d pose_eigen;
249  normalizeQuaternion(pose.orientation);
250  tf2::fromMsg(pose, pose_eigen);
251  if (!computePoseIK(robot_model_, req.group_name, req.goal_constraints.at(0).position_constraints.at(0).link_name,
252  pose_eigen, robot_model_->getModelFrame(), info.start_joint_position, info.goal_joint_position))
253  {
254  throw PtpNoIkSolutionForGoalPose("No IK solution for goal pose");
255  }
256  }
257 }
