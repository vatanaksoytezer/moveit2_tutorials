59 void planning_scene_monitor::TrajectoryMonitor::setSamplingFrequency(double sampling_frequency)
60 {
61  if (sampling_frequency != sampling_frequency_)
62  return; // silently return if nothing changes
63 
64  if (sampling_frequency <= std::numeric_limits<double>::epsilon())
65  RCLCPP_ERROR(LOGGER, "The sampling frequency for trajectory states should be positive");
66  else
67  RCLCPP_DEBUG(LOGGER, "Setting trajectory sampling frequency to %.1f", sampling_frequency);
68  sampling_frequency_ = sampling_frequency;

106 void planning_scene_monitor::TrajectoryMonitor::recordStates()
107 {
108  if (!current_state_monitor_)
109  return;
110 
111  rclcpp::Rate rate(sampling_frequency_);
112 
113  while (record_states_thread_)
114  {
115  rate.sleep();
116  std::pair<moveit::core::RobotStatePtr, rclcpp::Time> state = current_state_monitor_->getCurrentStateAndTime();
117  if (trajectory_.empty())
118  {
119  trajectory_.addSuffixWayPoint(state.first, 0.0);
120  trajectory_start_time_ = state.second;
121  last_recorded_state_time_ = state.second;
122  }
123  else
124  {
125  trajectory_.addSuffixWayPoint(state.first, (state.second - last_recorded_state_time_).seconds());
126  last_recorded_state_time_ = state.second;
127  }
128  if (state_add_callback_)
129  state_add_callback_(state.first, state.second);
130  }
131 }
