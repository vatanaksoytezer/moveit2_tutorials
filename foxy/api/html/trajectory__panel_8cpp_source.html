59  maximum_label_ = new QLabel(QString::number(slider_->maximum()));

93 void TrajectoryPanel::update(int way_point_count)
94 {
95  int max_way_point = std::max(0, way_point_count - 1);
96 
97  slider_->setEnabled(way_point_count != 0);
98  button_->setEnabled(way_point_count != 0);
99 
100  last_way_point_ = max_way_point;
101  paused_ = false;
102  slider_->setSliderPosition(0);
103  slider_->setMaximum(max_way_point);
104  maximum_label_->setText(QString::number(max_way_point));
105 }
