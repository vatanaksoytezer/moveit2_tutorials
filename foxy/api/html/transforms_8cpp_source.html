99 const Eigen::Isometry3d& Transforms::getTransform(const std::string& from_frame) const
100 {
101  if (!from_frame.empty())
102  {
103  FixedTransformsMap::const_iterator it = transforms_map_.find(from_frame);
104  if (it != transforms_map_.end())
105  return it->second;
106  // If no transform found in map, return identity
107  }
108 
109  RCLCPP_ERROR(LOGGER, "Unable to transform from frame '%s' to frame '%s'. Returning identity.", from_frame.c_str(),
110  target_frame_.c_str());
111 
112  // return identity
113  static const Eigen::Isometry3d IDENTITY = Eigen::Isometry3d::Identity();
114  return IDENTITY;
115 }

136 void Transforms::setTransform(const geometry_msgs::msg::TransformStamped& transform)
137 {
138  if (sameFrame(transform.child_frame_id, target_frame_))
139  {
140  Eigen::Isometry3d t = tf2::transformToEigen(transform);
141  setTransform(t, transform.header.frame_id);
142  }
143  else
144  {
145  RCLCPP_ERROR(LOGGER, "Given transform is to frame '%s', but frame '%s' was expected.",
146  transform.child_frame_id.c_str(), target_frame_.c_str());
147  }
148 }
