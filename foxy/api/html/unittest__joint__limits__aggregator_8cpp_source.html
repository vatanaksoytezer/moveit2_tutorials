88 TEST_F(JointLimitsAggregator, CorrectOverwriteByParamterPosition)
89 {
90  ros::NodeHandle nh("~/valid_1");
91 
92  pilz_industrial_motion_planner::JointLimitsContainer container =
93  pilz_industrial_motion_planner::JointLimitsAggregator::getAggregatedLimits(nh,
94  robot_model_->getActiveJointModels());
95 
96  for (std::pair<std::string, pilz_industrial_motion_planner::JointLimit> lim : container)
97  {
98  // Check for the overwrite
99  if (lim.first == "prbt_joint_1")
100  {
101  EXPECT_EQ(2, container.getLimit(lim.first).max_position);
102  EXPECT_EQ(-2, container.getLimit(lim.first).min_position);
103  }
104  // Check that nothing else changed
105  else
106  {
107  EXPECT_EQ(robot_model_->getJointModel(lim.first)->getVariableBounds()[0].min_position_,
108  container.getLimit(lim.first).min_position);
109  EXPECT_EQ(robot_model_->getJointModel(lim.first)->getVariableBounds()[0].max_position_,
110  container.getLimit(lim.first).max_position);
111  }
112  }
113 }
