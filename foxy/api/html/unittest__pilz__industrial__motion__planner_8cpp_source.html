65  void createPlannerInstance()
66  {
67  // Load planner name from parameter server
68  if (!ph_.getParam("planning_plugin", planner_plugin_name_))
69  {
70  ROS_FATAL_STREAM("Could not find planner plugin name");
71  FAIL();
72  }
73 
74  // Load the plugin
75  try
76  {
77  planner_plugin_loader_.reset(new pluginlib::ClassLoader<planning_interface::PlannerManager>(
78  "moveit_core", "planning_interface::PlannerManager"));
79  }
80  catch (pluginlib::PluginlibException& ex)
81  {
82  ROS_FATAL_STREAM("Exception while creating planning plugin loader " << ex.what());
83  FAIL();
84  }
85 
86  // Create planner
87  try
88  {
89  planner_instance_.reset(planner_plugin_loader_->createUnmanagedInstance(planner_plugin_name_));
90  ASSERT_TRUE(planner_instance_->initialize(robot_model_, ph_.getNamespace()))
91  << "Initialzing the planner instance failed.";
92  }
93  catch (pluginlib::PluginlibException& ex)
94  {
95  FAIL() << "Could not create planner " << ex.what() << "\n";
96  }
97  }
98 
99  void TearDown() override
100  {
101  planner_plugin_loader_->unloadLibraryForClass(planner_plugin_name_);
102  }
103 
104 protected:

124 TEST_P(CommandPlannerTest, ObtainLoadedPlanningAlgorithms)
125 {
126  // Check for the algorithms
127  std::vector<std::string> algs;
128 
129  planner_instance_->getPlanningAlgorithms(algs);
130  ASSERT_EQ(3u, algs.size()) << "Found more or less planning algorithms as expected! Found:"
131  << ::testing::PrintToString(algs);
132 
133  // Collect the algorithms, check for uniqueness
134  std::set<std::string> algs_set;
135  for (const auto& alg : algs)
136  {
137  algs_set.insert(alg);
138  }
139  ASSERT_EQ(algs.size(), algs_set.size()) << "There are two or more algorithms with the same name!";
140  ASSERT_TRUE(algs_set.find("LIN") != algs_set.end());
141  ASSERT_TRUE(algs_set.find("PTP") != algs_set.end());
142  ASSERT_TRUE(algs_set.find("CIRC") != algs_set.end());
143 }
