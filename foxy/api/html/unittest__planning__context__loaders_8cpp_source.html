139 TEST_P(PlanningContextLoadersTest, LoadContext)
140 {
141  planning_interface::PlanningContextPtr planning_context;
142 
143  // Without limits should return false
144  bool res = planning_context_loader_->loadContext(planning_context, "test", "test");
145  EXPECT_EQ(false, res) << "Context returned even when no limits where set";
146 
147  // After setting the limits this should work
148  pilz_industrial_motion_planner::JointLimitsContainer joint_limits =
149  testutils::createFakeLimits(robot_model_->getVariableNames());
150  pilz_industrial_motion_planner::LimitsContainer limits;
151  limits.setJointLimits(joint_limits);
152  pilz_industrial_motion_planner::CartesianLimit cart_limits;
153  cart_limits.setMaxRotationalVelocity(1 * M_PI);
154  cart_limits.setMaxTranslationalAcceleration(2);
155  cart_limits.setMaxTranslationalDeceleration(2);
156  cart_limits.setMaxTranslationalVelocity(1);
157  limits.setCartesianLimits(cart_limits);
158 
159  planning_context_loader_->setLimits(limits);
160  planning_context_loader_->setModel(robot_model_);
161 
162  try
163  {
164  res = planning_context_loader_->loadContext(planning_context, "test", "test");
165  }
166  catch (std::exception& ex)
167  {
168  FAIL() << "Exception!" << ex.what() << " " << typeid(ex).name();
169  }
170 
171  EXPECT_EQ(true, res) << "Context could not be loaded!";
172 }
