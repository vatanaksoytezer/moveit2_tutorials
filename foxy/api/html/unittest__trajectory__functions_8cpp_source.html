140 bool TrajectoryFunctionsTestBase::tfNear(const Eigen::Isometry3d& pose1, const Eigen::Isometry3d& pose2,
141  const double& epsilon)
142 {
143  for (std::size_t i = 0; i < 3; ++i)
144  for (std::size_t j = 0; j < 4; ++j)
145  {
146  if (fabs(pose1(i, j) - pose2(i, j)) > fabs(epsilon))
147  return false;
148  }
149  return true;
150 }

239  kinematics::KinematicsResult ik_result;

241  kinematics::KinematicsQueryOptions options = kinematics::KinematicsQueryOptions();
242 
243  // compute all ik solutions
244  EXPECT_TRUE(solver->getPositionIK(ik_poses, ik_seed, ik_solutions, ik_result, options));
245 
246  // compute one ik solution
247  EXPECT_TRUE(solver->getPositionIK(pose_expect, ik_seed, ik_actual, err_code));
248 
249  ASSERT_EQ(ik_expect.size(), ik_actual.size());
250 
251  for (std::size_t i = 0; i < ik_expect.size(); ++i)
252  {
253  EXPECT_NEAR(ik_actual.at(i), ik_expect.at(i), 4 * IK_SEED_OFFSET);
