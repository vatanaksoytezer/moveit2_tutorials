260 TEST_P(TrajectoryGeneratorPTPTest, testInsufficientLimit)
261 {
262  /**********/
263  /* Step 1 */
264  /**********/
265  const auto& joint_models = robot_model_->getActiveJointModels();
266  ASSERT_TRUE(joint_models.size());
267 
268  // joint limit with insufficient limits (no acc/dec limits)
269  JointLimit insufficient_limit;
270  insufficient_limit.has_position_limits = true;
271  insufficient_limit.max_position = 2.5;
272  insufficient_limit.min_position = -2.5;
273  insufficient_limit.has_velocity_limits = true;
274  insufficient_limit.max_velocity = 1.256;
275  insufficient_limit.has_acceleration_limits = false;
276  insufficient_limit.has_deceleration_limits = false;
277  JointLimitsContainer insufficient_joint_limits;
278  for (const auto& joint_model : joint_models)
279  {
280  ASSERT_TRUE(insufficient_joint_limits.addLimit(joint_model->getName(), insufficient_limit))
281  << "Failed to add the limits for joint " << joint_model->getName();
282  }
283  LimitsContainer insufficient_planner_limits;
284  insufficient_planner_limits.setJointLimits(insufficient_joint_limits);
285 
286  EXPECT_THROW(
287  {
288  std::unique_ptr<TrajectoryGeneratorPTP> ptp_error(
289  new TrajectoryGeneratorPTP(robot_model_, insufficient_planner_limits));
290  },
291  TrajectoryGeneratorInvalidLimitsException);
292 
293  /**********/
294  /* Step 2 */
295  /**********/
296  // joint limit with sufficient limits
297  JointLimit sufficient_limit;
298  sufficient_limit.has_position_limits = true;
299  sufficient_limit.max_position = 2.356;
300  sufficient_limit.min_position = -2.356;
301  sufficient_limit.has_velocity_limits = true;
302  sufficient_limit.max_velocity = 1;
303  sufficient_limit.has_acceleration_limits = true;
304  sufficient_limit.max_acceleration = 0.5;
305  sufficient_limit.has_deceleration_limits = true;
306  sufficient_limit.max_deceleration = -1;
307  JointLimitsContainer sufficient_joint_limits;
308  // fill joint limits container, such that it contains one sufficient limit and
309  // all others are insufficient
310  for (const auto& jmg : robot_model_->getJointModelGroups())
311  {
312  const auto& joint_names{ jmg->getActiveJointModelNames() };
313  ASSERT_FALSE(joint_names.empty());
314  ASSERT_TRUE(sufficient_joint_limits.addLimit(joint_names.front(), sufficient_limit))
315  << "Failed to add the limits for joint " << joint_names.front();
316 
317  for (auto it = std::next(joint_names.begin()); it != joint_names.end(); ++it)
318  {
319  ASSERT_TRUE(sufficient_joint_limits.addLimit((*it), insufficient_limit))
320  << "Failed to add the limits for joint " << (*it);
321  }
322  }
323  LimitsContainer sufficient_planner_limits;
324  sufficient_planner_limits.setJointLimits(sufficient_joint_limits);
325 
326  EXPECT_NO_THROW({
327  std::unique_ptr<TrajectoryGeneratorPTP> ptp_no_error(
328  new TrajectoryGeneratorPTP(robot_model_, sufficient_planner_limits));
329  });
330 }
