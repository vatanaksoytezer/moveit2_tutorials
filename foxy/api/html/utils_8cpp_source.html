50 moveit_msgs::msg::Constraints mergeConstraints(const moveit_msgs::msg::Constraints& first,
51  const moveit_msgs::msg::Constraints& second)
52 {
53  moveit_msgs::msg::Constraints r;
54 
55  // add all joint constraints that are in first but not in second
56  // and merge joint constraints that are for the same joint
57  for (const moveit_msgs::msg::JointConstraint& jc_first : first.joint_constraints)
58  {
59  bool add = true;
60  for (const moveit_msgs::msg::JointConstraint& jc_second : second.joint_constraints)
61  if (jc_second.joint_name == jc_first.joint_name)
62  {
63  add = false;
64  // now we merge
65  moveit_msgs::msg::JointConstraint m;
66  const moveit_msgs::msg::JointConstraint& a = jc_first;
67  const moveit_msgs::msg::JointConstraint& b = jc_second;
68  double low = std::max(a.position - a.tolerance_below, b.position - b.tolerance_below);
69  double high = std::min(a.position + a.tolerance_above, b.position + b.tolerance_above);
70  if (low > high)
71  {
72  RCLCPP_ERROR(LOGGER, "Attempted to merge incompatible constraints for joint '%s'. Discarding constraint.",
73  a.joint_name.c_str());
74  }
75  else
76  {
77  m.joint_name = a.joint_name;
78  m.position =
79  std::max(low, std::min((a.position * a.weight + b.position * b.weight) / (a.weight + b.weight), high));
80  m.weight = (a.weight + b.weight) / 2.0;
81  m.tolerance_above = std::max(0.0, high - m.position);
82  m.tolerance_below = std::max(0.0, m.position - low);
83  r.joint_constraints.push_back(m);
84  }
85  break;
86  }
87  if (add)
88  r.joint_constraints.push_back(jc_first);
89  }
90 
91  // add all joint constraints that are in second but not in first
92  for (const moveit_msgs::msg::JointConstraint& jc_second : second.joint_constraints)
93  {
94  bool add = true;
95  for (const moveit_msgs::msg::JointConstraint& jc_first : first.joint_constraints)
96  if (jc_second.joint_name == jc_first.joint_name)
97  {
98  add = false;
99  break;
100  }
101  if (add)
102  r.joint_constraints.push_back(jc_second);
103  }
104 
105  // merge rest of constraints
106  r.position_constraints = first.position_constraints;
107  for (const moveit_msgs::msg::PositionConstraint& position_constraint : second.position_constraints)
108  r.position_constraints.push_back(position_constraint);
109 
110  r.orientation_constraints = first.orientation_constraints;
111  for (const moveit_msgs::msg::OrientationConstraint& orientation_constraint : second.orientation_constraints)
112  r.orientation_constraints.push_back(orientation_constraint);
113 
114  r.visibility_constraints = first.visibility_constraints;
115  for (const moveit_msgs::msg::VisibilityConstraint& visibility_constraint : second.visibility_constraints)
116  r.visibility_constraints.push_back(visibility_constraint);
117 
118  return r;
119 }

138 moveit_msgs::msg::Constraints constructGoalConstraints(const moveit::core::RobotState& state,
139  const moveit::core::JointModelGroup* jmg, double tolerance_below,
140  double tolerance_above)
141 {
142  moveit_msgs::msg::Constraints goal;
143  std::vector<double> vals;
144  state.copyJointGroupPositions(jmg, vals);
145  goal.joint_constraints.resize(vals.size());
146  for (std::size_t i = 0; i < vals.size(); ++i)
147  {
148  goal.joint_constraints[i].joint_name = jmg->getVariableNames()[i];
149  goal.joint_constraints[i].position = vals[i];
150  goal.joint_constraints[i].tolerance_above = tolerance_below;
151  goal.joint_constraints[i].tolerance_below = tolerance_above;
152  goal.joint_constraints[i].weight = 1.0;
153  }
154 
155  return goal;
156 }

158 moveit_msgs::msg::Constraints constructGoalConstraints(const std::string& link_name,
159  const geometry_msgs::msg::PoseStamped& pose,
160  double tolerance_pos, double tolerance_angle)
161 {
162  moveit_msgs::msg::Constraints goal;
163 
164  goal.position_constraints.resize(1);
165  moveit_msgs::msg::PositionConstraint& pcm = goal.position_constraints[0];
166  pcm.link_name = link_name;
167  pcm.target_point_offset.x = 0;
168  pcm.target_point_offset.y = 0;
169  pcm.target_point_offset.z = 0;
170  pcm.constraint_region.primitives.resize(1);
171  shape_msgs::msg::SolidPrimitive& bv = pcm.constraint_region.primitives[0];
172  bv.type = shape_msgs::msg::SolidPrimitive::SPHERE;
173  bv.dimensions.resize(geometric_shapes::solidPrimitiveDimCount<shape_msgs::msg::SolidPrimitive::SPHERE>());
174  bv.dimensions[shape_msgs::msg::SolidPrimitive::SPHERE_RADIUS] = tolerance_pos;
175 
176  pcm.header = pose.header;
177  pcm.constraint_region.primitive_poses.resize(1);
178  pcm.constraint_region.primitive_poses[0].position = pose.pose.position;
179 
180  // orientation of constraint region does not affect anything, since it is a sphere
181  pcm.constraint_region.primitive_poses[0].orientation.x = 0.0;
182  pcm.constraint_region.primitive_poses[0].orientation.y = 0.0;
183  pcm.constraint_region.primitive_poses[0].orientation.z = 0.0;
184  pcm.constraint_region.primitive_poses[0].orientation.w = 1.0;
185  pcm.weight = 1.0;
186 
187  goal.orientation_constraints.resize(1);
188  moveit_msgs::msg::OrientationConstraint& ocm = goal.orientation_constraints[0];
189  ocm.link_name = link_name;
190  ocm.header = pose.header;
191  ocm.orientation = pose.pose.orientation;
192  ocm.absolute_x_axis_tolerance = tolerance_angle;
193  ocm.absolute_y_axis_tolerance = tolerance_angle;
194  ocm.absolute_z_axis_tolerance = tolerance_angle;
195  ocm.weight = 1.0;
196 
197  return goal;
198 }

286 static bool constructPoseStamped(const rclcpp::Node::SharedPtr& node, const std::string& pose_param,
287  geometry_msgs::msg::PoseStamped& pose)
288 {
289  if (!node->get_parameter(pose_param + ".frame_id", pose.header.frame_id))
290  return false;
291 
292  std::vector<double> orientation;
293  if (!node->get_parameter(pose_param + ".orientation", orientation) || orientation.size() != 3)
294  return false;
295 
296  tf2::Quaternion q;
297  q.setRPY(orientation[0], orientation[1], orientation[2]);
298  pose.pose.orientation = toMsg(q);
299 
300  std::vector<double> position;
301  if (!node->get_parameter(pose_param + ".position", position) || position.size() != 3)
302  return false;
303 
304  pose.pose.position.x = position[0];
305  pose.pose.position.y = position[1];
306  pose.pose.position.z = position[2];
307 
308  return true;
309 }

312 static bool constructConstraint(const rclcpp::Node::SharedPtr& node, const std::string& constraint_param,
313  moveit_msgs::msg::JointConstraint& constraint)
314 {
315  node->get_parameter(constraint_param + ".weight", constraint.weight);
316  node->get_parameter(constraint_param + ".joint_name", constraint.joint_name);
317  node->get_parameter(constraint_param + ".position", constraint.position);
318 
319  double tolerance;
320  if (node->get_parameter(constraint_param + ".tolerance", tolerance))
321  {
322  constraint.tolerance_below = tolerance;
323  constraint.tolerance_above = tolerance;
324  }
325  else if (node->has_parameter(constraint_param + ".tolerances"))
326  {
327  std::vector<double> tolerances;
328  node->get_parameter(constraint_param + ".tolerances", tolerances);
329  if (tolerances.size() != 2)
330  return false;
331 
332  constraint.tolerance_below = tolerances[0];
333  constraint.tolerance_above = tolerances[1];
334  }
335  else if (node->has_parameter(constraint_param + ".bounds"))
336  {
337  std::vector<double> bounds;
338  node->get_parameter(constraint_param + ".bounds", bounds);
339  if (bounds.size() != 2)
340  return false;
341 
342  const double lower_bound = bounds[0];
343  const double upper_bound = bounds[1];
344 
345  constraint.position = (lower_bound + upper_bound) / 2;
346  constraint.tolerance_below = constraint.position - lower_bound;
347  constraint.tolerance_above = upper_bound - constraint.position;
348  }
349 
350  return true;
351 }

354 static bool constructConstraint(const rclcpp::Node::SharedPtr& node, const std::string& constraint_param,
355  moveit_msgs::msg::PositionConstraint& constraint)
356 {
357  node->get_parameter(constraint_param + ".frame_id", constraint.header.frame_id);
358  node->get_parameter(constraint_param + ".weight", constraint.weight);
359  node->get_parameter(constraint_param + ".link_name", constraint.link_name);
360 
361  std::vector<double> target_offset;
362  if (node->get_parameter(constraint_param + ".target_offset", target_offset))
363  {
364  if (target_offset.size() != 3)
365  return false;
366 
367  constraint.target_point_offset.x = target_offset[0];
368  constraint.target_point_offset.y = target_offset[1];
369  constraint.target_point_offset.z = target_offset[2];
370  }
371  if (!node->list_parameters({ constraint_param + ".region" }, 1).names.empty()) // TODO(henningkayser): specify depth
372  {
373  geometry_msgs::msg::Pose region_pose;
374  region_pose.orientation.w = 1.0;
375 
376  shape_msgs::msg::SolidPrimitive region_primitive;
377  region_primitive.type = shape_msgs::msg::SolidPrimitive::BOX;
378  region_primitive.dimensions.resize(3);
379 
380  const auto parse_dimension = [&](const std::string& dimension_param, double& center, double& dimension) -> bool {
381  std::vector<double> dimension_limits;
382  if (!node->get_parameter(constraint_param + ".region." + dimension_param, dimension_limits) ||
383  dimension_limits.size() != 2)
384  return false;
385 
386  center = (dimension_limits[0] + dimension_limits[1]) / 2;
387  dimension = dimension_limits[1] - dimension_limits[2];
388  return true;
389  };
390 
391  if (!parse_dimension("x", region_pose.position.x,
392  region_primitive.dimensions[shape_msgs::msg::SolidPrimitive::BOX_X]) ||
393  !parse_dimension("y", region_pose.position.y,
394  region_primitive.dimensions[shape_msgs::msg::SolidPrimitive::BOX_Y]) ||
395  !parse_dimension("z", region_pose.position.z,
396  region_primitive.dimensions[shape_msgs::msg::SolidPrimitive::BOX_Z]))
397  return false;
398 
399  constraint.constraint_region.primitive_poses.push_back(region_pose);
400  constraint.constraint_region.primitives.emplace_back(region_primitive);
401  }
402 
403  return true;
404 }

468 static bool collectConstraints(const rclcpp::Node::SharedPtr& node, const std::vector<std::string>& constraint_ids,
469  moveit_msgs::msg::Constraints& constraints)
470 {
471  for (const auto& constraint_id : constraint_ids)
472  {
473  const auto constraint_param = ".constraints." + constraint_id;
474  if (!node->has_parameter(constraint_param + ".type"))
475  {
476  RCLCPP_ERROR(LOGGER, "constraint parameter does not specify its type");
477  return false;
478  }
479  std::string constraint_type;
480  node->get_parameter(constraint_param + ".type", constraint_type);
481  if (constraint_type == "joint")
482  {
483  constraints.joint_constraints.emplace_back();
484  if (!constructConstraint(node, constraint_param, constraints.joint_constraints.back()))
485  return false;
486  }
487  else if (constraint_type == "position")
488  {
489  constraints.position_constraints.emplace_back();
490  if (!constructConstraint(node, constraint_param, constraints.position_constraints.back()))
491  return false;
492  }
493  else if (constraint_type == "orientation")
494  {
495  constraints.orientation_constraints.emplace_back();
496  if (!constructConstraint(node, constraint_param, constraints.orientation_constraints.back()))
497  return false;
498  }
499  else if (constraint_type == "visibility")
500  {
501  constraints.visibility_constraints.emplace_back();
502  if (!constructConstraint(node, constraint_param, constraints.visibility_constraints.back()))
503  return false;
504  }
505  else
506  {
507  RCLCPP_ERROR_STREAM(LOGGER, "Unable to process unknown constraint type: " << constraint_type);
508  return false;
509  }
510  }
511 
512  return true;
513 }

517 {

527

532 bool kinematic_constraints::resolveConstraintFrames(const moveit::core::RobotState& state,
533  moveit_msgs::msg::Constraints& constraints)
534 {
535  for (auto& c : constraints.position_constraints)
536  {
537  bool frame_found;
538  const moveit::core::LinkModel* robot_link;
539  const Eigen::Isometry3d& transform = state.getFrameInfo(c.link_name, robot_link, frame_found);
540  if (!frame_found)
541  return false;
542 
543  // If the frame of the constraint is not part of the robot link model (but an attached body or subframe),
544  // the constraint needs to be expressed in the frame of a robot link.
545  if (c.link_name != robot_link->getName())
546  {
547  Eigen::Isometry3d robot_link_to_link_name = state.getGlobalLinkTransform(robot_link).inverse() * transform;
548  Eigen::Vector3d offset_link_name(c.target_point_offset.x, c.target_point_offset.y, c.target_point_offset.z);
549  Eigen::Vector3d offset_robot_link = robot_link_to_link_name * offset_link_name;
550 
551  c.link_name = robot_link->getName();
552  tf2::toMsg(offset_robot_link, c.target_point_offset);
553  }
554  }
555 
556  for (auto& c : constraints.orientation_constraints)
557  {
558  bool frame_found;
559  const moveit::core::LinkModel* robot_link;
560  // getFrameInfo() returns a valid isometry by contract
561  const Eigen::Isometry3d& transform = state.getFrameInfo(c.link_name, robot_link, frame_found);
562  if (!frame_found)
563  return false;
564 
565  // If the frame of the constraint is not part of the robot link model (but an attached body or subframe),
566  // the constraint needs to be expressed in the frame of a robot link.
567  if (c.link_name != robot_link->getName())
568  {
569  c.link_name = robot_link->getName();
570  Eigen::Quaterniond link_name_to_robot_link(transform.linear().transpose() *
571  state.getGlobalLinkTransform(robot_link).linear());
