70 void World::addToObject(const std::string& id, const std::vector<shapes::ShapeConstPtr>& shapes,
71  const EigenSTL::vector_Isometry3d& poses)
72 {
73  if (shapes.size() != poses.size())
74  {
75  RCLCPP_ERROR(LOGGER,
76  "Number of shapes and number of poses do not match. Not adding this object to collision world.");
77  return;
78  }
79 
80  if (shapes.empty())
81  return;
82 
83  int action = ADD_SHAPE;
84 
85  ObjectPtr& obj = objects_[id];
86  if (!obj)
87  {
88  obj.reset(new Object(id));
89  action |= CREATE;
90  }
91 
92  ensureUnique(obj);
93 
94  for (std::size_t i = 0; i < shapes.size(); ++i)
95  addToObjectInternal(obj, shapes[i], poses[i]);
96 
97  notify(obj, Action(action));
98 }

145 bool World::knowsTransform(const std::string& name) const
146 {
147  // Check object names first
148  std::map<std::string, ObjectPtr>::const_iterator it = objects_.find(name);
149  if (it != objects_.end())
150  // only accept object name as frame if it is associated to a unique shape
151  return !it->second->shape_poses_.empty();
152  else // Then objects' subframes
153  {
154  for (const std::pair<const std::string, ObjectPtr>& object : objects_)
155  {
156  // if "object name/" matches start of object_id, we found the matching object
157  if (boost::starts_with(name, object.first) && name[object.first.length()] == '/')
158  return object.second->subframe_poses_.find(name.substr(object.first.length() + 1)) !=
159  object.second->subframe_poses_.end();
160  }
161  }
162  return false;
163 }

174 const Eigen::Isometry3d& World::getTransform(const std::string& name, bool& frame_found) const
175 {
176  // assume found
177  frame_found = true;
178 
179  std::map<std::string, ObjectPtr>::const_iterator it = objects_.find(name);
180  if (it != objects_.end())
181  {
182  if (!it->second->shape_poses_.empty())
183  return it->second->shape_poses_[0];
184  }
185  else // Search within subframes
186  {
187  for (const std::pair<const std::string, ObjectPtr>& object : objects_)
188  {
189  // if "object name/" matches start of object_id, we found the matching object
190  if (boost::starts_with(name, object.first) && name[object.first.length()] == '/')
191  {
192  auto it = object.second->subframe_poses_.find(name.substr(object.first.length() + 1));
193  if (it != object.second->subframe_poses_.end())
194  return it->second;
195  }
196  }
197  }
198 
199  // we need a persisting isometry for the API
200  static const Eigen::Isometry3d IDENTITY_TRANSFORM = Eigen::Isometry3d::Identity();
201  frame_found = false;
202  return IDENTITY_TRANSFORM;
203 }

205 bool World::moveShapeInObject(const std::string& object_id, const shapes::ShapeConstPtr& shape,
206  const Eigen::Isometry3d& pose)
207 {
208  auto it = objects_.find(object_id);
209  if (it != objects_.end())
210  {
211  unsigned int n = it->second->shapes_.size();
212  for (unsigned int i = 0; i < n; ++i)
213  if (it->second->shapes_[i] == shape)
214  {
215  ensureUnique(it->second);
216  ASSERT_ISOMETRY(pose) // unsanitized input, could contain a non-isometry
217  it->second->shape_poses_[i] = pose;
218 
219  notify(it->second, MOVE_SHAPE);
220  return true;
221  }
222  }
223  return false;
224 }

226 bool World::moveObject(const std::string& object_id, const Eigen::Isometry3d& transform)
227 {
228  auto it = objects_.find(object_id);
229  if (it == objects_.end())
230  return false;
231  if (transform.isApprox(Eigen::Isometry3d::Identity()))
232  return true; // object already at correct location
233  ensureUnique(it->second);
234  for (size_t i = 0, n = it->second->shapes_.size(); i < n; ++i)
235  {
236  ASSERT_ISOMETRY(transform) // unsanitized input, could contain a non-isometry
237  it->second->shape_poses_[i] = transform * it->second->shape_poses_[i];
238  }
239  notify(it->second, MOVE_SHAPE);
240  return true;
241 }

243 bool World::removeShapeFromObject(const std::string& object_id, const shapes::ShapeConstPtr& shape)
244 {
245  auto it = objects_.find(object_id);
246  if (it != objects_.end())
247  {
248  unsigned int n = it->second->shapes_.size();
249  for (unsigned int i = 0; i < n; ++i)
250  if (it->second->shapes_[i] == shape)
251  {
252  ensureUnique(it->second);
253  it->second->shapes_.erase(it->second->shapes_.begin() + i);
254  it->second->shape_poses_.erase(it->second->shape_poses_.begin() + i);
255 
256  if (it->second->shapes_.empty())
257  {
258  notify(it->second, DESTROY);
259  objects_.erase(it);
260  }
261  else
262  {
263  notify(it->second, REMOVE_SHAPE);
264  }
265  return true;
266  }
267  }
268  return false;
269 }
