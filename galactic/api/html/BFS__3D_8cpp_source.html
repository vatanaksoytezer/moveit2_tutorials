39 inline int BFS_3D::getNode(int x, int y, int z)
40 {

42  {
43  // error "Invalid coordinates"

49 BFS_3D::BFS_3D(int width, int height, int length)
50 {
51  if (width <= 0 || height <= 0 || length <= 0)
52  {
53  // error "Invalid dimensions"
54  return;
55  }
56 
57  dim_x = width + 2;
58  dim_y = height + 2;
59  dim_z = length + 2;
60 
61  dim_xy = dim_x * dim_y;
62  dim_xyz = dim_xy * dim_z;
63 
64  distance_grid = new int[dim_xyz];
65  queue = new int[width * height * length];
66 
67  for (int node = 0; node < dim_xyz; ++node)
68  {
69  int x = node % dim_x, y = node / dim_x % dim_y, z = node / dim_xy;
70  if (x == 0 || x == dim_x - 1 || y == 0 || y == dim_y - 1 || z == 0 || z == dim_z - 1)
71  distance_grid[node] = WALL;
72  else
73  distance_grid[node] = UNDISCOVERED;
74  }
75 
76  running = false;
77 }

116 void BFS_3D::run(int x, int y, int z)
117 {
118  if (running)
119  {
120  // error "Search already running"
121  return;
122  }
123 
124  for (int i = 0; i < dim_xyz; ++i)
125  if (distance_grid[i] != WALL)
126  distance_grid[i] = UNDISCOVERED;
127 
128  origin = getNode(x, y, z);
129 
130  queue_head = 0;
131  queue_tail = 1;
132  queue[0] = origin;
133 
134  distance_grid[origin] = 0;
135 
136  search_thread_ = std::make_shared<boost::thread>(&BFS_3D::search, this, dim_x, dim_xy, distance_grid, queue,
137  queue_head, queue_tail);
138  running = true;
139 }
