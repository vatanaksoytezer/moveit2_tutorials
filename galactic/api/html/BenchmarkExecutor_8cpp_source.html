70 boost::posix_time::ptime toBoost(const std::chrono::time_point<Clock, Duration>& from)
71 {
72  typedef std::chrono::nanoseconds duration_t;
73  typedef long rep_t;
74  rep_t d = std::chrono::duration_cast<duration_t>(from.time_since_epoch()).count();
75  rep_t sec = d / 1000000000;
76  rep_t nsec = d % 1000000000;
77  namespace pt = boost::posix_time;
78 #ifdef BOOST_DATE_TIME_HAS_NANOSECONDS
79  return pt::from_time_t(sec) + pt::nanoseconds(nsec)
80 #else
81  return pt::from_time_t(sec) + pt::microseconds(nsec / 1000);
82 #endif
83 }

225 bool BenchmarkExecutor::runBenchmarks(const BenchmarkOptions& opts)
226 {
227  if (planning_pipelines_.empty())
228  {
229  RCLCPP_ERROR(LOGGER, "No planning pipelines configured. Did you call BenchmarkExecutor::initialize?");
230  return false;
231  }
232 
233  std::vector<BenchmarkRequest> queries;
234  moveit_msgs::msg::PlanningScene scene_msg;
235 
236  if (initializeBenchmarks(opts, scene_msg, queries))
237  {
238  if (!queriesAndPlannersCompatible(queries, opts.getPlanningPipelineConfigurations()))
239  return false;
240 
241  for (std::size_t i = 0; i < queries.size(); ++i)
242  {
243  // Configure planning scene
244  if (scene_msg.robot_model_name != planning_scene_->getRobotModel()->getName())
245  {
246  // Clear all geometry from the scene
247  planning_scene_->getWorldNonConst()->clearObjects();
248  planning_scene_->getCurrentStateNonConst().clearAttachedBodies();
249  planning_scene_->getCurrentStateNonConst().setToDefaultValues();
250 
251  planning_scene_->processPlanningSceneWorldMsg(scene_msg.world);
252  }
253  else
254  planning_scene_->usePlanningSceneMsg(scene_msg);
255 
256  // Calling query start events
257  for (QueryStartEventFunction& query_start_fn : query_start_fns_)
258  query_start_fn(queries[i].request, planning_scene_);
259 
260  RCLCPP_INFO(LOGGER, "Benchmarking query '%s' (%lu of %lu)", queries[i].name.c_str(), i + 1, queries.size());
261  std::chrono::system_clock::time_point start_time = std::chrono::system_clock::now();
262  runBenchmark(queries[i].request, options_.getPlanningPipelineConfigurations(), options_.getNumRuns());
263  std::chrono::duration<double> dt = std::chrono::system_clock::now() - start_time;
264  double duration = dt.count();
265 
266  for (QueryCompletionEventFunction& query_end_fn : query_end_fns_)
267  query_end_fn(queries[i].request, planning_scene_);
268 
269  writeOutput(queries[i], boost::posix_time::to_iso_extended_string(toBoost(start_time)), duration);
270  }
271 
272  return true;
273  }
274  return false;
275 }

289  request.name.c_str());
290  return false;
291  }
292  }
293  }

298 bool BenchmarkExecutor::initializeBenchmarks(const BenchmarkOptions& opts, moveit_msgs::msg::PlanningScene& scene_msg,
299  std::vector<BenchmarkRequest>& requests)
300 {
301  if (!plannerConfigurationsExist(opts.getPlanningPipelineConfigurations(), opts.getGroupName()))
302  return false;
303 
304  std::vector<StartState> start_states;
305  std::vector<PathConstraints> path_constraints;
306  std::vector<PathConstraints> goal_constraints;
307  std::vector<TrajectoryConstraints> traj_constraints;
308  std::vector<BenchmarkRequest> queries;
309 
310  if (!loadBenchmarkQueryData(opts, scene_msg, start_states, path_constraints, goal_constraints, traj_constraints,
311  queries))
312  {
313  RCLCPP_ERROR(LOGGER, "Failed to load benchmark query data");
314  return false;
315  }
316 
317  RCLCPP_INFO(
318  LOGGER,
319  "Benchmark loaded %lu starts, %lu goals, %lu path constraints, %lu trajectory constraints, and %lu queries",
320  start_states.size(), goal_constraints.size(), path_constraints.size(), traj_constraints.size(), queries.size());
321 
322  moveit_msgs::msg::WorkspaceParameters workspace_parameters = opts.getWorkspaceParameters();
323  // Make sure that workspace_parameters are set
324  if (workspace_parameters.min_corner.x == workspace_parameters.max_corner.x &&
325  workspace_parameters.min_corner.x == 0.0 &&
326  workspace_parameters.min_corner.y == workspace_parameters.max_corner.y &&
327  workspace_parameters.min_corner.y == 0.0 &&
328  workspace_parameters.min_corner.z == workspace_parameters.max_corner.z &&
329  workspace_parameters.min_corner.z == 0.0)
330  {
331  workspace_parameters.min_corner.x = workspace_parameters.min_corner.y = workspace_parameters.min_corner.z = -5.0;
332 
333  workspace_parameters.max_corner.x = workspace_parameters.max_corner.y = workspace_parameters.max_corner.z = 5.0;
334  }
335 
336  std::vector<double> goal_offset;
337  opts.getGoalOffsets(goal_offset);
338 
339  // Create the combinations of BenchmarkRequests
340 
341  // 1) Create requests for combinations of start states,
342  // goal constraints, and path constraints
343  for (PathConstraints& goal_constraint : goal_constraints)
344  {
345  // Common benchmark request properties
346  BenchmarkRequest brequest;
347  brequest.name = goal_constraint.name;
348  brequest.request.workspace_parameters = workspace_parameters;
349  brequest.request.goal_constraints = goal_constraint.constraints;
350  brequest.request.group_name = opts.getGroupName();
351  brequest.request.allowed_planning_time = opts.getTimeout();
352  brequest.request.num_planning_attempts = 1;
353 
354  if (brequest.request.goal_constraints.size() == 1 &&
355  brequest.request.goal_constraints[0].position_constraints.size() == 1 &&
356  brequest.request.goal_constraints[0].orientation_constraints.size() == 1 &&
357  brequest.request.goal_constraints[0].visibility_constraints.empty() &&
358  brequest.request.goal_constraints[0].joint_constraints.empty())
359  shiftConstraintsByOffset(brequest.request.goal_constraints[0], goal_offset);
360 
361  std::vector<BenchmarkRequest> request_combos;
362  createRequestCombinations(brequest, start_states, path_constraints, request_combos);
363  requests.insert(requests.end(), request_combos.begin(), request_combos.end());
364  }
365 
366  // 2) Existing queries are treated like goal constraints.
367  // Create all combos of query, start states, and path constraints
368  for (BenchmarkRequest& query : queries)
369  {
370  // Common benchmark request properties
371  BenchmarkRequest brequest;
372  brequest.name = query.name;
373  brequest.request = query.request;
374  brequest.request.group_name = opts.getGroupName();
375  brequest.request.allowed_planning_time = opts.getTimeout();
376  brequest.request.num_planning_attempts = 1;
377 
378  // Make sure that workspace_parameters are set
379  if (brequest.request.workspace_parameters.min_corner.x == brequest.request.workspace_parameters.max_corner.x &&
380  brequest.request.workspace_parameters.min_corner.x == 0.0 &&
381  brequest.request.workspace_parameters.min_corner.y == brequest.request.workspace_parameters.max_corner.y &&
382  brequest.request.workspace_parameters.min_corner.y == 0.0 &&
383  brequest.request.workspace_parameters.min_corner.z == brequest.request.workspace_parameters.max_corner.z &&
384  brequest.request.workspace_parameters.min_corner.z == 0.0)
385  {
386  // ROS_WARN("Workspace parameters are not set for request %s. Setting defaults", queries[i].name.c_str());
387  brequest.request.workspace_parameters = workspace_parameters;
388  }
389 
390  // Create all combinations of start states and path constraints
391  std::vector<BenchmarkRequest> request_combos;
392  createRequestCombinations(brequest, start_states, path_constraints, request_combos);
393  requests.insert(requests.end(), request_combos.begin(), request_combos.end());
394  }
395 
396  // 3) Trajectory constraints are also treated like goal constraints
397  for (TrajectoryConstraints& traj_constraint : traj_constraints)
398  {
399  // Common benchmark request properties
400  BenchmarkRequest brequest;
401  brequest.name = traj_constraint.name;
402  brequest.request.trajectory_constraints = traj_constraint.constraints;
403  brequest.request.group_name = opts.getGroupName();
404  brequest.request.allowed_planning_time = opts.getTimeout();
405  brequest.request.num_planning_attempts = 1;
406 
407  if (brequest.request.trajectory_constraints.constraints.size() == 1 &&
408  brequest.request.trajectory_constraints.constraints[0].position_constraints.size() == 1 &&
409  brequest.request.trajectory_constraints.constraints[0].orientation_constraints.size() == 1 &&
410  brequest.request.trajectory_constraints.constraints[0].visibility_constraints.empty() &&
411  brequest.request.trajectory_constraints.constraints[0].joint_constraints.empty())
412  shiftConstraintsByOffset(brequest.request.trajectory_constraints.constraints[0], goal_offset);
413 
414  std::vector<BenchmarkRequest> request_combos;
415  std::vector<PathConstraints> no_path_constraints;
416  createRequestCombinations(brequest, start_states, no_path_constraints, request_combos);
417  requests.insert(requests.end(), request_combos.begin(), request_combos.end());
418  }
419 
420  options_ = opts;
421  return true;
422 }

463 void BenchmarkExecutor::shiftConstraintsByOffset(moveit_msgs::msg::Constraints& constraints,
464  const std::vector<double>& offset)
465 {
466  Eigen::Isometry3d offset_tf(Eigen::AngleAxis<double>(offset[3], Eigen::Vector3d::UnitX()) *
467  Eigen::AngleAxis<double>(offset[4], Eigen::Vector3d::UnitY()) *
468  Eigen::AngleAxis<double>(offset[5], Eigen::Vector3d::UnitZ()));
469  offset_tf.translation() = Eigen::Vector3d(offset[0], offset[1], offset[2]);
470 
471  geometry_msgs::msg::Pose constraint_pose_msg;
472  constraint_pose_msg.position = constraints.position_constraints[0].constraint_region.primitive_poses[0].position;
473  constraint_pose_msg.orientation = constraints.orientation_constraints[0].orientation;
474  Eigen::Isometry3d constraint_pose;
475  tf2::fromMsg(constraint_pose_msg, constraint_pose);
476 
477  Eigen::Isometry3d new_pose = constraint_pose * offset_tf;
478  geometry_msgs::msg::Pose new_pose_msg;
479  new_pose_msg = tf2::toMsg(new_pose);
480 
481  constraints.position_constraints[0].constraint_region.primitive_poses[0].position = new_pose_msg.position;
482  constraints.orientation_constraints[0].orientation = new_pose_msg.orientation;
483 }

485 void BenchmarkExecutor::createRequestCombinations(const BenchmarkRequest& brequest,
486  const std::vector<StartState>& start_states,
487  const std::vector<PathConstraints>& path_constraints,
488  std::vector<BenchmarkRequest>& requests)
489 {
490  // Use default start state
491  if (start_states.empty())
492  {
493  // Adding path constraints
494  for (const PathConstraints& path_constraint : path_constraints)
495  {
496  BenchmarkRequest new_brequest = brequest;
497  new_brequest.request.path_constraints = path_constraint.constraints[0];
498  new_brequest.name = brequest.name + "_" + path_constraint.name;
499  requests.push_back(new_brequest);
500  }
501 
502  if (path_constraints.empty())
503  requests.push_back(brequest);
504  }
505  else // Create a request for each start state specified
506  {
507  for (const StartState& start_state : start_states)
508  {
509  // Skip start states that have the same name as the goal
510  if (start_state.name == brequest.name)
511  continue;
512 
513  BenchmarkRequest new_brequest = brequest;
514  new_brequest.request.start_state = start_state.state;
515 
516  // Duplicate the request for each of the path constraints
517  for (const PathConstraints& path_constraint : path_constraints)
518  {
519  new_brequest.request.path_constraints = path_constraint.constraints[0];
520  new_brequest.name = start_state.name + "_" + new_brequest.name + "_" + path_constraint.name;
521  requests.push_back(new_brequest);
522  }
523 
524  if (path_constraints.empty())
525  {
526  new_brequest.name = start_state.name + "_" + brequest.name;
527  requests.push_back(new_brequest);
528  }
529  }
530  }

533 bool BenchmarkExecutor::plannerConfigurationsExist(
534  const std::map<std::string, std::vector<std::string>>& pipeline_configurations, const std::string& group_name)
535 {
536  // Make sure planner plugins exist
537  for (const std::pair<const std::string, std::vector<std::string>>& pipeline_config_entry : pipeline_configurations)
538  {
539  bool pipeline_exists = false;
540  for (const std::pair<const std::string, planning_pipeline::PlanningPipelinePtr>& pipeline_entry :
541  planning_pipelines_)
542  {
543  pipeline_exists = pipeline_entry.first == pipeline_config_entry.first;
544  if (pipeline_exists)
545  break;
546  }
547 
548  if (!pipeline_exists)
549  {
550  RCLCPP_ERROR(LOGGER, "Planning pipeline '%s' does NOT exist", pipeline_config_entry.first.c_str());
551  return false;
552  }
553  }
554 
555  // Make sure planners exist within those pipelines
556  for (const std::pair<const std::string, std::vector<std::string>>& entry : pipeline_configurations)
557  {
558  planning_interface::PlannerManagerPtr pm = planning_pipelines_[entry.first]->getPlannerManager();
559  const planning_interface::PlannerConfigurationMap& config_map = pm->getPlannerConfigurations();
560 
561  // if the planner is chomp or stomp skip this function and return true for checking planner configurations for the
562  // planning group otherwise an error occurs, because for OMPL a specific planning algorithm needs to be defined for
563  // a planning group, whereas with STOMP and CHOMP this is not necessary
564  if (pm->getDescription().compare("stomp") || pm->getDescription().compare("chomp"))
565  continue;
566 
567  for (std::size_t i = 0; i < entry.second.size(); ++i)
568  {
569  bool planner_exists = false;
570  for (const std::pair<const std::string, planning_interface::PlannerConfigurationSettings>& config_entry :
571  config_map)
572  {
573  std::string planner_name = group_name + "[" + entry.second[i] + "]";
574  planner_exists = (config_entry.second.group == group_name && config_entry.second.name == planner_name);
575  }
576 
577  if (!planner_exists)
578  {
579  RCLCPP_ERROR(LOGGER, "Planner '%s' does NOT exist for group '%s' in pipeline '%s'", entry.second[i].c_str(),
580  group_name.c_str(), entry.first.c_str());
581  std::cout << "There are " << config_map.size() << " planner entries: " << '\n';
582  for (const auto& config_map_entry : config_map)
583  std::cout << config_map_entry.second.name << '\n';

796  planning_pipeline::PlanningPipelinePtr planning_pipeline = planning_pipelines_[pipeline_entry.first];

872 void BenchmarkExecutor::collectMetrics(PlannerRunData& metrics,
873  const planning_interface::MotionPlanDetailedResponse& mp_res, bool solved,
874  double total_time)
875 {
876  metrics["time REAL"] = moveit::core::toString(total_time);
877  metrics["solved BOOLEAN"] = boost::lexical_cast<std::string>(solved);
878 
879  if (solved)
880  {
881  // Analyzing the trajectory(ies) geometrically
882  double traj_len = 0.0; // trajectory length
883  double clearance = 0.0; // trajectory clearance (average)
884  double smoothness = 0.0; // trajectory smoothness (average)
885  bool correct = true; // entire trajectory collision free and in bounds
886 
887  double process_time = total_time;
888  for (std::size_t j = 0; j < mp_res.trajectory_.size(); ++j)
889  {
890  correct = true;
891  traj_len = 0.0;
892  clearance = 0.0;
893  smoothness = 0.0;
894  const robot_trajectory::RobotTrajectory& p = *mp_res.trajectory_[j];
895 
896  // compute path length
897  for (std::size_t k = 1; k < p.getWayPointCount(); ++k)
898  traj_len += p.getWayPoint(k - 1).distance(p.getWayPoint(k));
899 
900  // compute correctness and clearance
901  collision_detection::CollisionRequest req;
902  for (std::size_t k = 0; k < p.getWayPointCount(); ++k)
903  {
904  collision_detection::CollisionResult res;
905  planning_scene_->checkCollisionUnpadded(req, res, p.getWayPoint(k));
906  if (res.collision)
907  correct = false;
908  if (!p.getWayPoint(k).satisfiesBounds())
909  correct = false;
910  double d = planning_scene_->distanceToCollisionUnpadded(p.getWayPoint(k));
911  if (d > 0.0) // in case of collision, distance is negative
912  clearance += d;
913  }
914  clearance /= (double)p.getWayPointCount();
915 
916  // compute smoothness
917  if (p.getWayPointCount() > 2)
918  {
919  double a = p.getWayPoint(0).distance(p.getWayPoint(1));
920  for (std::size_t k = 2; k < p.getWayPointCount(); ++k)
921  {
922  // view the path as a sequence of segments, and look at the triangles it forms:
923  // s1
924  // /\ s4
925  // a / \ b |
926  // / \ |
927  // /......\_______|
928  // s0 c s2 s3
929  //
930 
931  // use Pythagoras generalized theorem to find the cos of the angle between segments a and b
932  double b = p.getWayPoint(k - 1).distance(p.getWayPoint(k));
933  double cdist = p.getWayPoint(k - 2).distance(p.getWayPoint(k));
934  double acos_value = (a * a + b * b - cdist * cdist) / (2.0 * a * b);
935  if (acos_value > -1.0 && acos_value < 1.0)
936  {
937  // the smoothness is actually the outside angle of the one we compute
938  double angle = (boost::math::constants::pi<double>() - acos(acos_value));
939 
940  // and we normalize by the length of the segments
941  double u = 2.0 * angle;
942  smoothness += u * u;
943  }
944  a = b;
945  }
946  smoothness /= (double)p.getWayPointCount();
947  }
948  metrics["path_" + mp_res.description_[j] + "_correct BOOLEAN"] = boost::lexical_cast<std::string>(correct);
949  metrics["path_" + mp_res.description_[j] + "_length REAL"] = moveit::core::toString(traj_len);
950  metrics["path_" + mp_res.description_[j] + "_clearance REAL"] = moveit::core::toString(clearance);
951  metrics["path_" + mp_res.description_[j] + "_smoothness REAL"] = moveit::core::toString(smoothness);
952  metrics["path_" + mp_res.description_[j] + "_time REAL"] = moveit::core::toString(mp_res.processing_time_[j]);
953 
954  if (j == mp_res.trajectory_.size() - 1)
955  {
956  metrics["final_path_correct BOOLEAN"] = boost::lexical_cast<std::string>(correct);
957  metrics["final_path_length REAL"] = moveit::core::toString(traj_len);
958  metrics["final_path_clearance REAL"] = moveit::core::toString(clearance);
959  metrics["final_path_smoothness REAL"] = moveit::core::toString(smoothness);
960  metrics["final_path_time REAL"] = moveit::core::toString(mp_res.processing_time_[j]);
961  }
962  process_time -= mp_res.processing_time_[j];
963  }
964  if (process_time <= 0.0)
965  process_time = 0.0;
966  metrics["process_time REAL"] = moveit::core::toString(process_time);
967  }
968 }

970 void BenchmarkExecutor::computeAveragePathSimilarities(
971  PlannerBenchmarkData& planner_data, const std::vector<planning_interface::MotionPlanDetailedResponse>& responses,
972  const std::vector<bool>& solved)
973 {
974  RCLCPP_INFO(LOGGER, "Computing result path similarity");
975  const size_t result_count = planner_data.size();
976  size_t unsolved = std::count_if(solved.begin(), solved.end(), [](bool s) { return !s; });
977  std::vector<double> average_distances(responses.size());
978  for (size_t first_traj_i = 0; first_traj_i < result_count; ++first_traj_i)
979  {
980  // If trajectory was not solved there is no valid average distance so it's set to max double only
981  if (!solved[first_traj_i])
982  {
983  average_distances[first_traj_i] = std::numeric_limits<double>::max();
984  continue;
985  }
986  // Iterate all result trajectories that haven't been compared yet
987  for (size_t second_traj_i = first_traj_i + 1; second_traj_i < result_count; ++second_traj_i)
988  {
989  // Ignore if other result has not been solved
990  if (!solved[second_traj_i])
991  continue;
992 
993  // Get final trajectories
994  const robot_trajectory::RobotTrajectory& traj_first = *responses[first_traj_i].trajectory_.back();
995  const robot_trajectory::RobotTrajectory& traj_second = *responses[second_traj_i].trajectory_.back();
996 
997  // Compute trajectory distance
998  double trajectory_distance;
999  if (!computeTrajectoryDistance(traj_first, traj_second, trajectory_distance))
1000  continue;
1001 
1002  // Add average distance to counters of both trajectories
1003  average_distances[first_traj_i] += trajectory_distance;
1004  average_distances[second_traj_i] += trajectory_distance;
1005  }
1006  // Normalize average distance by number of actual comparisons
1007  average_distances[first_traj_i] /= result_count - unsolved - 1;
1008  }
1009 
1010  // Store results in planner_data
1011  for (size_t i = 0; i < result_count; ++i)
1012  planner_data[i]["average_waypoint_distance REAL"] = moveit::core::toString(average_distances[i]);
1013 }

1015 bool BenchmarkExecutor::computeTrajectoryDistance(const robot_trajectory::RobotTrajectory& traj_first,
1016  const robot_trajectory::RobotTrajectory& traj_second,
1017  double& result_distance)
1018 {
1019  // Abort if trajectories are empty
1020  if (traj_first.empty() || traj_second.empty())
1021  return false;
1022 
1023  // Waypoint counter
1024  size_t pos_first = 0;
1025  size_t pos_second = 0;
1026  const size_t max_pos_first = traj_first.getWayPointCount() - 1;
1027  const size_t max_pos_second = traj_second.getWayPointCount() - 1;
1028 
1029  // Compute total distance between pairwise waypoints of both trajectories.
1030  // The selection of waypoint pairs is based on what steps results in the minimal distance between the next pair of
1031  // waypoints. We first check what steps are still possible or if we reached the end of the trajectories. Then we
1032  // compute the pairwise waypoint distances of the pairs from increasing both, the first, or the second trajectory.
1033  // Finally we select the pair that results in the minimal distance, summarize the total distance and iterate
1034  // accordingly. After that we compute the average trajectory distance by normalizing over the number of steps.
1035  double total_distance = 0;
1036  size_t steps = 0;
1037  double current_distance = traj_first.getWayPoint(pos_first).distance(traj_second.getWayPoint(pos_second));
1038  while (true)
1039  {
1040  // Keep track of total distance and number of comparisons
1041  total_distance += current_distance;
1042  ++steps;
1043  if (pos_first == max_pos_first && pos_second == max_pos_second) // end reached
1044  break;
1045 
1046  // Determine what steps are still possible
1047  bool can_up_first = pos_first < max_pos_first;
1048  bool can_up_second = pos_second < max_pos_second;
1049  bool can_up_both = can_up_first && can_up_second;
1050 
1051  // Compute pair-wise waypoint distances (increasing both, first, or second trajectories).
1052  double up_both = std::numeric_limits<double>::max();
1053  double up_first = std::numeric_limits<double>::max();
1054  double up_second = std::numeric_limits<double>::max();
1055  if (can_up_both)
1056  up_both = traj_first.getWayPoint(pos_first + 1).distance(traj_second.getWayPoint(pos_second + 1));
1057  if (can_up_first)
1058  up_first = traj_first.getWayPoint(pos_first + 1).distance(traj_second.getWayPoint(pos_second));
1059  if (can_up_second)
1060  up_second = traj_first.getWayPoint(pos_first).distance(traj_second.getWayPoint(pos_second + 1));
1061 
1062  // Select actual step, store new distance value and iterate trajectory positions
1063  if (can_up_both && up_both < up_first && up_both < up_second)
1064  {
1065  ++pos_first;
1066  ++pos_second;
1067  current_distance = up_both;
1068  }
1069  else if ((can_up_first && up_first < up_second) || !can_up_second)
1070  {
1071  ++pos_first;
1072  current_distance = up_first;
1073  }
1074  else if (can_up_second)
1075  {
1076  ++pos_second;
1077  current_distance = up_second;
1078  }
1079  }
1080  // Normalize trajectory distance by number of comparison steps
1081  result_distance = total_distance / static_cast<double>(steps);
1082  return true;
1083 }
