39 void get_bresenham3d_parameters(int p1x, int p1y, int p1z, int p2x, int p2y, int p2z, bresenham3d_param_t* params)
40 {
41  params->X1 = p1x;
42  params->Y1 = p1y;
43  params->Z1 = p1z;
44  params->X2 = p2x;
45  params->Y2 = p2y;
46  params->Z2 = p2z;
47 
48  params->XIndex = params->X1;
49  params->YIndex = params->Y1;
50  params->ZIndex = params->Z1;
51 
52  params->dx = fabs((double)(p2x - p1x));
53  params->dy = fabs((double)(p2y - p1y));
54  params->dz = fabs((double)(p2z - p1z));
55  params->dx2 = params->dx << 1;
56  params->dy2 = params->dy << 1;
57  params->dz2 = params->dz << 1;
58 
59  // get direction of slope
60  if ((double)(p2x - p1x) < 0)
61  params->IncX = -1;
62  else
63  params->IncX = 1;
64 
65  if ((double)(p2y - p1y) < 0)
66  params->IncY = -1;
67  else
68  params->IncY = 1;
69 
70  if ((double)(p2z - p1z) < 0)
71  params->IncZ = -1;
72  else
73  params->IncZ = 1;
74 
75  // choose which axis to use as the index
76  if (params->dx >= params->dy && params->dx >= params->dz)
77  {
78  params->UsingXYZIndex = 0;
79  params->err1 = params->dy2 - params->dx;
80  params->err2 = params->dz2 - params->dx;
81  }
82  else if (params->dy >= params->dx && params->dy >= params->dz)
83  {
84  params->UsingXYZIndex = 1;
85  params->err1 = params->dx2 - params->dy;
86  params->err2 = params->dz2 - params->dy;
87  }
88  else
89  {
90  params->UsingXYZIndex = 2;
91  params->err1 = params->dy2 - params->dz;
92  params->err2 = params->dx2 - params->dz;
93  }
94 }
