60 int main(int argc, char** argv)
61 {
62  rclcpp::init(argc, argv);
63  rclcpp::NodeOptions node_options;
64  node_options.allow_undeclared_parameters(true);
65  node_options.automatically_declare_parameters_from_overrides(true);
66  rclcpp::Node::SharedPtr node = rclcpp::Node::make_shared("publish_warehouse_data", node_options);
67 
68  // time to wait in between publishing messages
69  double delay = 0.001;
70 
71  boost::program_options::options_description desc;
72  desc.add_options()("help", "Show help message")("host", boost::program_options::value<std::string>(),
73  "Host for the "
74  "DB.")("port", boost::program_options::value<std::size_t>(),
75  "Port for the DB.")(
76  "scene", boost::program_options::value<std::string>(), "Name of scene to publish.")(
77  "planning_requests", "Also publish the planning requests that correspond to the scene")(
78  "planning_results", "Also publish the planning results that correspond to the scene")(
79  "constraint", boost::program_options::value<std::string>(), "Name of constraint to publish.")(
80  "state", boost::program_options::value<std::string>(),
81  "Name of the robot state to publish.")("delay", boost::program_options::value<double>()->default_value(delay),
82  "Time to wait in between publishing messages (s)");
83 
84  boost::program_options::variables_map vm;
85  boost::program_options::store(boost::program_options::parse_command_line(argc, argv, desc), vm);
86  boost::program_options::notify(vm);
87 
88  if (vm.count("help") || (!vm.count("scene") && !vm.count("constraint") && !vm.count("state")))
89  {
90  std::cout << desc << '\n';
91  return 1;
92  }
93  try
94  {
95  delay = vm["delay"].as<double>();
96  }
97  catch (...)
98  {
99  std::cout << desc << '\n';
100  return 2;
101  }
102  // Set up db
103  warehouse_ros::DatabaseConnection::Ptr conn = moveit_warehouse::loadDatabase(node);
104  if (vm.count("host") && vm.count("port"))
105  conn->setParams(vm["host"].as<std::string>(), vm["port"].as<std::size_t>());
106  if (!conn->connect())
107  return 1;
108 
109  rclcpp::executors::StaticSingleThreadedExecutor executor;
110  executor.add_node(node);
111 
112  rclcpp::Rate rate((int64_t)delay * 1000ms);
113 
114  // publish the scene
115  if (vm.count("scene"))
116  {
117  auto pub_scene = node->create_publisher<moveit_msgs::msg::PlanningScene>(PLANNING_SCENE_TOPIC, 10);
118  bool req = vm.count("planning_requests");
119  bool res = vm.count("planning_results");
120 
121  moveit_warehouse::PlanningSceneStorage pss(conn);
122  executor.spin_once(0ns);
123 
124  std::vector<std::string> scene_names;
125  pss.getPlanningSceneNames(vm["scene"].as<std::string>(), scene_names);
126 
127  for (const std::string& scene_name : scene_names)
128  {
129  moveit_warehouse::PlanningSceneWithMetadata pswm;
130  if (pss.getPlanningScene(pswm, scene_name))
131  {
132  RCLCPP_INFO(LOGGER, "Publishing scene '%s'",
133  pswm->lookupString(moveit_warehouse::PlanningSceneStorage::PLANNING_SCENE_ID_NAME).c_str());
134  pub_scene->publish(static_cast<const moveit_msgs::msg::PlanningScene&>(*pswm));
135  executor.spin_once(0ns);
136 
137  // publish optional data associated to the scene
138  if (req || res)
139  {
140  auto pub_req = node->create_publisher<moveit_msgs::msg::MotionPlanRequest>(PLANNING_REQUEST_TOPIC, 100);
141  auto pub_res = node->create_publisher<moveit_msgs::msg::RobotTrajectory>(PLANNING_RESULTS_TOPIC, 100);
142  std::vector<moveit_warehouse::MotionPlanRequestWithMetadata> planning_queries;
143  std::vector<std::string> query_names;
144  pss.getPlanningQueries(planning_queries, query_names, pswm->name);
145  RCLCPP_INFO(LOGGER, "There are %d planning queries associated to the scene",
146  static_cast<int>(planning_queries.size()));
147  rclcpp::sleep_for(500ms);
148  for (std::size_t i = 0; i < planning_queries.size(); ++i)
149  {
150  if (req)
151  {
152  RCLCPP_INFO(LOGGER, "Publishing query '%s'", query_names[i].c_str());
153  pub_req->publish(static_cast<const moveit_msgs::msg::MotionPlanRequest&>(*planning_queries[i]));
154  executor.spin_once(0ns);
155  }
156  if (res)
157  {
158  std::vector<moveit_warehouse::RobotTrajectoryWithMetadata> planning_results;
159  pss.getPlanningResults(planning_results, query_names[i], pswm->name);
160  for (moveit_warehouse::RobotTrajectoryWithMetadata& planning_result : planning_results)
161  {
162  pub_res->publish(static_cast<const moveit_msgs::msg::RobotTrajectory&>(*planning_result));
163  executor.spin_once(0ns);
164  }
165  }
166  }
167  }
168  rate.sleep();
169  }
170  }
171  }
172 
173  // publish constraints
174  if (vm.count("constraint"))
175  {
176  moveit_warehouse::ConstraintsStorage cs(conn);
177  auto pub_constr = node->create_publisher<moveit_msgs::msg::Constraints>(CONSTRAINTS_TOPIC, 100);
178  std::vector<std::string> cnames;
179  cs.getKnownConstraints(vm["constraint"].as<std::string>(), cnames);
180 
181  for (const std::string& cname : cnames)
182  {
183  moveit_warehouse::ConstraintsWithMetadata cwm;
184  if (cs.getConstraints(cwm, cname))
185  {
186  RCLCPP_INFO(LOGGER, "Publishing constraints '%s'",
187  cwm->lookupString(moveit_warehouse::ConstraintsStorage::CONSTRAINTS_ID_NAME).c_str());
188  pub_constr->publish(static_cast<const moveit_msgs::msg::Constraints&>(*cwm));
189  executor.spin_once(0ns);
190  rate.sleep();
191  }
192  }
193  }
194 
195  // publish constraints
196  if (vm.count("state"))
197  {
198  moveit_warehouse::RobotStateStorage rs(conn);
199  auto pub_state = node->create_publisher<moveit_msgs::msg::RobotState>(STATES_TOPIC, 100);
200  std::vector<std::string> rnames;
201  rs.getKnownRobotStates(vm["state"].as<std::string>(), rnames);
202 
203  for (const std::string& rname : rnames)
204  {
205  moveit_warehouse::RobotStateWithMetadata rswm;
206  if (rs.getRobotState(rswm, rname))
207  {
208  RCLCPP_INFO(LOGGER, "Publishing state '%s'",
209  rswm->lookupString(moveit_warehouse::RobotStateStorage::STATE_NAME).c_str());
210  pub_state->publish(static_cast<const moveit_msgs::msg::RobotState&>(*rswm));
211  executor.spin_once(0ns);
212  rate.sleep();
213  }
214  }
215  }
216 
217  rclcpp::sleep_for(1s);
218  RCLCPP_INFO(LOGGER, "Done.");
219 
220  return 0;
221 }
