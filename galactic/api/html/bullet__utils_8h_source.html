345 inline void getAverageSupport(const btConvexShape* shape, const btVector3& localNormal, float& outsupport,
346  btVector3& outpt)
347 {
348  btVector3 pt_sum(0, 0, 0);
349  float pt_count = 0;
350  float max_support = -1000;
351 
352  const btPolyhedralConvexShape* pshape = dynamic_cast<const btPolyhedralConvexShape*>(shape);
353  if (pshape)
354  {
355  int n_pts = pshape->getNumVertices();
356 
357  for (int i = 0; i < n_pts; ++i)
358  {
359  btVector3 pt;
360  pshape->getVertex(i, pt);
361 
362  float sup = pt.dot(localNormal);
363  if (sup > max_support + BULLET_EPSILON)
364  {
365  pt_count = 1;
366  pt_sum = pt;
367  max_support = sup;
368  }
369  else if (sup < max_support - BULLET_EPSILON)
370  {
371  }
372  else
373  {
374  pt_count += 1;
375  pt_sum += pt;
376  }
377  }
378  outsupport = max_support;
379  outpt = pt_sum / pt_count;
380  }
381  else
382  {
383  outpt = shape->localGetSupportingVertexWithoutMargin(localNormal);
384  outsupport = localNormal.dot(outpt);
385  }
386 }

389 inline btScalar addDiscreteSingleResult(btManifoldPoint& cp, const btCollisionObjectWrapper* colObj0Wrap,
390  const btCollisionObjectWrapper* colObj1Wrap, ContactTestData& collisions)
391 {
392  assert(dynamic_cast<const CollisionObjectWrapper*>(colObj0Wrap->getCollisionObject()) != nullptr);
393  assert(dynamic_cast<const CollisionObjectWrapper*>(colObj1Wrap->getCollisionObject()) != nullptr);
394  const CollisionObjectWrapper* cd0 = static_cast<const CollisionObjectWrapper*>(colObj0Wrap->getCollisionObject());
395  const CollisionObjectWrapper* cd1 = static_cast<const CollisionObjectWrapper*>(colObj1Wrap->getCollisionObject());
396 
397  std::pair<std::string, std::string> pc = getObjectPairKey(cd0->getName(), cd1->getName());
398 
399  const auto& it = collisions.res.contacts.find(pc);
400  bool found = (it != collisions.res.contacts.end());
401 
402  collision_detection::Contact contact;
403  contact.body_name_1 = cd0->getName();
404  contact.body_name_2 = cd1->getName();
405  contact.depth = static_cast<double>(cp.m_distance1);
406  contact.normal = convertBtToEigen(-1 * cp.m_normalWorldOnB);
407  contact.pos = convertBtToEigen(cp.m_positionWorldOnA);
408  contact.nearest_points[0] = contact.pos;
409  contact.nearest_points[1] = convertBtToEigen(cp.m_positionWorldOnB);
410 
411  contact.body_type_1 = cd0->getTypeID();
412  contact.body_type_2 = cd1->getTypeID();
413 
414  if (!processResult(collisions, contact, pc, found))
415  {
416  return 0;
417  }
418 
419  return 1;
420 }

422 inline btScalar addCastSingleResult(btManifoldPoint& cp, const btCollisionObjectWrapper* colObj0Wrap, int /*index0*/,
423  const btCollisionObjectWrapper* colObj1Wrap, int /*index1*/,
424  ContactTestData& collisions)
425 {
426  assert(dynamic_cast<const CollisionObjectWrapper*>(colObj0Wrap->getCollisionObject()) != nullptr);
427  assert(dynamic_cast<const CollisionObjectWrapper*>(colObj1Wrap->getCollisionObject()) != nullptr);
428 
429  const CollisionObjectWrapper* cd0 = static_cast<const CollisionObjectWrapper*>(colObj0Wrap->getCollisionObject());
430  const CollisionObjectWrapper* cd1 = static_cast<const CollisionObjectWrapper*>(colObj1Wrap->getCollisionObject());
431 
432  std::pair<std::string, std::string> pc = getObjectPairKey(cd0->getName(), cd1->getName());
433 
434  const auto& it = collisions.res.contacts.find(pc);
435  bool found = (it != collisions.res.contacts.end());
436 
437  collision_detection::Contact contact;
438  contact.body_name_1 = cd0->getName();
439  contact.body_name_2 = cd1->getName();
440  contact.depth = static_cast<double>(cp.m_distance1);
441  contact.normal = convertBtToEigen(-1 * cp.m_normalWorldOnB);
442  contact.pos = convertBtToEigen(cp.m_positionWorldOnA);
443 
444  contact.body_type_1 = cd0->getTypeID();
445  contact.body_type_2 = cd1->getTypeID();
446 
447  collision_detection::Contact* col = processResult(collisions, contact, pc, found);
448 
449  if (!col)
450  {
451  return 0;
452  }
453 
454  assert(!((cd0->m_collisionFilterGroup == btBroadphaseProxy::KinematicFilter) &&
455  (cd1->m_collisionFilterGroup == btBroadphaseProxy::KinematicFilter)));
456 
457  bool cast_shape_is_first = cd0->m_collisionFilterGroup == btBroadphaseProxy::KinematicFilter;
458 
459  btVector3 normal_world_from_cast = (cast_shape_is_first ? -1 : 1) * cp.m_normalWorldOnB;
460  const btCollisionObjectWrapper* first_col_obj_wrap = (cast_shape_is_first ? colObj0Wrap : colObj1Wrap);
461 
462  // we want the contact information of the non-casted object come first, therefore swap values
463  if (cast_shape_is_first)
464  {
465  std::swap(col->nearest_points[0], col->nearest_points[1]);
466  contact.pos = convertBtToEigen(cp.m_positionWorldOnB);
467  std::swap(col->body_name_1, col->body_name_2);
468  std::swap(col->body_type_1, col->body_type_2);
469  col->normal *= -1;
470  }
471 
472  btTransform tf_world0, tf_world1;
473  const CastHullShape* shape = static_cast<const CastHullShape*>(first_col_obj_wrap->getCollisionShape());
474 
475  tf_world0 = first_col_obj_wrap->getWorldTransform();
476  tf_world1 = first_col_obj_wrap->getWorldTransform() * shape->m_shape_transform;
477 
478  // transform normals into pose local coordinate systems
479  btVector3 normal_local0 = normal_world_from_cast * tf_world0.getBasis();
480  btVector3 normal_local1 = normal_world_from_cast * tf_world1.getBasis();
481 
482  btVector3 pt_local0;
483  float localsup0;
484  getAverageSupport(shape->m_shape, normal_local0, localsup0, pt_local0);
485  btVector3 pt_world0 = tf_world0 * pt_local0;
486  btVector3 pt_local1;
487  float localsup1;
488  getAverageSupport(shape->m_shape, normal_local1, localsup1, pt_local1);
489  btVector3 pt_world1 = tf_world1 * pt_local1;
490 
491  float sup0 = normal_world_from_cast.dot(pt_world0);
492  float sup1 = normal_world_from_cast.dot(pt_world1);
493 
494  // TODO: this section is potentially problematic. think hard about the math
495  if (sup0 - sup1 > BULLET_SUPPORT_FUNC_TOLERANCE)
496  {
497  col->percent_interpolation = 0;
498  }
499  else if (sup1 - sup0 > BULLET_SUPPORT_FUNC_TOLERANCE)
500  {
501  col->percent_interpolation = 1;
502  }
503  else
504  {
505  const btVector3& pt_on_cast = cast_shape_is_first ? cp.m_positionWorldOnA : cp.m_positionWorldOnB;
506  float l0c = (pt_on_cast - pt_world0).length();
507  float l1c = (pt_on_cast - pt_world1).length();
508 
509  if (l0c + l1c < BULLET_LENGTH_TOLERANCE)
510  {
511  col->percent_interpolation = .5;
512  }
513  else
514  {
515  col->percent_interpolation = static_cast<double>(l0c / (l0c + l1c));
516  }
517  }
518 
519  return 1;
520 }

585  void addContactPoint(const btVector3& normalOnBInWorld, const btVector3& pointInWorld, btScalar depth) override
586  {
587  if (result_callback_.collisions_.done || result_callback_.collisions_.pair_done ||
588  depth > static_cast<btScalar>(result_callback_.contact_distance_))
589  {

591  }

595  btVector3 local_a;
596  btVector3 local_b;

600  local_b = m_body0Wrap->getCollisionObject()->getWorldTransform().invXform(pointInWorld);
601  }

605  local_b = m_body1Wrap->getCollisionObject()->getWorldTransform().invXform(pointInWorld);
606  }
607

609  new_pt.m_positionWorldOnA = point_a;
610  new_pt.m_positionWorldOnB = pointInWorld;
611 
612  // BP mod, store contact triangles.
613  if (is_swapped)

615  new_pt.m_partId0 = m_partId1;
616  new_pt.m_partId1 = m_partId0;
617  new_pt.m_index0 = m_index1;
