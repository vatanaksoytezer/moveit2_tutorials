90 bool MoveGroupCartesianPathService::computeService(const std::shared_ptr<rmw_request_id_t> /* unused */,
91  const std::shared_ptr<moveit_msgs::srv::GetCartesianPath::Request> req,
92  std::shared_ptr<moveit_msgs::srv::GetCartesianPath::Response> res)
93 {
94  RCLCPP_INFO(LOGGER, "Received request to compute Cartesian path");
95  context_->planning_scene_monitor_->updateFrameTransforms();
96 
97  moveit::core::RobotState start_state =
98  planning_scene_monitor::LockedPlanningSceneRO(context_->planning_scene_monitor_)->getCurrentState();
99  moveit::core::robotStateMsgToRobotState(req->start_state, start_state);
100  if (const moveit::core::JointModelGroup* jmg = start_state.getJointModelGroup(req->group_name))
101  {
102  std::string link_name = req->link_name;
103  if (link_name.empty() && !jmg->getLinkModelNames().empty())
104  link_name = jmg->getLinkModelNames().back();
105 
106  bool ok = true;
107  EigenSTL::vector_Isometry3d waypoints(req->waypoints.size());
108  const std::string& default_frame = context_->planning_scene_monitor_->getRobotModel()->getModelFrame();
109  bool no_transform = req->header.frame_id.empty() ||
110  moveit::core::Transforms::sameFrame(req->header.frame_id, default_frame) ||
111  moveit::core::Transforms::sameFrame(req->header.frame_id, link_name);
112 
113  for (std::size_t i = 0; i < req->waypoints.size(); ++i)
114  {
115  if (no_transform)
116  tf2::fromMsg(req->waypoints[i], waypoints[i]);
117  else
118  {
119  geometry_msgs::msg::PoseStamped p;
120  p.header = req->header;
121  p.pose = req->waypoints[i];
122  if (performTransform(p, default_frame))
123  tf2::fromMsg(p.pose, waypoints[i]);
124  else
125  {
126  RCLCPP_ERROR(LOGGER, "Error encountered transforming waypoints to frame '%s'", default_frame.c_str());
127  ok = false;
128  break;
129  }
130  }
131  }
132 
133  if (ok)
134  {
135  if (req->max_step < std::numeric_limits<double>::epsilon())
136  {
137  RCLCPP_ERROR(LOGGER, "Maximum step to take between consecutive configurations along Cartesian path"
138  "was not specified (this value needs to be > 0)");
139  res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
140  }
141  else
142  {
143  if (!waypoints.empty())
144  {
145  moveit::core::GroupStateValidityCallbackFn constraint_fn;
146  std::unique_ptr<planning_scene_monitor::LockedPlanningSceneRO> ls;
147  std::unique_ptr<kinematic_constraints::KinematicConstraintSet> kset;
148  if (req->avoid_collisions || !moveit::core::isEmpty(req->path_constraints))
149  {
150  ls = std::make_unique<planning_scene_monitor::LockedPlanningSceneRO>(context_->planning_scene_monitor_);
151  kset = std::make_unique<kinematic_constraints::KinematicConstraintSet>((*ls)->getRobotModel());
152  kset->add(req->path_constraints, (*ls)->getTransforms());
153  constraint_fn = boost::bind(
154  &isStateValid,
155  req->avoid_collisions ? static_cast<const planning_scene::PlanningSceneConstPtr&>(*ls).get() : nullptr,
156  kset->empty() ? nullptr : kset.get(), boost::placeholders::_1, boost::placeholders::_2,
157  boost::placeholders::_3);
158  }
159  bool global_frame = !moveit::core::Transforms::sameFrame(link_name, req->header.frame_id);
160  RCLCPP_INFO(LOGGER,
161  "Attempting to follow %u waypoints for link '%s' using a step of %lf m "
162  "and jump threshold %lf (in %s reference frame)",
163  (unsigned int)waypoints.size(), link_name.c_str(), req->max_step, req->jump_threshold,
164  global_frame ? "global" : "link");
165  std::vector<moveit::core::RobotStatePtr> traj;
166  res->fraction = moveit::core::CartesianInterpolator::computeCartesianPath(
167  &start_state, jmg, traj, start_state.getLinkModel(link_name), waypoints, global_frame,
168  moveit::core::MaxEEFStep(req->max_step), moveit::core::JumpThreshold(req->jump_threshold), constraint_fn);
169  moveit::core::robotStateToRobotStateMsg(start_state, res->start_state);
170 
171  robot_trajectory::RobotTrajectory rt(context_->planning_scene_monitor_->getRobotModel(), req->group_name);
172  for (const moveit::core::RobotStatePtr& traj_state : traj)
173  rt.addSuffixWayPoint(traj_state, 0.0);
174 
175  // time trajectory
176  // \todo optionally compute timing to move the eef with constant speed
177  trajectory_processing::IterativeParabolicTimeParameterization time_param;
178  time_param.computeTimeStamps(rt, 1.0);
179 
180  rt.getRobotTrajectoryMsg(res->solution);
181  RCLCPP_INFO(LOGGER, "Computed Cartesian path with %u points (followed %lf%% of requested trajectory)",
182  (unsigned int)traj.size(), res->fraction * 100.0);
183  if (display_computed_paths_ && rt.getWayPointCount() > 0)
184  {
185  moveit_msgs::msg::DisplayTrajectory disp;
186  disp.model_id = context_->planning_scene_monitor_->getRobotModel()->getName();
187  disp.trajectory.resize(1, res->solution);
188  moveit::core::robotStateToRobotStateMsg(rt.getFirstWayPoint(), disp.trajectory_start);
189  display_path_->publish(disp);
190  }
191  }
192  res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
193  }
194  }
195  else
196  res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::FRAME_TRANSFORM_FAILURE;
197  }
198  else
199  res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
200 
201  return true;
202 }
