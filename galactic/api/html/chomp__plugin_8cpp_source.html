65  planning_interface::PlanningContextPtr
66  getPlanningContext(const planning_scene::PlanningSceneConstPtr& planning_scene,
67  const planning_interface::MotionPlanRequest& req,
68  moveit_msgs::msg::MoveItErrorCodes& error_code) const override
69  {
70  error_code.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
71 
72  if (req.group_name.empty())
73  {
74  RCLCPP_ERROR(LOGGER, "No group specified to plan for");
75  error_code.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
76  return planning_interface::PlanningContextPtr();
77  }
78 
79  if (!planning_scene)
80  {
81  RCLCPP_ERROR(LOGGER, "No planning scene supplied as input");
82  error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
83  return planning_interface::PlanningContextPtr();
84  }
85 
86  // create PlanningScene using hybrid collision detector
87  planning_scene::PlanningScenePtr ps = planning_scene->diff();
88  ps->allocateCollisionDetector(collision_detection::CollisionDetectorAllocatorHybrid::create());
89 
90  // retrieve and configure existing context
91  const CHOMPPlanningContextPtr& context = planning_contexts_.at(req.group_name);
92  context->setPlanningScene(ps);
93  context->setMotionPlanRequest(req);
94  error_code.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
95  return context;
96  }
