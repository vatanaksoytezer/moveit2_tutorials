72 {
73  // TODO - deal with changing resolution?
74  BodyDecompositionCache& cache = getBodyDecompositionCache();
75  shapes::ShapeConstWeakPtr wptr(shape);
76  {
77  boost::mutex::scoped_lock slock(cache.lock_);
78  BodyDecompositionCache::Map::const_iterator cache_it = cache.map_.find(wptr);
79  if (cache_it != cache.map_.end())
80  {
81  return cache_it->second;

84 
85  BodyDecompositionConstPtr bdcp(new BodyDecomposition(shape, resolution));
86  {
87  boost::mutex::scoped_lock slock(cache.lock_);
88  cache.map_[wptr] = bdcp;

137 void getBodySphereVisualizationMarkers(const GroupStateRepresentationConstPtr& gsr, const std::string& reference_frame,
138  visualization_msgs::msg::MarkerArray& body_marker_array)
139 {
140  // creating namespaces
141  std::string robot_ns = gsr->dfce_->group_name_ + "_sphere_decomposition";
142  std::string attached_ns = "attached_sphere_decomposition";
143 
144  // creating colors
145  std_msgs::msg::ColorRGBA robot_color;
146  robot_color.r = 0;
147  robot_color.b = 0.8f;
148  robot_color.g = 0;
149  robot_color.a = 0.5;
150 
151  std_msgs::msg::ColorRGBA attached_color;
152  attached_color.r = 1;
153  attached_color.g = 1;
154  attached_color.b = 0;
155  attached_color.a = 0.5;
156 
157  // creating sphere marker
158  visualization_msgs::msg::Marker sphere_marker;
159  sphere_marker.header.frame_id = reference_frame;
160  sphere_marker.header.stamp = rclcpp::Time(0);
161  sphere_marker.ns = robot_ns;
162  sphere_marker.id = 0;
163  sphere_marker.type = visualization_msgs::msg::Marker::SPHERE;
164  sphere_marker.action = visualization_msgs::msg::Marker::ADD;
165  sphere_marker.pose.orientation.x = 0;
166  sphere_marker.pose.orientation.y = 0;
167  sphere_marker.pose.orientation.z = 0;
168  sphere_marker.pose.orientation.w = 1;
169  sphere_marker.color = robot_color;
170  sphere_marker.lifetime = rclcpp::Duration(0, 0);
171 
172  const moveit::core::RobotState& state = *(gsr->dfce_->state_);
173  unsigned int id = 0;
174  for (unsigned int i = 0; i < gsr->dfce_->link_names_.size(); ++i)
175  {
176  const moveit::core::LinkModel* ls = state.getLinkModel(gsr->dfce_->link_names_[i]);
177  if (gsr->dfce_->link_has_geometry_[i])
178  {
179  gsr->link_body_decompositions_[i]->updatePose(state.getFrameTransform(ls->getName()));
180 
181  collision_detection::PosedBodySphereDecompositionConstPtr sphere_representation =
182  gsr->link_body_decompositions_[i];
183  for (unsigned int j = 0; j < sphere_representation->getCollisionSpheres().size(); ++j)
184  {
185  sphere_marker.pose.position = tf2::toMsg(sphere_representation->getSphereCenters()[j]);
186  sphere_marker.scale.x = sphere_marker.scale.y = sphere_marker.scale.z =
187  sphere_representation->getCollisionSpheres()[j].radius_;
188  sphere_marker.id = id;
189  id++;
190 
191  body_marker_array.markers.push_back(sphere_marker);
192  }
193  }
194  }
195 
196  sphere_marker.ns = attached_ns;
197  sphere_marker.color = attached_color;
198  for (unsigned int i = 0; i < gsr->dfce_->attached_body_names_.size(); ++i)
199  {
200  const moveit::core::AttachedBody* att = state.getAttachedBody(gsr->dfce_->attached_body_names_[i]);
201  if (!att)
202  {
203  RCLCPP_WARN(LOGGER,
204  "Attached body '%s' was not found, skipping sphere "
205  "decomposition visualization",
206  gsr->dfce_->attached_body_names_[i].c_str());
207  continue;
208  }
209 
210  if (gsr->attached_body_decompositions_[i]->getSize() != att->getShapes().size())
211  {
212  RCLCPP_WARN(LOGGER, "Attached body size discrepancy");
213  continue;
214  }
215 
216  for (unsigned int j = 0; j < att->getShapes().size(); ++j)
217  {
218  PosedBodySphereDecompositionVectorPtr sphere_decp = gsr->attached_body_decompositions_[i];
219  sphere_decp->updatePose(j, att->getGlobalCollisionBodyTransforms()[j]);
220 
221  sphere_marker.pose.position = tf2::toMsg(sphere_decp->getSphereCenters()[j]);
222  sphere_marker.scale.x = sphere_marker.scale.y = sphere_marker.scale.z =
223  sphere_decp->getCollisionSpheres()[j].radius_;
224  sphere_marker.id = id;
225  body_marker_array.markers.push_back(sphere_marker);
226  id++;
227  }
228  }
229 }
