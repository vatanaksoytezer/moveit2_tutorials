160  double getDistanceGradient(double x, double y, double z, double& gradient_x, double& gradient_y, double& gradient_z,
161  bool& in_bounds) const
162  {
163  // Transpose of a rotation matrix equals its inverse, but computationally cheaper
164  Eigen::Vector3d rel_pos = pose_.linear().transpose() * Eigen::Vector3d(x, y, z);
165  double gx, gy, gz;
166  double res = distance_field::PropagationDistanceField::getDistanceGradient(rel_pos.x(), rel_pos.y(), rel_pos.z(),
167  gx, gy, gz, in_bounds);
168  Eigen::Vector3d grad = pose_ * Eigen::Vector3d(gx, gy, gz);
169  gradient_x = grad.x();
170  gradient_y = grad.y();
171  gradient_z = grad.z();
172  return res;
173  }

392  void addToVector(PosedBodySphereDecompositionPtr& bd)
393  {

395  decomp_vector_.push_back(bd);
396  collision_spheres_.insert(collision_spheres_.end(), bd->getCollisionSpheres().begin(),
397  bd->getCollisionSpheres().end());
398  posed_collision_spheres_.insert(posed_collision_spheres_.end(), bd->getSphereCenters().begin(),
399  bd->getSphereCenters().end());
400  sphere_radii_.insert(sphere_radii_.end(), bd->getSphereRadii().begin(), bd->getSphereRadii().end());
401  }
402

404  {
405  return decomp_vector_.size();
406  }
407 
408  PosedBodySphereDecompositionConstPtr getPosedBodySphereDecomposition(unsigned int i) const
409  {
410  if (i >= decomp_vector_.size())
411  {
412  RCLCPP_INFO(LOGGER, "No body decomposition");
413  return empty_ptr_;
414  }
415  return decomp_vector_[i];
416  }
417 
418  void updatePose(unsigned int ind, const Eigen::Isometry3d& pose)
419  {
420  if (ind >= decomp_vector_.size())
421  {
422  RCLCPP_WARN(LOGGER, "Can't update pose");
423  return;
424  }
425  decomp_vector_[ind]->updatePose(pose);
426  for (unsigned int i = 0; i < decomp_vector_[ind]->getSphereCenters().size(); ++i)
427  {
428  posed_collision_spheres_[sphere_index_map_[ind] + i] = decomp_vector_[ind]->getSphereCenters()[i];
429  }
430  }
431 
432 private:
