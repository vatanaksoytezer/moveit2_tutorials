44 static inline bool validateScale(double scale)
45 {
46  if (scale < std::numeric_limits<double>::epsilon())
47  {
48  RCLCPP_ERROR(LOGGER, "Scale must be positive");
49  return false;
50  }
51  if (scale > std::numeric_limits<double>::max())
52  {
53  RCLCPP_ERROR(LOGGER, "Scale must be finite");
54  return false;
55  }
56  return true;
57 }

59 static inline bool validatePadding(double padding)

66  if (padding > std::numeric_limits<double>::max())
67  {

69  return false;
70  }
71  return true;

115 void CollisionEnv::setPadding(double padding)
116 {
117  if (!validatePadding(padding))
118  return;
119  std::vector<std::string> u;
120  const std::vector<const moveit::core::LinkModel*>& links = robot_model_->getLinkModelsWithCollisionGeometry();
121  for (auto link : links)
122  {
123  if (getLinkPadding(link->getName()) != padding)
124  u.push_back(link->getName());
125  link_padding_[link->getName()] = padding;
126  }
127  if (!u.empty())
128  updatedPaddingOrScaling(u);
129 }
