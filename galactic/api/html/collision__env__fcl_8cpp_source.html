56 void checkFCLCapabilities(const DistanceRequest& req)
57 {
58 #if MOVEIT_FCL_VERSION < FCL_VERSION_CHECK(0, 6, 0)
59  if (req.enable_nearest_points)

65  RCLCPP_ERROR_THROTTLE(LOGGER, steady_clock, 2000,
66  "You requested a distance check with enable_nearest_points=true, "
67  "but the FCL version MoveIt was compiled against (%d.%d.%d) "
68  "is known to return bogus nearest points. Please update your FCL "
69  "to at least 0.6.0.",

76 CollisionEnvFCL::CollisionEnvFCL(const moveit::core::RobotModelConstPtr& model, double padding, double scale)
77  : CollisionEnv(model, padding, scale)
78 {
79  const std::vector<const moveit::core::LinkModel*>& links = robot_model_->getLinkModelsWithCollisionGeometry();
80  std::size_t index;
81  robot_geoms_.resize(robot_model_->getLinkGeometryCount());
82  robot_fcl_objs_.resize(robot_model_->getLinkGeometryCount());
83  // we keep the same order of objects as what RobotState *::getLinkState() returns
84  for (auto link : links)
85  for (std::size_t j = 0; j < link->getShapes().size(); ++j)
86  {
87  FCLGeometryConstPtr link_geometry = createCollisionGeometry(link->getShapes()[j], getLinkScale(link->getName()),
88  getLinkPadding(link->getName()), link, j);
89  if (link_geometry)
90  {
91  index = link->getFirstCollisionBodyTransformIndex() + j;
92  robot_geoms_[index] = link_geometry;
93 
94  // Need to store the FCL object so the AABB does not get recreated every time.
95  // Every time this object is created, g->computeLocalAABB() is called which is
96  // very expensive and should only be calculated once. To update the AABB, use the
97  // collObj->setTransform and then call collObj->computeAABB() to transform the AABB.
98  robot_fcl_objs_[index] =
99  FCLCollisionObjectConstPtr(new fcl::CollisionObjectd(link_geometry->collision_geometry_));
100  }
101  else
102  RCLCPP_ERROR(LOGGER, "Unable to construct collision geometry for link '%s'", link->getName().c_str());
103  }
104 
105  manager_ = std::make_unique<fcl::DynamicAABBTreeCollisionManagerd>();
106 
107  // request notifications about changes to new world
108  observer_handle_ = getWorld()->addObserver(boost::bind(&CollisionEnvFCL::notifyObjectChange, this, _1, _2));
109 }

168 void CollisionEnvFCL::getAttachedBodyObjects(const moveit::core::AttachedBody* ab,
169  std::vector<FCLGeometryConstPtr>& geoms) const
170 {
171  const std::vector<shapes::ShapeConstPtr>& shapes = ab->getShapes();
172  const size_t num_shapes = shapes.size();
173  geoms.reserve(num_shapes);
174  for (std::size_t i = 0; i < num_shapes; ++i)
175  {
176  FCLGeometryConstPtr co = createCollisionGeometry(shapes[i], getLinkScale(ab->getAttachedLinkName()),

179  geoms.push_back(co);

183 void CollisionEnvFCL::constructFCLObjectWorld(const World::Object* obj, FCLObject& fcl_obj) const
184 {
185  for (std::size_t i = 0; i < obj->shapes_.size(); ++i)
186  {
187  FCLGeometryConstPtr g = createCollisionGeometry(obj->shapes_[i], obj);
188  if (g)
189  {
190  auto co = new fcl::CollisionObjectd(g->collision_geometry_, transform2fcl(obj->global_shape_poses_[i]));
191  fcl_obj.collision_objects_.push_back(FCLCollisionObjectPtr(co));
192  fcl_obj.collision_geometry_.push_back(g);
193  }

197 void CollisionEnvFCL::constructFCLObjectRobot(const moveit::core::RobotState& state, FCLObject& fcl_obj) const
198 {
199  fcl_obj.collision_objects_.reserve(robot_geoms_.size());
200  fcl::Transform3d fcl_tf;
201 
202  for (std::size_t i = 0; i < robot_geoms_.size(); ++i)
203  if (robot_geoms_[i] && robot_geoms_[i]->collision_geometry_)
204  {
205  transform2fcl(state.getCollisionBodyTransform(robot_geoms_[i]->collision_geometry_data_->ptr.link,
206  robot_geoms_[i]->collision_geometry_data_->shape_index),
207  fcl_tf);
208  auto coll_obj = new fcl::CollisionObjectd(*robot_fcl_objs_[i]);
209  coll_obj->setTransform(fcl_tf);
210  coll_obj->computeAABB();
211  fcl_obj.collision_objects_.push_back(FCLCollisionObjectPtr(coll_obj));
212  }
213 
214  // TODO: Implement a method for caching fcl::CollisionObject's for moveit::core::AttachedBody's
215  std::vector<const moveit::core::AttachedBody*> ab;
216  state.getAttachedBodies(ab);
217  for (auto& body : ab)
218  {
219  std::vector<FCLGeometryConstPtr> objs;
220  getAttachedBodyObjects(body, objs);
221  const EigenSTL::vector_Isometry3d& ab_t = body->getGlobalCollisionBodyTransforms();
222  for (std::size_t k = 0; k < objs.size(); ++k)
223  if (objs[k]->collision_geometry_)
224  {
225  transform2fcl(ab_t[k], fcl_tf);
226  fcl_obj.collision_objects_.push_back(
227  std::make_shared<fcl::CollisionObjectd>(objs[k]->collision_geometry_, fcl_tf));
228  // we copy the shared ptr to the CollisionGeometryData, as this is not stored by the class itself,
229  // and would be destroyed when objs goes out of scope.
230  fcl_obj.collision_geometry_.push_back(objs[k]);
231  }
232  }

255 void CollisionEnvFCL::checkSelfCollisionHelper(const CollisionRequest& req, CollisionResult& res,
256  const moveit::core::RobotState& state,
257  const AllowedCollisionMatrix* acm) const
258 {
259  FCLManager manager;
260  allocSelfCollisionBroadPhase(state, manager);
261  CollisionData cd(&req, &res, acm);
262  cd.enableGroup(getRobotModel());
263  manager.manager_->collide(&cd, &collisionCallback);
264  if (req.distance)
265  {
266  DistanceRequest dreq;
267  DistanceResult dres;
268 
269  dreq.group_name = req.group_name;
270  dreq.acm = acm;
271  dreq.enableGroup(getRobotModel());
272  distanceSelf(dreq, dres, state);
273  res.distance = dres.minimum_distance.distance;
274  }

305 void CollisionEnvFCL::checkRobotCollisionHelper(const CollisionRequest& req, CollisionResult& res,
306  const moveit::core::RobotState& state,
307  const AllowedCollisionMatrix* acm) const
308 {
309  FCLObject fcl_obj;
310  constructFCLObjectRobot(state, fcl_obj);
311 
312  CollisionData cd(&req, &res, acm);
313  cd.enableGroup(getRobotModel());
314  for (std::size_t i = 0; !cd.done_ && i < fcl_obj.collision_objects_.size(); ++i)
315  manager_->collide(fcl_obj.collision_objects_[i].get(), &cd, &collisionCallback);
316 
317  if (req.distance)
318  {
319  DistanceRequest dreq;
320  DistanceResult dres;
321 
322  dreq.group_name = req.group_name;
323  dreq.acm = acm;
324  dreq.enableGroup(getRobotModel());
325  distanceRobot(dreq, dres, state);
326  res.distance = dres.minimum_distance.distance;
327  }

355 void CollisionEnvFCL::updateFCLObject(const std::string& id)
356 {
357  // remove FCL objects that correspond to this object
358  auto jt = fcl_objs_.find(id);
359  if (jt != fcl_objs_.end())
360  {
361  jt->second.unregisterFrom(manager_.get());
362  jt->second.clear();
363  }
364 
365  // check to see if we have this object
366  auto it = getWorld()->find(id);
367  if (it != getWorld()->end())
368  {
369  // construct FCL objects that correspond to this object
370  if (jt != fcl_objs_.end())
371  {
372  constructFCLObjectWorld(it->second.get(), jt->second);
373  jt->second.registerTo(manager_.get());
374  }
375  else
376  {
377  constructFCLObjectWorld(it->second.get(), fcl_objs_[id]);
378  fcl_objs_[id].registerTo(manager_.get());
379  }
380  }
381  else
382  {
383  if (jt != fcl_objs_.end())
384  fcl_objs_.erase(jt);
385  }
386 
387  // manager_->update();
388 }

390 void CollisionEnvFCL::setWorld(const WorldPtr& world)
391 {
392  if (world == getWorld())
393  return;
394 
395  // turn off notifications about old world
396  getWorld()->removeObserver(observer_handle_);
397 
398  // clear out objects from old world
399  manager_->clear();
400  fcl_objs_.clear();
401  cleanCollisionGeometryCache();
402 
403  CollisionEnv::setWorld(world);
404 
405  // request notifications about changes to new world
406  observer_handle_ = getWorld()->addObserver(boost::bind(&CollisionEnvFCL::notifyObjectChange, this, _1, _2));
407 
408  // get notifications any objects already in the new world
409  getWorld()->notifyObserverAllObjects(observer_handle_, World::CREATE);
410 }

412 void CollisionEnvFCL::notifyObjectChange(const ObjectConstPtr& obj, World::Action action)
413 {
414  if (action == World::DESTROY)
415  {
416  auto it = fcl_objs_.find(obj->id_);
417  if (it != fcl_objs_.end())
418  {
419  it->second.unregisterFrom(manager_.get());
420  it->second.clear();
421  fcl_objs_.erase(it);
422  }
423  cleanCollisionGeometryCache();
424  }
425  else
426  {
427  updateFCLObject(obj->id_);
428  if (action & (World::DESTROY | World::REMOVE_SHAPE))
429  cleanCollisionGeometryCache();
430  }
