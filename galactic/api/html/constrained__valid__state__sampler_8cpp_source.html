106 bool ompl_interface::ValidConstrainedSampler::sampleNear(ompl::base::State* state, const ompl::base::State* near,
107  const double distance)
108 {
109  if (!sample(state))
110  return false;
111  double total_d = si_->distance(state, near);
112  if (total_d > distance)
113  {
114  double dist = pow(rng_.uniform01(), inv_dim_) * distance;
115  si_->getStateSpace()->interpolate(near, state, dist / total_d, state);
116  planning_context_->getOMPLStateSpace()->copyToRobotState(work_state_, state);
117  if (!kinematic_constraint_set_->decide(work_state_).satisfied)
118  return false;
119  }
120  return true;
121 }
