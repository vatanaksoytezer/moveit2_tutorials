129  void discover(bool force = false)
130  {
131  // Skip if controller stamp is too new for new discovery, enforce update if force==true
132  if (!force && ((node_->now() - controllers_stamp_) < CONTROLLER_INFORMATION_VALIDITY_AGE))
133  return;
134 
135  controllers_stamp_ = node_->now();
136 
137  auto request = std::make_shared<controller_manager_msgs::srv::ListControllers::Request>();
138  auto result_future = list_controllers_service_->async_send_request(request);
139  if (result_future.wait_for(std::chrono::duration<double>(SERVICE_CALL_TIMEOUT)) == std::future_status::timeout)
140  {
141  RCLCPP_WARN_STREAM(LOGGER, "Failed to read controllers from " << list_controllers_service_->get_service_name()
142  << " within " << SERVICE_CALL_TIMEOUT
143  << " seconds");
144  return;
145  }
146 
147  managed_controllers_.clear();
148  active_controllers_.clear();
149 
150  const auto& result = result_future.get();
151  for (const controller_manager_msgs::msg::ControllerState& controller : result->controller)
152  {
153  // If the controller is active, add it to the map of active controllers.
154  if (isActive(controller))
155  {
156  // Get the names of the interfaces currently claimed by the active controller.
157  auto& claimed_interfaces = active_controllers_.insert(std::make_pair(controller.name, controller))
158  .first->second.claimed_interfaces; // without namespace
159  // Modify the claimed interface names in-place to only include the name of the joint and not the command type
160  // (e.g. position, velocity, etc.).
161  std::transform(claimed_interfaces.cbegin(), claimed_interfaces.cend(), claimed_interfaces.begin(),
162  [](const std::string& claimed_interface) {
163  return parseJointNameFromResource(claimed_interface);
164  });
165  }
166 
167  // Instantiate a controller handle if one is available for this type of controller.
168  if (loader_.isClassAvailable(controller.type))
169  {
170  std::string absname = getAbsName(controller.name);
171  auto controller_it = managed_controllers_.insert(std::make_pair(absname, controller)).first; // with namespace
172  // Get the names of the interfaces that would be claimed by this currently-inactive controller if it was activated.
173  auto& required_interfaces = controller_it->second.required_command_interfaces;
174  // Modify the required interface names in-place to only include the name of the joint and not the command type
175  // (e.g. position, velocity, etc.).
176  std::transform(required_interfaces.cbegin(), required_interfaces.cend(), required_interfaces.begin(),
177  [](const std::string& required_interface) {
178  return parseJointNameFromResource(required_interface);
179  });
180  allocate(absname, controller_it->second);
181  }
182  }
183  }

191  void allocate(const std::string& name, const controller_manager_msgs::msg::ControllerState& controller)
192  {
193  if (handles_.find(name) == handles_.end())
194  {
195  const std::string& type = controller.type;
196  AllocatorsMap::iterator alloc_it = allocators_.find(type);
197  if (alloc_it == allocators_.end())
198  { // create allocator is needed
199  alloc_it = allocators_.insert(std::make_pair(type, loader_.createUniqueInstance(type))).first;
200  }
201 
202  std::vector<std::string> resources;
203  // Collect claimed resources across different hardware interfaces
204  for (const auto& resource : controller.claimed_interfaces)
205  {
206  resources.push_back(parseJointNameFromResource(resource));
207  }
208 
209  moveit_controller_manager::MoveItControllerHandlePtr handle =
210  alloc_it->second->alloc(node_, name, resources); // allocate handle
211  if (handle)
212  handles_.insert(std::make_pair(name, handle));
213  }
214  }

354  bool switchControllers(const std::vector<std::string>& activate, const std::vector<std::string>& deactivate) override
355  {
356  std::scoped_lock<std::mutex> lock(controllers_mutex_);
357  discover(true);
358 
359  typedef boost::bimap<std::string, std::string> resources_bimap;
360 
361  resources_bimap claimed_resources;
362 
363  // fill bimap with active controllers and their resources
364  for (std::pair<const std::string, controller_manager_msgs::msg::ControllerState>& active_controller :
365  active_controllers_)
366  {
367  for (const auto& resource : active_controller.second.claimed_interfaces)
368  {
369  claimed_resources.insert(resources_bimap::value_type(active_controller.second.name, resource));
370  }
371  }
372 
373  auto request = std::make_shared<controller_manager_msgs::srv::SwitchController::Request>();
374  for (const std::string& it : deactivate)
375  {
376  ControllersMap::iterator c = managed_controllers_.find(it);
377  if (c != managed_controllers_.end())
378  { // controller belongs to this manager
379  request->stop_controllers.push_back(c->second.name);
380  claimed_resources.right.erase(c->second.name); // remove resources
381  }
382  }
383 
384  // For each controller to activate, find conflicts between the interfaces required by that controller and the
385  // interfaces claimed by currently-active controllers.
386  for (const std::string& it : activate)
387  {
388  ControllersMap::iterator c = managed_controllers_.find(it);
389  if (c != managed_controllers_.end())
390  { // controller belongs to this manager
391  request->start_controllers.push_back(c->second.name);
392  for (const auto& required_resource : c->second.required_command_interfaces)
393  {
394  resources_bimap::right_const_iterator res = claimed_resources.right.find(required_resource);
395  if (res != claimed_resources.right.end())
396  { // resource is claimed
397  request->stop_controllers.push_back(res->second); // add claiming controller to stop list
398  claimed_resources.left.erase(res->second); // remove claimed resources
399  }
400  }
401  }
402  }
403 
404  // Setting level to STRICT means that the controller switch will only be committed if all controllers are
405  // successfully activated or deactivated.
406  request->strictness = controller_manager_msgs::srv::SwitchController::Request::STRICT;
407 
408  if (!request->start_controllers.empty() || !request->stop_controllers.empty())
409  { // something to switch?
410  auto result_future = switch_controller_service_->async_send_request(request);
411  if (result_future.wait_for(std::chrono::duration<double>(SERVICE_CALL_TIMEOUT)) == std::future_status::timeout)
412  {
413  RCLCPP_ERROR_STREAM(LOGGER, "Couldn't switch controllers at " << switch_controller_service_->get_service_name()
414  << " within " << SERVICE_CALL_TIMEOUT
415  << " seconds");
416  return false;
417  }
418  discover(true);
419  return result_future.get()->ok;
420  }
421  return true;
422  }
