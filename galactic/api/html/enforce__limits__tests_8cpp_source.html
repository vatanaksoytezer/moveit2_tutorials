47 void checkVelocityLimits(const moveit::core::JointModelGroup* joint_model_group, const Eigen::ArrayXd& delta_theta)
48 {
49  std::size_t joint_delta_index{ 0 };
50  for (const moveit::core::JointModel* joint : joint_model_group->getActiveJointModels())
51  {
52  const auto& bounds = joint->getVariableBounds(joint->getName());
53 
54  if (bounds.velocity_bounded_)
55  {
56  auto velocity = delta_theta(joint_delta_index) / PUBLISH_PERIOD;
57  EXPECT_GE(velocity, bounds.min_velocity_) << "Joint " << joint_delta_index << " violates velocity limit";
58  EXPECT_LE(velocity, bounds.max_velocity_) << "Joint " << joint_delta_index << " violates velocity limit";
59  }
60  ++joint_delta_index;
61  }
62 }

79 TEST_F(EnforceLimitsTests, VelocityScalingTest)
80 {
81  // Request joint angle changes that are too fast given the control period.
82  Eigen::ArrayXd delta_theta(7);
83  delta_theta[0] = 0; // rad
84  delta_theta[1] = 0.01;
85  delta_theta[2] = 0.02;
86  delta_theta[3] = 0.03;
87  delta_theta[4] = 0.04;
88  delta_theta[5] = 0.05;
89  delta_theta[6] = 0.06;
90 
91  // Store the original joint commands for comparison before applying velocity scaling.
92  auto result_delta_theta = moveit_servo::enforceVelocityLimits(joint_model_group_, PUBLISH_PERIOD, delta_theta);
93 
94  // Test that we don't violate velocity limits
95  checkVelocityLimits(joint_model_group_, result_delta_theta);
96 }
