48 class FixStartStateBounds : public planning_request_adapter::PlanningRequestAdapter
49 {
50 public:
51  static const std::string BOUNDS_PARAM_NAME;
52  static const std::string DT_PARAM_NAME;
53 
54  void initialize(const rclcpp::Node::SharedPtr& node, const std::string& parameter_namespace) override
55  {
56  node_ = node;
57  bounds_dist_ = getParam(node_, LOGGER, parameter_namespace, BOUNDS_PARAM_NAME, 0.05);
58  max_dt_offset_ = getParam(node_, LOGGER, parameter_namespace, DT_PARAM_NAME, 0.5);
59  }
60 
61  std::string getDescription() const override
62  {
63  return "Fix Start State Bounds";
64  }
65 
66  bool adaptAndPlan(const PlannerFn& planner, const planning_scene::PlanningSceneConstPtr& planning_scene,
67  const planning_interface::MotionPlanRequest& req, planning_interface::MotionPlanResponse& res,
68  std::vector<std::size_t>& added_path_index) const override
69  {
70  RCLCPP_DEBUG(LOGGER, "Running '%s'", getDescription().c_str());
71 
72  // get the specified start state
73  moveit::core::RobotState start_state = planning_scene->getCurrentState();
74  moveit::core::robotStateMsgToRobotState(planning_scene->getTransforms(), req.start_state, start_state);
75 
76  const std::vector<const moveit::core::JointModel*>& jmodels =
77  planning_scene->getRobotModel()->hasJointModelGroup(req.group_name) ?
78  planning_scene->getRobotModel()->getJointModelGroup(req.group_name)->getJointModels() :
79  planning_scene->getRobotModel()->getJointModels();
80 
81  bool change_req = false;
82  for (const moveit::core::JointModel* jm : jmodels)
83  {
84  // Check if we have a revolute, continuous joint. If we do, then we only need to make sure
85  // it is within de model's declared bounds (usually -Pi, Pi), since the values wrap around.
86  // It is possible that the encoder maintains values outside the range [-Pi, Pi], to inform
87  // how many times the joint was wrapped. Because of this, we remember the offsets for continuous
88  // joints, and we un-do them when the plan comes from the planner
89 
90  if (jm->getType() == moveit::core::JointModel::REVOLUTE)
91  {
92  if (static_cast<const moveit::core::RevoluteJointModel*>(jm)->isContinuous())
93  {
94  double initial = start_state.getJointPositions(jm)[0];
95  start_state.enforceBounds(jm);
96  double after = start_state.getJointPositions(jm)[0];
97  if (fabs(initial - after) > std::numeric_limits<double>::epsilon())
98  change_req = true;
99  }
100  }
101  else
102  // Normalize yaw; no offset needs to be remembered
103  if (jm->getType() == moveit::core::JointModel::PLANAR)
104  {
105  const double* p = start_state.getJointPositions(jm);
106  double copy[3] = { p[0], p[1], p[2] };
107  if (static_cast<const moveit::core::PlanarJointModel*>(jm)->normalizeRotation(copy))
108  {
109  start_state.setJointPositions(jm, copy);
110  change_req = true;
111  }
112  }
113  else
114  // Normalize quaternions
115  if (jm->getType() == moveit::core::JointModel::FLOATING)
116  {
117  const double* p = start_state.getJointPositions(jm);
118  double copy[7] = { p[0], p[1], p[2], p[3], p[4], p[5], p[6] };
119  if (static_cast<const moveit::core::FloatingJointModel*>(jm)->normalizeRotation(copy))
120  {
121  start_state.setJointPositions(jm, copy);
122  change_req = true;
123  }
124  }
125  }
126 
127  // pointer to a prefix state we could possibly add, if we detect we have to make changes
128  moveit::core::RobotStatePtr prefix_state;
129  for (const moveit::core::JointModel* jmodel : jmodels)
130  {
131  if (!start_state.satisfiesBounds(jmodel))
132  {
133  if (start_state.satisfiesBounds(jmodel, bounds_dist_))
134  {
135  if (!prefix_state)
136  prefix_state = std::make_shared<moveit::core::RobotState>(start_state);
137  start_state.enforceBounds(jmodel);
138  change_req = true;
139  RCLCPP_INFO(LOGGER, "Starting state is just outside bounds (joint '%s'). Assuming within bounds.",
140  jmodel->getName().c_str());
141  }
142  else
143  {
144  std::stringstream joint_values;
145  std::stringstream joint_bounds_low;
146  std::stringstream joint_bounds_hi;
147  const double* p = start_state.getJointPositions(jmodel);
148  for (std::size_t k = 0; k < jmodel->getVariableCount(); ++k)
149  joint_values << p[k] << " ";
150  const moveit::core::JointModel::Bounds& b = jmodel->getVariableBounds();
151  for (const moveit::core::VariableBounds& variable_bounds : b)
152  {
153  joint_bounds_low << variable_bounds.min_position_ << " ";
154  joint_bounds_hi << variable_bounds.max_position_ << " ";
155  }
156  RCLCPP_WARN(LOGGER,
157  "Joint '%s' from the starting state is outside bounds by a significant margin: [%s] should be in "
158  "the range [%s], [%s] but the error above the ~%s parameter (currently set to %f)",
159  jmodel->getName().c_str(), joint_values.str().c_str(), joint_bounds_low.str().c_str(),
160  joint_bounds_hi.str().c_str(), BOUNDS_PARAM_NAME.c_str(), bounds_dist_);
161  }
162  }
163  }
164 
165  bool solved;
166  // if we made any changes, use them
167  if (change_req)
168  {
169  planning_interface::MotionPlanRequest req2 = req;
170  moveit::core::robotStateToRobotStateMsg(start_state, req2.start_state);
171  solved = planner(planning_scene, req2, res);
172  }
173  else
174  solved = planner(planning_scene, req, res);
175 
176  // re-add the prefix state, if it was constructed
177  if (prefix_state && res.trajectory_ && !res.trajectory_->empty())
178  {
179  // heuristically decide a duration offset for the trajectory (induced by the additional point added as a prefix to
180  // the computed trajectory)
181  res.trajectory_->setWayPointDurationFromPrevious(0, std::min(max_dt_offset_,
182  res.trajectory_->getAverageSegmentDuration()));
183  res.trajectory_->addPrefixWayPoint(prefix_state, 0.0);
184  // we add a prefix point, so we need to bump any previously added index positions
185  for (std::size_t& added_index : added_path_index)
186  added_index++;
187  added_path_index.push_back(0);
188  }
189 
190  return solved;
191  }
192 
193 private:
194  rclcpp::Node::SharedPtr node_;
195  double bounds_dist_;
196  double max_dt_offset_;
197 };
