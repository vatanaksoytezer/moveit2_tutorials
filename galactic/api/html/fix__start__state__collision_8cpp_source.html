47 class FixStartStateCollision : public planning_request_adapter::PlanningRequestAdapter
48 {
49 public:
50  static const std::string DT_PARAM_NAME;
51  static const std::string JIGGLE_PARAM_NAME;
52  static const std::string ATTEMPTS_PARAM_NAME;
53 
54  void initialize(const rclcpp::Node::SharedPtr& node, const std::string& parameter_namespace) override
55  {
56  node_ = node;
57  max_dt_offset_ = getParam(node_, LOGGER, parameter_namespace, DT_PARAM_NAME, 0.5);
58  jiggle_fraction_ = getParam(node_, LOGGER, parameter_namespace, JIGGLE_PARAM_NAME, 0.02);
59  sampling_attempts_ = getParam(node_, LOGGER, parameter_namespace, ATTEMPTS_PARAM_NAME, 100);
60  if (sampling_attempts_ < 1)
61  {
62  sampling_attempts_ = 1;
63  RCLCPP_WARN(LOGGER, "Param '%s' needs to be at least 1.", ATTEMPTS_PARAM_NAME.c_str());
64  }
65  }
66 
67  std::string getDescription() const override
68  {
69  return "Fix Start State In Collision";
70  }
71 
72  bool adaptAndPlan(const PlannerFn& planner, const planning_scene::PlanningSceneConstPtr& planning_scene,
73  const planning_interface::MotionPlanRequest& req, planning_interface::MotionPlanResponse& res,
74  std::vector<std::size_t>& added_path_index) const override
75  {
76  RCLCPP_DEBUG(LOGGER, "Running '%s'", getDescription().c_str());
77 
78  // get the specified start state
79  moveit::core::RobotState start_state = planning_scene->getCurrentState();
80  moveit::core::robotStateMsgToRobotState(planning_scene->getTransforms(), req.start_state, start_state);
81 
82  collision_detection::CollisionRequest creq;
83  creq.group_name = req.group_name;
84  collision_detection::CollisionResult cres;
85  planning_scene->checkCollision(creq, cres, start_state);
86  if (cres.collision)
87  {
88  // Rerun in verbose mode
89  collision_detection::CollisionRequest vcreq = creq;
90  collision_detection::CollisionResult vcres;
91  vcreq.verbose = true;
92  planning_scene->checkCollision(vcreq, vcres, start_state);
93 
94  if (creq.group_name.empty())
95  RCLCPP_INFO(LOGGER, "Start state appears to be in collision");
96  else
97  RCLCPP_INFO(LOGGER, "Start state appears to be in collision with respect to group %s", creq.group_name.c_str());
98 
99  moveit::core::RobotStatePtr prefix_state(new moveit::core::RobotState(start_state));
100  random_numbers::RandomNumberGenerator& rng = prefix_state->getRandomNumberGenerator();
101 
102  const std::vector<const moveit::core::JointModel*>& jmodels =
103  planning_scene->getRobotModel()->hasJointModelGroup(req.group_name) ?
104  planning_scene->getRobotModel()->getJointModelGroup(req.group_name)->getJointModels() :
105  planning_scene->getRobotModel()->getJointModels();
106 
107  bool found = false;
108  for (int c = 0; !found && c < sampling_attempts_; ++c)
109  {
110  for (std::size_t i = 0; !found && i < jmodels.size(); ++i)
111  {
112  std::vector<double> sampled_variable_values(jmodels[i]->getVariableCount());
113  const double* original_values = prefix_state->getJointPositions(jmodels[i]);
114  jmodels[i]->getVariableRandomPositionsNearBy(rng, &sampled_variable_values[0], original_values,
115  jmodels[i]->getMaximumExtent() * jiggle_fraction_);
116  start_state.setJointPositions(jmodels[i], sampled_variable_values);
117  collision_detection::CollisionResult cres;
118  planning_scene->checkCollision(creq, cres, start_state);
119  if (!cres.collision)
120  {
121  found = true;
122  RCLCPP_INFO(LOGGER, "Found a valid state near the start state at distance %lf after %d attempts",
123  prefix_state->distance(start_state), c);
124  }
125  }
126  }
127 
128  if (found)
129  {
130  planning_interface::MotionPlanRequest req2 = req;
131  moveit::core::robotStateToRobotStateMsg(start_state, req2.start_state);
132  bool solved = planner(planning_scene, req2, res);
133  if (solved && !res.trajectory_->empty())
134  {
135  // heuristically decide a duration offset for the trajectory (induced by the additional point added as a
136  // prefix to the computed trajectory)
137  res.trajectory_->setWayPointDurationFromPrevious(0, std::min(max_dt_offset_,
138  res.trajectory_->getAverageSegmentDuration()));
139  res.trajectory_->addPrefixWayPoint(prefix_state, 0.0);
140  // we add a prefix point, so we need to bump any previously added index positions
141  for (std::size_t& added_index : added_path_index)
142  added_index++;
143  added_path_index.push_back(0);
144  }
145  return solved;
146  }
147  else
148  {
149  RCLCPP_WARN(LOGGER,
150  "Unable to find a valid state nearby the start state (using jiggle fraction of %lf and %u sampling "
151  "attempts). Passing the original planning request to the planner.",
152  jiggle_fraction_, sampling_attempts_);
153  return planner(planning_scene, req, res);
154  }
155  }
156  else
157  {
158  if (creq.group_name.empty())
159  RCLCPP_DEBUG(LOGGER, "Start state is valid");
160  else
161  RCLCPP_DEBUG(LOGGER, "Start state is valid with respect to group %s", creq.group_name.c_str());
162  return planner(planning_scene, req, res);
163  }
164  }
165 
166 private:
167  rclcpp::Node::SharedPtr node_;
168  double max_dt_offset_;
169  double jiggle_fraction_;
170  int sampling_attempts_;
171 };
