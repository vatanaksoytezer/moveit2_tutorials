50  void initialize(const rclcpp::Node::SharedPtr& node, const std::string& parameter_namespace) override
51  {
52  node_ = node;
53  workspace_extent_ = getParam(node_, LOGGER, parameter_namespace, WBOUNDS_PARAM_NAME, 10.0);
54  workspace_extent_ /= 2.0;
55  }

57  std::string getDescription() const override
58  {

62  bool adaptAndPlan(const PlannerFn& planner, const planning_scene::PlanningSceneConstPtr& planning_scene,
63  const planning_interface::MotionPlanRequest& req, planning_interface::MotionPlanResponse& res,
64  std::vector<std::size_t>& /*added_path_index*/) const override
65  {
66  RCLCPP_DEBUG(LOGGER, "Running '%s'", getDescription().c_str());
67  const moveit_msgs::msg::WorkspaceParameters& wparams = req.workspace_parameters;
68  if (wparams.min_corner.x == wparams.max_corner.x && wparams.min_corner.x == 0.0 &&
69  wparams.min_corner.y == wparams.max_corner.y && wparams.min_corner.y == 0.0 &&
70  wparams.min_corner.z == wparams.max_corner.z && wparams.min_corner.z == 0.0)
71  {
72  RCLCPP_DEBUG(LOGGER, "It looks like the planning volume was not specified. Using default values.");
73  planning_interface::MotionPlanRequest req2 = req;
74  moveit_msgs::msg::WorkspaceParameters& default_wp = req2.workspace_parameters;
75  default_wp.min_corner.x = default_wp.min_corner.y = default_wp.min_corner.z = -workspace_extent_;
76  default_wp.max_corner.x = default_wp.max_corner.y = default_wp.max_corner.z = workspace_extent_;
77  return planner(planning_scene, req2, res);
78  }
79  else
80  return planner(planning_scene, req, res);
81  }
82 
83 private:
84  rclcpp::Node::SharedPtr node_;
85  double workspace_extent_;
