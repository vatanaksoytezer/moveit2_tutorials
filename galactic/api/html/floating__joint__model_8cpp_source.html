100 double FloatingJointModel::getMaximumExtent(const Bounds& other_bounds) const
101 {
102  double dx = other_bounds[0].max_position_ - other_bounds[0].min_position_;
103  double dy = other_bounds[1].max_position_ - other_bounds[1].min_position_;
104  double dz = other_bounds[2].max_position_ - other_bounds[2].min_position_;
105  return sqrt(dx * dx + dy * dy + dz * dz) + boost::math::constants::pi<double>() * 0.5 * angular_distance_weight_;
106 }

121 double FloatingJointModel::distanceRotation(const double* values1, const double* values2) const
122 {
123  double dq =
124  fabs(values1[3] * values2[3] + values1[4] * values2[4] + values1[5] * values2[5] + values1[6] * values2[6]);
125  if (dq + std::numeric_limits<double>::epsilon() >= 1.0)
126  return 0.0;
127  else
128  return acos(dq);
129 }

131 void FloatingJointModel::interpolate(const double* from, const double* to, const double t, double* state) const
132 {
133  // interpolate position
134  state[0] = from[0] + (to[0] - from[0]) * t;
135  state[1] = from[1] + (to[1] - from[1]) * t;
136  state[2] = from[2] + (to[2] - from[2]) * t;
137 
138  double dq = fabs(from[3] * to[3] + from[4] * to[4] + from[5] * to[5] + from[6] * to[6]);
139  double theta = (dq + std::numeric_limits<double>::epsilon() >= 1.0) ? 0.0 : acos(dq);
140  if (theta > std::numeric_limits<double>::epsilon())
141  {
142  double d = 1.0 / sin(theta);
143  double s0 = sin((1.0 - t) * theta);
144  double s1 = sin(t * theta);
145  if (dq < 0) // Take care of long angle case see http://en.wikipedia.org/wiki/Slerp
146  s1 = -s1;
147  state[3] = (from[3] * s0 + to[3] * s1) * d;
148  state[4] = (from[4] * s0 + to[4] * s1) * d;
149  state[5] = (from[5] * s0 + to[5] * s1) * d;
150  state[6] = (from[6] * s0 + to[6] * s1) * d;
151  }
152  else
153  {
154  state[3] = from[3];
155  state[4] = from[4];
156  state[5] = from[5];
157  state[6] = from[6];
158  }
159 }

161 bool FloatingJointModel::satisfiesPositionBounds(const double* values, const Bounds& bounds, double margin) const
162 {
163  if (values[0] < bounds[0].min_position_ - margin || values[0] > bounds[0].max_position_ + margin)
164  return false;
165  if (values[1] < bounds[1].min_position_ - margin || values[1] > bounds[1].max_position_ + margin)
166  return false;
167  if (values[2] < bounds[2].min_position_ - margin || values[2] > bounds[2].max_position_ + margin)
168  return false;
169  double norm_sqr = values[3] * values[3] + values[4] * values[4] + values[5] * values[5] + values[6] * values[6];
170  return fabs(norm_sqr - 1.0) <= std::numeric_limits<float>::epsilon() * 10.0;
171 }

173 bool FloatingJointModel::normalizeRotation(double* values) const
174 {
175  // normalize the quaternion if we need to
176  double norm_sqr = values[3] * values[3] + values[4] * values[4] + values[5] * values[5] + values[6] * values[6];
177  if (fabs(norm_sqr - 1.0) > std::numeric_limits<double>::epsilon() * 100.0)
178  {
179  double norm = sqrt(norm_sqr);
180  if (norm < std::numeric_limits<double>::epsilon() * 100.0)
181  {
182  RCLCPP_WARN(LOGGER, "Quaternion is zero in RobotState representation. Setting to identity");
183  values[3] = 0.0;
184  values[4] = 0.0;
185  values[5] = 0.0;
186  values[6] = 1.0;
187  }
188  else
189  {
190  values[3] /= norm;
191  values[4] /= norm;
192  values[5] /= norm;
193  values[6] /= norm;
194  }
195  return true;
196  }
197  else
198  return false;
199 }

206 bool FloatingJointModel::enforcePositionBounds(double* values, const Bounds& bounds) const
207 {
208  bool result = normalizeRotation(values);
209  for (unsigned int i = 0; i < 3; ++i)
210  {
211  if (values[i] < bounds[i].min_position_)
212  {
213  values[i] = bounds[i].min_position_;
214  result = true;
215  }
216  else if (values[i] > bounds[i].max_position_)
217  {
218  values[i] = bounds[i].max_position_;
219  result = true;
220  }
221  }
222  return result;
223 }

232 void FloatingJointModel::computeVariablePositions(const Eigen::Isometry3d& transf, double* joint_values) const
233 {
234  joint_values[0] = transf.translation().x();
235  joint_values[1] = transf.translation().y();
236  joint_values[2] = transf.translation().z();
237  ASSERT_ISOMETRY(transf) // unsanitized input, could contain non-isometry
238  Eigen::Quaterniond q(transf.linear());
239  joint_values[3] = q.x();
240  joint_values[4] = q.y();

262 void FloatingJointModel::getVariableRandomPositions(random_numbers::RandomNumberGenerator& rng, double* values,
263  const Bounds& bounds) const
264 {

266  bounds[0].min_position_ <= -std::numeric_limits<double>::infinity())

289 void FloatingJointModel::getVariableRandomPositionsNearBy(random_numbers::RandomNumberGenerator& rng, double* values,
290  const Bounds& bounds, const double* near,
291  const double distance) const
292 {
293  if (bounds[0].max_position_ >= std::numeric_limits<double>::infinity() ||
294  bounds[0].min_position_ <= -std::numeric_limits<double>::infinity())
295  values[0] = 0.0;
296  else
297  values[0] = rng.uniformReal(std::max(bounds[0].min_position_, near[0] - distance),
298  std::min(bounds[0].max_position_, near[0] + distance));
299  if (bounds[1].max_position_ >= std::numeric_limits<double>::infinity() ||
300  bounds[1].min_position_ <= -std::numeric_limits<double>::infinity())
301  values[1] = 0.0;
302  else
303  values[1] = rng.uniformReal(std::max(bounds[1].min_position_, near[1] - distance),
304  std::min(bounds[1].max_position_, near[1] + distance));
305  if (bounds[2].max_position_ >= std::numeric_limits<double>::infinity() ||
306  bounds[2].min_position_ <= -std::numeric_limits<double>::infinity())
307  values[2] = 0.0;
308  else
309  values[2] = rng.uniformReal(std::max(bounds[2].min_position_, near[2] - distance),
310  std::min(bounds[2].max_position_, near[2] + distance));
311 
312  double da = angular_distance_weight_ * distance;
313  if (da >= .25 * boost::math::constants::pi<double>())
314  {
315  double q[4];
316  rng.quaternion(q);
317  values[3] = q[0];
318  values[4] = q[1];
319  values[5] = q[2];
320  values[6] = q[3];
321  }
322  else
323  {
324  // taken from OMPL
325  // sample angle & axis
326  double ax = rng.gaussian01();
327  double ay = rng.gaussian01();
328  double az = rng.gaussian01();
329  double angle = 2.0 * pow(rng.uniform01(), 1.0 / 3.0) * da;
330  // convert to quaternion
331  double q[4];
332  double norm = sqrt(ax * ax + ay * ay + az * az);
333  if (norm < 1e-6)
334  {
335  q[0] = q[1] = q[2] = 0.0;
336  q[3] = 1.0;
337  }
338  else
339  {
340  double s = sin(angle / 2.0);
341  q[0] = s * ax / norm;
342  q[1] = s * ay / norm;
343  q[2] = s * az / norm;
344  q[3] = cos(angle / 2.0);
345  }
346  // multiply quaternions: near * q
347  values[3] = near[6] * q[0] + near[3] * q[3] + near[4] * q[2] - near[5] * q[1];
348  values[4] = near[6] * q[1] + near[4] * q[3] + near[5] * q[0] - near[3] * q[2];
349  values[5] = near[6] * q[2] + near[5] * q[3] + near[3] * q[1] - near[4] * q[0];
350  values[6] = near[6] * q[3] - near[3] * q[0] - near[4] * q[1] - near[5] * q[2];
351  }
352 }
