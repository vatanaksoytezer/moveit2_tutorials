520 TEST_F(IntegrationTestCommandListManager, TestExecutionTime)
521 {
522  Sequence seq{ data_loader_->getSequence("ComplexSequence") };
523  ASSERT_GE(seq.size(), 2u);
524  RobotTrajCont res_single_vec{ manager_->solve(scene_, pipeline_, seq.toRequest()) };
525  EXPECT_EQ(res_single_vec.size(), 1u);
526  EXPECT_GT(res_single_vec.front()->getWayPointCount(), 0u);
527 
528  moveit_msgs::msg::MotionSequenceRequest req{ seq.toRequest() };
529  // Create large request by making copies of the original sequence commands
530  // and adding them to the end of the original sequence.
531  const size_t n{ req.items.size() };
532  for (size_t i = 0; i < n; ++i)
533  {
534  moveit_msgs::msg::MotionSequenceItem item{ req.items.at(i) };
535  if (i == 0)
536  {
537  // Remove start state because only the first request
538  // is allowed to have a start state in a sequence.
539  item.req.start_state = moveit_msgs::msg::RobotState();
540  }
541  req.items.push_back(item);
542  }
543 
544  RobotTrajCont res_n_vec{ manager_->solve(scene_, pipeline_, req) };
545  EXPECT_EQ(res_n_vec.size(), 1u);
546  EXPECT_GT(res_n_vec.front()->getWayPointCount(), 0u);
547 
548  const double trajectory_time_1 =
549  res_single_vec.front()->getWayPointDurationFromStart(res_single_vec.front()->getWayPointCount() - 1);
550  const double trajectory_time_n =
551  res_n_vec.front()->getWayPointDurationFromStart(res_n_vec.front()->getWayPointCount() - 1);
552  double multiplicator = req.items.size() / n;
553  EXPECT_LE(trajectory_time_n, trajectory_time_1 * multiplicator);
554  EXPECT_GE(trajectory_time_n, trajectory_time_1 * multiplicator * 0.5);
555 }
