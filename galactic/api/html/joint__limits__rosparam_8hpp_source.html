244 inline bool get_joint_limits(const std::string& joint_name, const rclcpp::Node::SharedPtr& node,
245  const std::string& param_ns, SoftJointLimits& soft_limits)
246 {
247  const std::string param_base_name = (param_ns.empty() ? "" : param_ns + ".") + "joint_limits." + joint_name;
248  try
249  {
250  if (!node->has_parameter(param_base_name + ".has_soft_limits") &&
251  !node->has_parameter(param_base_name + ".k_velocity") &&
252  !node->has_parameter(param_base_name + ".k_position") &&
253  !node->has_parameter(param_base_name + ".soft_lower_limit") &&
254  !node->has_parameter(param_base_name + ".soft_upper_limit"))
255  {
256  RCLCPP_DEBUG(node->get_logger(),
257  "No soft joint limits specification found for joint '%s' in the parameter server "
258  "(node: %s param name: %s).",
259  joint_name.c_str(), node->get_name(), param_base_name.c_str());
260  return false;
261  }
262  }
263  catch (const std::exception& ex)
264  {
265  RCLCPP_ERROR(node->get_logger(), "%s", ex.what());
266  return false;
267  }
268 
269  // Override soft limits if complete specification is found
270  bool has_soft_limits;
271  if (node->get_parameter(param_base_name + ".has_soft_limits", has_soft_limits))
272  {
273  if (has_soft_limits && node->has_parameter(param_base_name + ".k_position") &&
274  node->has_parameter(param_base_name + ".k_velocity") &&
275  node->has_parameter(param_base_name + ".soft_lower_limit") &&
276  node->has_parameter(param_base_name + ".soft_upper_limit"))
277  {
278  node->get_parameter(param_base_name + ".k_position", soft_limits.k_position);
279  node->get_parameter(param_base_name + ".k_velocity", soft_limits.k_velocity);
280  node->get_parameter(param_base_name + ".soft_lower_limit", soft_limits.min_position);
281  node->get_parameter(param_base_name + ".soft_upper_limit", soft_limits.max_position);
282  return true;
283  }
284  }
285 
286  return false;
287 }
