55 // check if a parent or ancestor of joint is included in this group
56 bool includesParent(const JointModel* joint, const JointModelGroup* group)
57 {
58  bool found = false;
59  // if we find that an ancestor is also in the group, then the joint is not a root
60  while (joint->getParentLinkModel() != nullptr)
61  {
62  joint = joint->getParentLinkModel()->getParentJointModel();
63  if (group->hasJointModel(joint->getName()) && joint->getVariableCount() > 0 && joint->getMimic() == nullptr)
64  {
65  found = true;
66  break;
67  }
68  else if (joint->getMimic() != nullptr)
69  {
70  const JointModel* mjoint = joint->getMimic();
71  if (group->hasJointModel(mjoint->getName()) && mjoint->getVariableCount() > 0 && mjoint->getMimic() == nullptr)
72  found = true;
73  else if (includesParent(mjoint, group))
74  found = true;
75  if (found)
76  break;
77  }
78  }
79  return found;
80 }

82 // check if joint a is right below b, in the kinematic chain, with no active DOF missing
83 bool jointPrecedes(const JointModel* a, const JointModel* b)
84 {
85  if (!a->getParentLinkModel())
86  return false;
87  const JointModel* p = a->getParentLinkModel()->getParentJointModel();
88  while (p)
89  {
90  if (p == b)
91  return true;
92  if (p->getType() == JointModel::FIXED)
93  p = p->getParentLinkModel() ? p->getParentLinkModel()->getParentJointModel() : nullptr;
94  else
95  break;
96  }
97 
98  return false;
99 }

102 JointModelGroup::JointModelGroup(const std::string& group_name, const srdf::Model::Group& config,
103  const std::vector<const JointModel*>& unsorted_group_joints,
104  const RobotModel* parent_model)
105  : parent_model_(parent_model)
106  , name_(group_name)
107  , common_root_(nullptr)
108  , variable_count_(0)
109  , active_variable_count_(0)
110  , is_contiguous_index_list_(true)
111  , is_chain_(false)
112  , is_single_dof_(true)
113  , config_(config)
114 {
115  // sort joints in Depth-First order
116  joint_model_vector_ = unsorted_group_joints;
117  std::sort(joint_model_vector_.begin(), joint_model_vector_.end(), OrderJointsByIndex());
118  joint_model_name_vector_.reserve(joint_model_vector_.size());
119 
120  // figure out active joints, mimic joints, fixed joints
121  // construct index maps, list of variables
122  for (const JointModel* joint_model : joint_model_vector_)
123  {
124  joint_model_name_vector_.push_back(joint_model->getName());
125  joint_model_map_[joint_model->getName()] = joint_model;
126  unsigned int vc = joint_model->getVariableCount();
127  if (vc > 0)
128  {
129  if (vc > 1)
130  is_single_dof_ = false;
131  const std::vector<std::string>& name_order = joint_model->getVariableNames();
132  if (joint_model->getMimic() == nullptr)
133  {
134  active_joint_model_vector_.push_back(joint_model);
135  active_joint_model_name_vector_.push_back(joint_model->getName());
136  active_joint_model_start_index_.push_back(variable_count_);
137  active_joint_models_bounds_.push_back(&joint_model->getVariableBounds());
138  active_variable_count_ += vc;
139  }
140  else
141  mimic_joints_.push_back(joint_model);
142  for (const std::string& name : name_order)
143  {
144  variable_names_.push_back(name);
145  variable_names_set_.insert(name);
146  }
147 
148  int first_index = joint_model->getFirstVariableIndex();
149  for (std::size_t j = 0; j < name_order.size(); ++j)
150  {
151  variable_index_list_.push_back(first_index + j);
152  joint_variables_index_map_[name_order[j]] = variable_count_ + j;
153  }
154  joint_variables_index_map_[joint_model->getName()] = variable_count_;
155 
156  if (joint_model->getType() == JointModel::REVOLUTE &&
157  static_cast<const RevoluteJointModel*>(joint_model)->isContinuous())
158  continuous_joint_model_vector_.push_back(joint_model);
159 
160  variable_count_ += vc;
161  }
162  else
163  fixed_joints_.push_back(joint_model);
164  }
165 
166  // now we need to find all the set of joints within this group
167  // that root distinct subtrees
168  for (const JointModel* active_joint_model : active_joint_model_vector_)
169  {
170  // if we find that an ancestor is also in the group, then the joint is not a root
171  if (!includesParent(active_joint_model, this))
172  joint_roots_.push_back(active_joint_model);
173  }
174 
175  // when updating this group within a state, it is useful to know
176  // if the full state of a group is contiguous within the full state of the robot
177  if (variable_index_list_.empty())
178  is_contiguous_index_list_ = false;
179  else
180  for (std::size_t i = 1; i < variable_index_list_.size(); ++i)
181  if (variable_index_list_[i] != variable_index_list_[i - 1] + 1)
182  {
183  is_contiguous_index_list_ = false;
184  break;
185  }
186 
187  // when updating/sampling a group state only, only mimic joints that have their parent within the group get updated.
188  for (const JointModel* mimic_joint : mimic_joints_)
189  // if the joint we mimic is also in this group, we will need to do updates when sampling
190  if (hasJointModel(mimic_joint->getMimic()->getName()))
191  {
192  int src = joint_variables_index_map_[mimic_joint->getMimic()->getName()];
193  int dest = joint_variables_index_map_[mimic_joint->getName()];
194  GroupMimicUpdate mu(src, dest, mimic_joint->getMimicFactor(), mimic_joint->getMimicOffset());
195  group_mimic_update_.push_back(mu);
196  }
197 
198  // now we need to make another pass for group links (we include the fixed joints here)
199  std::set<const LinkModel*> group_links_set;
200  for (const JointModel* joint_model : joint_model_vector_)
201  group_links_set.insert(joint_model->getChildLinkModel());
202  for (const LinkModel* group_link : group_links_set)
203  link_model_vector_.push_back(group_link);
204  std::sort(link_model_vector_.begin(), link_model_vector_.end(), OrderLinksByIndex());
205 
206  for (const LinkModel* link_model : link_model_vector_)
207  {
208  link_model_map_[link_model->getName()] = link_model;
209  link_model_name_vector_.push_back(link_model->getName());
210  if (!link_model->getShapes().empty())
211  {
212  link_model_with_geometry_vector_.push_back(link_model);
213  link_model_with_geometry_name_vector_.push_back(link_model->getName());
214  }
215  }
216 
217  // compute the common root of this group
218  if (!joint_roots_.empty())
219  {
220  common_root_ = joint_roots_[0];
221  for (std::size_t i = 1; i < joint_roots_.size(); ++i)
222  common_root_ = parent_model->getCommonRoot(joint_roots_[i], common_root_);
223  }
224 
225  // compute updated links
226  for (const JointModel* joint_root : joint_roots_)
227  {
228  const std::vector<const LinkModel*>& links = joint_root->getDescendantLinkModels();
229  updated_link_model_set_.insert(links.begin(), links.end());
230  }
231  for (const LinkModel* updated_link_model : updated_link_model_set_)
232  {
233  updated_link_model_name_set_.insert(updated_link_model->getName());
234  updated_link_model_vector_.push_back(updated_link_model);
235  if (!updated_link_model->getShapes().empty())
236  {
237  updated_link_model_with_geometry_vector_.push_back(updated_link_model);
238  updated_link_model_with_geometry_set_.insert(updated_link_model);
239  updated_link_model_with_geometry_name_set_.insert(updated_link_model->getName());
240  }
241  }
242  std::sort(updated_link_model_vector_.begin(), updated_link_model_vector_.end(), OrderLinksByIndex());
243  std::sort(updated_link_model_with_geometry_vector_.begin(), updated_link_model_with_geometry_vector_.end(),
244  OrderLinksByIndex());
245  for (const LinkModel* updated_link_model : updated_link_model_vector_)
246  updated_link_model_name_vector_.push_back(updated_link_model->getName());
247  for (const LinkModel* updated_link_model_with_geometry : updated_link_model_with_geometry_vector_)
248  updated_link_model_with_geometry_name_vector_.push_back(updated_link_model_with_geometry->getName());
249 
250  // check if this group should actually be a chain
251  if (joint_roots_.size() == 1 && !active_joint_model_vector_.empty())
252  {
253  bool chain = true;
254  // due to our sorting, the joints are sorted in a DF fashion, so looking at them in reverse,
255  // we should always get to the parent.
256  for (std::size_t k = joint_model_vector_.size() - 1; k > 0; --k)
257  if (!jointPrecedes(joint_model_vector_[k], joint_model_vector_[k - 1]))
258  {
259  chain = false;
260  break;
261  }
262  if (chain)
263  is_chain_ = true;
264  }
265 }

503 bool JointModelGroup::getEndEffectorTips(std::vector<const LinkModel*>& tips) const
504 {
505  tips.clear();
506  for (const std::string& name : getAttachedEndEffectorNames())
507  {
508  const JointModelGroup* eef = parent_model_->getEndEffector(name);
509  if (!eef)
510  {
511  RCLCPP_ERROR(LOGGER, "Unable to find joint model group for eef");
512  return false;
513  }
514  const std::string& eef_parent = eef->getEndEffectorParentGroup().second;
515 
516  const LinkModel* eef_link = parent_model_->getLinkModel(eef_parent);
517  if (!eef_link)
518  {
519  RCLCPP_ERROR(LOGGER, "Unable to find end effector link for eef");
520  return false;
521  }
522  // insert eef_link into tips, maintaining a *sorted* vector, thus enabling use of std::lower_bound
523  const auto insert_it = std::lower_bound(tips.cbegin(), tips.cend(), eef_link);
524  if (insert_it == tips.end() || eef_link != *insert_it) // only insert if not a duplicate
525  tips.insert(insert_it, eef_link);
526  }
527  return true;
528 }

570  joint_bijection.clear();
571  for (const std::string& ik_jname : ik_jnames)
572  {
573  VariableIndexMap::const_iterator it = joint_variables_index_map_.find(ik_jname);
574  if (it == joint_variables_index_map_.end())

578  continue;

581  "but group '%s' does not contain such a joint.",
582  ik_jname.c_str(), getName().c_str());
583  return false;
584  }
585  const JointModel* jm = getJointModel(ik_jname);
586  for (unsigned int k = 0; k < jm->getVariableCount(); ++k)
587  joint_bijection.push_back(it->second + k);
588  }

592 void JointModelGroup::setSolverAllocators(const std::pair<SolverAllocatorFn, SolverAllocatorMapFn>& solvers)
593 {
594  if (solvers.first)
595  {

597  group_kinematics_.first.solver_instance_ = solvers.first(this);

604  }
605  }
606  else

614  ks.default_ik_timeout_ = group_kinematics_.first.default_ik_timeout_;
615  if (!computeIKIndexBijection(ks.solver_instance_->getJointNames(), ks.bijection_))

623 bool JointModelGroup::canSetStateFromIK(const std::string& tip) const
624 {
625  const kinematics::KinematicsBaseConstPtr& solver = getSolverInstance();
626  if (!solver || tip.empty())
627  return false;
628 
629  const std::vector<std::string>& tip_frames = solver->getTipFrames();
630 
631  if (tip_frames.empty())
632  {
633  RCLCPP_WARN(LOGGER, "Group %s has no tip frame(s)", name_.c_str());
634  return false;
635  }
636 
637  // loop through all tip frames supported by the JMG
638  for (const std::string& tip_frame : tip_frames)
639  {
640  // remove frame reference, if specified
641  const std::string& tip_local = tip[0] == '/' ? tip.substr(1) : tip;
642  const std::string& tip_frame_local = tip_frame[0] == '/' ? tip_frame.substr(1) : tip_frame;
643  RCLCPP_DEBUG(LOGGER, "comparing input tip: %s to this groups tip: %s ", tip_local.c_str(), tip_frame_local.c_str());
644 
645  // Check if the IK solver's tip is the same as the frame of inquiry
646  if (tip_local != tip_frame_local)
647  {
648  // If not the same, check if this planning group includes the frame of inquiry
649  if (hasLinkModel(tip_frame_local))
650  {
651  const LinkModel* lm = getLinkModel(tip_frame_local);
652  const LinkTransformMap& fixed_links = lm->getAssociatedFixedTransforms();
653  // Check if our frame of inquiry is located anywhere further down the chain (towards the tip of the arm)
654  for (const std::pair<const LinkModel* const, Eigen::Isometry3d>& fixed_link : fixed_links)
655  {
656  if (fixed_link.first->getName() == tip_local)
657  return true;
658  }
659  }
660  }
661  else
662  return true;
663  }
664 
665  // Did not find any valid tip frame links to use
666  return false;
667 }

740 bool JointModelGroup::isValidVelocityMove(const double* from_joint_pose, const double* to_joint_pose,
741  std::size_t array_size, double dt) const
742 {
743  const std::vector<const JointModel::Bounds*>& bounds = getActiveJointModelsBounds();
744  const std::vector<unsigned int>& bij = getKinematicsSolverJointBijection();
745 
746  for (std::size_t i = 0; i < array_size; ++i)
747  {
748  double dtheta = std::abs(from_joint_pose[i] - to_joint_pose[i]);
749  const std::vector<moveit::core::VariableBounds>* var_bounds = bounds[bij[i]];
750 
751  if (var_bounds->size() != 1)
752  {
753  // TODO(davetcoleman) Support multiple variables
754  RCLCPP_ERROR(LOGGER, "Attempting to check velocity bounds for waypoint move with joints that have multiple "
755  "variables");
756  return false;
757  }
758  const double max_velocity = (*var_bounds)[0].max_velocity_;
759 
760  double max_dtheta = dt * max_velocity;
761  if (dtheta > max_dtheta)
762  {
763  RCLCPP_DEBUG(LOGGER, "Not valid velocity move because of joint %lu", i);
764  return false;
765  }
766  }
767 
768  return true;
769 }
