49 bool robot_interaction::KinematicOptions::setStateFromIK(moveit::core::RobotState& state, const std::string& group,
50  const std::string& tip,
51  const geometry_msgs::msg::Pose& pose) const
52 {
53  const moveit::core::JointModelGroup* jmg = state.getJointModelGroup(group);
54  if (!jmg)
55  {
56  RCLCPP_ERROR(LOGGER, "No getJointModelGroup('%s') found", group.c_str());
57  return false;
58  }
59  bool result = state.setFromIK(jmg, pose, tip,
60  // limit timeout to 0.1s if set from JMG's default, i.e. when timeout_seconds_ == 0
61  timeout_seconds_ > 0.0 ? timeout_seconds_ : std::min(0.1, jmg->getDefaultIKTimeout()),
62  state_validity_callback_, options_);
63  state.update();
64  return result;
65 }

67 void robot_interaction::KinematicOptions::setOptions(const KinematicOptions& source, OptionBitmask fields)
68 {
69 // This function is implemented with the O_FIELDS and QO_FIELDS macros to
70 // ensure that any fields added to robot_interaction::KinematicOptions or
71 // kinematics::KinematicsQueryOptions are also added here and to the
72 // KinematicOptions::OptionBitmask enum.
73 
74 // This needs to represent all the fields in
75 // robot_interaction::KinematicOptions except options_
76 #define O_FIELDS(F) \
77  F(double, timeout_seconds_, TIMEOUT) \
78  F(moveit::core::GroupStateValidityCallbackFn, state_validity_callback_, STATE_VALIDITY_CALLBACK)
79 
80 // This needs to represent all the fields in
81 // kinematics::KinematicsQueryOptions
82 #define QO_FIELDS(F) \
83  F(bool, lock_redundant_joints, LOCK_REDUNDANT_JOINTS) \
84  F(bool, return_approximate_solution, RETURN_APPROXIMATE_SOLUTION) \
85  F(::kinematics::DiscretizationMethods::DiscretizationMethod, discretization_method, DISCRETIZATION_METHOD)
86 
87  // This structure should be identical to kinematics::KinematicsQueryOptions
88  // This is only used in the BOOST_STATIC_ASSERT below.
89  struct DummyKinematicsQueryOptions
90  {
91 #define F(type, member, enumval) type member;
92  QO_FIELDS(F)
93 #undef F
94  };
95  // This structure should be identical to robot_interaction::KinematicOptions
96  // This is only used in the BOOST_STATIC_ASSERT below.
97  struct DummyKinematicOptions
98  {
99 #define F(type, member, enumval) type member;
100  O_FIELDS(F)
101 #undef F
102  DummyKinematicsQueryOptions options_;
103  };
104 
105  // If these asserts fails it means that fields were added to
106  // kinematics::KinematicsQueryOptions or robot_interaction::KinematicOptions
107  // and not added to the O_FIELDS and QO_FIELDS definitions above. To fix add
108  // any new fields to the definitions above.
109  BOOST_STATIC_ASSERT(sizeof(kinematics::KinematicsQueryOptions) == sizeof(DummyKinematicsQueryOptions));
110  BOOST_STATIC_ASSERT(sizeof(KinematicOptions) == sizeof(DummyKinematicOptions));
111 
112 // copy fields from other to this if its bit is set in fields
113 #define F(type, member, enumval) \
114  if (fields & KinematicOptions::enumval) \
115  member = source.member;
116  O_FIELDS(F)
117 #undef F
118 
119 // copy fields from other.options_ to this.options_ if its bit is set in
120 // fields
121 #define F(type, member, enumval) \
122  if (fields & KinematicOptions::enumval) \
123  options_.member = source.options_.member;
124  QO_FIELDS(F)
125 #undef F
126 }
