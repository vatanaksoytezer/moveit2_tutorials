52 KinematicsConstraintAware::KinematicsConstraintAware(const moveit::core::RobotModelConstPtr& kinematic_model,
53  const std::string& group_name)
54 {
55  if (!kinematic_model->hasJointModelGroup(group_name))
56  {
57  ROS_ERROR_NAMED("kinematics_constraint_aware", "The group %s does not exist", group_name.c_str());
58  joint_model_group_ = NULL;
59  return;
60  }
61  kinematic_model_ = kinematic_model;
62  group_name_ = group_name;
63  joint_model_group_ = kinematic_model_->getJointModelGroup(group_name);
64  if (joint_model_group_->getSolverInstance())
65  {
66  has_sub_groups_ = false;
67  sub_groups_names_.push_back(group_name_);
68  }
69  else
70  {
71  ROS_DEBUG_NAMED("kinematics_constraint_aware", "No kinematics solver instance defined for group %s",
72  group_name.c_str());
73  bool is_solvable_group = true;
74  if (!(joint_model_group_->getSubgroupNames().empty()))
75  {
76  const std::vector<std::string> sub_groups_names = joint_model_group_->getSubgroupNames();
77  for (std::size_t i = 0; i < sub_groups_names.size(); ++i)
78  {
79  if (!kinematic_model_->getJointModelGroup(sub_groups_names[i])->getSolverInstance())
80  {
81  is_solvable_group = false;
82  break;
83  }
84  }
85  if (is_solvable_group)
86  {
87  ROS_DEBUG_NAMED("kinematics_constraint_aware", "Group %s is a group for which we can solve IK",
88  joint_model_group_->getName().c_str());
89  sub_groups_names_ = sub_groups_names;
90  }
91  else
92  {
93  joint_model_group_ = NULL;
94  return;
95  }
96  }
97  else
98  {
99  joint_model_group_ = NULL;
100  ROS_INFO_NAMED("kinematics_constraint_aware", "No solver allocated for group %s", group_name.c_str());
101  }
102  has_sub_groups_ = true;
103  }
104  ik_attempts_ = 10;
105 }

107 bool KinematicsConstraintAware::getIK(const planning_scene::PlanningSceneConstPtr& planning_scene,
108  const kinematics_constraint_aware::KinematicsRequest& request,
109  kinematics_constraint_aware::KinematicsResponse& response) const
110 {
111  if (!joint_model_group_)
112  {
113  ROS_ERROR_NAMED("kinematics_constraint_aware", "This solver has not been constructed properly");
114  return false;
115  }
116 
117  if (!planning_scene)
118  {
119  ROS_ERROR_NAMED("kinematics_constraint_aware", "Planning scene must be allocated");
120  return false;
121  }
122 
123  if (!response.solution_)
124  {
125  response.solution_ = std::make_shared<moveit::core::RobotState>(planning_scene->getCurrentState());
126  }
127 
128  ros::WallTime start_time = ros::WallTime::now();
129  if (request.group_name_ != group_name_)
130  {
131  response.error_code_.val = response.error_code_.INVALID_GROUP_NAME;
132  return false;
133  }
134 
135  // Setup the seed and the values for all other joints in the robot
136  moveit::core::RobotState kinematic_state = *request.robot_state_;
137  std::vector<std::string> ik_link_names = request.ik_link_names_;
138 
139  // Transform request to tip frame if necessary
140  if (!request.ik_link_names_.empty())
141  {
142  for (std::size_t i = 0; i < request.pose_stamped_vector_.size(); ++i)
143  {
144  geometry_msgs::PoseStamped tmp_pose = request.pose_stamped_vector_[i];
145  // The assumption is that this new link is rigidly attached to the tip link for the group
146  if (!kinematic_model_->getJointModelGroup(sub_groups_names_[i])->hasLinkModel(request.ik_link_names_[i]) &&
147  kinematic_model_->getJointModelGroup(sub_groups_names_[i])->isLinkUpdated(request.ik_link_names_[i]))
148  {
149  tmp_pose.pose = getTipFramePose(planning_scene, kinematic_state, request.pose_stamped_vector_[i].pose,
150  request.ik_link_names_[i], i);
151  ik_link_names[i] =
152  kinematic_model_->getJointModelGroup(sub_groups_names_[i])->getSolverInstance()->getTipFrame();
153  }
154  else if (!kinematic_model_->getJointModelGroup(sub_groups_names_[i])->canSetStateFromIK(request.ik_link_names_[i]))
155  {
156  ROS_ERROR_NAMED("kinematics_constraint_aware", "Could not find IK solver for link %s for group %s",
157  request.ik_link_names_[i].c_str(), sub_groups_names_[i].c_str());
158  return false;
159  }
160  }
161  }
162 
163  // Transform the requests to the base frame of the kinematic model
164  EigenSTL::vector_Isometry3d goals =
165  transformPoses(planning_scene, kinematic_state, request.pose_stamped_vector_, kinematic_model_->getModelFrame());
166 
167  moveit::core::StateValidityCallbackFn constraint_callback_fn =
168  boost::bind(&KinematicsConstraintAware::validityCallbackFn, this, planning_scene, request, response, _1, _2);
169 
170  bool result = false;
171  if (has_sub_groups_)
172  {
173  result = kinematic_state.getJointStateGroup(group_name_)
174  ->setFromIK(goals, ik_link_names, ik_attempts_, request.timeout_.toSec(), constraint_callback_fn);
175  }
176  else
177  {
178  result =
179  ik_link_names.empty() ?
180  kinematic_state.getJointStateGroup(group_name_)
181  ->setFromIK(goals[0], ik_attempts_, request.timeout_.toSec(), constraint_callback_fn) :
182  kinematic_state.getJointStateGroup(group_name_)
183  ->setFromIK(goals[0], ik_link_names[0], ik_attempts_, request.timeout_.toSec(), constraint_callback_fn);
184  }
185 
186  if (result)
187  {
188  std::vector<double> solution_values;
189  kinematic_state.getJointStateGroup(group_name_)->getVariableValues(solution_values);
190  response.solution_->getJointStateGroup(group_name_)->setVariableValues(solution_values);
191  response.error_code_.val = response.error_code_.SUCCESS;
192  }
193 
194  if (response.error_code_.val == 0)
195  {
196  response.error_code_.val = response.error_code_.NO_IK_SOLUTION;
197  }
198  return result;
199 }

201 bool KinematicsConstraintAware::validityCallbackFn(const planning_scene::PlanningSceneConstPtr& planning_scene,
202  const kinematics_constraint_aware::KinematicsRequest& request,
203  kinematics_constraint_aware::KinematicsResponse& response,
204  moveit::core::JointStateGroup* joint_state_group,
205  const std::vector<double>& joint_group_variable_values) const
206 {
207  joint_state_group->setVariableValues(joint_group_variable_values);
208 
209  // Now check for collisions
210  if (request.check_for_collisions_)
211  {
212  collision_detection::CollisionRequest collision_request;
213  collision_detection::CollisionResult collision_result;
214  collision_request.group_name = request.group_name_;
215  planning_scene->checkCollision(collision_request, collision_result, *joint_state_group->getRobotState());
216  if (collision_result.collision)
217  {
218  ROS_DEBUG_NAMED("kinematics_constraint_aware", "IK solution is in collision");
219  response.error_code_.val = response.error_code_.GOAL_IN_COLLISION;
220  return false;
221  }
222  }
223 
224  // Now check for constraints
225  if (request.constraints_)
226  {
227  kinematic_constraints::ConstraintEvaluationResult constraint_result;
228  constraint_result =
229  request.constraints_->decide(*joint_state_group->getRobotState(), response.constraint_eval_results_);
230  if (!constraint_result.satisfied)
231  {
232  ROS_DEBUG_NAMED("kinematics_constraint_aware", "IK solution violates constraints");
233  response.error_code_.val = response.error_code_.GOAL_VIOLATES_PATH_CONSTRAINTS;
234  return false;
235  }
236  }
237 
238  // Now check for user specified constraints
239  if (request.constraint_callback_)
240  {
241  if (!request.constraint_callback_(joint_state_group, joint_group_variable_values))
242  {
243  ROS_DEBUG_NAMED("kinematics_constraint_aware", "IK solution violates user specified constraints");
244  response.error_code_.val = response.error_code_.GOAL_VIOLATES_PATH_CONSTRAINTS;
245  return false;
246  }
247  }
248 
249  return true;
250 }

252 bool KinematicsConstraintAware::getIK(const planning_scene::PlanningSceneConstPtr& planning_scene,
253  const moveit_msgs::msg::GetConstraintAwarePositionIK::Request& request,
254  moveit_msgs::msg::GetConstraintAwarePositionIK::Response& response) const
255 {
256  if (!joint_model_group_)
257  {
258  ROS_ERROR_NAMED("kinematics_constraint_aware", "This solver has not been constructed properly");
259  return false;
260  }
261 
262  if (!planning_scene)
263  {
264  ROS_ERROR_NAMED("kinematics_constraint_aware", "Planning scene must be allocated");
265  return false;
266  }
267 
268  kinematics_constraint_aware::KinematicsRequest kinematics_request;
269  kinematics_constraint_aware::KinematicsResponse kinematics_response;
270 
271  if (!convertServiceRequest(planning_scene, request, kinematics_request, kinematics_response))
272  {
273  response.error_code = kinematics_response.error_code_;
274  return false;
275  }
276 
277  bool result = getIK(planning_scene, kinematics_request, kinematics_response);
278  response.error_code = kinematics_response.error_code_;
279  kinematics_response.solution_->getStateValues(response.solution.joint_state);
280  return result;
281 }

283 bool KinematicsConstraintAware::convertServiceRequest(
284  const planning_scene::PlanningSceneConstPtr& planning_scene,
285  const moveit_msgs::msg::GetConstraintAwarePositionIK::Request& request,
286  kinematics_constraint_aware::KinematicsRequest& kinematics_request,
287  kinematics_constraint_aware::KinematicsResponse& kinematics_response) const
288 {
289  if (request.ik_request.group_name != group_name_)
290  {
291  ROS_ERROR_NAMED("kinematics_constraint_aware", "This kinematics solver does not support requests for group: %s",
292  request.ik_request.group_name.c_str());
293  kinematics_response.error_code_.val = kinematics_response.error_code_.INVALID_GROUP_NAME;
294  return false;
295  }
296 
297  if (!request.ik_request.pose_stamped_vector.empty() &&
298  request.ik_request.pose_stamped_vector.size() != sub_groups_names_.size())
299  {
300  ROS_ERROR_NAMED("kinematics_constraint_aware",
301  "Number of poses in request: %d must match number of sub groups %d in this group",
302  request.ik_request.pose_stamped_vector.size(), sub_groups_names_.size());
303  kinematics_response.error_code_.val = kinematics_response.error_code_.INVALID_GROUP_NAME;
304  return false;
305  }
306 
307  if (!request.ik_request.ik_link_names.empty() && request.ik_request.ik_link_names.size() != sub_groups_names_.size())
308  {
309  ROS_ERROR_NAMED("kinematics_constraint_aware",
310  "Number of ik_link_names in request: "
311  "%d must match number of sub groups %d in this group or must be zero",
312  request.ik_request.ik_link_names.size(), sub_groups_names_.size());
313  kinematics_response.error_code_.val = kinematics_response.error_code_.INVALID_GROUP_NAME;
314  return false;
315  }
316 
317  if (request.ik_request.ik_link_names.empty() && request.ik_request.ik_link_name != "")
318  kinematics_request.ik_link_names_.push_back(request.ik_request.ik_link_name);
319  else
320  kinematics_request.ik_link_names_ = request.ik_request.ik_link_names;
321 
322  if (request.ik_request.pose_stamped_vector.empty())
323  kinematics_request.pose_stamped_vector_.push_back(request.ik_request.pose_stamped);
324  else
325  kinematics_request.pose_stamped_vector_ = request.ik_request.pose_stamped_vector;
326 
327  kinematics_request.robot_state_ = std::make_shared<moveit::core::RobotState>(planning_scene->getCurrentState());
328  kinematics_request.robot_state_->setStateValues(request.ik_request.robot_state.joint_state);
329  kinematics_request.constraints_ = std::make_shared<kinematic_constraints::KinematicConstraintSet>(
330  kinematic_model_, planning_scene->getTransforms());
331  kinematics_request.constraints_->add(request.constraints);
332  kinematics_request.timeout_ = request.ik_request.timeout;
333  kinematics_request.group_name_ = request.ik_request.group_name;
334  kinematics_request.check_for_collisions_ = true;
335 
336  kinematics_response.solution_ = std::make_shared<moveit::core::RobotState>(planning_scene->getCurrentState());
337 
338  return true;
339 }

341 EigenSTL::vector_Isometry3d KinematicsConstraintAware::transformPoses(
342  const planning_scene::PlanningSceneConstPtr& planning_scene, const moveit::core::RobotState& kinematic_state,
343  const std::vector<geometry_msgs::PoseStamped>& poses, const std::string& target_frame) const
344 {
345  Eigen::Isometry3d eigen_pose, eigen_pose_2;
346  EigenSTL::vector_Isometry3d result(poses.size());
347  bool target_frame_is_root_frame = (target_frame == kinematic_state.getRobotModel()->getModelFrame());
348  for (std::size_t i = 0; i < poses.size(); ++i)
349  {
350  geometry_msgs::Pose pose = poses[i].pose;
351  tf2::fromMsg(pose, eigen_pose_2);
352  planning_scene->getTransforms()->transformPose(kinematic_state, poses[i].header.frame_id, eigen_pose_2, eigen_pose);
353  if (!target_frame_is_root_frame)
354  {
355  eigen_pose_2 = planning_scene->getTransforms()->getTransform(kinematic_state, target_frame);
356  eigen_pose = eigen_pose_2.inverse() * eigen_pose;
357  }
358  result[i] = eigen_pose;
359  }
360  return result;
361 }

363 geometry_msgs::Pose KinematicsConstraintAware::getTipFramePose(
364  const planning_scene::PlanningSceneConstPtr& planning_scene, const moveit::core::RobotState& kinematic_state,
365  const geometry_msgs::Pose& pose, const std::string& link_name, unsigned int sub_group_index) const
366 {
367  geometry_msgs::Pose result;
368  Eigen::Isometry3d eigen_pose_in, eigen_pose_link, eigen_pose_tip;
369  std::string tip_name =
370  kinematic_model_->getJointModelGroup(sub_groups_names_[sub_group_index])->getSolverInstance()->getTipFrame();
371  tf2::fromMsg(pose, eigen_pose_in);
372  eigen_pose_link = planning_scene->getTransforms()->getTransform(kinematic_state, link_name);
373  eigen_pose_tip = planning_scene->getTransforms()->getTransform(kinematic_state, tip_name);
374  eigen_pose_in = eigen_pose_in * (eigen_pose_link.inverse() * eigen_pose_tip);
375  result = tf2::toMsg(eigen_pose_in, result);
376  return result;
377 }
