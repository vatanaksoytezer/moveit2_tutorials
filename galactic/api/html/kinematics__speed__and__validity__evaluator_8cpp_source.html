45 int main(int argc, char** argv)
46 {

48 
49  auto node = rclcpp::Node::make_shared("inverse_kinematics_test");
50  rclcpp::executors::MultiThreadedExecutor executor;
51  executor.add_node(node);
52 
53  if (argc <= 1)
54  RCLCPP_ERROR(LOGGER, "An argument specifying the group name is needed");
55  else
56  {
57  robot_model_loader::RobotModelLoader rml(node, ROBOT_DESCRIPTION);
58  std::string group = argv[1];
59  RCLCPP_INFO(LOGGER, "Evaluating IK for %s", group.c_str());
60 
61  const moveit::core::JointModelGroup* jmg = rml.getModel()->getJointModelGroup(group);
62  if (jmg)
63  {
64  const kinematics::KinematicsBaseConstPtr& solver = jmg->getSolverInstance();
65  if (solver)
66  {
67  const std::string& tip = solver->getTipFrame();
68  moveit::core::RobotState state(rml.getModel());
69  state.setToDefaultValues();
70 
71  RCLCPP_INFO(LOGGER, "Tip Frame: %s", solver->getTipFrame().c_str());
72  RCLCPP_INFO(LOGGER, "Base Frame: %s", solver->getBaseFrame().c_str());
73  RCLCPP_INFO(LOGGER, "IK Timeout: %f", solver->getDefaultTimeout());
74  RCLCPP_INFO(LOGGER, "Search res: %f", solver->getSearchDiscretization());
75 
76  unsigned int test_count = 1000;
77  if (argc > 2)
78  try
79  {
80  test_count = boost::lexical_cast<unsigned int>(argv[2]);
81  }
82  catch (...)
83  {
84  }
85 
86  RCLCPP_INFO(LOGGER, "Running %u tests", test_count);
87 
88  moveit::tools::Profiler::Start();
89  for (unsigned int i = 0; i < test_count; ++i)
90  {
91  state.setToRandomPositions(jmg);
92  // getGlobalLinkTransform() returns a valid isometry by contract
93  Eigen::Isometry3d pose = state.getGlobalLinkTransform(tip);
94  state.setToRandomPositions(jmg);
95  moveit::tools::Profiler::Begin("IK");
96  state.setFromIK(jmg, pose);
97  moveit::tools::Profiler::End("IK");
98  // getGlobalLinkTransform() returns a valid isometry by contract
99  const Eigen::Isometry3d& pose_upd = state.getGlobalLinkTransform(tip);
100  Eigen::Isometry3d diff = pose_upd * pose.inverse(); // valid isometry
101  double rot_err = (diff.linear() - Eigen::Matrix3d::Identity()).norm();
102  double trans_err = diff.translation().norm();
103  moveit::tools::Profiler::Average("Rotation error", rot_err);
104  moveit::tools::Profiler::Average("Translation error", trans_err);
105  if (rot_err < 1e-3 && trans_err < 1e-3)
106  {
107  moveit::tools::Profiler::Event("Valid IK");
108  moveit::tools::Profiler::Average("Success Rate", 100);
109  }
110  else
111  {
112  moveit::tools::Profiler::Event("Invalid IK");
113  moveit::tools::Profiler::Average("Success Rate", 0);
114  }
115  }
116  moveit::tools::Profiler::Stop();
117  moveit::tools::Profiler::Status();
118  }
119  else
120  RCLCPP_ERROR(LOGGER, "No kinematics solver specified for group %s", group.c_str());
121  }
122  }
123  executor.spin();
124  rclcpp::shutdown();
125  return 0;
126 }
