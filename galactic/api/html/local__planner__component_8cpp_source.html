68 bool LocalPlannerComponent::initialize()
69 {
70  // Load planner parameter
71  config_.load(node_);
72 
73  // Validate config
74  if (config_.local_solution_topic_type == "std_msgs/Float64MultiArray")
75  {
76  if ((config_.publish_joint_positions && config_.publish_joint_velocities) ||
77  (!config_.publish_joint_positions && !config_.publish_joint_velocities))
78  {
79  RCLCPP_ERROR(LOGGER, "When publishing a std_msgs/Float64MultiArray, you must select positions OR velocities. "
80  "Enabling both or none is not possible!");
81  return false;
82  }
83  }
84 
85  // Configure planning scene monitor
86  planning_scene_monitor_ = std::make_shared<planning_scene_monitor::PlanningSceneMonitor>(
87  node_, "robot_description", tf_buffer_, "local_planner/planning_scene_monitor");
88  if (!planning_scene_monitor_->getPlanningScene())
89  {
90  RCLCPP_ERROR(LOGGER, "Unable to configure planning scene monitor");
91  return false;
92  }
93 
94  // Start state and scene monitors
95  RCLCPP_INFO(LOGGER, "Starting planning scene monitors");
96  planning_scene_monitor_->startSceneMonitor();
97  planning_scene_monitor_->startWorldGeometryMonitor();
98  planning_scene_monitor_->startStateMonitor();
99 
100  // Load trajectory operator plugin
101  try
102  {
103  trajectory_operator_loader_ = std::make_unique<pluginlib::ClassLoader<TrajectoryOperatorInterface>>(
104  "moveit_hybrid_planning", "moveit::hybrid_planning::TrajectoryOperatorInterface");
105  }
106  catch (pluginlib::PluginlibException& ex)
107  {
108  RCLCPP_ERROR(LOGGER, "Exception while creating trajectory operator plugin loader: '%s'", ex.what());
109  return false;
110  }
111  try
112  {
113  trajectory_operator_instance_ =
114  trajectory_operator_loader_->createUniqueInstance(config_.trajectory_operator_plugin_name);
115  if (!trajectory_operator_instance_->initialize(node_, planning_scene_monitor_->getRobotModel(),
116  config_.group_name)) // TODO(sjahr) add default group param
117  throw std::runtime_error("Unable to initialize trajectory operator plugin");
118  RCLCPP_INFO(LOGGER, "Using trajectory operator interface '%s'", config_.trajectory_operator_plugin_name.c_str());
119  }
120  catch (pluginlib::PluginlibException& ex)
121  {
122  RCLCPP_ERROR(LOGGER, "Exception while loading trajectory operator '%s': '%s'",
123  config_.trajectory_operator_plugin_name.c_str(), ex.what());
124  return false;
125  }
126 
127  // Load local constraint solver
128  try
129  {
130  local_constraint_solver_plugin_loader_ = std::make_unique<pluginlib::ClassLoader<LocalConstraintSolverInterface>>(
131  "moveit_hybrid_planning", "moveit::hybrid_planning::LocalConstraintSolverInterface");
132  }
133  catch (pluginlib::PluginlibException& ex)
134  {
135  RCLCPP_ERROR(LOGGER, "Exception while creating constraint solver plugin loader '%s'", ex.what());
136  return false;
137  }
138  try
139  {
140  local_constraint_solver_instance_ =
141  local_constraint_solver_plugin_loader_->createUniqueInstance(config_.local_constraint_solver_plugin_name);
142  if (!local_constraint_solver_instance_->initialize(node_, planning_scene_monitor_, config_.group_name))
143  throw std::runtime_error("Unable to initialize constraint solver plugin");
144  RCLCPP_INFO(LOGGER, "Using constraint solver interface '%s'", config_.local_constraint_solver_plugin_name.c_str());
145  }
146  catch (pluginlib::PluginlibException& ex)
147  {
148  RCLCPP_ERROR(LOGGER, "Exception while loading constraint solver '%s': '%s'",
149  config_.local_constraint_solver_plugin_name.c_str(), ex.what());
150  return false;
151  }
152 
153  // Initialize local planning request action server
154  using namespace std::placeholders;
155  local_planning_request_server_ = rclcpp_action::create_server<moveit_msgs::action::LocalPlanner>(
156  node_, "local_planning_action",
157  [](const rclcpp_action::GoalUUID& /*unused*/,
158  std::shared_ptr<const moveit_msgs::action::LocalPlanner::Goal> /*unused*/) {
159  RCLCPP_INFO(LOGGER, "Received local planning goal request");
160  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
161  },
162  [](const std::shared_ptr<rclcpp_action::ServerGoalHandle<moveit_msgs::action::LocalPlanner>>& /*unused*/) {
163  RCLCPP_INFO(LOGGER, "Received request to cancel local planning goal");
164  return rclcpp_action::CancelResponse::ACCEPT;
165  },
166  [this](std::shared_ptr<rclcpp_action::ServerGoalHandle<moveit_msgs::action::LocalPlanner>> goal_handle) {
167  local_planning_goal_handle_ = std::move(goal_handle);
168  // Start local planning loop when an action request is received
169  timer_ = node_->create_wall_timer(1s / config_.local_planning_frequency,
170  std::bind(&LocalPlannerComponent::executeIteration, this));
171  });
172 
173  // Initialize global trajectory listener
174  global_solution_subscriber_ = node_->create_subscription<moveit_msgs::msg::MotionPlanResponse>(
175  config_.global_solution_topic, 1, [this](const moveit_msgs::msg::MotionPlanResponse::SharedPtr msg) {
176  // Add received trajectory to internal reference trajectory
177  robot_trajectory::RobotTrajectory new_trajectory(planning_scene_monitor_->getRobotModel(), msg->group_name);
178  moveit::core::RobotState start_state(planning_scene_monitor_->getRobotModel());
179  moveit::core::robotStateMsgToRobotState(msg->trajectory_start, start_state);
180  new_trajectory.setRobotTrajectoryMsg(start_state, msg->trajectory);
181  *local_planner_feedback_ = trajectory_operator_instance_->addTrajectorySegment(new_trajectory);
182 
183  // Feedback is only send when the hybrid planning architecture should react to a discrete event that occurred
184  // when the reference trajectory is updated
185  if (!local_planner_feedback_->feedback.empty())
186  {
187  local_planning_goal_handle_->publish_feedback(local_planner_feedback_);
188  }
189 
190  // Update local planner state
191  state_ = LocalPlannerState::LOCAL_PLANNING_ACTIVE;
192  });
193 
194  // Initialize local solution publisher
195  RCLCPP_INFO(LOGGER, "Using '%s' as local solution topic type", config_.local_solution_topic_type.c_str());
196  if (config_.local_solution_topic_type == "trajectory_msgs/JointTrajectory")
197  {
198  local_trajectory_publisher_ =
199  node_->create_publisher<trajectory_msgs::msg::JointTrajectory>(config_.local_solution_topic, 1);
200  }
201  else if (config_.local_solution_topic_type == "std_msgs/Float64MultiArray")
202  {
203  local_solution_publisher_ =
204  node_->create_publisher<std_msgs::msg::Float64MultiArray>(config_.local_solution_topic, 1);
205  }
206  else if (config_.local_solution_topic_type == "CUSTOM")
207  {
208  // Local solution publisher is defined by the local constraint solver plugin
209  }
210 
211  state_ = LocalPlannerState::AWAIT_GLOBAL_TRAJECTORY;
212  return true;
213 }

215 void LocalPlannerComponent::executeIteration()
216 {
217  auto result = std::make_shared<moveit_msgs::action::LocalPlanner::Result>();
218 
219  // Do different things depending on the planner's internal state
220  switch (state_)
221  {
222  // Wait for global solution to be published
223  case LocalPlannerState::AWAIT_GLOBAL_TRAJECTORY:
224  // Do nothing
225  break;
226  // Notify action client that local planning failed
227  case LocalPlannerState::ABORT:
228  {
229  result->error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
230  result->error_message = "Local planner is in an aborted state. Resetting.";
231  local_planning_goal_handle_->abort(result);
232  reset();
233  break;
234  }
235  // If the planner received an action request and a global solution it starts to plan locally
236  case LocalPlannerState::LOCAL_PLANNING_ACTIVE:
237  {
238  // Read current planning scene
239  planning_scene_monitor_->updateSceneWithCurrentState();
240  planning_scene_monitor_->lockSceneRead(); // LOCK planning scene
241  planning_scene::PlanningScenePtr planning_scene = planning_scene_monitor_->getPlanningScene();
242  planning_scene_monitor_->unlockSceneRead(); // UNLOCK planning scene
243 
244  // Get current state
245  auto current_robot_state = planning_scene->getCurrentStateNonConst();
246 
247  // Check if the global goal is reached
248  if (trajectory_operator_instance_->getTrajectoryProgress(current_robot_state) > PROGRESS_THRESHOLD)
249  {
250  local_planning_goal_handle_->succeed(result);
251  reset();
252  break;
253  }
254 
255  // Get local goal trajectory to follow
256  robot_trajectory::RobotTrajectory local_trajectory =
257  robot_trajectory::RobotTrajectory(planning_scene_monitor_->getRobotModel(), config_.group_name);
258  *local_planner_feedback_ =
259  trajectory_operator_instance_->getLocalTrajectory(current_robot_state, local_trajectory);
260 
261  // Feedback is only sent when the hybrid planning architecture should react to a discrete event that occurred
262  // during the identification of the local planning problem
263  if (!local_planner_feedback_->feedback.empty())
264  {
265  local_planning_goal_handle_->publish_feedback(local_planner_feedback_);
266  }
267 
268  // Solve local planning problem
269  trajectory_msgs::msg::JointTrajectory local_solution;
270 
271  // Feedback is only send when the hybrid planning architecture should react to a discrete event that occurred
272  // while computing a local solution
273  *local_planner_feedback_ = local_constraint_solver_instance_->solve(
274  local_trajectory, local_planning_goal_handle_->get_goal(), local_solution);
275 
276  // Feedback is only send when the hybrid planning architecture should react to a discrete event
277  if (!local_planner_feedback_->feedback.empty())
278  {
279  local_planning_goal_handle_->publish_feedback(local_planner_feedback_);
280  }
281 
282  // Use a configurable message interface like MoveIt servo
283  // (See https://github.com/ros-planning/moveit2/blob/main/moveit_ros/moveit_servo/src/servo_calcs.cpp)
284  // Format outgoing msg in the right format
285  // (trajectory_msgs/JointTrajectory or joint positions/velocities in form of std_msgs/Float64MultiArray).
286  if (config_.local_solution_topic_type == "trajectory_msgs/JointTrajectory")
287  {
288  local_trajectory_publisher_->publish(local_solution);
289  }
290  else if (config_.local_solution_topic_type == "std_msgs/Float64MultiArray")
291  {
292  // Transform "trajectory_msgs/JointTrajectory" to "std_msgs/Float64MultiArray"
293  auto joints = std::make_unique<std_msgs::msg::Float64MultiArray>();
294  if (!local_solution.points.empty())
295  {
296  if (config_.publish_joint_positions)
297  {
298  joints->data = local_solution.points[0].positions;
299  }
300  else if (config_.publish_joint_velocities)
301  {
302  joints->data = local_solution.points[0].velocities;
303  }
304  }
305  local_solution_publisher_->publish(std::move(joints));
306  }
307  else if (config_.local_solution_topic_type == "CUSTOM")
308  {
309  // Local solution publisher is defined by the local constraint solver plugin
310  }
311  break;
312  }
313  default:
314  {
315  result->error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
316  result->error_message = "Unexpected failure.";
317  local_planning_goal_handle_->abort(result);
318  RCLCPP_ERROR(LOGGER, "Local planner somehow failed"); // TODO(sjahr) Add more detailed failure information
319  reset();
320  break;
321  }
322  }

325 void LocalPlannerComponent::reset()
326 {
327  local_constraint_solver_instance_->reset();
328  trajectory_operator_instance_->reset();
329  timer_->cancel();
