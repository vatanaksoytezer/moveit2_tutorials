82  sensor_parameters_->getNearClippingPlaneDistance(),

85  mesh_renderer_->setShadersFromString(render_vertex_shader, render_fragment_shader);
86  depth_filter_->setShadersFromString(filter_vertex_shader, filter_fragment_shader);

88  depth_filter_->begin();

90  glGenTextures(1, &sensor_depth_texture_);
91 
92  glUniform1i(glGetUniformLocation(depth_filter_->getProgramID(), "sensor"), 0);

94  glUniform1i(glGetUniformLocation(depth_filter_->getProgramID(), "label"), 4);
95 
96  shadow_threshold_location_ = glGetUniformLocation(depth_filter_->getProgramID(), "shadow_threshold");
97 
98  depth_filter_->end();
99 
100  canvas_ = glGenLists(1);
101  glNewList(canvas_, GL_COMPILE);
102  glBegin(GL_QUADS);

104  glColor3f(1, 1, 1);
105  glTexCoord2f(0, 0);

170 mesh_filter::MeshHandle mesh_filter::MeshFilterBase::addMesh(const shapes::Mesh& mesh)
171 {
172  std::unique_lock<std::mutex> _(meshes_mutex_);
173 
174  JobPtr job(new FilterJob<void>(std::bind(&MeshFilterBase::addMeshHelper, this, next_handle_, &mesh)));
175  addJob(job);
176  job->wait();
177  mesh_filter::MeshHandle ret = next_handle_;
178  const std::size_t sz = min_handle_ + meshes_.size() + 1;
179  for (std::size_t i = min_handle_; i < sz; ++i)
180  if (meshes_.find(i) == meshes_.end())
181  {
182  next_handle_ = i;
183  break;
184  }
185  min_handle_ = next_handle_;
186  return ret;
187 }

262 void mesh_filter::MeshFilterBase::run(const std::string& render_vertex_shader,
263  const std::string& render_fragment_shader,
264  const std::string& filter_vertex_shader,
265  const std::string& filter_fragment_shader)
266 {
267  initialize(render_vertex_shader, render_fragment_shader, filter_vertex_shader, filter_fragment_shader);
268 
269  while (!stop_)
270  {
271  std::unique_lock<std::mutex> lock(jobs_mutex_);
272  // check if we have new sensor data to be processed. If not, wait until we get notified.
273  if (jobs_queue_.empty())
274  jobs_condition_.wait(lock);
275 
276  if (!jobs_queue_.empty())
277  {
278  JobPtr job = jobs_queue_.front();
279  jobs_queue_.pop();
280  lock.unlock();
281  job->execute();
282  lock.lock();
283  }
284  }
285  deInitialize();
286 }

303 void mesh_filter::MeshFilterBase::doFilter(const void* sensor_data, const int encoding) const
304 {
305  std::unique_lock<std::mutex> _(transform_callback_mutex_);
306 
307  mesh_renderer_->begin();
308  sensor_parameters_->setRenderParameters(*mesh_renderer_);
309 
310  glEnable(GL_TEXTURE_2D);
311  glEnable(GL_DEPTH_TEST);
312  glDepthFunc(GL_LESS);
313  glEnable(GL_CULL_FACE);
314  glCullFace(GL_FRONT);
315  glDisable(GL_ALPHA_TEST);
316  glDisable(GL_BLEND);
317 
318  GLuint padding_coefficients_id = glGetUniformLocation(mesh_renderer_->getProgramID(), "padding_coefficients");
319  Eigen::Vector3f padding_coefficients =
320  sensor_parameters_->getPaddingCoefficients() * padding_scale_ + Eigen::Vector3f(0, 0, padding_offset_);
321  glUniform3f(padding_coefficients_id, padding_coefficients[0], padding_coefficients[1], padding_coefficients[2]);
322 
323  Eigen::Isometry3d transform;
324  for (const std::pair<const MeshHandle, GLMeshPtr>& mesh : meshes_)
325  if (transform_callback_(mesh.first, transform))
326  mesh.second->render(transform);
327 
328  mesh_renderer_->end();
329 
330  // now filter the depth_map with the second rendering stage
331  // depth_filter_.setBufferSize (width, height);
332  // depth_filter_.setCameraParameters (fx, fy, cx, cy);
333  depth_filter_->begin();
334  sensor_parameters_->setFilterParameters(*depth_filter_);
335  glEnable(GL_TEXTURE_2D);
336  glEnable(GL_DEPTH_TEST);
337  glDepthFunc(GL_ALWAYS);
338  glDisable(GL_CULL_FACE);
339  glDisable(GL_ALPHA_TEST);
340  glDisable(GL_BLEND);
341 
342  // glUniform1f (near_location_, depth_filter_.getNearClippingDistance ());
343  // glUniform1f (far_location_, depth_filter_.getFarClippingDistance ());
344  glUniform1f(shadow_threshold_location_, shadow_threshold_);
345 
346  GLuint depth_texture = mesh_renderer_->getDepthTexture();
347  GLuint color_texture = mesh_renderer_->getColorTexture();
348 
349  // bind sensor depth
350  glActiveTexture(GL_TEXTURE0);
351  glBindTexture(GL_TEXTURE_2D, sensor_depth_texture_);
352 
353  float scale =
354  1.0 / (sensor_parameters_->getFarClippingPlaneDistance() - sensor_parameters_->getNearClippingPlaneDistance());
355 
356  if (encoding == GL_UNSIGNED_SHORT)
357  // unsigned shorts shorts will be mapped to the range 0-1 during transfer. Afterwards we can apply another scale +
358  // offset to
359  // map the values between near and far clipping plane to 0 - 1. -> scale = (65535 * depth - near ) / (far - near)
360  // we have: [0 - 65535] -> [0 - 1]
361  // we want: [near - far] -> [0 - 1]
362  glPixelTransferf(GL_DEPTH_SCALE, scale * 65.535);
363  else
364  glPixelTransferf(GL_DEPTH_SCALE, scale);
365  glPixelTransferf(GL_DEPTH_BIAS, -scale * sensor_parameters_->getNearClippingPlaneDistance());
366 
367  glTexImage2D(GL_TEXTURE_2D, 0, GL_DEPTH_COMPONENT, sensor_parameters_->getWidth(), sensor_parameters_->getHeight(), 0,
368  GL_DEPTH_COMPONENT, encoding, sensor_data);
369  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
370  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
371  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
372  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
373 
374  // bind depth map
375  glActiveTexture(GL_TEXTURE2);
376  glBindTexture(GL_TEXTURE_2D, depth_texture);
377 
378  // bind labels
379  glActiveTexture(GL_TEXTURE4);
380  glBindTexture(GL_TEXTURE_2D, color_texture);
381  glCallList(canvas_);
382  depth_filter_->end();
383 }
