264 void MotionPlanningFrame::allowExternalProgramCommunication(bool enable)
265 {
266  // This is needed to prevent UI event (resuming the options) triggered
267  // before getRobotInteraction() is loaded and ready
268  if (first_time_)
269  return;
270 
271  planning_display_->getRobotInteraction()->toggleMoveInteractiveMarkerTopic(enable);
272  planning_display_->toggleSelectPlanningGroupSubscription(enable);
273  if (enable)
274  {
275  using std::placeholders::_1;
276  plan_subscriber_ = node_->create_subscription<std_msgs::msg::Empty>(
277  "/rviz/moveit/plan", 1, std::bind(&MotionPlanningFrame::remotePlanCallback, this, _1));
278  execute_subscriber_ = node_->create_subscription<std_msgs::msg::Empty>(
279  "/rviz/moveit/execute", 1, std::bind(&MotionPlanningFrame::remoteExecuteCallback, this, _1));
280  stop_subscriber_ = node_->create_subscription<std_msgs::msg::Empty>(
281  "/rviz/moveit/stop", 1, std::bind(&MotionPlanningFrame::remoteStopCallback, this, _1));
282  update_start_state_subscriber_ = node_->create_subscription<std_msgs::msg::Empty>(
283  "/rviz/moveit/update_start_state", 1, std::bind(&MotionPlanningFrame::remoteUpdateStartStateCallback, this, _1));
284  update_goal_state_subscriber_ = node_->create_subscription<std_msgs::msg::Empty>(
285  "/rviz/moveit/update_goal_state", 1, std::bind(&MotionPlanningFrame::remoteUpdateGoalStateCallback, this, _1));
286  update_custom_start_state_subscriber_ = node_->create_subscription<moveit_msgs::msg::RobotState>(
287  "/rviz/moveit/update_custom_start_state", 1,
288  std::bind(&MotionPlanningFrame::remoteUpdateCustomStartStateCallback, this, _1));
289  update_custom_goal_state_subscriber_ = node_->create_subscription<moveit_msgs::msg::RobotState>(
290  "/rviz/moveit/update_custom_goal_state", 1,
291  std::bind(&MotionPlanningFrame::remoteUpdateCustomGoalStateCallback, this, _1));
292  }
293  else
294  { // disable
295  plan_subscriber_.reset();
296  execute_subscriber_.reset();
297  stop_subscriber_.reset();
298  update_start_state_subscriber_.reset();
299  update_goal_state_subscriber_.reset();
300  update_custom_start_state_subscriber_.reset();
301  update_custom_goal_state_subscriber_.reset();
302  }

315 void MotionPlanningFrame::fillStateSelectionOptions()
316 {
317  const QSignalBlocker start_state_blocker(ui_->start_state_combo_box);
318  const QSignalBlocker goal_state_blocker(ui_->goal_state_combo_box);
319  ui_->start_state_combo_box->clear();
320  ui_->goal_state_combo_box->clear();
321 
322  if (!planning_display_->getPlanningSceneMonitor())
323  return;
324 
325  const moveit::core::RobotModelConstPtr& robot_model = planning_display_->getRobotModel();
326  std::string group = planning_display_->getCurrentPlanningGroup();
327  if (group.empty())
328  return;
329  const moveit::core::JointModelGroup* jmg = robot_model->getJointModelGroup(group);
330  if (jmg)
331  {
332  ui_->start_state_combo_box->addItem(QString("<random valid>"));
333  ui_->start_state_combo_box->addItem(QString("<random>"));
334  ui_->start_state_combo_box->addItem(QString("<current>"));
335  ui_->start_state_combo_box->addItem(QString("<same as goal>"));
336  ui_->start_state_combo_box->addItem(QString("<previous>"));
337 
338  ui_->goal_state_combo_box->addItem(QString("<random valid>"));
339  ui_->goal_state_combo_box->addItem(QString("<random>"));
340  ui_->goal_state_combo_box->addItem(QString("<current>"));
341  ui_->goal_state_combo_box->addItem(QString("<same as start>"));
342  ui_->goal_state_combo_box->addItem(QString("<previous>"));
343 
344  const std::vector<std::string>& known_states = jmg->getDefaultStateNames();
345  if (!known_states.empty())
346  {
347  ui_->start_state_combo_box->insertSeparator(ui_->start_state_combo_box->count());
348  ui_->goal_state_combo_box->insertSeparator(ui_->goal_state_combo_box->count());
349  for (const std::string& known_state : known_states)
350  {
351  ui_->start_state_combo_box->addItem(QString::fromStdString(known_state));
352  ui_->goal_state_combo_box->addItem(QString::fromStdString(known_state));
353  }
354  }
355 
356  ui_->start_state_combo_box->setCurrentIndex(2); // default to 'current'
357  ui_->goal_state_combo_box->setCurrentIndex(2); // default to 'current'
358  }

361 void MotionPlanningFrame::changePlanningGroupHelper()
362 {
363  if (!planning_display_->getPlanningSceneMonitor())
364  return;
365 
366  planning_display_->addMainLoopJob(boost::bind(&MotionPlanningFrame::fillStateSelectionOptions, this));
367  planning_display_->addMainLoopJob(
368  boost::bind(&MotionPlanningFrame::populateConstraintsList, this, std::vector<std::string>()));
369 
370  const moveit::core::RobotModelConstPtr& robot_model = planning_display_->getRobotModel();
371  std::string group = planning_display_->getCurrentPlanningGroup();
372  planning_display_->addMainLoopJob(
373  boost::bind(&MotionPlanningParamWidget::setGroupName, ui_->planner_param_treeview, group));
374  planning_display_->addMainLoopJob(
375  [=]() { ui_->planning_group_combo_box->setCurrentText(QString::fromStdString(group)); });
376 
377  if (!group.empty() && robot_model)
378  {
379  RCLCPP_INFO(LOGGER, "group %s", group.c_str());
380  if (move_group_ && move_group_->getName() == group)
381  return;
382  RCLCPP_INFO(LOGGER, "Constructing new MoveGroup connection for group '%s' in namespace '%s'", group.c_str(),
383  planning_display_->getMoveGroupNS().c_str());
384  moveit::planning_interface::MoveGroupInterface::Options opt(
385  group, moveit::planning_interface::MoveGroupInterface::ROBOT_DESCRIPTION, planning_display_->getMoveGroupNS());
386  opt.robot_model_ = robot_model;
387  opt.robot_description_.clear();
388  try
389  {
390 #ifdef RVIZ_TF1
391  std::shared_ptr<tf2_ros::Buffer> tf_buffer = moveit::planning_interface::getSharedTF();
392 #else
393  //@note: tf2 no longer accessible?
394  // /std::shared_ptr<tf2_ros::Buffer> tf_buffer = context_->getFrameManager()->getTF2BufferPtr();
395  std::shared_ptr<tf2_ros::Buffer> tf_buffer = moveit::planning_interface::getSharedTF();
396 #endif
397  move_group_ = std::make_shared<moveit::planning_interface::MoveGroupInterface>(
398  node_, opt, tf_buffer, rclcpp::Duration::from_seconds(30));
399  if (planning_scene_storage_)
400  move_group_->setConstraintsDatabase(ui_->database_host->text().toStdString(), ui_->database_port->value());
401  }
402  catch (std::exception& ex)
403  {
404  RCLCPP_ERROR(LOGGER, "%s", ex.what());
405  }
406  planning_display_->addMainLoopJob(
407  boost::bind(&MotionPlanningParamWidget::setMoveGroup, ui_->planner_param_treeview, move_group_));
408  if (move_group_)
409  {
410  move_group_->allowLooking(ui_->allow_looking->isChecked());
411  move_group_->allowReplanning(ui_->allow_replanning->isChecked());
412  bool has_unique_endeffector = !move_group_->getEndEffectorLink().empty();
413  planning_display_->addMainLoopJob([=]() { ui_->use_cartesian_path->setEnabled(has_unique_endeffector); });
414  std::vector<moveit_msgs::msg::PlannerInterfaceDescription> desc;
415  if (move_group_->getInterfaceDescriptions(desc))
416  planning_display_->addMainLoopJob(boost::bind(&MotionPlanningFrame::populatePlannersList, this, desc));
417  planning_display_->addBackgroundJob(boost::bind(&MotionPlanningFrame::populateConstraintsList, this),
418  "populateConstraintsList");
419 
420  if (first_time_)
421  {
422  first_time_ = false;
423  const planning_scene_monitor::LockedPlanningSceneRO& ps = planning_display_->getPlanningSceneRO();
424  if (ps)
425  {
426  planning_display_->setQueryStartState(ps->getCurrentState());
427  planning_display_->setQueryGoalState(ps->getCurrentState());
428  }
429  // This ensures saved UI settings applied after planning_display_ is ready
430  planning_display_->useApproximateIK(ui_->approximate_ik->isChecked());
431  if (ui_->allow_external_program->isChecked())
432  planning_display_->addMainLoopJob(
433  boost::bind(&MotionPlanningFrame::allowExternalProgramCommunication, this, true));
434  }
435  }
436  }

454 void MotionPlanningFrame::addSceneObject()
455 {
456  static const double MIN_VAL = 1e-6;
457 
458  if (!planning_display_->getPlanningSceneMonitor())
459  {
460  return;
461  }
462 
463  // get size values
464  double x_length = ui_->shape_size_x_spin_box->isEnabled() ? ui_->shape_size_x_spin_box->value() : MIN_VAL;
465  double y_length = ui_->shape_size_y_spin_box->isEnabled() ? ui_->shape_size_y_spin_box->value() : MIN_VAL;
466  double z_length = ui_->shape_size_z_spin_box->isEnabled() ? ui_->shape_size_z_spin_box->value() : MIN_VAL;
467  if (x_length < MIN_VAL || y_length < MIN_VAL || z_length < MIN_VAL)
468  {
469  QMessageBox::warning(this, QString("Dimension is too small"), QString("Size values need to be >= %1").arg(MIN_VAL));
470  return;
471  }
472 
473  // by default, name object by shape type
474  std::string selected_shape = ui_->shapes_combo_box->currentText().toStdString();
475  shapes::ShapeConstPtr shape;
476  switch (ui_->shapes_combo_box->currentData().toInt()) // fetch shape ID from current combobox item
477  {
478  case shapes::BOX:
479  shape = std::make_shared<shapes::Box>(x_length, y_length, z_length);
480  break;
481  case shapes::SPHERE:
482  shape = std::make_shared<shapes::Sphere>(0.5 * x_length);
483  break;
484  case shapes::CONE:
485  shape = std::make_shared<shapes::Cone>(0.5 * x_length, z_length);
486  break;
487  case shapes::CYLINDER:
488  shape = std::make_shared<shapes::Cylinder>(0.5 * x_length, z_length);
489  break;
490  case shapes::MESH:
491  {
492  QUrl url;
493  if (ui_->shapes_combo_box->currentText().contains("file")) // open from file
494  url = QFileDialog::getOpenFileUrl(this, tr("Import Object Mesh"), QString(),
495  "CAD files (*.stl *.obj *.dae);;All files (*.*)");
496  else // open from URL
497  url = QInputDialog::getText(this, tr("Import Object Mesh"), tr("URL for file to import from:"),
498  QLineEdit::Normal, QString("http://"));
499  if (!url.isEmpty())
500  shape = loadMeshResource(url.toString().toStdString());
501  if (!shape)
502  return;
503  // name mesh objects by their file name
504  selected_shape = url.fileName().toStdString();
505  break;
506  }
507  default:
508  QMessageBox::warning(this, QString("Unsupported shape"),
509  QString("The '%1' is not supported.").arg(ui_->shapes_combo_box->currentText()));
510  }
511 
512  // find available (initial) name of object
513  int idx = 0;
514  std::string shape_name = selected_shape + "_" + std::to_string(idx);
515  while (planning_display_->getPlanningSceneRO()->getWorld()->hasObject(shape_name))
516  {
517  idx++;
518  shape_name = selected_shape + "_" + std::to_string(idx);
519  }
520 
521  // Actually add object to the plugin's PlanningScene
522  {
523  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
524  ps->getWorldNonConst()->addToObject(shape_name, shape, Eigen::Isometry3d::Identity());
525  }
526  setLocalSceneEdited();
527 
528  planning_display_->addMainLoopJob(boost::bind(&MotionPlanningFrame::populateCollisionObjectsList, this));
529 
530  // Automatically select the inserted object so that its IM is displayed
531  planning_display_->addMainLoopJob(
532  boost::bind(&MotionPlanningFrame::setItemSelectionInList, this, shape_name, true, ui_->collision_objects_list));
533 
534  planning_display_->queueRenderSceneGeometry();
535 }

537 shapes::ShapePtr MotionPlanningFrame::loadMeshResource(const std::string& url)
538 {
539  shapes::Mesh* mesh = shapes::createMeshFromResource(url);
540  if (mesh)
541  {
542  // If the object is very large, ask the user if the scale should be reduced.
543  bool object_is_very_large = false;
544  for (unsigned int i = 0; i < mesh->vertex_count; ++i)
545  {
546  if ((abs(mesh->vertices[i * 3 + 0]) > LARGE_MESH_THRESHOLD) ||
547  (abs(mesh->vertices[i * 3 + 1]) > LARGE_MESH_THRESHOLD) ||
548  (abs(mesh->vertices[i * 3 + 2]) > LARGE_MESH_THRESHOLD))
549  {
550  object_is_very_large = true;
551  break;
552  }
553  }
554  if (object_is_very_large)
555  {
556  QMessageBox msg_box;
557  msg_box.setText(
558  "The object is very large (greater than 10 m). The file may be in millimeters instead of meters.");
559  msg_box.setInformativeText("Attempt to fix the size by shrinking the object?");
560  msg_box.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
561  msg_box.setDefaultButton(QMessageBox::Yes);
562  if (msg_box.exec() == QMessageBox::Yes)
563  {
564  for (unsigned int i = 0; i < mesh->vertex_count; ++i)
565  {
566  unsigned int i3 = i * 3;
567  mesh->vertices[i3] *= 0.001;
568  mesh->vertices[i3 + 1] *= 0.001;
569  mesh->vertices[i3 + 2] *= 0.001;
570  }
571  }
572  }
573 
574  return shapes::ShapePtr(mesh);
575  }
576  else
577  {
578  QMessageBox::warning(this, QString("Import error"), QString("Unable to import object"));
579  return shapes::ShapePtr();
580  }
581 }
