114 void MotionPlanningFrame::computeDatabaseConnectButtonClicked()
115 {
116  RCLCPP_INFO(LOGGER, "Connect to database: {host: %s, port: %d}", ui_->database_host->text().toStdString().c_str(),
117  ui_->database_port->value());
118  if (planning_scene_storage_)
119  {
120  planning_scene_storage_.reset();
121  robot_state_storage_.reset();
122  constraints_storage_.reset();
123  planning_display_->addMainLoopJob(
124  boost::bind(&MotionPlanningFrame::computeDatabaseConnectButtonClickedHelper, this, 1));
125  }
126  else
127  {
128  planning_display_->addMainLoopJob(
129  boost::bind(&MotionPlanningFrame::computeDatabaseConnectButtonClickedHelper, this, 2));
130  try
131  {
132  warehouse_ros::DatabaseConnection::Ptr conn = moveit_warehouse::loadDatabase(node_);
133  conn->setParams(ui_->database_host->text().toStdString(), ui_->database_port->value(), 5.0);
134  if (conn->connect())
135  {
136  planning_scene_storage_ = std::make_shared<moveit_warehouse::PlanningSceneStorage>(conn);
137  robot_state_storage_ = std::make_shared<moveit_warehouse::RobotStateStorage>(conn);
138  constraints_storage_ = std::make_shared<moveit_warehouse::ConstraintsStorage>(conn);
139  }
140  else
141  {
142  planning_display_->addMainLoopJob(
143  boost::bind(&MotionPlanningFrame::computeDatabaseConnectButtonClickedHelper, this, 3));
144  return;
145  }
146  }
147  catch (std::exception& ex)
148  {
149  planning_display_->addMainLoopJob(
150  boost::bind(&MotionPlanningFrame::computeDatabaseConnectButtonClickedHelper, this, 3));
151  RCLCPP_ERROR(LOGGER, "%s", ex.what());
152  return;
153  }
154  planning_display_->addMainLoopJob(
155  boost::bind(&MotionPlanningFrame::computeDatabaseConnectButtonClickedHelper, this, 4));
156  }
157 }
