411  bool setJointValueTarget(const geometry_msgs::msg::Pose& eef_pose, const std::string& end_effector_link,
412  const std::string& frame, bool approx)
413  {
414  const std::string& eef = end_effector_link.empty() ? getEndEffectorLink() : end_effector_link;
415  // this only works if we have an end-effector
416  if (!eef.empty())
417  {
418  // first we set the goal to be the same as the start state
419  moveit::core::RobotStatePtr c = getStartState();
420  if (c)
421  {
422  setTargetType(JOINT);
423  c->enforceBounds();
424  getTargetRobotState() = *c;
425  if (!getTargetRobotState().satisfiesBounds(getGoalJointTolerance()))
426  return false;
427  }
428  else
429  return false;
430 
431  // we may need to do approximate IK
432  kinematics::KinematicsQueryOptions o;
433  o.return_approximate_solution = approx;
434 
435  // if no frame transforms are needed, call IK directly
436  if (frame.empty() || moveit::core::Transforms::sameFrame(frame, getRobotModel()->getModelFrame()))
437  return getTargetRobotState().setFromIK(getJointModelGroup(), eef_pose, eef, 0.0,
438  moveit::core::GroupStateValidityCallbackFn(), o);
439  else
440  {
441  // transform the pose into the model frame, then do IK
442  bool frame_found;
443  const Eigen::Isometry3d& t = getTargetRobotState().getFrameTransform(frame, &frame_found);
444  if (frame_found)
445  {
446  Eigen::Isometry3d p;
447  tf2::fromMsg(eef_pose, p);
448  return getTargetRobotState().setFromIK(getJointModelGroup(), t * p, eef, 0.0,
449  moveit::core::GroupStateValidityCallbackFn(), o);
450  }
451  else
452  {
453  RCLCPP_ERROR(LOGGER, "Unable to transform from frame '%s' to frame '%s'", frame.c_str(),
454  getRobotModel()->getModelFrame().c_str());
455  return false;
456  }
457  }
458  }
459  else
460  return false;
461  }
