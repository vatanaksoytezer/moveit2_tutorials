171 template <typename T>
172 class IkSolution : public IkSolutionBase<T>
173 {
174 public:
175  IkSolution(const std::vector<IkSingleDOFSolutionBase<T> >& vinfos, const std::vector<int>& vfree)
176  {
177  _vbasesol = vinfos;
178  _vfree = vfree;
179  }
180 
181  virtual void GetSolution(T* solution, const T* freevalues) const
182  {
183  for (std::size_t i = 0; i < _vbasesol.size(); ++i)
184  {
185  if (_vbasesol[i].freeind < 0)
186  solution[i] = _vbasesol[i].foffset;
187  else
188  {
189  solution[i] = freevalues[_vbasesol[i].freeind] * _vbasesol[i].fmul + _vbasesol[i].foffset;
190  if (solution[i] > T(3.14159265358979))
191  {
192  solution[i] -= T(6.28318530717959);
193  }
194  else if (solution[i] < T(-3.14159265358979))
195  {
196  solution[i] += T(6.28318530717959);
197  }
198  }
199  }
200  }
201 
202  virtual void GetSolution(std::vector<T>& solution, const std::vector<T>& freevalues) const
203  {
204  solution.resize(GetDOF());
205  GetSolution(&solution.at(0), freevalues.size() > 0 ? &freevalues.at(0) : NULL);
206  }
207 
208  virtual const std::vector<int>& GetFree() const
209  {
210  return _vfree;
211  }
212  virtual int GetDOF() const
213  {
214  return static_cast<int>(_vbasesol.size());
215  }
216 
217  virtual void Validate() const
218  {
219  for (size_t i = 0; i < _vbasesol.size(); ++i)
220  {
221  if (_vbasesol[i].maxsolutions == (unsigned char)-1)
222  {
223  throw std::runtime_error("max solutions for joint not initialized");
224  }
225  if (_vbasesol[i].maxsolutions > 0)
226  {
227  if (_vbasesol[i].indices[0] >= _vbasesol[i].maxsolutions)
228  {
229  throw std::runtime_error("index >= max solutions for joint");
230  }
231  if (_vbasesol[i].indices[1] != (unsigned char)-1 && _vbasesol[i].indices[1] >= _vbasesol[i].maxsolutions)
232  {
233  throw std::runtime_error("2nd index >= max solutions for joint");
234  }
235  }
236  }
237  }
238 
239  virtual void GetSolutionIndices(std::vector<unsigned int>& v) const
240  {
241  v.resize(0);
242  v.push_back(0);
243  for (int i = static_cast<int>(_vbasesol.size()) - 1; i >= 0; --i)
244  {
245  if (_vbasesol[i].maxsolutions != (unsigned char)-1 && _vbasesol[i].maxsolutions > 1)
246  {
247  for (size_t j = 0; j < v.size(); ++j)
248  {
249  v[j] *= _vbasesol[i].maxsolutions;
250  }
251  size_t orgsize = v.size();
252  if (_vbasesol[i].indices[1] != (unsigned char)-1)
253  {
254  for (size_t j = 0; j < orgsize; ++j)
255  {
256  v.push_back(v[j] + _vbasesol[i].indices[1]);
257  }
258  }
259  if (_vbasesol[i].indices[0] != (unsigned char)-1)
260  {
261  for (size_t j = 0; j < orgsize; ++j)
262  {
263  v[j] += _vbasesol[i].indices[0];
264  }
265  }
266  }
267  }
268  }

271  std::vector<int> _vfree;
272 };
