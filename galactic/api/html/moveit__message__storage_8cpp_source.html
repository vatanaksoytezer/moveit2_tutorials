48 void moveit_warehouse::MoveItMessageStorage::filterNames(const std::string& regex, std::vector<std::string>& names) const
49 {
50  if (!regex.empty())
51  {
52  std::vector<std::string> fnames;
53  boost::regex r(regex);
54  for (std::string& name : names)
55  {
56  boost::cmatch match;
57  if (boost::regex_match(name.c_str(), match, r))
58  fnames.push_back(name);
59  }
60  names.swap(fnames);
61  }
62 }
