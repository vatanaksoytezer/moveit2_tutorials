83 {
84  planning_interface::PlannerConfigurationMap pconfig2 = pconfig;
85 
86  // construct default configurations for planning groups that don't have configs already passed in
87  for (const moveit::core::JointModelGroup* group : robot_model_->getJointModelGroups())
88  {
89  if (pconfig.find(group->getName()) == pconfig.end())

91  planning_interface::PlannerConfigurationSettings empty;

94  }
95  }

97  context_manager_.setPlannerConfigurations(pconfig2);

134 bool OMPLInterface::loadPlannerConfiguration(const std::string& group_name, const std::string& planner_id,
135  const std::map<std::string, std::string>& group_params,
136  planning_interface::PlannerConfigurationSettings& planner_config)
137 {
138  rcl_interfaces::msg::ListParametersResult planner_params_result =
139  node_->list_parameters({ parameter_namespace_ + ".planner_configs." + planner_id }, 2);
140 
141  if (planner_params_result.names.empty())
142  {
143  RCLCPP_ERROR(LOGGER, "Could not find the planner configuration '%s' on the param server", planner_id.c_str());
144  return false;
145  }
146 
147  planner_config.name = group_name + "[" + planner_id + "]";
148  planner_config.group = group_name;
149 
150  // default to specified parameters of the group (overridden by configuration specific parameters)
151  planner_config.config = group_params;
152 
153  // read parameters specific for this configuration
154  for (const auto& planner_param : planner_params_result.names)
155  {
156  const rclcpp::Parameter param = node_->get_parameter(planner_param);
157  auto param_name = planner_param.substr(planner_param.find(planner_id) + planner_id.size() + 1);
158  planner_config.config[param_name] = param.value_to_string();
159  }
160 
161  return true;
162 }

164 void OMPLInterface::loadPlannerConfigurations()
165 {
166  // read the planning configuration for each group
167  planning_interface::PlannerConfigurationMap pconfig;
168  pconfig.clear();
169 
170  for (const std::string& group_name : robot_model_->getJointModelGroupNames())
171  {
172  // the set of planning parameters that can be specific for the group (inherited by configurations of that group)
173  // with their expected parameter type
174  static const std::pair<std::string, rclcpp::ParameterType> KNOWN_GROUP_PARAMS[] = {
175  { "projection_evaluator", rclcpp::ParameterType::PARAMETER_STRING },
176  { "longest_valid_segment_fraction", rclcpp::ParameterType::PARAMETER_DOUBLE },
177  { "enforce_joint_model_state_space", rclcpp::ParameterType::PARAMETER_BOOL },
178  { "enforce_constrained_state_space", rclcpp::ParameterType::PARAMETER_BOOL }
179  };
180 
181  const std::string group_name_param = parameter_namespace_ + "." + group_name;
182 
183  // get parameters specific for the robot planning group
184  std::map<std::string, std::string> specific_group_params;
185  for (const auto& k : KNOWN_GROUP_PARAMS)
186  {
187  std::string param_name{ group_name_param };
188  param_name += ".";
189  param_name += k.first;
190  if (node_->has_parameter(param_name))
191  {
192  const rclcpp::Parameter parameter = node_->get_parameter(param_name);
193  if (parameter.get_type() != k.second)
194  {
195  RCLCPP_ERROR_STREAM(LOGGER, "Invalid type for parameter '" << k.first << "' expected ["
196  << rclcpp::to_string(k.second) << "] got ["
197  << rclcpp::to_string(parameter.get_type()) << "]");
198  continue;
199  }
200  if (parameter.get_type() == rclcpp::ParameterType::PARAMETER_STRING)
201  specific_group_params[k.first] = parameter.as_string();
202  else if (parameter.get_type() == rclcpp::ParameterType::PARAMETER_DOUBLE)
203  specific_group_params[k.first] = moveit::core::toString(parameter.as_double());
204  else if (parameter.get_type() == rclcpp::ParameterType::PARAMETER_INTEGER)
205  specific_group_params[k.first] = std::to_string(parameter.as_int());
206  else if (parameter.get_type() == rclcpp::ParameterType::PARAMETER_BOOL)
207  specific_group_params[k.first] = std::to_string(parameter.as_bool());
208  }
209  }
210 
211  // add default planner configuration
212  planning_interface::PlannerConfigurationSettings default_pc;
213  std::string default_planner_id;
214  if (node_->get_parameter(group_name_param + ".default_planner_config", default_planner_id))
215  {
216  if (!loadPlannerConfiguration(group_name, default_planner_id, specific_group_params, default_pc))
217  {
218  default_planner_id = "";
219  }
220  }
221 
222  if (default_planner_id.empty())
223  {
224  default_pc.group = group_name;
225  default_pc.config = specific_group_params;
226  default_pc.config["type"] = "geometric::RRTConnect";
227  }
228 
229  default_pc.name = group_name; // this is the name of the default config
230  pconfig[default_pc.name] = default_pc;
231 
232  // get parameters specific to each planner type
233  std::vector<std::string> config_names;
234  if (node_->get_parameter(group_name_param + ".planner_configs", config_names))
235  {
236  for (const auto& planner_id : config_names)
237  {
238  planning_interface::PlannerConfigurationSettings pc;
239  if (loadPlannerConfiguration(group_name, planner_id, specific_group_params, pc))
240  {
241  pconfig[pc.name] = pc;
242  }
243  }
244  }
245  }
246 
247  for (const std::pair<const std::string, planning_interface::PlannerConfigurationSettings>& config : pconfig)
248  {
249  RCLCPP_DEBUG(LOGGER, "Parameters for configuration '%s'", config.first.c_str());
250 
251  for (const std::pair<const std::string, std::string>& parameters : config.second.config)
252  {
253  RCLCPP_DEBUG(LOGGER, " - %s = %s", parameters.first.c_str(), parameters.second.c_str());
254  }
255  }
256 
257  setPlannerConfigurations(pconfig);
258 }
