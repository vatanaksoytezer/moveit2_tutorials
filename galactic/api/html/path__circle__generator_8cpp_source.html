39 std::unique_ptr<KDL::Path> PathCircleGenerator::circleFromCenter(const KDL::Frame& start_pose,
40  const KDL::Frame& goal_pose,
41  const KDL::Vector& center_point, double eqradius)
42 {
43  double a = (start_pose.p - center_point).Norm();
44  double b = (goal_pose.p - center_point).Norm();
45  double c = (start_pose.p - goal_pose.p).Norm();
46 
47  if (fabs(a - b) > MAX_RADIUS_DIFF)
48  {
49  throw ErrorMotionPlanningCenterPointDifferentRadius();
50  }
51 
52  // compute the rotation angle
53  double alpha = cosines(a, b, c);
54 
55  KDL::RotationalInterpolation* rot_interpo = new KDL::RotationalInterpolation_SingleAxis();
56  double old_kdl_epsilon = KDL::epsilon;
57  try
58  {
59  KDL::epsilon = MAX_COLINEAR_NORM;
60  auto circle = std::unique_ptr<KDL::Path>(
61  new KDL::Path_Circle(start_pose, center_point, goal_pose.p, goal_pose.M, alpha, rot_interpo, eqradius,
62  true /* take ownership of RotationalInterpolation */));
63  KDL::epsilon = old_kdl_epsilon;
64  return circle;
65  }
66  catch (KDL::Error_MotionPlanning&)
67  {
68  delete rot_interpo; // in case we could not construct the Path object, avoid
69  // a memory leak
70  KDL::epsilon = old_kdl_epsilon;
71  throw; // and pass the exception on to the caller
72  }
73 }

75 std::unique_ptr<KDL::Path> PathCircleGenerator::circleFromInterim(const KDL::Frame& start_pose,
76  const KDL::Frame& goal_pose,
77  const KDL::Vector& interim_point, double eqradius)
78 {
79  // compute the center point from interim point
80  // triangle edges
81  const KDL::Vector t = interim_point - start_pose.p;
82  const KDL::Vector u = goal_pose.p - start_pose.p;
83  const KDL::Vector v = goal_pose.p - interim_point;
84  // triangle normal
85  const KDL::Vector w = t * u; // cross product
86 
87  // circle center
88  if (w.Norm() < MAX_COLINEAR_NORM)
89  {
90  throw KDL::Error_MotionPlanning_Circle_No_Plane();
91  }
92  const KDL::Vector center_point =
93  start_pose.p + (u * dot(t, t) * dot(u, v) - t * dot(u, u) * dot(t, v)) * 0.5 / pow(w.Norm(), 2);
94 
95  // compute the rotation angle
96  // triangle edges
97  const KDL::Vector t_center = center_point - start_pose.p;
98  const KDL::Vector v_center = goal_pose.p - center_point;
99  double a = t_center.Norm();
100  double b = v_center.Norm();
101  double c = u.Norm();
102  double alpha = cosines(a, b, c);
103 
104  KDL::Vector kdl_aux_point(interim_point);
105 
106  // if the angle at the interim is an acute angle (<90deg), rotation angle is
107  // an obtuse angle (>90deg)
108  // in this case using the interim as auxiliary point for KDL::Path_Circle can
109  // lead to a path in the wrong direction
110  double interim_angle = cosines(t.Norm(), v.Norm(), u.Norm());
111  if (interim_angle < M_PI / 2)
112  {
113  alpha = 2 * M_PI - alpha;
114 
115  // exclude that the goal is not colinear with start and center, then use the
116  // opposite of the goal as auxiliary point
117  if ((t_center * v_center).Norm() > MAX_COLINEAR_NORM)
118  {
119  kdl_aux_point = 2 * center_point - goal_pose.p;
120  }
121  }
122 
123  KDL::RotationalInterpolation* rot_interpo = new KDL::RotationalInterpolation_SingleAxis();
124  try
125  {
126  return std::unique_ptr<KDL::Path>(new KDL::Path_Circle(start_pose, center_point, kdl_aux_point, goal_pose.M, alpha,
127  rot_interpo, eqradius,
128  true /* take ownership of RotationalInterpolation */));
129  }
130  catch (KDL::Error_MotionPlanning&) // LCOV_EXCL_START // The cases where
131  // KDL would throw are already handled
132  // above,
133  // we keep these lines to be safe
134  {
135  delete rot_interpo; // in case we could not construct the Path object, avoid
136  // a memory leak
137  throw; // and pass the exception on to the caller
138  // LCOV_EXCL_STOP
139  }
140 }

142 double PathCircleGenerator::cosines(const double a, const double b, const double c)
143 {
144  return acos(std::max(std::min((pow(a, 2) + pow(b, 2) - pow(c, 2)) / (2.0 * a * b), 1.0), -1.0));
145 }
