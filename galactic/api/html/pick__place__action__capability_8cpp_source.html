88 void move_group::MoveGroupPickPlaceAction::executePickupCallbackPlanOnly(
89  const moveit_msgs::action::PickupGoalConstPtr& goal, moveit_msgs::action::PickupResult& action_res)
90 {
91  pick_place::PickPlanPtr plan;
92  try
93  {
94  planning_scene_monitor::LockedPlanningSceneRO ps(context_->planning_scene_monitor_);
95  plan = pick_place_->planPick(ps, *goal);
96  }
97  catch (std::exception& ex)
98  {
99  ROS_ERROR_NAMED("manipulation", "Pick&place threw an exception: %s", ex.what());
100  }
101 
102  if (plan)
103  {
104  const std::vector<pick_place::ManipulationPlanPtr>& success = plan->getSuccessfulManipulationPlans();
105  if (success.empty())
106  {
107  action_res.error_code = plan->getErrorCode();
108  }
109  else
110  {
111  const pick_place::ManipulationPlanPtr& result = success.back();
112  convertToMsg(result->trajectories_, action_res.trajectory_start, action_res.trajectory_stages);
113  action_res.trajectory_descriptions.resize(result->trajectories_.size());
114  for (std::size_t i = 0; i < result->trajectories_.size(); ++i)
115  action_res.trajectory_descriptions[i] = result->trajectories_[i].description_;
116  if (result->id_ < goal->possible_grasps.size())
117  action_res.grasp = goal->possible_grasps[result->id_];
118  action_res.error_code.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
119  action_res.planning_time = plan->getLastPlanTime();
120  }
121  }
122  else
123  {
124  action_res.error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
125  }
126 }

315 void move_group::MoveGroupPickPlaceAction::executePickupCallback(
316  const moveit_msgs::action::PickupGoalConstPtr& input_goal)
317 {
318  setPickupState(PLANNING);
319 
320  // before we start planning, ensure that we have the latest robot state received...
321  auto node = context_->moveit_cpp_->getNode();
322  context_->planning_scene_monitor_->waitForCurrentRobotState(node->get_clock()->now());
323  context_->planning_scene_monitor_->updateFrameTransforms();
324 
325  moveit_msgs::action::PickupGoalConstPtr goal;
326  if (input_goal->possible_grasps.empty())
327  {
328  moveit_msgs::action::PickupGoal* copy(new moveit_msgs::action::PickupGoal(*input_goal));
329  goal.reset(copy);
330  fillGrasps(*copy);
331  }
332  else
333  goal = input_goal;
334 
335  moveit_msgs::action::PickupResult action_res;
336 
337  if (goal->planning_options.plan_only || !context_->allow_trajectory_execution_)
338  {
339  if (!goal->planning_options.plan_only)
340  ROS_WARN_NAMED("manipulation", "This instance of MoveGroup is not allowed to execute trajectories but the pick "
341  "goal request has plan_only set to false. Only a motion plan will be computed "
342  "anyway.");
343  executePickupCallbackPlanOnly(goal, action_res);
344  }
345  else
346  executePickupCallbackPlanAndExecute(goal, action_res);
347 
348  bool planned_trajectory_empty = action_res.trajectory_stages.empty();
349  std::string response =
350  getActionResultString(action_res.error_code, planned_trajectory_empty, goal->planning_options.plan_only);
351  if (action_res.error_code.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
352  pickup_action_server_->setSucceeded(action_res, response);
353  else
354  {
355  if (action_res.error_code.val == moveit_msgs::msg::MoveItErrorCodes::PREEMPTED)
356  pickup_action_server_->setPreempted(action_res, response);
357  else
358  pickup_action_server_->setAborted(action_res, response);
359  }
360 
361  setPickupState(IDLE);
362 }

424 void move_group::MoveGroupPickPlaceAction::fillGrasps(moveit_msgs::action::PickupGoal& goal)
425 {
426  planning_scene_monitor::LockedPlanningSceneRO lscene(context_->planning_scene_monitor_);
427 
428  ROS_DEBUG_NAMED("manipulation", "Using default grasp poses");
429  goal.minimize_object_distance = true;
430 
431  // add a number of default grasp points
432  // \todo add more!
433  moveit_msgs::msg::Grasp g;
434  g.grasp_pose.header.frame_id = goal.target_name;
435  g.grasp_pose.pose.position.x = -0.2;
436  g.grasp_pose.pose.position.y = 0.0;
437  g.grasp_pose.pose.position.z = 0.0;
438  g.grasp_pose.pose.orientation.x = 0.0;
439  g.grasp_pose.pose.orientation.y = 0.0;
440  g.grasp_pose.pose.orientation.z = 0.0;
441  g.grasp_pose.pose.orientation.w = 1.0;
442 
443  g.pre_grasp_approach.direction.header.frame_id = lscene->getPlanningFrame();
444  g.pre_grasp_approach.direction.vector.x = 1.0;
445  g.pre_grasp_approach.min_distance = 0.1;
446  g.pre_grasp_approach.desired_distance = 0.2;
447 
448  g.post_grasp_retreat.direction.header.frame_id = lscene->getPlanningFrame();
449  g.post_grasp_retreat.direction.vector.z = 1.0;
450  g.post_grasp_retreat.min_distance = 0.1;
451  g.post_grasp_retreat.desired_distance = 0.2;
452 
453  if (lscene->getRobotModel()->hasEndEffector(goal.end_effector))
454  {
455  g.pre_grasp_posture.joint_names = lscene->getRobotModel()->getEndEffector(goal.end_effector)->getJointModelNames();
456  g.pre_grasp_posture.points.resize(1);
457  g.pre_grasp_posture.points[0].positions.resize(g.pre_grasp_posture.joint_names.size(),
458  std::numeric_limits<double>::max());
459 
460  g.grasp_posture.joint_names = g.pre_grasp_posture.joint_names;
461  g.grasp_posture.points.resize(1);
462  g.grasp_posture.points[0].positions.resize(g.grasp_posture.joint_names.size(), -std::numeric_limits<double>::max());
463  }
464  goal.possible_grasps.push_back(g);
465 }
