72 bool transformToEndEffectorGoal(const geometry_msgs::PoseStamped& goal_pose,
73  const moveit::core::AttachedBody* attached_body, geometry_msgs::PoseStamped& place_pose)
74 {
75  const EigenSTL::vector_Isometry3d& fixed_transforms = attached_body->getShapePosesInLinkFrame();
76  if (fixed_transforms.empty())
77  return false;
78 
79  Eigen::Isometry3d end_effector_transform;
80  tf2::fromMsg(goal_pose.pose, end_effector_transform);
81  end_effector_transform = end_effector_transform * fixed_transforms[0].inverse();
82  place_pose.header = goal_pose.header;
83  place_pose.pose = tf2::toMsg(end_effector_transform);
84  return true;
85 }

88 bool PlacePlan::plan(const planning_scene::PlanningSceneConstPtr& planning_scene,
89  const moveit_msgs::action::PlaceGoal& goal)
90 {
91  double timeout = goal.allowed_planning_time;
92  ros::WallTime endtime = ros::WallTime::now() + ros::WallDuration(timeout);
93  std::string attached_object_name = goal.attached_object_name;
94  const moveit::core::JointModelGroup* jmg = nullptr;
95  const moveit::core::JointModelGroup* eef = nullptr;
96 
97  // if the group specified is actually an end-effector, we use it as such
98  if (planning_scene->getRobotModel()->hasEndEffector(goal.group_name))
99  {
100  eef = planning_scene->getRobotModel()->getEndEffector(goal.group_name);
101  if (eef)
102  { // if we correctly found the eef, then we try to find out what the planning group is
103  const std::string& eef_parent = eef->getEndEffectorParentGroup().first;
104  if (eef_parent.empty())
105  {
106  ROS_ERROR_STREAM_NAMED("manipulation", "No parent group to plan in was identified based on end-effector '"
107  << goal.group_name
108  << "'. Please define a parent group in the SRDF.");
109  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
110  return false;
111  }
112  else
113  jmg = planning_scene->getRobotModel()->getJointModelGroup(eef_parent);
114  }
115  }
116  else
117  {
118  // if a group name was specified, try to use it
119  jmg = goal.group_name.empty() ? nullptr : planning_scene->getRobotModel()->getJointModelGroup(goal.group_name);
120  if (jmg)
121  {
122  // we also try to find the corresponding eef
123  const std::vector<std::string>& eef_names = jmg->getAttachedEndEffectorNames();
124  if (eef_names.empty())
125  {
126  ROS_ERROR_STREAM_NAMED("manipulation",
127  "There are no end-effectors specified for group '" << goal.group_name << "'");
128  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
129  return false;
130  }
131  else
132  // check to see if there is an end effector that has attached objects associaded, so we can complete the place
133  for (const std::string& eef_name : eef_names)
134  {
135  std::vector<const moveit::core::AttachedBody*> attached_bodies;
136  const moveit::core::JointModelGroup* eg = planning_scene->getRobotModel()->getEndEffector(eef_name);
137  if (eg)
138  {
139  // see if there are objects attached to links in the eef
140  planning_scene->getCurrentState().getAttachedBodies(attached_bodies, eg);
141 
142  // is is often possible that the objects are attached to the same link that the eef itself is attached,
143  // so we check for attached bodies there as well
144  const moveit::core::LinkModel* attached_link_model =
145  planning_scene->getRobotModel()->getLinkModel(eg->getEndEffectorParentGroup().second);
146  if (attached_link_model)
147  {
148  std::vector<const moveit::core::AttachedBody*> attached_bodies2;
149  planning_scene->getCurrentState().getAttachedBodies(attached_bodies2, attached_link_model);
150  attached_bodies.insert(attached_bodies.end(), attached_bodies2.begin(), attached_bodies2.end());
151  }
152  }
153 
154  // if this end effector has attached objects, we go on
155  if (!attached_bodies.empty())
156  {
157  // if the user specified the name of the attached object to place, we check that indeed
158  // the group contains this attached body
159  if (!attached_object_name.empty())
160  {
161  bool found = false;
162  for (const moveit::core::AttachedBody* attached_body : attached_bodies)
163  if (attached_body->getName() == attached_object_name)
164  {
165  found = true;
166  break;
167  }
168  // if the attached body this group has is not the same as the one specified,
169  // we cannot use this eef
170  if (!found)
171  continue;
172  }
173 
174  // if we previoulsy have set the eef it means we have more options we could use, so things are ambiguous
175  if (eef)
176  {
177  ROS_ERROR_STREAM_NAMED("manipulation", "There are multiple end-effectors for group '"
178  << goal.group_name
179  << "' that are currently holding objects. It is ambiguous "
180  "which end-effector to use. Please specify it explicitly.");
181  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
182  return false;
183  }
184  // set the end effector (this was initialized to NULL above)
185  eef = planning_scene->getRobotModel()->getEndEffector(eef_name);
186  }
187  }
188  }
189  }
190 
191  // if we know the attached object, but not the eef, we can try to identify that
192  if (!attached_object_name.empty() && !eef)
193  {
194  const moveit::core::AttachedBody* attached_body =
195  planning_scene->getCurrentState().getAttachedBody(attached_object_name);
196  if (attached_body)
197  {
198  // get the robot model link this attached body is associated to
199  const moveit::core::LinkModel* link = attached_body->getAttachedLink();
200  // check to see if there is a unique end effector containing the link
201  const std::vector<const moveit::core::JointModelGroup*>& eefs =
202  planning_scene->getRobotModel()->getEndEffectors();
203  for (const moveit::core::JointModelGroup* end_effector : eefs)
204  if (end_effector->hasLinkModel(link->getName()))
205  {
206  if (eef)
207  {
208  ROS_ERROR_STREAM_NAMED("manipulation", "There are multiple end-effectors that include the link '"
209  << link->getName() << "' which is where the body '"
210  << attached_object_name
211  << "' is attached. It is unclear which end-effector to use.");
212  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
213  return false;
214  }
215  eef = end_effector;
216  }
217  }
218  // if the group is also unknown, but we just found out the eef
219  if (!jmg && eef)
220  {
221  const std::string& eef_parent = eef->getEndEffectorParentGroup().first;
222  if (eef_parent.empty())
223  {
224  ROS_ERROR_STREAM_NAMED("manipulation", "No parent group to plan in was identified based on end-effector '"
225  << goal.group_name
226  << "'. Please define a parent group in the SRDF.");
227  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
228  return false;
229  }
230  else
231  jmg = planning_scene->getRobotModel()->getJointModelGroup(eef_parent);
232  }
233  }
234 
235  if (!jmg || !eef)
236  {
237  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
238  return false;
239  }
240 
241  // try to infer attached body name if possible
242  int loop_count = 0;
243  while (attached_object_name.empty() && loop_count < 2)
244  {
245  // in the first try, look for objects attached to the eef, if the eef is known;
246  // otherwise, look for attached bodies in the planning group itself
247  std::vector<const moveit::core::AttachedBody*> attached_bodies;
248  planning_scene->getCurrentState().getAttachedBodies(attached_bodies, loop_count == 0 ? eef : jmg);
249 
250  loop_count++;
251  if (attached_bodies.size() > 1)
252  {
253  ROS_ERROR_NAMED("manipulation",
254  "Multiple attached bodies for group '%s' but no explicit attached object to place was specified",
255  goal.group_name.c_str());
256  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_OBJECT_NAME;
257  return false;
258  }
259  else
260  attached_object_name = attached_bodies[0]->getName();
261  }
262 
263  const moveit::core::AttachedBody* attached_body =
264  planning_scene->getCurrentState().getAttachedBody(attached_object_name);
265  if (!attached_body)
266  {
267  ROS_ERROR_NAMED("manipulation", "There is no object to detach for place action");
268  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_OBJECT_NAME;
269  return false;
270  }
271 
272  ros::WallTime start_time = ros::WallTime::now();
273 
274  // construct common data for possible manipulation plans
275  ManipulationPlanSharedDataPtr plan_data(new ManipulationPlanSharedData());
276  ManipulationPlanSharedDataConstPtr const_plan_data = plan_data;
277  plan_data->planning_group_ = jmg;
278  plan_data->end_effector_group_ = eef;
279  plan_data->ik_link_ = planning_scene->getRobotModel()->getLinkModel(eef->getEndEffectorParentGroup().second);
280 
281  plan_data->timeout_ = endtime;
282  plan_data->path_constraints_ = goal.path_constraints;
283  plan_data->planner_id_ = goal.planner_id;
284  plan_data->minimize_object_distance_ = false;
285  plan_data->max_goal_sampling_attempts_ = 2;
286  moveit_msgs::msg::AttachedCollisionObject& detach_object_msg = plan_data->diff_attached_object_;
287 
288  // construct the attached object message that will change the world to what it would become after a placement
289  detach_object_msg.link_name = attached_body->getAttachedLinkName();
290  detach_object_msg.object.id = attached_object_name;
291  detach_object_msg.object.operation = moveit_msgs::msg::CollisionObject::REMOVE;
292 
293  collision_detection::AllowedCollisionMatrixPtr approach_place_acm(
294  new collision_detection::AllowedCollisionMatrix(planning_scene->getAllowedCollisionMatrix()));
295 
296  // we are allowed to touch certain other objects with the gripper
297  approach_place_acm->setEntry(eef->getLinkModelNames(), goal.allowed_touch_objects, true);
298 
299  // we are allowed to touch the target object slightly while retreating the end effector
300  std::vector<std::string> touch_links(attached_body->getTouchLinks().begin(), attached_body->getTouchLinks().end());
301  approach_place_acm->setEntry(attached_object_name, touch_links, true);
302 
303  if (!goal.support_surface_name.empty())
304  {
305  // we are allowed to have contact between the target object and the support surface before the place
306  approach_place_acm->setEntry(goal.support_surface_name, attached_object_name, true);
307 
308  // optionally, it may be allowed to touch the support surface with the gripper
309  if (goal.allow_gripper_support_collision)
310  approach_place_acm->setEntry(goal.support_surface_name, eef->getLinkModelNames(), true);
311  }
312 
313  // configure the manipulation pipeline
314  pipeline_.reset();
315 
316  ManipulationStagePtr stage1(
317  new ReachableAndValidPoseFilter(planning_scene, approach_place_acm, pick_place_->getConstraintsSamplerManager()));
318  ManipulationStagePtr stage2(new ApproachAndTranslateStage(planning_scene, approach_place_acm));
319  ManipulationStagePtr stage3(new PlanStage(planning_scene, pick_place_->getPlanningPipeline()));
320  pipeline_.addStage(stage1).addStage(stage2).addStage(stage3);
321 
322  initialize();
323 
324  pipeline_.start();
325 
326  // order the place locations by quality
327  std::vector<std::size_t> place_locations_order(goal.place_locations.size());
328  for (std::size_t i = 0; i < goal.place_locations.size(); ++i)
329  place_locations_order[i] = i;
330  OrderPlaceLocationQuality oq(goal.place_locations);
331  // using stable_sort to preserve order of place locations with equal quality
332  std::stable_sort(place_locations_order.begin(), place_locations_order.end(), oq);
333 
334  // add possible place locations
335  for (std::size_t i = 0; i < goal.place_locations.size(); ++i)
336  {
337  ManipulationPlanPtr p(new ManipulationPlan(const_plan_data));
338  const moveit_msgs::action::PlaceLocation& pl = goal.place_locations[place_locations_order[i]];
339 
340  if (goal.place_eef)
341  p->goal_pose_ = pl.place_pose;
342  else
343  // The goals are specified for the attached body
344  // but we want to transform them into goals for the end-effector instead
345  if (!transformToEndEffectorGoal(pl.place_pose, attached_body, p->goal_pose_))
346  {
347  p->goal_pose_ = pl.place_pose;
348  ROS_ERROR_NAMED("manipulation", "Unable to transform the desired pose of the object to the pose of the "
349  "end-effector");
350  }
351 
352  p->approach_ = pl.pre_place_approach;
353  p->retreat_ = pl.post_place_retreat;
354  p->retreat_posture_ = pl.post_place_posture;
355  p->id_ = place_locations_order[i];
356  if (p->retreat_posture_.joint_names.empty())
357  p->retreat_posture_ = attached_body->getDetachPosture();
358  pipeline_.push(p);
359  }
360  ROS_INFO_NAMED("manipulation", "Added %d place locations", static_cast<int>(goal.place_locations.size()));
361 
362  // wait till we're done
363  waitForPipeline(endtime);
364 
365  pipeline_.stop();
366 
367  last_plan_time_ = (ros::WallTime::now() - start_time).toSec();
368 
369  if (!getSuccessfulManipulationPlans().empty())
370  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
371  else
372  {
373  if (last_plan_time_ > timeout)
374  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::TIMED_OUT;
375  else
376  {
377  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED;
378  if (!goal.place_locations.empty())
379  {
380  ROS_WARN_NAMED("manipulation", "All supplied place locations failed. Retrying last location in verbose mode.");
381  // everything failed. we now start the pipeline again in verbose mode for one grasp
382  initialize();
383  pipeline_.setVerbose(true);
384  pipeline_.start();
385  pipeline_.reprocessLastFailure();
386  waitForPipeline(ros::WallTime::now() + ros::WallDuration(1.0));
387  pipeline_.stop();
388  pipeline_.setVerbose(false);
389  }
390  }
391  }
392  ROS_INFO_NAMED("manipulation", "Place planning completed after %lf seconds", last_plan_time_);
393 
394  return error_code_.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
395 }
