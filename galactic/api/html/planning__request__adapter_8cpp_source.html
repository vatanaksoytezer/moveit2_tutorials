132 bool PlanningRequestAdapterChain::adaptAndPlan(const planning_interface::PlannerManagerPtr& planner,
133  const planning_scene::PlanningSceneConstPtr& planning_scene,
134  const planning_interface::MotionPlanRequest& req,
135  planning_interface::MotionPlanResponse& res,
136  std::vector<std::size_t>& added_path_index) const
137 {
138  // if there are no adapters, run the planner directly
139  if (adapters_.empty())
140  {
141  added_path_index.clear();
142  return callPlannerInterfaceSolve(planner.get(), planning_scene, req, res);
143  }
144  else
145  {
146  // the index values added by each adapter
147  std::vector<std::vector<std::size_t> > added_path_index_each(adapters_.size());
148 
149  // if there are adapters, construct a function pointer for each, in order,
150  // so that in the end we have a nested sequence of function pointers that call the adapters in the correct order.
151  PlanningRequestAdapter::PlannerFn fn = boost::bind(&callAdapter1, adapters_.back().get(), planner, _1, _2, _3,
152  boost::ref(added_path_index_each.back()));
153  for (int i = adapters_.size() - 2; i >= 0; --i)
154  fn = boost::bind(&callAdapter2, adapters_[i].get(), fn, _1, _2, _3, boost::ref(added_path_index_each[i]));
155  bool result = fn(planning_scene, req, res);
156  added_path_index.clear();
157 
158  // merge the index values from each adapter
159  for (std::vector<std::size_t>& added_states_by_each_adapter : added_path_index_each)
160  for (std::size_t& added_index : added_states_by_each_adapter)
161  {
162  for (std::size_t& index_in_path : added_path_index)
163  if (added_index <= index_in_path)
164  index_in_path++;
165  added_path_index.push_back(added_index);
166  }
167  std::sort(added_path_index.begin(), added_path_index.end());
168  return result;
169  }
170 }
