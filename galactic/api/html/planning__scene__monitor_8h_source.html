668  void initialize(bool read_only)

672  }
673 
674  MOVEIT_STRUCT_FORWARD(SingleUnlock);
675 
676  // we use this struct so that lock/unlock are called only once
677  // even if the LockedPlanningScene instance is copied around
678  struct SingleUnlock
679  {
680  SingleUnlock(PlanningSceneMonitor* planning_scene_monitor, bool read_only)
681  : planning_scene_monitor_(planning_scene_monitor), read_only_(read_only)
682  {
683  if (read_only)
684  planning_scene_monitor_->lockSceneRead();
685  else
686  planning_scene_monitor_->lockSceneWrite();
687  }
688  ~SingleUnlock()
689  {
690  if (read_only_)
691  planning_scene_monitor_->unlockSceneRead();
692  else
693  planning_scene_monitor_->unlockSceneWrite();
694  }
695  PlanningSceneMonitor* planning_scene_monitor_;
696  bool read_only_;
697  };
698 
699  PlanningSceneMonitorPtr planning_scene_monitor_;
700  SingleUnlockPtr lock_;
701 };
