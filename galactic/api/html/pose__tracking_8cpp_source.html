111 PoseTrackingStatusCode PoseTracking::moveToPose(const Eigen::Vector3d& positional_tolerance,
112  const double angular_tolerance, const double target_pose_timeout)
113 {
114  // Reset stop requested flag before starting motions
115  stop_requested_ = false;
116  // Wait a bit for a target pose message to arrive.
117  // The target pose may get updated by new messages as the robot moves (in a callback function).
118  const rclcpp::Time start_time = node_->now();
119  rclcpp::Clock clock;
120 
121  while ((!haveRecentTargetPose(target_pose_timeout) || !haveRecentEndEffectorPose(target_pose_timeout)) &&
122  ((node_->now() - start_time).seconds() < target_pose_timeout))
123  {
124  if (servo_->getCommandFrameTransform(command_frame_transform_))
125  {
126  command_frame_transform_stamp_ = node_->now();
127  }
128  std::this_thread::sleep_for(1ms);
129  }
130 
131  if (!haveRecentTargetPose(target_pose_timeout))
132  {
133  RCLCPP_ERROR_STREAM(LOGGER, "The target pose was not updated recently. Aborting.");
134  return PoseTrackingStatusCode::NO_RECENT_TARGET_POSE;
135  }
136 
137  // Continue sending PID controller output to Servo until one of the following conditions is met:
138  // - Goal tolerance is satisfied
139  // - Target pose becomes outdated
140  // - Command frame transform becomes outdated
141  // - Another thread requested a stop
142  while (rclcpp::ok())
143  {
144  if (satisfiesPoseTolerance(positional_tolerance, angular_tolerance))
145  {
146  RCLCPP_INFO_STREAM(LOGGER, "The target pose is achieved!");
147  break;
148  }
149  // Attempt to update robot pose
150  if (servo_->getCommandFrameTransform(command_frame_transform_))
151  {
152  command_frame_transform_stamp_ = node_->now();
153  }
154 
155  // Check that end-effector pose (command frame transform) is recent enough.
156  if (!haveRecentEndEffectorPose(target_pose_timeout))
157  {
158  RCLCPP_ERROR_STREAM(LOGGER, "The end effector pose was not updated in time. Aborting.");
159  doPostMotionReset();
160  return PoseTrackingStatusCode::NO_RECENT_END_EFFECTOR_POSE;
161  }
162 
163  if (stop_requested_)
164  {
165  RCLCPP_INFO_STREAM(LOGGER, "Halting servo motion, a stop was requested.");
166  doPostMotionReset();
167  return PoseTrackingStatusCode::STOP_REQUESTED;
168  }
169 
170  // Compute servo command from PID controller output and send it to the Servo object, for execution
171  twist_stamped_pub_->publish(*calculateTwistCommand());
172 
173  if (!loop_rate_.sleep())
174  {
175  RCLCPP_WARN_STREAM_THROTTLE(LOGGER, clock, LOG_THROTTLE_PERIOD, "Target control rate was missed");
176  }
177  }
178 
179  doPostMotionReset();
180  return PoseTrackingStatusCode::SUCCESS;
181 }

244 bool PoseTracking::satisfiesPoseTolerance(const Eigen::Vector3d& positional_tolerance, const double angular_tolerance)
245 {
246  std::lock_guard<std::mutex> lock(target_pose_mtx_);
247  double x_error = target_pose_.pose.position.x - command_frame_transform_.translation()(0);
248  double y_error = target_pose_.pose.position.y - command_frame_transform_.translation()(1);
249  double z_error = target_pose_.pose.position.z - command_frame_transform_.translation()(2);
250 
251  // If uninitialized, likely haven't received the target pose yet.
252  if (!angular_error_)
253  return false;
254 
255  return ((std::abs(x_error) < positional_tolerance(0)) && (std::abs(y_error) < positional_tolerance(1)) &&
256  (std::abs(z_error) < positional_tolerance(2)) && (std::abs(*angular_error_) < angular_tolerance));
257 }

259 void PoseTracking::targetPoseCallback(const geometry_msgs::msg::PoseStamped::ConstSharedPtr msg)
260 {
261  std::lock_guard<std::mutex> lock(target_pose_mtx_);
262  target_pose_ = *msg;
263  // If the target pose is not defined in planning frame, transform the target pose.
264  if (target_pose_.header.frame_id != planning_frame_)
265  {
266  try
267  {
268  geometry_msgs::msg::TransformStamped target_to_planning_frame = transform_buffer_.lookupTransform(
269  planning_frame_, target_pose_.header.frame_id, rclcpp::Time(0), rclcpp::Duration(100ms));
270  tf2::doTransform(target_pose_, target_pose_, target_to_planning_frame);
271 
272  // Prevent doTransform from copying a stamp of 0, which will cause the haveRecentTargetPose check to fail servo motions
273  target_pose_.header.stamp = node_->now();
274  }
275  catch (const tf2::TransformException& ex)
276  {
277  RCLCPP_WARN_STREAM(LOGGER, ex.what());
278  return;
279  }
280  }

283 geometry_msgs::msg::TwistStamped::ConstSharedPtr PoseTracking::calculateTwistCommand()
284 {
285  // use the shared pool to create a message more efficiently
286  auto msg = moveit::util::make_shared_from_pool<geometry_msgs::msg::TwistStamped>();
287 
288  // Get twist components from PID controllers
289  geometry_msgs::msg::Twist& twist = msg->twist;
290  Eigen::Quaterniond q_desired;
291 
292  // Scope mutex locking only to operations which require access to target pose.
293  {
294  std::lock_guard<std::mutex> lock(target_pose_mtx_);
295  msg->header.frame_id = target_pose_.header.frame_id;
296 
297  // Position
298  twist.linear.x = cartesian_position_pids_[0].computeCommand(
299  target_pose_.pose.position.x - command_frame_transform_.translation()(0), loop_rate_.period().count());
300  twist.linear.y = cartesian_position_pids_[1].computeCommand(
301  target_pose_.pose.position.y - command_frame_transform_.translation()(1), loop_rate_.period().count());
302  twist.linear.z = cartesian_position_pids_[2].computeCommand(
303  target_pose_.pose.position.z - command_frame_transform_.translation()(2), loop_rate_.period().count());
304 
305  // Orientation algorithm:
306  // - Find the orientation error as a quaternion: q_error = q_desired * q_current ^ -1
307  // - Use the angle-axis PID controller to calculate an angular rate
308  // - Convert to angular velocity for the TwistStamped message
309  q_desired = Eigen::Quaterniond(target_pose_.pose.orientation.w, target_pose_.pose.orientation.x,
310  target_pose_.pose.orientation.y, target_pose_.pose.orientation.z);
311  }
312 
313  Eigen::Quaterniond q_current(command_frame_transform_.rotation());
314  Eigen::Quaterniond q_error = q_desired * q_current.inverse();
315 
316  // Convert axis-angle to angular velocity
317  Eigen::AngleAxisd axis_angle(q_error);
318  // Cache the angular error, for rotation tolerance checking
319  angular_error_ = axis_angle.angle();
320 
321  double ang_vel_magnitude =
322  cartesian_orientation_pids_[0].computeCommand(*angular_error_, loop_rate_.period().count());
323  twist.angular.x = ang_vel_magnitude * axis_angle.axis()[0];
324  twist.angular.y = ang_vel_magnitude * axis_angle.axis()[1];
325  twist.angular.z = ang_vel_magnitude * axis_angle.axis()[2];
326 
327  msg->header.stamp = node_->now();
328 
329  return msg;
330 }

372  y_pid_config_.k_i = y_integral_gain;
373  y_pid_config_.k_d = y_derivative_gain;

376  z_pid_config_.k_d = z_derivative_gain;
377 
378  angular_pid_config_.k_p = angular_proportional_gain;
379  angular_pid_config_.k_i = angular_integral_gain;
380  angular_pid_config_.k_d = angular_derivative_gain;
381 
382  cartesian_position_pids_.clear();
383  cartesian_orientation_pids_.clear();
384  initializePID(x_pid_config_, cartesian_position_pids_);
385  initializePID(y_pid_config_, cartesian_position_pids_);
386  initializePID(z_pid_config_, cartesian_position_pids_);
387  initializePID(angular_pid_config_, cartesian_orientation_pids_);
