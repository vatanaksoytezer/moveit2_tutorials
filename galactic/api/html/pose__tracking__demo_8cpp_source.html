52 // Class for monitoring status of moveit_servo
53 class StatusMonitor
54 {
55 public:
56  StatusMonitor(const rclcpp::Node::SharedPtr& node, const std::string& topic)
57  {
58  sub_ = node->create_subscription<std_msgs::msg::Int8>(
59  topic, rclcpp::SystemDefaultsQoS(), std::bind(&StatusMonitor::statusCB, this, std::placeholders::_1));
60  }
61 
62 private:
63  void statusCB(const std_msgs::msg::Int8::ConstSharedPtr msg)
64  {
65  moveit_servo::StatusCode latest_status = static_cast<moveit_servo::StatusCode>(msg->data);
66  if (latest_status != status_)
67  {
68  status_ = latest_status;
69  const auto& status_str = moveit_servo::SERVO_STATUS_CODE_MAP.at(status_);
70  RCLCPP_INFO_STREAM(LOGGER, "Servo status: " << status_str);
71  }
72  }
73 
74  moveit_servo::StatusCode status_ = moveit_servo::StatusCode::INVALID;
75  rclcpp::Subscription<std_msgs::msg::Int8>::SharedPtr sub_;
76 };

93

97  exit(EXIT_FAILURE);

99 
100  // Load the planning scene monitor
101  planning_scene_monitor::PlanningSceneMonitorPtr planning_scene_monitor;
102  planning_scene_monitor = std::make_shared<planning_scene_monitor::PlanningSceneMonitor>(node, "robot_description");
103  if (!planning_scene_monitor->getPlanningScene())
104  {
105  RCLCPP_ERROR_STREAM(LOGGER, "Error in setting up the PlanningSceneMonitor.");
106  exit(EXIT_FAILURE);
107  }
108 
109  planning_scene_monitor->providePlanningSceneService();
110  planning_scene_monitor->startSceneMonitor();
111  planning_scene_monitor->startWorldGeometryMonitor(
112  planning_scene_monitor::PlanningSceneMonitor::DEFAULT_COLLISION_OBJECT_TOPIC,
113  planning_scene_monitor::PlanningSceneMonitor::DEFAULT_PLANNING_SCENE_WORLD_TOPIC,
114  false /* skip octomap monitor */);
115  planning_scene_monitor->startStateMonitor(servo_parameters->joint_topic);
116  planning_scene_monitor->startPublishingPlanningScene(planning_scene_monitor::PlanningSceneMonitor::UPDATE_SCENE);
117 
118  // Wait for Planning Scene Monitor to setup
119  if (!planning_scene_monitor->waitForCurrentRobotState(node->now(), 5.0 /* seconds */))
120  {
121  RCLCPP_ERROR_STREAM(LOGGER, "Error waiting for current robot state in PlanningSceneMonitor.");
122  exit(EXIT_FAILURE);

124

127 
128  // Make a publisher for sending pose commands
129  auto target_pose_pub =
130  node->create_publisher<geometry_msgs::msg::PoseStamped>("target_pose", rclcpp::SystemDefaultsQoS());
131 
132  // Subscribe to servo status (and log it when it changes)
133  StatusMonitor status_monitor(node, servo_parameters->status_topic);
134 
135  Eigen::Vector3d lin_tol{ 0.001, 0.001, 0.001 };
136  double rot_tol = 0.01;

138  // Get the current EE transform

140  tracker.getCommandFrameTransform(current_ee_tf);
141 
142  // Convert it to a Pose

144  target_pose.header.frame_id = current_ee_tf.header.frame_id;
145  target_pose.pose.position.x = current_ee_tf.transform.translation.x;
146  target_pose.pose.position.y = current_ee_tf.transform.translation.y;
147  target_pose.pose.position.z = current_ee_tf.transform.translation.z;
148  target_pose.pose.orientation = current_ee_tf.transform.rotation;
149 
150  // Modify it a little bit
151  target_pose.pose.position.x += 0.1;
152 
153  // resetTargetPose() can be used to clear the target pose and wait for a new one, e.g. when moving between multiple
154  // waypoints
155  tracker.resetTargetPose();
156 
157  // Publish target pose
158  target_pose.header.stamp = node->now();
159  target_pose_pub->publish(target_pose);
160 
161  // Run the pose tracking in a new thread
162  std::thread move_to_pose_thread([&tracker, &lin_tol, &rot_tol] {
163  moveit_servo::PoseTrackingStatusCode tracking_status =
164  tracker.moveToPose(lin_tol, rot_tol, 0.1 /* target pose timeout */);
165  RCLCPP_INFO_STREAM(LOGGER, "Pose tracker exited with status: "
166  << moveit_servo::POSE_TRACKING_STATUS_CODE_MAP.at(tracking_status));
167  });
168 
169  rclcpp::Rate loop_rate(50);
170  for (size_t i = 0; i < 500; ++i)
171  {
172  // Modify the pose target a little bit each cycle
173  // This is a dynamic pose target
174  target_pose.pose.position.z += 0.0004;
175  target_pose.header.stamp = node->now();
176  target_pose_pub->publish(target_pose);
177 
178  loop_rate.sleep();
179  }
180 
181  // Make sure the tracker is stopped and clean up
182  move_to_pose_thread.join();
183 
184  // Kill executor thread before shutdown
185  executor.cancel();

187 
188  rclcpp::shutdown();

190 }
