384 bool IKFastKinematicsPlugin::computeRelativeTransform(const std::string& from, const std::string& to,
385  Eigen::Isometry3d& transform, bool& differs_from_identity)
386 {
387  RobotStatePtr robot_state;
388  robot_state.reset(new RobotState(robot_model_));
389  robot_state->setToDefaultValues();
390 
391  auto* from_link = robot_model_->getLinkModel(from);
392  auto* to_link = robot_model_->getLinkModel(to);
393  if (!from_link || !to_link)
394  return false;
395 
396  if (robot_model_->getRigidlyConnectedParentLinkModel(from_link) !=
397  robot_model_->getRigidlyConnectedParentLinkModel(to_link))
398  {
399  RCLCPP_ERROR_STREAM(LOGGER, "Link frames " << from << " and " << to << " are not rigidly connected.");
400  return false;
401  }
402 
403  transform = robot_state->getGlobalLinkTransform(from_link).inverse() * robot_state->getGlobalLinkTransform(to_link);
404  differs_from_identity = !transform.matrix().isIdentity();
405  return true;
406 }

559 size_t IKFastKinematicsPlugin::solve(KDL::Frame& pose_frame, const std::vector<double>& vfree,
560  IkSolutionList<IkReal>& solutions) const
561 {
562  // IKFast56/61
563  solutions.Clear();
564 
565  double trans[3];
566  trans[0] = pose_frame.p[0]; //-.18;
567  trans[1] = pose_frame.p[1];
568  trans[2] = pose_frame.p[2];
569 
570  KDL::Rotation mult;
571  KDL::Vector direction;
572 
573  switch (GetIkType())
574  {
575  case IKP_Transform6D:
576  case IKP_Translation3D:
577  // For **Transform6D**, eerot is 9 values for the 3x3 rotation matrix. For **Translation3D**, these are ignored.
578 
579  mult = pose_frame.M;
580 
581  double vals[9];
582  vals[0] = mult(0, 0);
583  vals[1] = mult(0, 1);
584  vals[2] = mult(0, 2);
585  vals[3] = mult(1, 0);
586  vals[4] = mult(1, 1);
587  vals[5] = mult(1, 2);
588  vals[6] = mult(2, 0);
589  vals[7] = mult(2, 1);
590  vals[8] = mult(2, 2);
591 
592  // IKFast56/61
593  ComputeIk(trans, vals, vfree.size() > 0 ? &vfree[0] : nullptr, solutions);
594  return solutions.GetNumSolutions();
595 
596  case IKP_Direction3D:
597  case IKP_Ray4D:
598  case IKP_TranslationDirection5D:
599  // For **Direction3D**, **Ray4D**, and **TranslationDirection5D**, the first 3 values represent the target
600  // direction.
601 
602  direction = pose_frame.M * KDL::Vector(0, 0, 1);
603  ComputeIk(trans, direction.data, vfree.size() > 0 ? &vfree[0] : nullptr, solutions);
604  return solutions.GetNumSolutions();
605 
606  case IKP_TranslationXAxisAngle4D:
607  case IKP_TranslationYAxisAngle4D:
608  case IKP_TranslationZAxisAngle4D:
609  // For *TranslationXAxisAngle4D*, *TranslationYAxisAngle4D*, *TranslationZAxisAngle4D* - end effector origin
610  // reaches desired 3D translation, manipulator direction makes a specific angle with x/y/z-axis (defined in the
611  // manipulator base linkâs coordinate system)
612  RCLCPP_ERROR(LOGGER, "IK for this IkParameterizationType not implemented yet.");
613  return 0;
614 
615  case IKP_TranslationLocalGlobal6D:
616  // For **TranslationLocalGlobal6D**, the diagonal elements ([0],[4],[8]) are the local translation inside the end
617  // effector coordinate system.
618  RCLCPP_ERROR(LOGGER, "IK for this IkParameterizationType not implemented yet.");
619  return 0;
620 
621  case IKP_Rotation3D:
622  case IKP_Lookat3D:
623  case IKP_TranslationXY2D:
624  case IKP_TranslationXYOrientation3D:
625  RCLCPP_ERROR(LOGGER, "IK for this IkParameterizationType not implemented yet.");
626  return 0;
627 
628  case IKP_TranslationXAxisAngleZNorm4D:
629  double roll, pitch, yaw;
630  // For **TranslationXAxisAngleZNorm4D** - end effector origin reaches desired 3D translation, manipulator
631  // direction needs to be orthogonal to z axis and be rotated at a certain angle starting from the x axis (defined
632  // in the manipulator base linkâs coordinate system)
633  pose_frame.M.GetRPY(roll, pitch, yaw);
634  ComputeIk(trans, &yaw, vfree.size() > 0 ? &vfree[0] : nullptr, solutions);
635  return solutions.GetNumSolutions();
636 
637  case IKP_TranslationYAxisAngleXNorm4D:
638  // For **TranslationYAxisAngleXNorm4D** - end effector origin reaches desired 3D translation, manipulator
639  // direction needs to be orthogonal to x axis and be rotated at a certain angle starting from the y axis (defined
640  // in the manipulator base linkâs coordinate system)
641  pose_frame.M.GetRPY(roll, pitch, yaw);
642  ComputeIk(trans, &roll, vfree.size() > 0 ? &vfree[0] : nullptr, solutions);
643  return solutions.GetNumSolutions();
644 
645  case IKP_TranslationZAxisAngleYNorm4D:
646  // For **TranslationZAxisAngleYNorm4D** - end effector origin reaches desired 3D translation, manipulator
647  // direction needs to be orthogonal to y axis and be rotated at a certain angle starting from the z axis (defined
