127 void Profiler::status(std::ostream& out, bool merge)
128 {
129  stop();
130  lock_.lock();
131  printOnDestroy_ = false;
132 
133  out << '\n';
134  out << " *** Profiling statistics. Total counted time : " << to_seconds(tinfo_.total) << " seconds\n";
135 
136  if (merge)
137  {
138  PerThread combined;
139  for (std::map<boost::thread::id, PerThread>::const_iterator it = data_.begin(); it != data_.end(); ++it)
140  {
141  for (const std::pair<const std::string, unsigned long int>& event : it->second.events)
142  combined.events[event.first] += event.second;
143  for (const std::pair<const std::string, AvgInfo>& iavg : it->second.avg)
144  {
145  combined.avg[iavg.first].total += iavg.second.total;
146  combined.avg[iavg.first].totalSqr += iavg.second.totalSqr;
147  combined.avg[iavg.first].parts += iavg.second.parts;
148  }
149  for (const std::pair<const std::string, TimeInfo>& itm : it->second.time)
150  {
151  TimeInfo& tc = combined.time[itm.first];
152  tc.total = tc.total + itm.second.total;
153  tc.parts = tc.parts + itm.second.parts;
154  if (tc.shortest > itm.second.shortest)
155  tc.shortest = itm.second.shortest;
156  if (tc.longest < itm.second.longest)
157  tc.longest = itm.second.longest;
158  }
159  }
160  printThreadInfo(out, combined);
161  }
162  else
163  for (std::map<boost::thread::id, PerThread>::const_iterator it = data_.begin(); it != data_.end(); ++it)
164  {
165  out << "Thread " << it->first << ":\n";
166  printThreadInfo(out, it->second);
167  }
168  lock_.unlock();
169 }

212 void Profiler::printThreadInfo(std::ostream& out, const PerThread& data)
213 {
214  double total = to_seconds(tinfo_.total);
215 
216  std::vector<DataIntVal> events;
217  for (const std::pair<const std::string, unsigned long>& event : data.events)
218  {
219  DataIntVal next = { event.first, event.second };
220  events.push_back(next);
221  }
222  std::sort(events.begin(), events.end(), SortIntByValue());
223  if (!events.empty())
224  out << "Events:\n";
225  for (const DataIntVal& event : events)
226  out << event.name_ << ": " << event.value_ << '\n';
227 
228  std::vector<DataDoubleVal> avg;
229  for (const std::pair<const std::string, AvgInfo>& ia : data.avg)
230  {
231  DataDoubleVal next = { ia.first, ia.second.total / (double)ia.second.parts };
232  avg.push_back(next);
233  }
234  std::sort(avg.begin(), avg.end(), SortDoubleByValue());
235  if (!avg.empty())
236  out << "Averages:\n";
237  for (const DataDoubleVal& average : avg)
238  {
239  const AvgInfo& a = data.avg.find(average.name_)->second;
240  out << average.name_ << ": " << average.value_ << " (stddev = "
241  << sqrt(fabs(a.totalSqr - (double)a.parts * average.value_ * average.value_) / ((double)a.parts - 1.)) << ")\n";
242  }
243 
244  std::vector<DataDoubleVal> time;
245 
246  for (const std::pair<const std::string, TimeInfo>& itm : data.time)
247  {
248  DataDoubleVal next = { itm.first, to_seconds(itm.second.total) };
249  time.push_back(next);
250  }
251 
252  std::sort(time.begin(), time.end(), SortDoubleByValue());
253  if (!time.empty())
254  out << "Blocks of time:\n";
255 
256  double unaccounted = total;
257  for (DataDoubleVal& time_block : time)
258  {
259  const TimeInfo& d = data.time.find(time_block.name_)->second;
260 
261  double t_s = to_seconds(d.shortest);
262  double t_l = to_seconds(d.longest);
263  out << time_block.name_ << ": " << time_block.value_ << "s (" << (100.0 * time_block.value_ / total) << "%), ["
264  << t_s << "s --> " << t_l << " s], " << d.parts << " parts";
265  if (d.parts > 0)
266  {
267  double pavg = to_seconds(d.total) / (double)d.parts;
268  out << ", " << pavg << " s on average";
269  if (pavg < 1.0)
270  out << " (" << 1.0 / pavg << " /s)";
271  }
272  out << '\n';
273  unaccounted -= time_block.value_;
274  }
275  // if we do not appear to have counted time multiple times, print the unaccounted time too
276  if (unaccounted >= 0.0)
277  {
278  out << "Unaccounted time : " << unaccounted;
279  if (total > 0.0)
280  out << " (" << (100.0 * unaccounted / total) << " %)";
281  out << '\n';
282  }
283 
284  out << '\n';
285 }
