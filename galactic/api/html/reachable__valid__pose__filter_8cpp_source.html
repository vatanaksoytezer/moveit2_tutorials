60 bool isStateCollisionFree(const planning_scene::PlanningScene* planning_scene,
61  const collision_detection::AllowedCollisionMatrix* collision_matrix, bool verbose,
62  const pick_place::ManipulationPlan* manipulation_plan, moveit::core::RobotState* state,
63  const moveit::core::JointModelGroup* group, const double* joint_group_variable_values)
64 {
65  collision_detection::CollisionRequest req;
66  req.verbose = verbose;
67  req.group_name = manipulation_plan->shared_data_->planning_group_->getName();
68 
69  state->setJointGroupPositions(group, joint_group_variable_values);
70 
71  if (manipulation_plan->approach_posture_.points.empty())
72  {
73  state->update();
74  collision_detection::CollisionResult res;
75  planning_scene->checkCollision(req, res, *state, *collision_matrix);
76  if (res.collision)
77  return false;
78  }
79  else
80  // apply approach posture for the end effector (we always apply it here since it could be the case the sampler
81  // changes this posture)
82  for (std::size_t i = 0; i < manipulation_plan->approach_posture_.points.size(); ++i)
83  {
84  state->setVariablePositions(manipulation_plan->approach_posture_.joint_names,
85  manipulation_plan->approach_posture_.points[i].positions);
86  state->update();
87  collision_detection::CollisionResult res;
88  planning_scene->checkCollision(req, res, *state, *collision_matrix);
89  if (res.collision)
90  return false;
91  }
92  return planning_scene->isStateFeasible(*state);
93 }

96 bool pick_place::ReachableAndValidPoseFilter::isEndEffectorFree(const ManipulationPlanPtr& plan,
97  moveit::core::RobotState& token_state) const
98 {
99  tf2::fromMsg(plan->goal_pose_.pose, plan->transformed_goal_pose_);
100  plan->transformed_goal_pose_ =
101  planning_scene_->getFrameTransform(token_state, plan->goal_pose_.header.frame_id) * plan->transformed_goal_pose_;
102  token_state.updateStateWithLinkAt(plan->shared_data_->ik_link_, plan->transformed_goal_pose_);
103  collision_detection::CollisionRequest req;
104  req.verbose = verbose_;
105  collision_detection::CollisionResult res;
106  req.group_name = plan->shared_data_->end_effector_group_->getName();
107  planning_scene_->checkCollision(req, res, token_state, *collision_matrix_);
108  return !res.collision;
109 }

111 bool pick_place::ReachableAndValidPoseFilter::evaluate(const ManipulationPlanPtr& plan) const
112 {
113  // initialize with scene state
114  moveit::core::RobotStatePtr token_state(new moveit::core::RobotState(planning_scene_->getCurrentState()));
115  if (isEndEffectorFree(plan, *token_state))
116  {
117  // update the goal pose message if anything has changed; this is because the name of the frame in the input goal
118  // pose
119  // can be that of objects in the collision world but most components are unaware of those transforms,
120  // so we convert to a frame that is certainly known
121  if (!moveit::core::Transforms::sameFrame(planning_scene_->getPlanningFrame(), plan->goal_pose_.header.frame_id))
122  {
123  plan->goal_pose_.pose = tf2::toMsg(plan->transformed_goal_pose_);
124  plan->goal_pose_.header.frame_id = planning_scene_->getPlanningFrame();
125  }
126 
127  // convert the pose we want to reach to a set of constraints
128  plan->goal_constraints_ =
129  kinematic_constraints::constructGoalConstraints(plan->shared_data_->ik_link_->getName(), plan->goal_pose_);
130 
131  const std::string& planning_group = plan->shared_data_->planning_group_->getName();
132 
133  // construct a sampler for the specified constraints; this can end up calling just IK, but it is more general
134  // and allows for robot-specific samplers, producing samples that also change the base position if needed, etc
135  plan->goal_sampler_ =
136  constraints_sampler_manager_->selectSampler(planning_scene_, planning_group, plan->goal_constraints_);
137  if (plan->goal_sampler_)
138  {
139  plan->goal_sampler_->setGroupStateValidityCallback(boost::bind(
140  &isStateCollisionFree, planning_scene_.get(), collision_matrix_.get(), verbose_, plan.get(), _1, _2, _3));
141  plan->goal_sampler_->setVerbose(verbose_);
142  if (plan->goal_sampler_->sample(*token_state, plan->shared_data_->max_goal_sampling_attempts_))
143  {
144  plan->possible_goal_states_.push_back(token_state);
145  return true;
146  }
147  else if (verbose_)
148  ROS_INFO_NAMED("manipulation", "Sampler failed to produce a state");
149  }
150  else
151  ROS_ERROR_THROTTLE_NAMED(1, "manipulation", "No sampler was constructed");
152  }
153  plan->error_code_.val = moveit_msgs::msg::MoveItErrorCodes::GOAL_IN_COLLISION;
154  return false;
155 }
