125 void RevoluteJointModel::interpolate(const double* from, const double* to, const double t, double* state) const
126 {
127  if (continuous_)
128  {
129  double diff = to[0] - from[0];
130  if (fabs(diff) <= boost::math::constants::pi<double>())
131  state[0] = from[0] + diff * t;
132  else
133  {
134  if (diff > 0.0)
135  diff = 2.0 * boost::math::constants::pi<double>() - diff;
136  else
137  diff = -2.0 * boost::math::constants::pi<double>() - diff;
138  state[0] = from[0] - diff * t;
139  // input states are within bounds, so the following check is sufficient
140  if (state[0] > boost::math::constants::pi<double>())
141  state[0] -= 2.0 * boost::math::constants::pi<double>();
142  else if (state[0] < -boost::math::constants::pi<double>())
143  state[0] += 2.0 * boost::math::constants::pi<double>();
144  }
145  }
146  else
147  state[0] = from[0] + (to[0] - from[0]) * t;
148 }

171 bool RevoluteJointModel::harmonizePosition(double* values, const JointModel::Bounds& other_bounds) const
172 {
173  bool modified = false;
174  if (*values < other_bounds[0].min_position_)
175  {
176  while (*values + 2 * M_PI <= other_bounds[0].max_position_)
177  {
178  *values += 2 * M_PI;
179  modified = true;
180  }
181  }
182  else if (*values > other_bounds[0].max_position_)
183  {
184  while (*values - 2 * M_PI >= other_bounds[0].min_position_)
185  {
186  *values -= 2 * M_PI;
187  modified = true;
188  }
189  }

193 bool RevoluteJointModel::enforcePositionBounds(double* values, const Bounds& bounds) const
194 {
195  if (continuous_)
196  {
197  double& v = values[0];
198  if (v <= -boost::math::constants::pi<double>() || v > boost::math::constants::pi<double>())
199  {
200  v = fmod(v, 2.0 * boost::math::constants::pi<double>());
201  if (v <= -boost::math::constants::pi<double>())
202  v += 2.0 * boost::math::constants::pi<double>();
203  else if (v > boost::math::constants::pi<double>())
204  v -= 2.0 * boost::math::constants::pi<double>();
205  return true;
206  }
207  }
208  else
209  {
210  if (values[0] < bounds[0].min_position_)
211  {
212  values[0] = bounds[0].min_position_;
213  return true;
214  }
215  else if (values[0] > bounds[0].max_position_)
216  {
217  values[0] = bounds[0].max_position_;
218  return true;
219  }
220  }
221  return false;
222 }

224 void RevoluteJointModel::computeTransform(const double* joint_values, Eigen::Isometry3d& transf) const
225 {
226  const double c = cos(joint_values[0]);
227  const double s = sin(joint_values[0]);
228  const double t = 1.0 - c;
229  const double txy = t * xy_;
230  const double txz = t * xz_;
231  const double tyz = t * yz_;
232 
233  const double zs = axis_.z() * s;
234  const double ys = axis_.y() * s;
235  const double xs = axis_.x() * s;
236 
237  // column major
238  double* d = transf.data();
239 
240  d[0] = t * x2_ + c;
241  d[1] = txy + zs;
242  d[2] = txz - ys;
243  d[3] = 0.0;
244 
245  d[4] = txy - zs;
246  d[5] = t * y2_ + c;
247  d[6] = tyz + xs;
248  d[7] = 0.0;
249 
250  d[8] = txz + ys;
251  d[9] = tyz - xs;
252  d[10] = t * z2_ + c;
253  d[11] = 0.0;
254 
255  d[12] = 0.0;
256  d[13] = 0.0;
257  d[14] = 0.0;

263 void RevoluteJointModel::computeVariablePositions(const Eigen::Isometry3d& transf, double* joint_values) const
264 {
265  ASSERT_ISOMETRY(transf) // unsanitized input, could contain a non-isometry
266  Eigen::Quaterniond q(transf.linear());
267  q.normalize();
268  size_t max_idx;
269  axis_.array().abs().maxCoeff(&max_idx);
270  joint_values[0] = 2. * atan2(q.vec()[max_idx] / axis_[max_idx], q.w());
