111  def move(self, position, wait=True):
112  """
113  @param position [float]: List of joint angles to achieve.
114  @param wait bool: If false, the commands gets operated asynchronously.
115  """
116  group = self._robot.get_default_owner_group(self.name())
117  if group is None:
118  raise MoveItCommanderException(
119  "There is no known group containing joint %s. Cannot move."
120  % self._name
121  )
122  gc = self._robot.get_group(group)
123  if gc is not None:
124  gc.set_joint_value_target(gc.get_current_joint_values())
125  gc.set_joint_value_target(self._name, position)
126  return gc.go(wait)
127  return False

167  def get_robot_markers(self, *args):
168  """Get a MarkerArray of the markers that make up this robot
169 
170  Usage:
171  (): gets all markers for current state
172  state (RobotState): gets markers for a particular state
173  values (dict): get markers with given values
174  values, links (dict, list): get markers with given values and these links
175  group (string): get all markers for a group
176  group, values (string, dict): get all markers for a group with desired values
177  """
178  mrkr = MarkerArray()
179  if not args:
180  conversions.msg_from_string(mrkr, self._r.get_robot_markers())
181  else:
182  if isinstance(args[0], RobotState):
183  msg_str = conversions.msg_to_string(args[0])
184  conversions.msg_from_string(mrkr, self._r.get_robot_markers(msg_str))
185  elif isinstance(args[0], dict):
186  conversions.msg_from_string(mrkr, self._r.get_robot_markers(*args))
187  elif isinstance(args[0], str):
188  conversions.msg_from_string(mrkr, self._r.get_group_markers(*args))
189  else:
190  raise MoveItCommanderException("Unexpected type")
191  return mrkr

297  def get_default_owner_group(self, joint_name):
298  """
299  Get the name of the smallest group (fewest joints) that includes
300  the joint name specified as argument.
301  """
302  if not joint_name in self._joint_owner_groups:
303  group = None
304  for g in self.get_group_names():
305  if joint_name in self.get_joint_names(g):
306  if group is None:
307  group = g
308  else:
309  if len(self.get_link_names(g)) < len(
310  self.get_link_names(group)
311  ):
312  group = g
313  self._joint_owner_groups[joint_name] = group
314  return self._joint_owner_groups[joint_name]
