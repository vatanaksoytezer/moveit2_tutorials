207 // ******************************************************************************************
208 void ROSControllersWidget::loadControllersTree()
209 {
210  // Disable Tree
211  controllers_tree_->setUpdatesEnabled(false); // prevent table from updating until we are completely done
212  controllers_tree_->setDisabled(true); // make sure we disable it so that the cellChanged event is not called
213  controllers_tree_->clear(); // reset the tree
214 
215  // Display all controllers by looping through them
216  for (ROSControlConfig& controller : config_data_->getROSControllers())
217  {
218  loadToControllersTree(controller);
219  }
220 
221  // Re-enable Tree
222  controllers_tree_->setUpdatesEnabled(true); // prevent table from updating until we are completely done
223  controllers_tree_->setDisabled(false); // make sure we disable it so that the cellChanged event is not called
224  current_edit_controller_.clear(); // no controller is being edited
225  alterTree("expand");
226 }

228 // ******************************************************************************************
229 // Displays data in the controller_config_ data structure into a QtTableWidget
230 // ******************************************************************************************
231 void ROSControllersWidget::loadToControllersTree(const moveit_setup_assistant::ROSControlConfig& controller_it)
232 {
233  // Fonts for tree
234  const QFont top_level_font(QFont().defaultFamily(), 11, QFont::Bold);
235  const QFont type_font(QFont().defaultFamily(), 11, QFont::Normal, QFont::StyleItalic);
236 
237  QTreeWidgetItem* controller;
238 
239  controller = new QTreeWidgetItem();
240 
241  // First column
242  controller->setText(0, controller_it.name_.c_str());
243  controller->setFont(0, top_level_font);
244  controller->setData(0, Qt::UserRole, QVariant::fromValue(0));
245 
246  // Second column
247  controller->setText(1, controller_it.type_.c_str());
248  controller->setFont(1, type_font);
249  controller->setData(1, Qt::UserRole, QVariant::fromValue(4));
250  controllers_tree_->addTopLevelItem(controller);
251 
252  if (!controller_it.joints_.empty())
253  {
254  // Joints --------------------------------------------------------------
255  QTreeWidgetItem* joints = new QTreeWidgetItem(controller);
256  joints->setText(0, "Joints");
257  joints->setFont(0, type_font);
258  joints->setData(0, Qt::UserRole, QVariant::fromValue(1));
259  controller->addChild(joints);
260 
261  // Loop through all available joints
262  for (const std::string& joint : controller_it.joints_)
263  {
264  QTreeWidgetItem* joint_item = new QTreeWidgetItem(joints);
265  joint_item->setData(0, Qt::UserRole, QVariant::fromValue(2));
266 
267  // Add to tree
268  joint_item->setText(0, joint.c_str());
269  joints->addChild(joint_item);
270  }
271  }
272 }

345 // ******************************************************************************************
346 // Delete a controller
347 // ******************************************************************************************
348 void ROSControllersWidget::deleteController()
349 {
350  std::string controller_name = current_edit_controller_;
351 
352  if (controller_name.empty())
353  {
354  QTreeWidgetItem* item = controllers_tree_->currentItem();
355  // Check that something was actually selected
356  if (item == nullptr)
357  return;
358 
359  // Get the user custom properties of the currently selected row
360  int type = item->data(0, Qt::UserRole).value<int>();
361  if (type == 0)
362  controller_name = item->text(0).toUtf8().constData();
363  }
364 
365  // Confirm user wants to delete controller
366  if (QMessageBox::question(
367  this, "Confirm Controller Deletion",
368  QString("Are you sure you want to delete the controller '").append(controller_name.c_str()).append(" ?"),
369  QMessageBox::Ok | QMessageBox::Cancel) == QMessageBox::Cancel)
370  {
371  return;
372  }
373  // Delete actual controller
374  if (config_data_->deleteROSController(controller_name))
375  {
376  ROS_INFO_STREAM_NAMED("Setup Assistant", "Controller " << controller_name << " deleted succefully");
377  }
378  else
379  {
380  ROS_WARN_STREAM_NAMED("Setup Assistant", "Couldn't delete Controller " << controller_name);
381  }
382 
383  current_edit_controller_.clear();
384 
385  // Switch to main screen
386  showMainScreen();
387 
388  // Reload main screen table
389  loadControllersTree();
390 }

487

491  if (link.empty())
492  {
493  continue;
494  }
495 
496  // Highlight link

648 // ******************************************************************************************
649 // Call when controller edit screen is done and needs to be saved
650 // ******************************************************************************************
651 bool ROSControllersWidget::saveControllerScreen()
652 {
653  // Get a reference to the supplied strings
654  const std::string& controller_name = controller_edit_widget_->getControllerName();
655  const std::string& controller_type = controller_edit_widget_->getControllerType();
656 
657  // Used for editing existing controllers
658  moveit_setup_assistant::ROSControlConfig* searched_controller = nullptr;
659 
660  std::smatch invalid_name_match;
661  std::regex invalid_reg_ex("[^a-z|^1-9|^_]");
662 
663  // Check that a valid controller name has been given
664  if (controller_name.empty() || std::regex_search(controller_name, invalid_name_match, invalid_reg_ex))
665  {
666  QMessageBox::warning(this, "Error Saving", "Invalid ROS controller name");
667  return false;
668  }
669 
670  // Check if this is an existing controller
671  if (!current_edit_controller_.empty())
672  {
