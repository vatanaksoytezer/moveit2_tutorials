58 bool RuckigSmoothing::applySmoothing(robot_trajectory::RobotTrajectory& trajectory,
59  const double max_velocity_scaling_factor,
60  const double max_acceleration_scaling_factor)
61 {
62  const moveit::core::JointModelGroup* group = trajectory.getGroup();
63  if (!group)
64  {
65  RCLCPP_ERROR(LOGGER, "It looks like the planner did not set the group the plan was computed for");
66  return false;
67  }
68 
69  const size_t num_waypoints = trajectory.getWayPointCount();
70  if (num_waypoints < 2)
71  {
72  RCLCPP_ERROR(LOGGER, "Trajectory does not have enough points to smooth with Ruckig");
73  return false;
74  }
75 
76  const size_t num_dof = group->getVariableCount();
77 
78  // This lib does not actually work properly when angles wrap around, so we need to unwind the path first
79  trajectory.unwind();
80 
81  // Instantiate the smoother
82  double timestep = trajectory.getAverageSegmentDuration();
83  std::unique_ptr<ruckig::Ruckig<0>> ruckig_ptr;
84  ruckig_ptr = std::make_unique<ruckig::Ruckig<0>>(num_dof, timestep);
85  ruckig::InputParameter<0> ruckig_input{ num_dof };
86  ruckig::OutputParameter<0> ruckig_output{ num_dof };
87 
88  // Initialize the smoother
89  const std::vector<int>& idx = group->getVariableIndexList();
90  initializeRuckigState(ruckig_input, ruckig_output, *trajectory.getFirstWayPointPtr(), num_dof, idx);
91 
92  // Kinematic limits (vel/accel/jerk)
93  const std::vector<std::string>& vars = group->getVariableNames();
94  const moveit::core::RobotModel& rmodel = group->getParentModel();
95  for (size_t i = 0; i < num_dof; ++i)
96  {
97  // TODO(andyz): read this from the joint group if/when jerk limits are added to the JointModel
98  ruckig_input.max_jerk.at(i) = DEFAULT_MAX_JERK;
99 
100  const moveit::core::VariableBounds& bounds = rmodel.getVariableBounds(vars.at(i));
101 
102  // This assumes min/max bounds are symmetric
103  if (bounds.velocity_bounded_)
104  {
105  ruckig_input.max_velocity.at(i) = max_velocity_scaling_factor * bounds.max_velocity_;
106  }
107  else
108  {
109  ruckig_input.max_velocity.at(i) = max_velocity_scaling_factor * DEFAULT_MAX_VELOCITY;
110  }
111  if (bounds.acceleration_bounded_)
112  {
113  ruckig_input.max_acceleration.at(i) = max_acceleration_scaling_factor * bounds.max_acceleration_;
114  }
115  else
116  {
117  ruckig_input.max_acceleration.at(i) = max_acceleration_scaling_factor * DEFAULT_MAX_ACCELERATION;
118  }
119  }
120 
121  ruckig::Result ruckig_result;
122  bool smoothing_complete = false;
123  double duration_extension_factor = 1;
124  while ((duration_extension_factor < MAX_DURATION_EXTENSION_FACTOR) && !smoothing_complete)
125  {
126  for (size_t waypoint_idx = 0; waypoint_idx < num_waypoints - 1; ++waypoint_idx)
127  {
128  moveit::core::RobotStatePtr next_waypoint = trajectory.getWayPointPtr(waypoint_idx + 1);
129 
130  getNextRuckigInput(ruckig_output, next_waypoint, num_dof, idx, ruckig_input);
131 
132  // Run Ruckig
133  ruckig_result = ruckig_ptr->update(ruckig_input, ruckig_output);
134 
135  // If the requested velocity is too great, a joint can actually "move backward" to give itself more time to
136  // accelerate to the target velocity. Iterate and decrease velocities until that behavior is gone.
137  bool backward_motion_detected = checkForLaggingMotion(num_dof, ruckig_input, ruckig_output);
138 
139  double velocity_magnitude = getTargetVelocityMagnitude(ruckig_input, num_dof);
140  while (backward_motion_detected && (velocity_magnitude > MINIMUM_VELOCITY_SEARCH_MAGNITUDE))
141  {
142  // Skip repeated waypoints with no change in position. Ruckig does not handle this well and there's really no
143  // need to smooth it Simply set it equal to the previous (identical) waypoint.
144  if (checkForIdenticalWaypoints(*trajectory.getWayPointPtr(waypoint_idx), *next_waypoint, trajectory.getGroup()))
145  {
146  *next_waypoint = trajectory.getWayPoint(waypoint_idx);
147  continue;
148  }
149 
150  // decrease target velocity
151  for (size_t joint = 0; joint < num_dof; ++joint)
152  {
153  ruckig_input.target_velocity.at(joint) *= 0.9;
154  // Propagate the change in velocity to acceleration, too.
155  // We don't change the position to ensure the exact target position is achieved.
156  ruckig_input.target_acceleration.at(joint) =
157  (ruckig_input.target_velocity.at(joint) - ruckig_output.new_velocity.at(joint)) / timestep;
158  }
159  velocity_magnitude = getTargetVelocityMagnitude(ruckig_input, num_dof);
160  // Run Ruckig
161  ruckig_result = ruckig_ptr->update(ruckig_input, ruckig_output);
162 
163  // check for backward motion
164  backward_motion_detected = checkForLaggingMotion(num_dof, ruckig_input, ruckig_output);
165  }
166 
167  // Overwrite pos/vel/accel of the target waypoint
168  for (size_t joint = 0; joint < num_dof; ++joint)
169  {
170  next_waypoint->setVariablePosition(idx.at(joint), ruckig_output.new_position.at(joint));
171  next_waypoint->setVariableVelocity(idx.at(joint), ruckig_output.new_velocity.at(joint));
172  next_waypoint->setVariableAcceleration(idx.at(joint), ruckig_output.new_acceleration.at(joint));
173  }
174  next_waypoint->update();
175  }
176 
177  // If ruckig failed, the duration of the seed trajectory likely wasn't long enough.
178  // Try duration extension several times.
179  // TODO: see issue 767. (https://github.com/ros-planning/moveit2/issues/767)
180  if (ruckig_result == ruckig::Result::Working)
181  {
182  smoothing_complete = true;
183  }
184  else
185  {
186  // If Ruckig failed, it's likely because the original seed trajectory did not have a long enough duration when
187  // jerk is taken into account. Extend the duration and try again.
188  initializeRuckigState(ruckig_input, ruckig_output, *trajectory.getFirstWayPointPtr(), num_dof, idx);
189  duration_extension_factor *= DURATION_EXTENSION_FRACTION;
190  for (size_t waypoint_idx = 1; waypoint_idx < num_waypoints; ++waypoint_idx)
191  {
192  trajectory.setWayPointDurationFromPrevious(
193  waypoint_idx, DURATION_EXTENSION_FRACTION * trajectory.getWayPointDurationFromPrevious(waypoint_idx));
194  // TODO(andyz): re-calculate waypoint velocity and acceleration here?
195  }
196 
197  timestep = trajectory.getAverageSegmentDuration();
198  ruckig_ptr = std::make_unique<ruckig::Ruckig<0>>(num_dof, timestep);
199  }
200  }
201 
202  // Either of these results is acceptable.
203  // Working means smoothing worked well but the final target position wasn't exactly achieved (I think) -- Andy Z.
204  if ((ruckig_result != ruckig::Result::Working) && (ruckig_result != ruckig::Result::Finished))
205  {
206  RCLCPP_ERROR_STREAM(LOGGER, "Ruckig trajectory smoothing failed. Ruckig error: " << ruckig_result);
207  return false;
208  }
209 
210  return true;
211 }

213 void RuckigSmoothing::initializeRuckigState(ruckig::InputParameter<0>& ruckig_input,
214  ruckig::OutputParameter<0>& ruckig_output,
215  const moveit::core::RobotState& first_waypoint, size_t num_dof,
216  const std::vector<int>& idx)
217 {
218  std::vector<double> current_positions_vector(num_dof);
219  std::vector<double> current_velocities_vector(num_dof);
220  std::vector<double> current_accelerations_vector(num_dof);
221 
222  for (size_t i = 0; i < num_dof; ++i)
223  {
224  current_positions_vector.at(i) = first_waypoint.getVariablePosition(idx.at(i));
225  current_velocities_vector.at(i) = first_waypoint.getVariableVelocity(idx.at(i));
226  current_accelerations_vector.at(i) = first_waypoint.getVariableAcceleration(idx.at(i));
227  }
228  std::copy_n(current_positions_vector.begin(), num_dof, ruckig_input.current_position.begin());
229  std::copy_n(current_velocities_vector.begin(), num_dof, ruckig_input.current_velocity.begin());
230  std::copy_n(current_accelerations_vector.begin(), num_dof, ruckig_input.current_acceleration.begin());
231  // Initialize output data struct
232  ruckig_output.new_position = ruckig_input.current_position;
233  ruckig_output.new_velocity = ruckig_input.current_velocity;
234  ruckig_output.new_acceleration = ruckig_input.current_acceleration;
235 }

237 bool RuckigSmoothing::checkForIdenticalWaypoints(const moveit::core::RobotState& prev_waypoint,
238  const moveit::core::RobotState& next_waypoint,
239  const moveit::core::JointModelGroup* joint_group)
240 {
241  double magnitude_position_difference = prev_waypoint.distance(next_waypoint, joint_group);
242 
243  return (magnitude_position_difference <= IDENTICAL_POSITION_EPSILON);
244 }

246 double RuckigSmoothing::getTargetVelocityMagnitude(const ruckig::InputParameter<0>& ruckig_input, size_t num_dof)
247 {
248  double vel_magnitude = 0;
249  for (size_t joint = 0; joint < num_dof; ++joint)
250  {
251  vel_magnitude += ruckig_input.target_velocity.at(joint) * ruckig_input.target_velocity.at(joint);
252  }
253  return sqrt(vel_magnitude);
254 }

256 bool RuckigSmoothing::checkForLaggingMotion(const size_t num_dof, const ruckig::InputParameter<0>& ruckig_input,
257  const ruckig::OutputParameter<0>& ruckig_output)
258 {
259  // Check for backward motion of any joint
260  for (size_t joint = 0; joint < num_dof; ++joint)
261  {
262  // This indicates the jerk-limited output lags the target output
263  if ((ruckig_output.new_velocity.at(joint) / ruckig_input.target_velocity.at(joint)) < 1)
264  {
265  return true;
266  }
267  }
268  return false;
269 }

271 void RuckigSmoothing::getNextRuckigInput(const ruckig::OutputParameter<0>& ruckig_output,
272  const moveit::core::RobotStatePtr& next_waypoint, size_t num_dof,
273  const std::vector<int>& idx, ruckig::InputParameter<0>& ruckig_input)
274 {
275  // TODO(andyz): https://github.com/ros-planning/moveit2/issues/766
276  // ruckig_output.pass_to_input(ruckig_input);
277 
278  for (size_t joint = 0; joint < num_dof; ++joint)
279  {
280  // Feed output from the previous timestep back as input
281  ruckig_input.current_position.at(joint) = ruckig_output.new_position.at(joint);
282  ruckig_input.current_velocity.at(joint) = ruckig_output.new_velocity.at(joint);
283  ruckig_input.current_acceleration.at(joint) = ruckig_output.new_acceleration.at(joint);
284 
285  // Target state is the next waypoint
286  ruckig_input.target_position.at(joint) = next_waypoint->getVariablePosition(idx.at(joint));
287  ruckig_input.target_velocity.at(joint) = next_waypoint->getVariableVelocity(idx.at(joint));
288  ruckig_input.target_acceleration.at(joint) = next_waypoint->getVariableAcceleration(idx.at(joint));
289  }
