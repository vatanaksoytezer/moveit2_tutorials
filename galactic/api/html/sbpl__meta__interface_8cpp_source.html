47 bool SBPLMetaInterface::solve(const planning_scene::PlanningSceneConstPtr& planning_scene,

49  moveit_msgs::srv::GetMotionPlan::Response& res)
50 {
51  first_ok_ = false;
52  first_done_ = false;
53  second_ok_ = false;
54  second_done_ = false;
55 
56  PlanningParameters param_bfs;
57  param_bfs.use_bfs_ = true;
58  PlanningParameters param_no_bfs;
59  param_no_bfs.use_bfs_ = false;
60  moveit_msgs::srv::GetMotionPlan::Response res1, res2;
61  boost::thread thread1(boost::bind(&SBPLMetaInterface::runSolver, this, true, boost::cref(planning_scene),
62  boost::cref(req), boost::ref(res1), param_bfs));
63  boost::thread thread2(boost::bind(&SBPLMetaInterface::runSolver, this, false, boost::cref(planning_scene),
64  boost::cref(req), boost::ref(res2), param_no_bfs));
65  boost::mutex::scoped_lock lock(planner_done_mutex_);
66  planner_done_condition_.wait(lock);
67 
68  if (first_done_)
69  {
70  std::cerr << "FIRST DONE" << '\n';
71  if (first_ok_)
72  {
73  std::cerr << "First ok, interrupting second" << '\n';
74  if (!second_done_)
75  {
76  thread2.interrupt();
77  thread2.join();
78  }
79  }
80  else
81  {
82  if (!second_done_)
83  {
84  planner_done_condition_.wait(lock);
85  }
86  }
87  }
88  if (second_done_)
89  {
90  std::cerr << "Second done" << '\n';
91  if (second_ok_)
92  {
93  std::cerr << "Second ok, interrupting first" << '\n';
94  if (!first_done_)
95  {
96  thread1.interrupt();
97  thread1.join();
98  }
99  }
100  else
101  {
102  if (!first_done_)
103  {
104  planner_done_condition_.wait(lock);
105  }
106  }
107  }
108 
109  if (!first_ok_ && !second_ok_)
110  {
111  std::cerr << "Both planners failed" << '\n';
112  res = res1;
113  return false;
114  }
115  if (!first_ok_ && second_ok_)
116  {
117  std::cerr << "Sbpl interface no bfs reports time "
118  << sbpl_interface_second_->getLastPlanningStatistics().total_planning_time_ << '\n';
119  last_planning_statistics_ = sbpl_interface_second_->getLastPlanningStatistics();
120  res = res2;
121  return true;
122  }
123  else if (first_ok_ && !second_ok_)
124  {
125  std::cerr << "Sbpl interface bfs reports time "
126  << sbpl_interface_first_->getLastPlanningStatistics().total_planning_time_ << '\n';
127  last_planning_statistics_ = sbpl_interface_first_->getLastPlanningStatistics();
128  res = res1;
129  return true;
130  }
131  std::cerr << "Sbpl interface bfs reports time "
132  << sbpl_interface_first_->getLastPlanningStatistics().total_planning_time_ << '\n';
133  std::cerr << "Sbpl interface no bfs reports time "
134  << sbpl_interface_second_->getLastPlanningStatistics().total_planning_time_ << '\n';
135  if (sbpl_interface_first_->getLastPlanningStatistics().total_planning_time_ <
136  sbpl_interface_second_->getLastPlanningStatistics().total_planning_time_)
137  {
138  last_planning_statistics_ = sbpl_interface_first_->getLastPlanningStatistics();
139  res = res1;
140  }

144  res = res2;
