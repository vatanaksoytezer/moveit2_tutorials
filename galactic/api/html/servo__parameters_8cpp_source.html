55 rcl_interfaces::msg::SetParametersResult
56 ServoParameters::CallbackHandler::setParametersCallback(const std::vector<rclcpp::Parameter>& parameters)
57 {
58  const std::lock_guard<std::mutex> guard(mutex_);
59  auto result = rcl_interfaces::msg::SetParametersResult();
60  result.successful = true;
61 
62  for (const auto& parameter : parameters)
63  {
64  auto search = set_parameter_callbacks_.find(parameter.get_name());
65  if (search != set_parameter_callbacks_.end())
66  {
67  RCLCPP_INFO_STREAM(LOGGER, "setParametersCallback - "
68  << parameter.get_name() << "<" << parameter.get_type_name()
69  << ">: " << rclcpp::to_string(parameter.get_parameter_value()));
70  for (const auto& callback : search->second)
71  {
72  result = callback(parameter);
73 
74  if (!result.successful)
75  {
76  // Handle automatic parameter set failure
77  return result;
78  }
79  }
80  }
81  }
82 
83  return result;
84 }

320 std::optional<ServoParameters> ServoParameters::validate(ServoParameters parameters)
321 {
322  // Begin input checking
323  if (parameters.publish_period <= 0.)
324  {
325  RCLCPP_WARN(LOGGER, "Parameter 'publish_period' should be "
326  "greater than zero. Check yaml file.");
327  return std::nullopt;
328  }
329  if (parameters.num_outgoing_halt_msgs_to_publish < 0)
330  {
331  RCLCPP_WARN(LOGGER, "Parameter 'num_outgoing_halt_msgs_to_publish' should be greater than zero. Check yaml file.");
332  return std::nullopt;
333  }
334  if (parameters.hard_stop_singularity_threshold <= parameters.lower_singularity_threshold)
335  {
336  RCLCPP_WARN(LOGGER, "Parameter 'hard_stop_singularity_threshold' "
337  "should be greater than 'lower_singularity_threshold.' "
338  "Check yaml file.");
339  return std::nullopt;
340  }
341  if ((parameters.hard_stop_singularity_threshold <= 0.) || (parameters.lower_singularity_threshold <= 0.))
342  {
343  RCLCPP_WARN(LOGGER, "Parameters 'hard_stop_singularity_threshold' "
344  "and 'lower_singularity_threshold' should be "
345  "greater than zero. Check yaml file.");
346  return std::nullopt;
347  }
348  if (parameters.smoothing_filter_plugin_name.empty())
349  {
350  RCLCPP_WARN(LOGGER, "A smoothing plugin is required.");
351  return std::nullopt;
352  }
353  if (parameters.joint_limit_margin < 0.)
354  {
355  RCLCPP_WARN(LOGGER, "Parameter 'joint_limit_margin' should usually be greater than or equal to zero, "
356  "although negative values can be used if the specified joint limits are actually soft. "
357  "Check yaml file.");
358  }
359  if (parameters.command_in_type != "unitless" && parameters.command_in_type != "speed_units")
360  {
361  RCLCPP_WARN(LOGGER, "command_in_type should be 'unitless' or "
362  "'speed_units'. Check yaml file.");
363  return std::nullopt;
364  }
365  if (parameters.command_out_type != "trajectory_msgs/JointTrajectory" &&
366  parameters.command_out_type != "std_msgs/Float64MultiArray")
367  {
368  RCLCPP_WARN(LOGGER, "Parameter command_out_type should be "
369  "'trajectory_msgs/JointTrajectory' or "
370  "'std_msgs/Float64MultiArray'. Check yaml file.");
371  return std::nullopt;
372  }
373  if (!parameters.publish_joint_positions && !parameters.publish_joint_velocities &&
374  !parameters.publish_joint_accelerations)
375  {
376  RCLCPP_WARN(LOGGER, "At least one of publish_joint_positions / "
377  "publish_joint_velocities / "
378  "publish_joint_accelerations must be true. Check "
379  "yaml file.");
380  return std::nullopt;
381  }
382  if ((parameters.command_out_type == "std_msgs/Float64MultiArray") && parameters.publish_joint_positions &&
383  parameters.publish_joint_velocities)
384  {
385  RCLCPP_WARN(LOGGER, "When publishing a std_msgs/Float64MultiArray, "
386  "you must select positions OR velocities.");
387  return std::nullopt;
388  }
389  // Collision checking
390  if (parameters.self_collision_proximity_threshold <= 0.)
391  {
392  RCLCPP_WARN(LOGGER, "Parameter 'self_collision_proximity_threshold' should be "
393  "greater than zero. Check yaml file.");
394  return std::nullopt;
395  }
396  if (parameters.scene_collision_proximity_threshold <= 0.)
397  {
398  RCLCPP_WARN(LOGGER, "Parameter 'scene_collision_proximity_threshold' should be "
399  "greater than zero. Check yaml file.");
400  return std::nullopt;
401  }
402  if (parameters.scene_collision_proximity_threshold < parameters.self_collision_proximity_threshold)
403  {
404  RCLCPP_WARN(LOGGER, "Parameter 'self_collision_proximity_threshold' should probably be less "
405  "than or equal to 'scene_collision_proximity_threshold'. Check yaml file.");
406  }
407  if (parameters.collision_check_rate <= 0)
408  {
409  RCLCPP_WARN(LOGGER, "Parameter 'collision_check_rate' should be "
410  "greater than zero. Check yaml file.");
411  return std::nullopt;
412  }
413  return parameters;
414 }

416 ServoParameters::SharedConstPtr ServoParameters::makeServoParameters(const rclcpp::Node::SharedPtr& node,
417  std::string ns, /* = "moveit_servo"*/
418  bool dynamic_parameters /* = true */)
419 {
420  auto node_parameters = node->get_node_parameters_interface();
421 
422  // Get the parameters
423  declare(ns, node_parameters);
424  auto parameters = validate(get(ns, node_parameters));
425 
426  if (parameters)
427  {
428  auto parameters_ptr = std::make_shared<ServoParameters>(parameters.value());
429  parameters_ptr->callback_handler_ = std::make_shared<ServoParameters::CallbackHandler>();
430 
431  // register parameter change callback
432  if (dynamic_parameters)
433  {
434  using std::placeholders::_1;
435  parameters_ptr->callback_handler_->on_set_parameters_callback_handler_ =
436  node->add_on_set_parameters_callback(std::bind(&ServoParameters::CallbackHandler::setParametersCallback,
437  parameters_ptr->callback_handler_.get(), _1));
438  }
439 
440  return parameters_ptr;
441  }
442  return nullptr;
443 }
