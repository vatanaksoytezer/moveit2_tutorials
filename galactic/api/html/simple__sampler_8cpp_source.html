88 moveit_msgs::action::LocalPlanner::Feedback
89 SimpleSampler::getLocalTrajectory(const moveit::core::RobotState& current_state,
90  robot_trajectory::RobotTrajectory& local_trajectory)
91 {
92  // Delete previous local trajectory
93  local_trajectory.clear();
94 
95  // Determine current local trajectory based on configured behavior
96  if (pass_through_)
97  {
98  // Use reference_trajectory as local trajectory
99  local_trajectory.append(*reference_trajectory_, 0.0, 0, reference_trajectory_->getWayPointCount());
100  }
101  else
102  {
103  // Get next desired robot state
104  moveit::core::RobotState next_desired_goal_state = reference_trajectory_->getWayPoint(next_waypoint_index_);
105 
106  // Check if state is reached
107  if (next_desired_goal_state.distance(current_state, joint_group_) <= WAYPOINT_RADIAN_TOLERANCE)
108  {
109  // Update index (and thus desired robot state)
110  next_waypoint_index_ = std::min(next_waypoint_index_ + 1, reference_trajectory_->getWayPointCount() - 1);
111  }
112 
113  // Construct local trajectory containing the next global trajectory waypoint
114  local_trajectory.addSuffixWayPoint(reference_trajectory_->getWayPoint(next_waypoint_index_),
115  reference_trajectory_->getWayPointDurationFromPrevious(next_waypoint_index_));
116  }
117 
118  // Return empty feedback
119  return feedback_;
120 }
