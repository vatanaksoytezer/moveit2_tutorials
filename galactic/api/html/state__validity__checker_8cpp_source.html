77 bool StateValidityChecker::isValid(const ompl::base::State* state, bool verbose) const
78 {
79  assert(state != nullptr);
80  // Use cached validity if it is available
81  if (state->as<ModelBasedStateSpace::StateType>()->isValidityKnown())
82  {
83  return state->as<ModelBasedStateSpace::StateType>()->isMarkedValid();
84  }
85 
86  if (!si_->satisfiesBounds(state))
87  {
88  if (verbose)
89  {
90  RCLCPP_INFO(LOGGER, "State outside bounds");
91  }
92  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markInvalid();
93  return false;
94  }
95 
96  moveit::core::RobotState* robot_state = tss_.getStateStorage();
97  planning_context_->getOMPLStateSpace()->copyToRobotState(*robot_state, state);
98 
99  // check path constraints
100  const kinematic_constraints::KinematicConstraintSetPtr& kset = planning_context_->getPathConstraints();
101  if (kset && !kset->decide(*robot_state, verbose).satisfied)
102  {
103  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markInvalid();
104  return false;
105  }
106 
107  // check feasibility
108  if (!planning_context_->getPlanningScene()->isStateFeasible(*robot_state, verbose))
109  {
110  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markInvalid();
111  return false;
112  }
113 
114  // check collision avoidance
115  collision_detection::CollisionResult res;
116  planning_context_->getPlanningScene()->checkCollision(
117  verbose ? collision_request_simple_verbose_ : collision_request_simple_, res, *robot_state);
118  if (!res.collision)
119  {
120  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markValid();
121  }
122  else
123  {
124  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markInvalid();
125  }
126  return !res.collision;
127 }

129 bool StateValidityChecker::isValid(const ompl::base::State* state, double& dist, bool verbose) const
130 {
131  assert(state != nullptr);
132  // Use cached validity and distance if they are available
133  if (state->as<ModelBasedStateSpace::StateType>()->isValidityKnown() &&
134  state->as<ModelBasedStateSpace::StateType>()->isGoalDistanceKnown())
135  {
136  dist = state->as<ModelBasedStateSpace::StateType>()->distance;
137  return state->as<ModelBasedStateSpace::StateType>()->isMarkedValid();
138  }
139 
140  if (!si_->satisfiesBounds(state))
141  {
142  if (verbose)
143  {
144  RCLCPP_INFO(LOGGER, "State outside bounds");
145  }
146  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markInvalid(0.0);
147  return false;
148  }
149 
150  moveit::core::RobotState* robot_state = tss_.getStateStorage();
151  planning_context_->getOMPLStateSpace()->copyToRobotState(*robot_state, state);
152 
153  // check path constraints
154  const kinematic_constraints::KinematicConstraintSetPtr& kset = planning_context_->getPathConstraints();
155  if (kset)
156  {
157  kinematic_constraints::ConstraintEvaluationResult cer = kset->decide(*robot_state, verbose);
158  if (!cer.satisfied)
159  {
160  dist = cer.distance;
161  const_cast<ob::State*>(state)->as<ModelBasedStateSpace::StateType>()->markInvalid(dist);
162  return false;
163  }
164  }
165 
166  // check feasibility
167  if (!planning_context_->getPlanningScene()->isStateFeasible(*robot_state, verbose))
168  {
169  dist = 0.0;
170  return false;
171  }
172 
173  // check collision avoidance
174  collision_detection::CollisionResult res;
175  planning_context_->getPlanningScene()->checkCollision(
176  verbose ? collision_request_with_distance_verbose_ : collision_request_with_distance_, res, *robot_state);
177  dist = res.distance;
178  return !res.collision;
179 }

181 double StateValidityChecker::cost(const ompl::base::State* state) const
182 {
183  assert(state != nullptr);
184  double cost = 0.0;
185 
186  moveit::core::RobotState* robot_state = tss_.getStateStorage();
187  planning_context_->getOMPLStateSpace()->copyToRobotState(*robot_state, state);
188 
189  // Calculates cost from a summation of distance to obstacles times the size of the obstacle
190  collision_detection::CollisionResult res;
191  planning_context_->getPlanningScene()->checkCollision(collision_request_with_cost_, res, *robot_state);
192 
193  for (const collision_detection::CostSource& cost_source : res.cost_sources)
194  {
195  cost += cost_source.cost * cost_source.getVolume();
196  }
197 
198  return cost;
199 }

201 double StateValidityChecker::clearance(const ompl::base::State* state) const
202 {
203  assert(state != nullptr);
204  moveit::core::RobotState* robot_state = tss_.getStateStorage();
205  planning_context_->getOMPLStateSpace()->copyToRobotState(*robot_state, state);
206 
207  collision_detection::CollisionResult res;
208  planning_context_->getPlanningScene()->checkCollision(collision_request_with_distance_, res, *robot_state);
209  return res.collision ? 0.0 : (res.distance < 0.0 ? std::numeric_limits<double>::infinity() : res.distance);
210 }
