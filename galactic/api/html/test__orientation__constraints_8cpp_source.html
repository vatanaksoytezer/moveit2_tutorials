134 inline Eigen::Quaterniond xyz_intrinsix_to_quat(double rx, double ry, double rz)
135 {
136  return Eigen::AngleAxisd(rx, Eigen::Vector3d::UnitX()) * Eigen::AngleAxisd(ry, Eigen::Vector3d::UnitY()) *
137  Eigen::AngleAxisd(rz, Eigen::Vector3d::UnitZ());
138 }
139 
141 inline Eigen::Quaterniond rotation_vector_to_quat(double rx, double ry, double rz)
142 {
143  Eigen::Vector3d v{ rx, ry, rz };
144  Eigen::Matrix3d m{ Eigen::AngleAxisd(v.norm(), v.normalized()) };
145  return Eigen::Quaterniond{ m };
146 }
