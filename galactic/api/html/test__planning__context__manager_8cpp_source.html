82  void testSimpleRequest(const std::vector<double>& start, const std::vector<double>& goal)
83  {
84  SCOPED_TRACE("testSimpleRequest");
85 
86  // create all the test specific input necessary to make the getPlanningContext call possible
87  planning_interface::PlannerConfigurationSettings pconfig_settings;
88  pconfig_settings.group = group_name_;
89  pconfig_settings.name = group_name_;
90  pconfig_settings.config = { { "enforce_joint_model_state_space", "0" } };
91 
92  planning_interface::PlannerConfigurationMap pconfig_map{ { pconfig_settings.name, pconfig_settings } };
93  moveit_msgs::msg::MoveItErrorCodes error_code;
94  planning_interface::MotionPlanRequest request = createRequest(start, goal);
95 
96  // setup the planning context manager
97  ompl_interface::PlanningContextManager pcm(robot_model_, constraint_sampler_manager_);
98  pcm.setPlannerConfigurations(pconfig_map);
99 
100  // see if it returns the expected planning context
101  auto pc = pcm.getPlanningContext(planning_scene_, request, error_code, node_, false);
102 
103  // the planning context should have a simple setup created
104  EXPECT_NE(pc->getOMPLSimpleSetup(), nullptr);
105 
106  // the OMPL state space in the planning context should be of type JointModelStateSpace
107  EXPECT_NE(dynamic_cast<ompl_interface::JointModelStateSpace*>(pc->getOMPLStateSpace().get()), nullptr);
108 
109  // there did not magically appear path constraints in the planning context
110  EXPECT_TRUE(pc->getPathConstraints()->empty());
111 
112  // solve the planning problem
113  planning_interface::MotionPlanDetailedResponse res;
114  ASSERT_TRUE(pc->solve(res));
115  }

117  void testPathConstraints(const std::vector<double>& start, const std::vector<double>& goal)
118  {
119  SCOPED_TRACE("testPathConstraints");
120 
121  // create all the test specific input necessary to make the getPlanningContext call possible
122  planning_interface::PlannerConfigurationSettings pconfig_settings;
123  pconfig_settings.group = group_name_;
124  pconfig_settings.name = group_name_;
125  pconfig_settings.config = { { "enforce_joint_model_state_space", "0" } };
126 
127  planning_interface::PlannerConfigurationMap pconfig_map{ { pconfig_settings.name, pconfig_settings } };
128  moveit_msgs::msg::MoveItErrorCodes error_code;
129  planning_interface::MotionPlanRequest request = createRequest(start, goal);
130 
131  // give it some more time, as rejection sampling of the path constraints occasionally takes some time
132  request.allowed_planning_time = 10.0;
133 
134  // create path constraints around start state, to make sure they are satisfied
135  robot_state_->setJointGroupPositions(joint_model_group_, start);
136  Eigen::Isometry3d ee_pose = robot_state_->getGlobalLinkTransform(ee_link_name_);
137  geometry_msgs::msg::Quaternion ee_orientation = tf2::toMsg(Eigen::Quaterniond(ee_pose.rotation()));
138 
139  // setup the planning context manager
140  ompl_interface::PlanningContextManager pcm(robot_model_, constraint_sampler_manager_);
141  pcm.setPlannerConfigurations(pconfig_map);
142 
143  // ORIENTATION CONSTRAINTS
144  // ***********************
145  request.path_constraints.orientation_constraints.push_back(createOrientationConstraint(ee_orientation));
146 
147  // See if the planning context manager returns the expected planning context
148  auto pc = pcm.getPlanningContext(planning_scene_, request, error_code, node_, false);
149 
150  EXPECT_NE(pc->getOMPLSimpleSetup(), nullptr);
151 
152  // As the joint_model_group_ has no IK solver initialized, we expect a joint model state space
153  EXPECT_NE(dynamic_cast<ompl_interface::JointModelStateSpace*>(pc->getOMPLStateSpace().get()), nullptr);
154 
155  planning_interface::MotionPlanDetailedResponse response;
156  ASSERT_TRUE(pc->solve(response));
157 
158  // Are the path constraints created in the planning context?
159  auto path_constraints = pc->getPathConstraints();
160  EXPECT_FALSE(path_constraints->empty());
161  EXPECT_EQ(path_constraints->getOrientationConstraints().size(), 1u);
162  EXPECT_TRUE(path_constraints->getPositionConstraints().empty());
163  EXPECT_TRUE(path_constraints->getJointConstraints().empty());
164  EXPECT_TRUE(path_constraints->getVisibilityConstraints().empty());
165 
166  // Check if all the states in the solution satisfy the path constraints.
167  // A detailed response returns 3 solutions: the ompl solution, the simplified solution and the interpolated
168  // solution. We test all of them here.
169  for (const robot_trajectory::RobotTrajectoryPtr& trajectory : response.trajectory_)
170  {
171  for (std::size_t pt_index = 0; pt_index < trajectory->getWayPointCount(); ++pt_index)
172  {
173  EXPECT_TRUE(path_constraints->decide(trajectory->getWayPoint(pt_index)).satisfied);
174  }
175  }
176 
177  // POSITION CONSTRAINTS
178  // ***********************
179  request.path_constraints.orientation_constraints.clear();
180  request.path_constraints.position_constraints.push_back(createPositionConstraint(
181  { ee_pose.translation().x(), ee_pose.translation().y(), ee_pose.translation().z() }, { 0.1, 0.1, 0.1 }));
182 
183  // See if the planning context manager returns the expected planning context
184  pc = pcm.getPlanningContext(planning_scene_, request, error_code, node_, false);
185 
186  EXPECT_NE(pc->getOMPLSimpleSetup(), nullptr);
187 
188  // As the joint_model_group_ has no IK solver initialized, we expect a joint model state space
189  EXPECT_NE(dynamic_cast<ompl_interface::JointModelStateSpace*>(pc->getOMPLStateSpace().get()), nullptr);
190 
191  // Create a new response, because the solve method does not clear the given respone
192  planning_interface::MotionPlanDetailedResponse response2;
193  ASSERT_TRUE(pc->solve(response2));
194 
195  // Are the path constraints created in the planning context?
196  path_constraints = pc->getPathConstraints();
197  EXPECT_FALSE(path_constraints->empty());
198  EXPECT_EQ(path_constraints->getPositionConstraints().size(), 1u);
199  EXPECT_TRUE(path_constraints->getOrientationConstraints().empty());
200 
201  // Check if all the states in the solution satisfy the path constraints.
202  // A detailed response returns 3 solutions: the ompl solution, the simplified solution and the interpolated
203  // solution. We test all of them here.
204  for (const robot_trajectory::RobotTrajectoryPtr& trajectory : response2.trajectory_)
205  {
206  for (std::size_t pt_index = 0; pt_index < trajectory->getWayPointCount(); ++pt_index)
207  {
208  EXPECT_TRUE(path_constraints->decide(trajectory->getWayPoint(pt_index)).satisfied);
209  }
210  }
211  }

222  planning_interface::MotionPlanRequest createRequest(const std::vector<double>& start,
223  const std::vector<double>& goal) const
224  {
225  planning_interface::MotionPlanRequest request;
226  request.group_name = group_name_;
227  request.allowed_planning_time = 5.0;
228 
229  // fill out start state in request
230  moveit::core::RobotState start_state(robot_model_);
231  start_state.setToDefaultValues();
232  start_state.setJointGroupPositions(joint_model_group_, start);
233  moveit::core::robotStateToRobotStateMsg(start_state, request.start_state);
234 
235  // fill out goal state in request
236  moveit::core::RobotState goal_state(robot_model_);
237  goal_state.setToDefaultValues();
238  goal_state.setJointGroupPositions(joint_model_group_, goal);
239 
240  moveit_msgs::msg::Constraints joint_goal =
241  kinematic_constraints::constructGoalConstraints(goal_state, joint_model_group_, 0.001);
242 
243  request.goal_constraints.push_back(joint_goal);
244 
245  return request;
246  }

249  moveit_msgs::msg::PositionConstraint createPositionConstraint(std::array<double, 3> position,
250  std::array<double, 3> dimensions)
251  {
252  shape_msgs::msg::SolidPrimitive box;
253  box.type = shape_msgs::msg::SolidPrimitive::BOX;
254  box.dimensions.resize(3);
255  box.dimensions[shape_msgs::msg::SolidPrimitive::BOX_X] = dimensions[0];
256  box.dimensions[shape_msgs::msg::SolidPrimitive::BOX_Y] = dimensions[1];
257  box.dimensions[shape_msgs::msg::SolidPrimitive::BOX_Z] = dimensions[2];
258 
259  geometry_msgs::msg::Pose box_pose;
260  box_pose.position.x = position[0];
261  box_pose.position.y = position[1];
262  box_pose.position.z = position[2];
263  box_pose.orientation.w = 1.0;
264 
265  moveit_msgs::msg::PositionConstraint pc;
266  pc.header.frame_id = base_link_name_;
267  pc.link_name = ee_link_name_;
268  pc.constraint_region.primitives.push_back(box);
269  pc.constraint_region.primitive_poses.push_back(box_pose);
270 
271  return pc;
272  }
