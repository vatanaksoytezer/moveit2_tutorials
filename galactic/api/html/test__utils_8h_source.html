104 inline sensor_msgs::msg::JointState generateJointState(const std::vector<double>& pos, const std::vector<double>& vel,
105  const std::string& joint_prefix = testutils::JOINT_NAME_PREFIX)
106 {
107  sensor_msgs::msg::JointState state;
108  auto posit = pos.cbegin();
109  size_t i = 0;
110 
111  while (posit != pos.cend())
112  {
113  state.name.push_back(testutils::getJointName(i + 1, joint_prefix));
114  state.position.push_back(*posit);
115 
116  i++;
117  posit++;
118  }
119  for (const auto& one_vel : vel)
120  {
121  state.velocity.push_back(one_vel);
122  }
123  return state;
124 }

132 inline moveit_msgs::msg::Constraints
133 generateJointConstraint(const std::vector<double>& pos_list,
134  const std::string& joint_prefix = testutils::JOINT_NAME_PREFIX)
135 {
136  moveit_msgs::msg::Constraints gc;
137 
138  auto pos_it = pos_list.begin();
139 
140  for (size_t i = 0; i < pos_list.size(); ++i)
141  {
142  moveit_msgs::msg::JointConstraint jc;
143  jc.joint_name = testutils::getJointName(i + 1, joint_prefix);
144  jc.position = *pos_it;
145  gc.joint_constraints.push_back(jc);
146 
147  ++pos_it;
148  }
149 
150  return gc;
151 }

382 inline geometry_msgs::msg::Quaternion fromEuler(double a, double b, double c)
383 {
384  tf2::Vector3 qvz(0., 0., 1.);
385  tf2::Vector3 qvy(0., 1., 0.);
386  tf2::Quaternion q1(qvz, a);
387 
388  q1 = q1 * tf2::Quaternion(qvy, b);
389  q1 = q1 * tf2::Quaternion(qvz, c);
390 
391  return tf2::toMsg(q1);
392 }
