206 struct TestAction
207 {
208  collision_detection::World::Object obj_;
209  collision_detection::World::Action action_;
210  int cnt_;
211  TestAction() : obj_(""), cnt_(0)
212  {
213  reset();
214  }
215  void reset()
216  {
217  obj_.id_ = "";
218  obj_.shapes_.clear();
219  obj_.shape_poses_.clear();
220  action_ = 0x7f;
221  }
222 };

225 static void TrackChangesNotify(TestAction* ta, const collision_detection::World::ObjectConstPtr& obj,
226  collision_detection::World::Action action)
227 {
228  ta->obj_ = *obj;
229  ta->action_ = action;
230  ta->cnt_++;

368 TEST(World, ObjectPoseAndSubframes)
369 {
370  collision_detection::World world;
371 
372  TestAction ta;
373  collision_detection::World::ObserverHandle observer_ta;
374  observer_ta = world.addObserver(boost::bind(TrackChangesNotify, &ta, _1, _2));
375 
376  // Create shapes
377  shapes::ShapePtr ball(new shapes::Sphere(1.0));
378  shapes::ShapePtr box(new shapes::Box(1, 1, 1));
379  shapes::ShapePtr cyl(new shapes::Cylinder(0.5, 3)); // radius, length
380 
381  // Confirm that setting object pose creates an object
382  world.setObjectPose("mix1", Eigen::Isometry3d::Identity());
383 
384  EXPECT_EQ(1, ta.cnt_);
385  EXPECT_EQ("mix1", ta.obj_.id_);
386  EXPECT_EQ(collision_detection::World::CREATE, ta.action_);
387 
388  // Move multi-shape objects, use object pose, use subframes
389  world.addToObject("mix1", box, Eigen::Isometry3d::Identity());
390  world.addToObject("mix1", cyl, Eigen::Isometry3d(Eigen::Translation3d(0, 0, 2)));
391 
392  moveit::core::FixedTransformsMap subframes;
393  subframes["frame1"] = Eigen::Isometry3d(Eigen::Translation3d(0, 0, 2));
394  subframes["frame2"] = Eigen::Isometry3d(Eigen::Translation3d(0, 1, 0));
395  world.setSubframesOfObject("mix1", subframes);
396 
397  // Check subframes and shape poses
398  bool found_ok, found_bad;
399  Eigen::Isometry3d pose = world.getTransform("mix1/frame1", found_ok);
400  EXPECT_TRUE(found_ok);
401  EXPECT_EQ(2.0, pose(2, 3)); // check translation.z
402 
403  pose = world.getTransform("mix1/frame2", found_ok);
404  EXPECT_TRUE(found_ok);
405  EXPECT_EQ(1.0, pose(1, 3)); // check translation.y
406  EXPECT_EQ(0.0, pose(2, 3)); // z
407 
408  pose = world.getTransform("mix1/frame3", found_bad);
409  EXPECT_FALSE(found_bad);
410 
411  // Set new object pose, check that all shapes and subframes moved
412  world.setObjectPose("mix1", Eigen::Isometry3d(Eigen::Translation3d(0, 0, 1)));
413 
414  pose = world.getTransform("mix1/frame1", found_ok);
415  EXPECT_TRUE(found_ok);
416  EXPECT_EQ(3.0, pose(2, 3)); // z
417 
418  pose = world.getTransform("mix1/frame2", found_ok);
419  EXPECT_TRUE(found_ok);
420  EXPECT_EQ(1.0, pose(1, 3)); // y
421  EXPECT_EQ(1.0, pose(2, 3)); // z
422 
423  pose = world.getGlobalShapeTransform("mix1", 0);
424  EXPECT_TRUE(found_ok);
425  EXPECT_EQ(1.0, pose(2, 3)); // z
426 
427  collision_detection::World::ObjectConstPtr obj = world.getObject("mix1");
428  EXPECT_EQ(0.0, obj->shape_poses_[0](2, 3)); // Internal shape poses do *not* change
429  EXPECT_EQ(2.0, obj->shape_poses_[1](2, 3));
430 
431  // Shift object, check that object pose changed
432  world.moveObject("mix1", Eigen::Isometry3d(Eigen::Translation3d(0, 0, 1)));
433 
434  pose = world.getTransform("mix1", found_ok);
435  EXPECT_TRUE(found_ok);
436  EXPECT_EQ(2.0, pose(2, 3)); // z
437 
438  pose = world.getTransform("mix1/frame1", found_ok);
439  EXPECT_TRUE(found_ok);
440  EXPECT_EQ(4.0, pose(2, 3)); // z
441 
442  EXPECT_EQ(0.0, obj->shape_poses_[0](2, 3)); // Internal shape poses should still be constant
443  EXPECT_EQ(2.0, obj->shape_poses_[1](2, 3));
444 
445  // Move object absolute, check object pose
446  world.setObjectPose("mix1", Eigen::Isometry3d(Eigen::Translation3d(0, 0, 1)));
447  pose = world.getTransform("mix1", found_ok);
448  EXPECT_TRUE(found_ok);
449  EXPECT_EQ(1.0, pose(2, 3)); // z
450 }
