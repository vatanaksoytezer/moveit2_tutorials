49 class TrajOptPlannerManager : public planning_interface::PlannerManager
50 {
51 public:
52  TrajOptPlannerManager() : planning_interface::PlannerManager()
53  {
54  }
55 
56  bool initialize(const moveit::core::RobotModelConstPtr& model, const std::string& ns) override
57  {
58  ROS_INFO(" ======================================= initialize gets called");
59 
60  if (!ns.empty())
61  nh_ = ros::NodeHandle(ns);
62  std::string trajopt_ns = ns.empty() ? "trajopt" : ns + "/trajopt";
63 
64  for (const std::string& gpName : model->getJointModelGroupNames())
65  {
66  ROS_INFO(" ======================================= group name: %s, robot model: %s", gpName.c_str(),
67  model->getName().c_str());
68  planning_contexts_[gpName] =
69  TrajOptPlanningContextPtr(new TrajOptPlanningContext("trajopt_planning_context", gpName, model));
70  }
71 
72  return true;
73  }
74 
75  bool canServiceRequest(const moveit_msgs::MotionPlanRequest& req) const override
76  {
77  return req.trajectory_constraints.constraints.empty();
78  }
79 
80  std::string getDescription() const override
81  {
82  return "TrajOpt";
83  }
84 
85  void getPlanningAlgorithms(std::vector<std::string>& algs) const override
86  {
87  algs.clear();
88  algs.push_back("trajopt");
89  }
90 
91  planning_interface::PlanningContextPtr getPlanningContext(const planning_scene::PlanningSceneConstPtr& planning_scene,
92  const planning_interface::MotionPlanRequest& req,
93  moveit_msgs::MoveItErrorCodes& error_code) const override
94  {
95  ROS_INFO(" ======================================= getPlanningContext() is called ");
96  error_code.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
97 
98  if (req.group_name.empty())
99  {
100  ROS_ERROR("No group specified to plan for");
101  error_code.val = moveit_msgs::MoveItErrorCodes::INVALID_GROUP_NAME;
102  return planning_interface::PlanningContextPtr();
103  }
104 
105  if (!planning_scene)
106  {
107  ROS_ERROR("No planning scene supplied as input");
108  error_code.val = moveit_msgs::MoveItErrorCodes::FAILURE;
109  return planning_interface::PlanningContextPtr();
110  }
111 
112  // create PlanningScene using hybrid collision detector
113  planning_scene::PlanningScenePtr ps = planning_scene->diff();
114 
115  // set FCL for the collision
116  ps->allocateCollisionDetector(collision_detection::CollisionDetectorAllocatorFCL::create());
117 
118  // retrieve and configure existing context
119  const TrajOptPlanningContextPtr& context = planning_contexts_.at(req.group_name);
120 
121  ROS_INFO(" ======================================= context is made ");
122 
123  context->setPlanningScene(ps);
124  context->setMotionPlanRequest(req);
125 
126  error_code.val = moveit_msgs::MoveItErrorCodes::SUCCESS;
127 
128  return context;
129  }
130 
131 private:
132  ros::NodeHandle nh_;
133 
134 protected:
135  std::map<std::string, TrajOptPlanningContextPtr> planning_contexts_;
136 };
