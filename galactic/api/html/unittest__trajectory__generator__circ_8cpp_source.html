133  void checkCircResult(const planning_interface::MotionPlanRequest& req,
134  const planning_interface::MotionPlanResponse& res)
135  {
136  moveit_msgs::msg::MotionPlanResponse res_msg;
137  res.getMessage(res_msg);
138  EXPECT_TRUE(testutils::isGoalReached(res.trajectory_->getFirstWayPointPtr()->getRobotModel(),
139  res_msg.trajectory.joint_trajectory, req, other_tolerance_));
140 
141  EXPECT_TRUE(
142  testutils::checkJointTrajectory(res_msg.trajectory.joint_trajectory, planner_limits_.getJointLimitContainer()));
143 
144  EXPECT_EQ(req.path_constraints.position_constraints.size(), 1u);
145  EXPECT_EQ(req.path_constraints.position_constraints.at(0).constraint_region.primitive_poses.size(), 1u);
146 
147  // Check that all point have the equal distance to the center
148  Eigen::Vector3d circ_center;
149  getCircCenter(req, res, circ_center);
150 
151  for (std::size_t i = 0; i < res.trajectory_->getWayPointCount(); ++i)
152  {
153  Eigen::Affine3d waypoint_pose = res.trajectory_->getWayPointPtr(i)->getFrameTransform(target_link_);
154  EXPECT_NEAR(
155  (res.trajectory_->getFirstWayPointPtr()->getFrameTransform(target_link_).translation() - circ_center).norm(),
156  (circ_center - waypoint_pose.translation()).norm(), cartesian_position_tolerance_);
157  }
158 
159  // check translational and rotational paths
160  ASSERT_TRUE(testutils::checkCartesianTranslationalPath(res.trajectory_, target_link_, acceleration_tolerance_));
161  ASSERT_TRUE(testutils::checkCartesianRotationalPath(res.trajectory_, target_link_, angular_acc_tolerance_,
162  rot_axis_norm_tolerance_));
163 
164  for (size_t idx = 0; idx < res.trajectory_->getLastWayPointPtr()->getVariableCount(); ++idx)
165  {
166  EXPECT_NEAR(0.0, res.trajectory_->getLastWayPointPtr()->getVariableVelocity(idx), other_tolerance_);
167  EXPECT_NEAR(0.0, res.trajectory_->getLastWayPointPtr()->getVariableAcceleration(idx), other_tolerance_);
168  }
169  }

171  void getCircCenter(const planning_interface::MotionPlanRequest& req,
172  const planning_interface::MotionPlanResponse& res, Eigen::Vector3d& circ_center)
173  {
174  if (req.path_constraints.name == "center")
175  {
176  tf2::fromMsg(req.path_constraints.position_constraints.at(0).constraint_region.primitive_poses.at(0).position,
177  circ_center);
178  }
179  else if (req.path_constraints.name == "interim")
180  {
181  Eigen::Vector3d interim;
182  tf2::fromMsg(req.path_constraints.position_constraints.at(0).constraint_region.primitive_poses.at(0).position,
183  interim);
184  Eigen::Vector3d start = res.trajectory_->getFirstWayPointPtr()->getFrameTransform(target_link_).translation();
185  Eigen::Vector3d goal = res.trajectory_->getLastWayPointPtr()->getFrameTransform(target_link_).translation();
186 
187  const Eigen::Vector3d t = interim - start;
188  const Eigen::Vector3d u = goal - start;
189  const Eigen::Vector3d v = goal - interim;
190 
191  const Eigen::Vector3d w = t.cross(u);
192 
193  ASSERT_GT(w.norm(), 1e-8) << "Circle center not well defined for given start, interim and goal.";
194 
195  circ_center = start + (u * t.dot(t) * u.dot(v) - t * u.dot(u) * t.dot(v)) * 0.5 / pow(w.norm(), 2);
196  }
197  }
