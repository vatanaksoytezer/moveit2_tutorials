86 bool getState(const std::shared_ptr<moveit_msgs::srv::GetRobotStateFromWarehouse::Request> request,
87  std::shared_ptr<moveit_msgs::srv::GetRobotStateFromWarehouse::Response> response,
88  moveit_warehouse::RobotStateStorage* rs)
89 {
90  if (!rs->hasRobotState(request->name, request->robot))
91  {
92  RCLCPP_ERROR_STREAM(LOGGER, "No state called '" << request->name << "' for robot '" << request->robot << "'.");
93  moveit_msgs::msg::RobotState dummy;
94  response->state = dummy;
95  return false;
96  }
97  moveit_warehouse::RobotStateWithMetadata state_buffer;
98  rs->getRobotState(state_buffer, request->name, request->robot);
99  response->state = static_cast<const moveit_msgs::msg::RobotState&>(*state_buffer);
100  return true;
101 }
