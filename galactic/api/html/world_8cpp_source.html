71 void World::addToObject(const std::string& object_id, const Eigen::Isometry3d& pose,
72  const std::vector<shapes::ShapeConstPtr>& shapes,
73  const EigenSTL::vector_Isometry3d& shape_poses)
74 {
75  if (shapes.size() != shape_poses.size())
76  {
77  RCLCPP_ERROR(LOGGER,
78  "Number of shapes and number of poses do not match. Not adding this object to collision world.");
79  return;
80  }
81 
82  if (shapes.empty())
83  return;
84 
85  int action = ADD_SHAPE;
86 
87  ObjectPtr& obj = objects_[object_id];
88  if (!obj)
89  {
90  obj = std::make_shared<Object>(object_id);
91  action |= CREATE;
92  obj->pose_ = pose;
93  }
94  else
95  ensureUnique(obj);
96 
97  for (std::size_t i = 0; i < shapes.size(); ++i)
98  addToObjectInternal(obj, shapes[i], shape_poses[i]);
99 
100  notify(obj, Action(action));
101 }

131 bool World::knowsTransform(const std::string& name) const
132 {
133  // Check object names first
134  std::map<std::string, ObjectPtr>::const_iterator it = objects_.find(name);
135  if (it != objects_.end())
136  return true;
137  else // Then objects' subframes
138  {
139  for (const std::pair<const std::string, ObjectPtr>& object : objects_)
140  {
141  // if "object name/" matches start of object_id, we found the matching object
142  if (boost::starts_with(name, object.first) && name[object.first.length()] == '/')
143  {
144  return object.second->global_subframe_poses_.find(name.substr(object.first.length() + 1)) !=
145  object.second->global_subframe_poses_.end();
146  }
147  }
148  }
149  return false;
150 }

161 const Eigen::Isometry3d& World::getTransform(const std::string& name, bool& frame_found) const
162 {
163  // assume found
164  frame_found = true;
165 
166  std::map<std::string, ObjectPtr>::const_iterator it = objects_.find(name);
167  if (it != objects_.end())
168  {
169  return it->second->pose_;
170  }
171  else // Search within subframes
172  {
173  for (const std::pair<const std::string, ObjectPtr>& object : objects_)
174  {
175  // if "object name/" matches start of object_id, we found the matching object
176  if (boost::starts_with(name, object.first) && name[object.first.length()] == '/')
177  {
178  auto it = object.second->global_subframe_poses_.find(name.substr(object.first.length() + 1));
179  if (it != object.second->global_subframe_poses_.end())
180  {
181  return it->second;
182  }
183  }
184  }
185  }
186 
187  // we need a persisting isometry for the API
188  static const Eigen::Isometry3d IDENTITY_TRANSFORM = Eigen::Isometry3d::Identity();
189  frame_found = false;
190  return IDENTITY_TRANSFORM;
191 }

223 bool World::moveShapeInObject(const std::string& object_id, const shapes::ShapeConstPtr& shape,
224  const Eigen::Isometry3d& shape_pose)
225 {
226  auto it = objects_.find(object_id);
227  if (it != objects_.end())
228  {
229  unsigned int n = it->second->shapes_.size();
230  for (unsigned int i = 0; i < n; ++i)
231  if (it->second->shapes_[i] == shape)
232  {
233  ensureUnique(it->second);
234  ASSERT_ISOMETRY(shape_pose) // unsanitized input, could contain a non-isometry
235  it->second->shape_poses_[i] = shape_pose;
236  it->second->global_shape_poses_[i] = it->second->pose_ * shape_pose;
237 
238  notify(it->second, MOVE_SHAPE);
239  return true;
240  }
241  }
242  return false;
243 }

245 bool World::moveObject(const std::string& object_id, const Eigen::Isometry3d& transform)
246 {
247  auto it = objects_.find(object_id);
248  if (it == objects_.end())
249  return false;
250  if (transform.isApprox(Eigen::Isometry3d::Identity()))
251  return true; // object already at correct location
252 
253  ASSERT_ISOMETRY(transform) // unsanitized input, could contain a non-isometry
254  return setObjectPose(object_id, transform * it->second->pose_);
255 }

257 bool World::setObjectPose(const std::string& object_id, const Eigen::Isometry3d& pose)
258 {
259  ASSERT_ISOMETRY(pose); // unsanitized input, could contain a non-isometry
260  ObjectPtr& obj = objects_[object_id];
261  int action;
262  if (!obj)
263  {
264  obj = std::make_shared<Object>(object_id);
265  action = CREATE;
266  }
267  else
268  {
269  ensureUnique(obj);
270  action = obj->shapes_.empty() ? 0 : MOVE_SHAPE;
271  }
272 
273  obj->pose_ = pose;
274  updateGlobalPosesInternal(obj);
275  notify(obj, Action(action));
276  return true;

279 bool World::removeShapeFromObject(const std::string& object_id, const shapes::ShapeConstPtr& shape)
280 {
281  auto it = objects_.find(object_id);
282  if (it != objects_.end())
283  {
284  unsigned int n = it->second->shapes_.size();
285  for (unsigned int i = 0; i < n; ++i)
286  if (it->second->shapes_[i] == shape)
287  {
288  ensureUnique(it->second);
289  it->second->shapes_.erase(it->second->shapes_.begin() + i);
290  it->second->shape_poses_.erase(it->second->shape_poses_.begin() + i);
291  it->second->global_shape_poses_.erase(it->second->global_shape_poses_.begin() + i);
292 
293  if (it->second->shapes_.empty())
294  {
295  notify(it->second, DESTROY);
296  objects_.erase(it);
297  }
298  else
299  {
300  notify(it->second, REMOVE_SHAPE);
301  }
302  return true;
303  }
304  }
305  return false;
306 }

326 bool World::setSubframesOfObject(const std::string& object_id, const moveit::core::FixedTransformsMap& subframe_poses)
327 {
328  auto obj_pair = objects_.find(object_id);
329  if (obj_pair == objects_.end())
330  {
331  return false;
332  }
333  for (const auto& t : subframe_poses)
334  {
335  ASSERT_ISOMETRY(t.second) // unsanitized input, could contain a non-isometry
336  }
337  obj_pair->second->subframe_poses_ = subframe_poses;
338  obj_pair->second->global_subframe_poses_ = subframe_poses;
339  updateGlobalPosesInternal(obj_pair->second, false, true);
340  return true;
341 }

343 void World::updateGlobalPosesInternal(ObjectPtr& obj, bool update_shape_poses, bool update_subframe_poses)
344 {
345  // Update global shape poses
346  if (update_shape_poses)
347  for (unsigned int i = 0; i < obj->global_shape_poses_.size(); ++i)
348  obj->global_shape_poses_[i] = obj->pose_ * obj->shape_poses_[i];
349 
350  // Update global subframe poses
351  if (update_subframe_poses)
352  {
353  for (auto it_global_pose = obj->global_subframe_poses_.begin(), it_local_pose = obj->subframe_poses_.begin(),
354  end_poses = obj->global_subframe_poses_.end();
355  it_global_pose != end_poses; ++it_global_pose, ++it_local_pose)
356  {
357  it_global_pose->second = obj->pose_ * it_local_pose->second;
358  }
359  }
360 }
