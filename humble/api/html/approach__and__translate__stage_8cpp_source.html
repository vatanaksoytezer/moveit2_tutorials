63 bool isStateCollisionFree(const planning_scene::PlanningScene* planning_scene,
64  const collision_detection::AllowedCollisionMatrix* collision_matrix, bool verbose,
65  const trajectory_msgs::JointTrajectory* grasp_posture, moveit::core::RobotState* state,
66  const moveit::core::JointModelGroup* group, const double* joint_group_variable_values)
67 {
68  state->setJointGroupPositions(group, joint_group_variable_values);
69 
70  collision_detection::CollisionRequest req;
71  req.verbose = verbose;
72  req.group_name = group->getName();
73 
74  if (!grasp_posture->joint_names.empty())
75  {
76  // apply the grasp posture for the end effector (we always apply it here since it could be the case the sampler
77  // changes this posture)
78  for (std::size_t i = 0; i < grasp_posture->points.size(); ++i)
79  {
80  state->setVariablePositions(grasp_posture->joint_names, grasp_posture->points[i].positions);
81  collision_detection::CollisionResult res;
82  planning_scene->checkCollision(req, res, *state, *collision_matrix);
83  if (res.collision)
84  return false;
85  }
86  }
87  else
88  {
89  collision_detection::CollisionResult res;
90  planning_scene->checkCollision(req, res, *state, *collision_matrix);
91  if (res.collision)
92  return false;
93  }
94  return planning_scene->isStateFeasible(*state);
95 }

97 bool samplePossibleGoalStates(const ManipulationPlanPtr& plan, const moveit::core::RobotState& reference_state,
98  double min_distance, unsigned int attempts)
99 {
100  // initialize with scene state
101  auto token_state = std::make_shared<moveit::core::RobotState>(reference_state);
102  for (unsigned int j = 0; j < attempts; ++j)
103  {
104  double min_d = std::numeric_limits<double>::infinity();
105 
106  // Samples given the constraints, populating the joint state group
107  if (plan->goal_sampler_->sample(*token_state, plan->shared_data_->max_goal_sampling_attempts_))
108  {
109  // Check if this new sampled state is closest we've found so far
110  for (std::size_t i = 0; i < plan->possible_goal_states_.size(); ++i)
111  {
112  double d = plan->possible_goal_states_[i]->distance(*token_state, plan->shared_data_->planning_group_);
113  if (d < min_d)
114  min_d = d;
115  }
116  if (min_d >= min_distance)
117  {
118  plan->possible_goal_states_.push_back(token_state);
119  return true;
120  }
121  }
122  }
123  return false;
124 }

126 // This function is called during trajectory execution, after the gripper is closed, to attach the currently gripped
127 // object
128 bool executeAttachObject(const ManipulationPlanSharedDataConstPtr& shared_plan_data,
129  const trajectory_msgs::JointTrajectory& detach_posture,
130  const plan_execution::ExecutableMotionPlan& motion_plan)
131 {
132  if (shared_plan_data->diff_attached_object_.object.id.empty())
133  {
134  // the user did not provide an object to attach, so just return successfully
135  return true;
136  }
137 
138  ROS_DEBUG_NAMED("manipulation", "Applying attached object diff to maintained planning scene (attaching/detaching "
139  "object to end effector)");
140  bool ok = false;
141  {
142  planning_scene_monitor::LockedPlanningSceneRW ps(motion_plan.planning_scene_monitor_);
143  moveit_msgs::msg::AttachedCollisionObject msg = shared_plan_data->diff_attached_object_;
144  // remember the configuration of the gripper before the grasp;
145  // this configuration will be set again when releasing the object
146  msg.detach_posture = detach_posture;
147  ok = ps->processAttachedCollisionObjectMsg(msg);
148  }
149  motion_plan.planning_scene_monitor_->triggerSceneUpdateEvent(
150  (planning_scene_monitor::PlanningSceneMonitor::
151  SceneUpdateType)(planning_scene_monitor::PlanningSceneMonitor::UPDATE_GEOMETRY +
152  planning_scene_monitor::PlanningSceneMonitor::UPDATE_STATE));
153  return ok;
154 }

156 // Add the close end effector trajectory to the overall plan (after the approach trajectory, before the retreat)
157 void addGripperTrajectory(const ManipulationPlanPtr& plan,
158  const collision_detection::AllowedCollisionMatrixConstPtr& collision_matrix,
159  const std::string& name)
160 {
161  // Check if a "closed" end effector configuration was specified
162  if (!plan->retreat_posture_.joint_names.empty())
163  {
164  moveit::core::RobotStatePtr ee_closed_state(
165  new moveit::core::RobotState(plan->trajectories_.back().trajectory_->getLastWayPoint()));
166 
167  auto ee_closed_traj = std::make_shared<robot_trajectory::RobotTrajectory>(
168  ee_closed_state->getRobotModel(), plan->shared_data_->end_effector_group_->getName()));
169  ee_closed_traj->setRobotTrajectoryMsg(*ee_closed_state, plan->retreat_posture_);
170  // If user has defined a time for it's gripper movement time, don't add the
171  // DEFAULT_GRASP_POSTURE_COMPLETION_DURATION
172  if (!plan->retreat_posture_.points.empty() &&
173  plan->retreat_posture_.points.back().time_from_start > ros::Duration(0.0))
174  {
175  ee_closed_traj->addPrefixWayPoint(ee_closed_state, 0.0);
176  }
177  else
178  { // Do what was done before
179  ROS_INFO_STREAM("Adding default duration of " << PickPlace::DEFAULT_GRASP_POSTURE_COMPLETION_DURATION
180  << " seconds to the grasp closure time. Assign time_from_start to "
181  << "your trajectory to avoid this.");
182  ee_closed_traj->addPrefixWayPoint(ee_closed_state, PickPlace::DEFAULT_GRASP_POSTURE_COMPLETION_DURATION);
183  }
184 
185  plan_execution::ExecutableTrajectory et(ee_closed_traj, name);
186 
187  // Add a callback to attach the object to the EE after closing the gripper
188  et.effect_on_success_ = [plan](const plan_execution::ExecutableMotionPlan* motion_plan) {
189  return executeAttachObject(plan->shared_data_, plan->approach_posture_, *motion_plan);
190  };
191  et.allowed_collision_matrix_ = collision_matrix;
192  plan->trajectories_.push_back(et);
193  }
194  else
195  {
196  ROS_WARN_NAMED("manipulation", "No joint states of grasp postures have been defined in the pick place action.");
197  }
198 }

202 bool ApproachAndTranslateStage::evaluate(const ManipulationPlanPtr& plan) const
203 {
204  const moveit::core::JointModelGroup* jmg = plan->shared_data_->planning_group_;
205  // compute what the maximum distance reported between any two states in the planning group could be, and keep 1% of
206  // that;
207  // this is the minimum distance between sampled goal states
208  const double min_distance = 0.01 * jmg->getMaximumExtent();
209 
210  // convert approach direction and retreat direction to Eigen structures
211  Eigen::Vector3d approach_direction, retreat_direction;
212  tf2::fromMsg(plan->approach_.direction.vector, approach_direction);
213  tf2::fromMsg(plan->retreat_.direction.vector, retreat_direction);
214 
215  // if translation vectors are specified in the frame of the ik link name, then we assume the frame is local;
216  // otherwise, the frame is global
217  bool approach_direction_is_global_frame = !moveit::core::Transforms::sameFrame(
218  plan->approach_.direction.header.frame_id, plan->shared_data_->ik_link_->getName());
219  bool retreat_direction_is_global_frame = !moveit::core::Transforms::sameFrame(
220  plan->retreat_.direction.header.frame_id, plan->shared_data_->ik_link_->getName());
221 
222  // transform the input vectors in accordance to frame specified in the header;
223  if (approach_direction_is_global_frame)
224  // getFrameTransform() returns a valid isometry by contract
225  approach_direction =
226  planning_scene_->getFrameTransform(plan->approach_.direction.header.frame_id).linear() * approach_direction;
227  if (retreat_direction_is_global_frame)
228  // getFrameTransform() returns a valid isometry by contract
229  retreat_direction =
230  planning_scene_->getFrameTransform(plan->retreat_.direction.header.frame_id).linear() * retreat_direction;
231 
232  // state validity checking during the approach must ensure that the gripper posture is that for pre-grasping
233  moveit::core::GroupStateValidityCallbackFn approach_valid_callback =
234  [scene = planning_scene_.get(), acm = collision_matrix_.get(), verbose = this->verbose_,
235  approach_posture = &plan->approach_posture_](moveit::core::RobotState* robot_state,
236  const moveit::core::JointModelGroup* joint_group,
237  const double* joint_group_variable_values) {
238  return isStateCollisionFree(scene, acm, verbose, approach_posture, robot_state, joint_group,
239  joint_group_variable_values);
240  };
241  plan->goal_sampler_->setVerbose(verbose_);
242  std::size_t attempted_possible_goal_states = 0;
243  do // continuously sample possible goal states
244  {
245  for (std::size_t i = attempted_possible_goal_states; i < plan->possible_goal_states_.size() && !signal_stop_;
246  ++i, ++attempted_possible_goal_states)
247  {
248  // if we are trying to get as close as possible to the goal (maximum one meter)
249  if (plan->shared_data_->minimize_object_distance_)
250  {
251  static const double MAX_CLOSE_UP_DIST = 1.0;
252  auto close_up_state = std::make_shared<moveit::core::RobotState>(*plan->possible_goal_states_[i]);
253  std::vector<moveit::core::RobotStatePtr> close_up_states;
254  double d_close_up = moveit::core::CartesianInterpolator::computeCartesianPath(
255  close_up_state.get(), plan->shared_data_->planning_group_, close_up_states, plan->shared_data_->ik_link_,
256  approach_direction, approach_direction_is_global_frame, MAX_CLOSE_UP_DIST,
257  moveit::core::MaxEEFStep(max_step_), moveit::core::JumpThreshold(jump_factor_), approach_valid_callback);
258  // if progress towards the object was made, update the desired goal state
259  if (d_close_up > 0.0 && close_up_states.size() > 1)
260  *plan->possible_goal_states_[i] = *close_up_states[close_up_states.size() - 2];
261  }
262 
263  // try to compute a straight line path that arrives at the goal using the specified approach direction
264  auto first_approach_state = std::make_shared<moveit::core::RobotState>(*plan->possible_goal_states_[i]);
265 
266  std::vector<moveit::core::RobotStatePtr> approach_states;
267  double d_approach = moveit::core::CartesianInterpolator::computeCartesianPath(
268  first_approach_state.get(), plan->shared_data_->planning_group_, approach_states,
269  plan->shared_data_->ik_link_, -approach_direction, approach_direction_is_global_frame,
270  plan->approach_.desired_distance, moveit::core::MaxEEFStep(max_step_),
271  moveit::core::JumpThreshold(jump_factor_), approach_valid_callback);
272 
273  // if we were able to follow the approach direction for sufficient length, try to compute a retreat direction
274  if (d_approach > plan->approach_.min_distance && !signal_stop_)
275  {
276  if (plan->retreat_.desired_distance > 0.0)
277  {
278  // construct a planning scene that is just a diff on top of our actual planning scene
279  planning_scene::PlanningScenePtr planning_scene_after_approach = planning_scene_->diff();
280 
281  // assume the current state of the diff world is the one we plan to reach
282  planning_scene_after_approach->getCurrentStateNonConst() = *plan->possible_goal_states_[i];
283 
284  // apply the difference message to this world that virtually attaches the object we are manipulating
285  planning_scene_after_approach->processAttachedCollisionObjectMsg(plan->shared_data_->diff_attached_object_);
286 
287  // state validity checking during the retreat after the grasp must ensure the gripper posture is that of the
288  // actual grasp
289  moveit::core::GroupStateValidityCallbackFn retreat_valid_callback =
290  [scene = planning_scene_after_approach.get(), acm = collision_matrix_.get(), verbose = verbose_,
291  retreat_posture = &plan->retreat_posture_](moveit::core::RobotState* robot_state,
292  const moveit::core::JointModelGroup* joint_group,
293  const double* joint_group_variable_values) {
294  return isStateCollisionFree(scene, acm, verbose, retreat_posture, robot_state, joint_group,
295  joint_group_variable_values);
296  };
297 
298  // try to compute a straight line path that moves from the goal in a desired direction
299  moveit::core::RobotStatePtr last_retreat_state(
300  new moveit::core::RobotState(planning_scene_after_approach->getCurrentState()));
301  std::vector<moveit::core::RobotStatePtr> retreat_states;
302  double d_retreat = moveit::core::CartesianInterpolator::computeCartesianPath(
303  last_retreat_state.get(), plan->shared_data_->planning_group_, retreat_states,
304  plan->shared_data_->ik_link_, retreat_direction, retreat_direction_is_global_frame,
305  plan->retreat_.desired_distance, moveit::core::MaxEEFStep(max_step_),
306  moveit::core::JumpThreshold(jump_factor_), retreat_valid_callback);
307 
308  // if sufficient progress was made in the desired direction, we have a goal state that we can consider for
309  // future stages
310  if (d_retreat > plan->retreat_.min_distance && !signal_stop_)
311  {
312  // Create approach trajectory
313  std::reverse(approach_states.begin(), approach_states.end());
314  auto approach_traj = std::make_shared<robot_trajectory::RobotTrajectory>(
315  planning_scene_->getRobotModel(), plan->shared_data_->planning_group_->getName()));
316  for (const moveit::core::RobotStatePtr& approach_state : approach_states)
317  approach_traj->addSuffixWayPoint(approach_state, 0.0);
318 
319  // Create retreat trajectory
320  auto retreat_traj = std::make_shared<robot_trajectory::RobotTrajectory>(
321  planning_scene_->getRobotModel(), plan->shared_data_->planning_group_->getName()));
322  for (const moveit::core::RobotStatePtr& retreat_state : retreat_states)
323  retreat_traj->addSuffixWayPoint(retreat_state, 0.0);
324 
325  // Add timestamps to approach|retreat trajectories
326  time_param_.computeTimeStamps(*approach_traj);
327  time_param_.computeTimeStamps(*retreat_traj);
328 
329  // Convert approach trajectory to an executable trajectory
330  plan_execution::ExecutableTrajectory et_approach(approach_traj, "approach");
331  et_approach.allowed_collision_matrix_ = collision_matrix_;
332  plan->trajectories_.push_back(et_approach);
333 
334  // Add gripper close trajectory
335  addGripperTrajectory(plan, collision_matrix_, "grasp");
336 
337  // Convert retreat trajectory to an executable trajectory
338  plan_execution::ExecutableTrajectory et_retreat(retreat_traj, "retreat");
339  et_retreat.allowed_collision_matrix_ = collision_matrix_;
340  plan->trajectories_.push_back(et_retreat);
341 
342  plan->approach_state_ = approach_states.front();
343  return true;
344  }
345  }
346  else // No retreat was specified, so package up approach and grip trajectories.
347  {
348  // Reset the approach_state_ RobotStatePtr so that we can retry computing the cartesian path
349  plan->approach_state_.swap(first_approach_state);
350 
351  // Create approach trajectory
352  std::reverse(approach_states.begin(), approach_states.end());
353  auto approach_traj = std::make_shared<robot_trajectory::RobotTrajectory>(
354  planning_scene_->getRobotModel(), plan->shared_data_->planning_group_->getName()));
355  for (const moveit::core::RobotStatePtr& approach_state : approach_states)
356  approach_traj->addSuffixWayPoint(approach_state, 0.0);
357 
358  // Add timestamps to approach trajectories
359  time_param_.computeTimeStamps(*approach_traj);
360 
361  // Convert approach trajectory to an executable trajectory
362  plan_execution::ExecutableTrajectory et_approach(approach_traj, "approach");
363  et_approach.allowed_collision_matrix_ = collision_matrix_;
364  plan->trajectories_.push_back(et_approach);
365 
366  // Add gripper close trajectory
367  addGripperTrajectory(plan, collision_matrix_, "grasp");
368 
369  plan->approach_state_ = approach_states.front();
370 
371  return true;
372  }
373  }
374  }
375  } while (plan->possible_goal_states_.size() < max_goal_count_ && !signal_stop_ &&
376  samplePossibleGoalStates(plan, planning_scene_->getCurrentState(), min_distance, max_fail_));
377 
378  plan->error_code_.val = moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED;
379 
380  return false;
381 }
