49 int main(int argc, char* argv[])
50 {

53  unsigned int num;
54  bool reset_to_default;
55  po::options_description desc("Options");
56  // clang-format off
57  desc.add_options()
58  ("help", "show help message")
59  ("group", po::value<std::string>(&group)->default_value("all"), "name of planning group")
60  ("tip", po::value<std::string>(&tip)->default_value("default"), "name of the end effector in the planning group")
61  ("num", po::value<unsigned int>(&num)->default_value(100000), "number of IK solutions to compute")
62  ("reset_to_default", po::value<bool>(&reset_to_default)->default_value(true),
63  "whether to reset IK seed to default state. If set to false, the seed is the "
64  "correct IK solution (to accelerate filling the cache).");
65  // clang-format on
66 
67  po::variables_map vm;
68  po::store(po::parse_command_line(argc, argv, desc), vm);
69  po::notify(vm);
70 
71  if (vm.count("help") != 0u)
72  {
73  std::cout << desc << "\n";
74  return 1;
75  }
76 
77  rclcpp::init(argc, argv);
78  rclcpp::Node::SharedPtr node = rclcpp::Node::make_shared("benchmark_ik");
79  rclcpp::executors::MultiThreadedExecutor executor;
80  executor.add_node(node);
81 
82  // TODO(henningkayser): Load robot model from robot_description, fix kinematic param config
83  // robot_model_loader::RobotModelLoader robot_model_loader(node);
84  // const moveit::core::RobotModelPtr& kinematic_model = robot_model_loader.getModel();
85  const moveit::core::RobotModelPtr& kinematic_model = moveit::core::loadTestingRobotModel("panda");
86  planning_scene::PlanningScene planning_scene(kinematic_model);
87  moveit::core::RobotState& kinematic_state = planning_scene.getCurrentStateNonConst();
88  collision_detection::CollisionRequest collision_request;
89  collision_detection::CollisionResult collision_result;
90  std::chrono::duration<double> ik_time(0);
91  std::chrono::time_point<std::chrono::system_clock> start;
92  std::vector<moveit::core::JointModelGroup*> groups;
93  std::vector<std::string> end_effectors;
94 
95  if (group == "all")
96  groups = kinematic_model->getJointModelGroups();
97  else
98  groups.push_back(kinematic_model->getJointModelGroup(group));
99 
100  for (const auto& group : groups)
101  {
102  // skip group if there's no IK solver
103  if (group->getSolverInstance() == nullptr)
104  {
105  RCLCPP_WARN_STREAM(LOGGER, "No kinematic solver configured for group '" << group->getName() << "' - skipping");
106  continue;
107  }
108 
109  if (tip == "default")
110  group->getEndEffectorTips(end_effectors);
111  else
112  end_effectors = std::vector<std::string>(1, tip);
113 
114  // perform first IK call to load the cache, so that the time for loading is not included in
115  // average IK call time
116  kinematic_state.setToDefaultValues();
117  EigenSTL::vector_Isometry3d default_eef_states;
118  for (const auto& end_effector : end_effectors)
119  default_eef_states.push_back(kinematic_state.getGlobalLinkTransform(end_effector));
120  if (end_effectors.size() == 1)
121  kinematic_state.setFromIK(group, default_eef_states[0], end_effectors[0], 0.1);
122  else
123  kinematic_state.setFromIK(group, default_eef_states, end_effectors, 0.1);
124 
125  bool found_ik;
126  unsigned int num_failed_calls = 0, num_self_collisions = 0;
127  EigenSTL::vector_Isometry3d end_effector_states(end_effectors.size());
128  unsigned int i = 0;
129  while (i < num)
130  {
131  kinematic_state.setToRandomPositions(group);
132  collision_result.clear();
133  planning_scene.checkSelfCollision(collision_request, collision_result);
134  if (collision_result.collision)
135  {
136  ++num_self_collisions;
137  continue;
138  }
139  for (unsigned j = 0; j < end_effectors.size(); ++j)
140  end_effector_states[j] = kinematic_state.getGlobalLinkTransform(end_effectors[j]);
141  if (reset_to_default)
142  kinematic_state.setToDefaultValues();
143  start = std::chrono::system_clock::now();
144  if (end_effectors.size() == 1)
145  found_ik = kinematic_state.setFromIK(group, end_effector_states[0], end_effectors[0], 0.1);
146  else
147  found_ik = kinematic_state.setFromIK(group, end_effector_states, end_effectors, 0.1);
148  ik_time += std::chrono::system_clock::now() - start;
149  if (!found_ik)
150  num_failed_calls++;
151  ++i;
152  if (i % 100 == 0)
153  RCLCPP_INFO(LOGGER,
154  "Avg. time per IK solver call is %g after %d calls. %g%% of calls failed to return a solution. "
155  "%g%% of random joint configurations were ignored due to self-collisions.",
156  ik_time.count() / (double)i, i, 100. * num_failed_calls / i,
157  100. * num_self_collisions / (num_self_collisions + i));
158  }
159  RCLCPP_INFO(LOGGER, "Summary for group %s: %g %g %g", group->getName().c_str(), ik_time.count() / (double)i,
160  100. * num_failed_calls / i, 100. * num_self_collisions / (num_self_collisions + i));
161  }
162 
163  rclcpp::shutdown();
164  return 0;
165 }
