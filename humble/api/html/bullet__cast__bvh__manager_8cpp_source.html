42 BulletCastBVHManagerPtr BulletCastBVHManager::clone() const
43 {
44  BulletCastBVHManagerPtr manager = std::make_shared<BulletCastBVHManager>();
45 
46  for (const std::pair<const std::string, collision_detection_bullet::CollisionObjectWrapperPtr>& cow : link2cow_)
47  {
48  CollisionObjectWrapperPtr new_cow = cow.second->clone();
49

55  manager->addCollisionObject(new_cow);

57 
58  manager->setActiveCollisionObjects(active_);

60

64 void BulletCastBVHManager::setCastCollisionObjectsTransform(const std::string& name, const Eigen::Isometry3d& pose1,
65  const Eigen::Isometry3d& pose2)
66 {
67  // TODO: Find a way to remove this check. Need to store information in Tesseract EnvState indicating transforms with
68  // geometry
69  auto it = link2cow_.find(name);
70  if (it != link2cow_.end())
71  {
72  CollisionObjectWrapperPtr& cow = it->second;
73  assert(cow->m_collisionFilterGroup == btBroadphaseProxy::KinematicFilter);
74 
75  btTransform tf1 = convertEigenToBt(pose1);
76  btTransform tf2 = convertEigenToBt(pose2);
77 
78  cow->setWorldTransform(tf1);
79  link2cow_[name]->setWorldTransform(tf1);
80 
81  // If collision object is disabled don't proceed
82  if (cow->m_enabled)
83  {
84  if (btBroadphaseProxy::isConvex(cow->getCollisionShape()->getShapeType()))
85  {
86  static_cast<CastHullShape*>(cow->getCollisionShape())->updateCastTransform(tf1.inverseTimes(tf2));
87  }
88  else if (btBroadphaseProxy::isCompound(cow->getCollisionShape()->getShapeType()))
89  {
90  btCompoundShape* compound = static_cast<btCompoundShape*>(cow->getCollisionShape());
91 
92  for (int i = 0; i < compound->getNumChildShapes(); ++i)
93  {
94  if (btBroadphaseProxy::isConvex(compound->getChildShape(i)->getShapeType()))
95  {
96  const btTransform& local_tf = compound->getChildTransform(i);
97 
98  btTransform delta_tf = (tf1 * local_tf).inverseTimes(tf2 * local_tf);
99  static_cast<CastHullShape*>(compound->getChildShape(i))->updateCastTransform(delta_tf);
100  compound->updateChildTransform(i, local_tf, false); // This is required to update the BVH tree
101  }
102  else if (btBroadphaseProxy::isCompound(compound->getChildShape(i)->getShapeType()))
103  {
104  btCompoundShape* second_compound = static_cast<btCompoundShape*>(compound->getChildShape(i));
105 
106  for (int j = 0; j < second_compound->getNumChildShapes(); ++j)
107  {
108  assert(!btBroadphaseProxy::isCompound(second_compound->getChildShape(j)->getShapeType()));
109  const btTransform& local_tf = second_compound->getChildTransform(j);
110 
111  btTransform delta_tf = (tf1 * local_tf).inverseTimes(tf2 * local_tf);
112  static_cast<CastHullShape*>(second_compound->getChildShape(j))->updateCastTransform(delta_tf);
113  second_compound->updateChildTransform(j, local_tf, false); // This is required to update the BVH tree
114  }
115  second_compound->recalculateLocalAabb();
116  }
117  }
118  compound->recalculateLocalAabb();
119  }
120  else
121  {
122  RCLCPP_ERROR_STREAM(BULLET_LOGGER, "I can only continuous collision check convex shapes and "
123  "compound shapes made of convex shapes");
124  throw std::runtime_error(
125  "I can only continuous collision check convex shapes and compound shapes made of convex shapes");
126  }
127 
128  // Now update Broadphase AABB (See BulletWorld updateSingleAabb function)
129  updateBroadphaseAABB(cow, broadphase_, dispatcher_);
130  }
131  }
132 }

134 void BulletCastBVHManager::contactTest(collision_detection::CollisionResult& collisions,
135  const collision_detection::CollisionRequest& req,
136  const collision_detection::AllowedCollisionMatrix* acm, bool /*self*/ = false)
137 {
138  ContactTestData cdata(active_, contact_distance_, collisions, req);
139  broadphase_->calculateOverlappingPairs(dispatcher_.get());
140  btOverlappingPairCache* pair_cache = broadphase_->getOverlappingPairCache();
141 
142  RCLCPP_DEBUG_STREAM(BULLET_LOGGER, "Number overlapping candidates " << pair_cache->getNumOverlappingPairs());
143 
144  BroadphaseContactResultCallback cc(cdata, contact_distance_, acm, false, true);
145  TesseractCollisionPairCallback collision_callback(dispatch_info_, dispatcher_.get(), cc);
146  pair_cache->processAllOverlappingPairs(&collision_callback, dispatcher_.get());
147 }

149 void BulletCastBVHManager::addCollisionObject(const CollisionObjectWrapperPtr& cow)
150 {
151  std::string name = cow->getName();
152  if (cow->m_collisionFilterGroup == btBroadphaseProxy::KinematicFilter)
153  {
154  CollisionObjectWrapperPtr cast_cow = makeCastCollisionObject(cow);
155  link2cow_[name] = cast_cow;
156  }
157  else
158  {
159  link2cow_[name] = cow;
160  }
161 
162  btVector3 aabb_min, aabb_max;
163  link2cow_[name]->getAABB(aabb_min, aabb_max);
164 
165  int type = link2cow_[name]->getCollisionShape()->getShapeType();
166  link2cow_[name]->setBroadphaseHandle(
167  broadphase_->createProxy(aabb_min, aabb_max, type, link2cow_[name].get(), link2cow_[name]->m_collisionFilterGroup,
168  link2cow_[name]->m_collisionFilterMask, dispatcher_.get()));
169 }
