48 ButterworthFilter::ButterworthFilter(double low_pass_filter_coeff)
49  : previous_measurements_{ 0., 0. }
50  , previous_filtered_measurement_(0.)
51  , scale_term_(1. / (1. + low_pass_filter_coeff))
52  , feedback_term_(1. - low_pass_filter_coeff)
53 {
54  // guarantee this doesn't change because the logic below depends on this length implicitly
55  static_assert(ButterworthFilter::FILTER_LENGTH == 2,
56  "online_signal_smoothing::ButterworthFilter::FILTER_LENGTH should be 2");
57 
58  if (std::isinf(feedback_term_))
59  throw std::length_error("online_signal_smoothing::ButterworthFilter: infinite feedback_term_");
60 
61  if (std::isinf(scale_term_))
62  throw std::length_error("online_signal_smoothing::ButterworthFilter: infinite scale_term_");
63 
64  if (low_pass_filter_coeff < 1)
65  throw std::length_error(
66  "online_signal_smoothing::ButterworthFilter: Filter coefficient < 1. makes the lowpass filter unstable");
67 
68  if (std::abs(feedback_term_) < EPSILON)
69  throw std::length_error(
70  "online_signal_smoothing::ButterworthFilter: Filter coefficient value resulted in feedback term of 0");
71 }

73 double ButterworthFilter::filter(double new_measurement)
74 {
75  // Push in the new measurement
76  previous_measurements_[1] = previous_measurements_[0];
77  previous_measurements_[0] = new_measurement;
78 
79  previous_filtered_measurement_ = scale_term_ * (previous_measurements_[1] + previous_measurements_[0] -
80  feedback_term_ * previous_filtered_measurement_);
81 
82  return previous_filtered_measurement_;
83 }

85 void ButterworthFilter::reset(const double data)
86 {
87  previous_measurements_[0] = data;
88  previous_measurements_[1] = data;
89  previous_filtered_measurement_ = data;
90 }

107 bool ButterworthFilterPlugin::doSmoothing(std::vector<double>& position_vector)
108 {
109  if (position_vector.size() != position_filters_.size())
110  {
111  RCLCPP_ERROR_THROTTLE(node_->get_logger(), *node_->get_clock(), 1000,
112  "Position vector to be smoothed does not have the right length.");
113  return false;
114  }
115  for (size_t i = 0; i < position_vector.size(); ++i)
116  {
117  // Lowpass filter the position command
118  position_vector[i] = position_filters_.at(i).filter(position_vector[i]);
119  }
120  return true;
121 };
