47 void checkVelocityLimits(const moveit::core::JointModelGroup* joint_model_group, const Eigen::ArrayXd& velocity)
48 {
49  std::size_t joint_index{ 0 };
50  for (const moveit::core::JointModel* joint : joint_model_group->getActiveJointModels())
51  {
52  const auto& bounds = joint->getVariableBounds(joint->getName());
53 
54  if (bounds.velocity_bounded_)
55  {
56  EXPECT_GE(velocity(joint_index), bounds.min_velocity_) << "Joint " << joint_index << " violates velocity limit";
57  EXPECT_LE(velocity(joint_index), bounds.max_velocity_) << "Joint " << joint_index << " violates velocity limit";
58  }
59  ++joint_index;
60  }
61 }
