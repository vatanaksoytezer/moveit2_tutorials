228 void GroupEditWidget::setSelected(const std::string& group_name)
229 {
230  group_name_field_->setText(QString(group_name.c_str()));
231 
232  // Load properties from moveit_config_data.cpp ----------------------------------------------
233 
234  // Load resolution
235  double* resolution = &config_data_->group_meta_data_[group_name].kinematics_solver_search_resolution_;
236  if (*resolution == 0)
237  {
238  // Set default value
239  *resolution = DEFAULT_KIN_SOLVER_SEARCH_RESOLUTION;
240  }
241  kinematics_resolution_field_->setText(QString::number(*resolution));
242 
243  // Load timeout
244  double* timeout = &config_data_->group_meta_data_[group_name].kinematics_solver_timeout_;
245  if (*timeout == 0)
246  {
247  // Set default value
248  *timeout = DEFAULT_KIN_SOLVER_TIMEOUT;
249  }
250  kinematics_timeout_field_->setText(QString::number(*timeout));
251 
252  // Set kin solver
253  std::string kin_solver = config_data_->group_meta_data_[group_name].kinematics_solver_;
254 
255  // If this group doesn't have a solver, reset it to 'None'
256  if (kin_solver.empty())
257  {
258  kin_solver = "None";
259  }
260 
261  // Set the kin solver combo box
262  int index = kinematics_solver_field_->findText(kin_solver.c_str());
263  if (index == -1)
264  {
265  QMessageBox::warning(this, "Missing Kinematic Solvers",
266  QString("Unable to find the kinematic solver '")
267  .append(kin_solver.c_str())
268  .append("'. Trying running rosmake for this package. Until fixed, this setting will be "
269  "lost the next time the MoveIt configuration files are generated"));
270  return;
271  }
272  else
273  {
274  kinematics_solver_field_->setCurrentIndex(index);
275  }
276 
277  kinematics_parameters_file_field_->setText(
278  config_data_->group_meta_data_[group_name].kinematics_parameters_file_.c_str());
279 
280  // Set default planner
281  std::string default_planner = config_data_->group_meta_data_[group_name].default_planner_;
282 
283  // If this group doesn't have a solver, reset it to 'None'
284  if (default_planner.empty())
285  {
286  default_planner = "None";
287  }
288 
289  index = default_planner_field_->findText(default_planner.c_str());
290  if (index == -1)
291  {
292  QMessageBox::warning(this, "Missing Default Planner",
293  QString("Unable to find the default planner '%1'").arg(default_planner.c_str()));
294  }
295  else
296  {
297  default_planner_field_->setCurrentIndex(index);
298  }
299 }

304 void GroupEditWidget::loadKinematicPlannersComboBox()
305 {
306  // Only load this combo box once
307  static bool has_loaded = false;
308  if (has_loaded)
309  return;
310  has_loaded = true;
311 
312  // Remove all old items
313  kinematics_solver_field_->clear();
314  default_planner_field_->clear();
315 
316  // Add none option, the default
317  kinematics_solver_field_->addItem("None");
318  default_planner_field_->addItem("None");
319 
320  // load all avail kin planners
321  std::unique_ptr<pluginlib::ClassLoader<kinematics::KinematicsBase>> loader;
322  try
323  {
324  loader = std::make_unique<pluginlib::ClassLoader<kinematics::KinematicsBase>>("moveit_core",
325  "kinematics::KinematicsBase");
326  }
327  catch (pluginlib::PluginlibException& ex)
328  {
329  QMessageBox::warning(this, "Missing Kinematic Solvers",
330  "Exception while creating class loader for kinematic "
331  "solver plugins");
332  ROS_ERROR_STREAM(ex.what());
333  return;
334  }
335 
336  // Get classes
337  const std::vector<std::string>& classes = loader->getDeclaredClasses();
338 
339  // Warn if no plugins are found
340  if (classes.empty())
341  {
342  QMessageBox::warning(this, "Missing Kinematic Solvers",
343  "No MoveIt-compatible kinematics solvers found. Try "
344  "installing moveit_kinematics (sudo apt-get install "
345  "ros-${ROS_DISTRO}-moveit-kinematics)");
346  return;
347  }
348 
349  // Loop through all planners and add to combo box
350  for (const std::string& kinematics_plugin_name : classes)
351  {
352  kinematics_solver_field_->addItem(kinematics_plugin_name.c_str());
353  }
354 
355  std::vector<OMPLPlannerDescription> planners = config_data_->getOMPLPlanners();
356  for (OMPLPlannerDescription& planner : planners)
357  {
358  std::string planner_name = planner.name_;
359  default_planner_field_->addItem(planner_name.c_str());
360  }
361 }

363 void GroupEditWidget::selectKinematicsFile()
364 {
365  QString filename = QFileDialog::getOpenFileName(this, "Select a parameter file", "", "YAML files (*.yaml)");
366 
367  if (filename.isEmpty())
368  {
369  return;
370  }
371 
372  std::string package_name;
373  std::string relative_filename;
374  bool package_found =
375  config_data_->extractPackageNameFromPath(filename.toStdString(), package_name, relative_filename);
376 
377  QString lookup_path = filename;
378  if (package_found)
379  {
380  lookup_path = QString("$(find %1)/%2").arg(package_name.c_str()).arg(relative_filename.c_str());
381  }
382  kinematics_parameters_file_field_->setText(lookup_path);
383 }
