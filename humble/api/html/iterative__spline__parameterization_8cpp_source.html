78 bool IterativeSplineParameterization::computeTimeStamps(robot_trajectory::RobotTrajectory& trajectory,
79  const double max_velocity_scaling_factor,
80  const double max_acceleration_scaling_factor) const
81 {
82  if (trajectory.empty())
83  return true;
84 
85  const moveit::core::JointModelGroup* group = trajectory.getGroup();
86  if (!group)
87  {
88  RCLCPP_ERROR(LOGGER, "It looks like the planner did not set "
89  "the group the plan was computed for");
90  return false;
91  }
92  const moveit::core::RobotModel& rmodel = group->getParentModel();
93  const std::vector<int>& idx = group->getVariableIndexList();
94  const std::vector<std::string>& vars = group->getVariableNames();
95  double velocity_scaling_factor = 1.0;
96  double acceleration_scaling_factor = 1.0;
97  unsigned int num_points = trajectory.getWayPointCount();
98  unsigned int num_joints = group->getVariableCount();
99 
100  // Set scaling factors
101  if (max_velocity_scaling_factor > 0.0 && max_velocity_scaling_factor <= 1.0)
102  velocity_scaling_factor = max_velocity_scaling_factor;
103  else if (max_velocity_scaling_factor == 0.0)
104  {
105  RCLCPP_DEBUG(LOGGER, "A max_velocity_scaling_factor of 0.0 was specified, defaulting to %f instead.",
106  velocity_scaling_factor);
107  }
108  else
109  {
110  RCLCPP_WARN(LOGGER, "Invalid max_velocity_scaling_factor %f specified, defaulting to %f instead.",
111  max_velocity_scaling_factor, velocity_scaling_factor);
112  }
113  if (max_acceleration_scaling_factor > 0.0 && max_acceleration_scaling_factor <= 1.0)
114  acceleration_scaling_factor = max_acceleration_scaling_factor;
115  else if (max_acceleration_scaling_factor == 0.0)
116  {
117  RCLCPP_DEBUG(LOGGER, "A max_acceleration_scaling_factor of 0.0 was specified, defaulting to %f instead.",
118  acceleration_scaling_factor);
119  }
120  else
121  {
122  RCLCPP_WARN(LOGGER, "Invalid max_acceleration_scaling_factor %f specified, defaulting to %f instead.",
123  max_acceleration_scaling_factor, acceleration_scaling_factor);
124  }
125  // No wrapped angles.
126  trajectory.unwind();
127 
128  if (add_points_)
129  {
130  // Insert 2nd and 2nd-last points
131  // (required to force acceleration to specified values at endpoints)
132  if (trajectory.getWayPointCount() >= 2)
133  {
134  moveit::core::RobotState point = trajectory.getWayPoint(1);
135  moveit::core::RobotStatePtr p0, p1;
136 
137  // 2nd point is 90% of p0, and 10% of p1
138  p0 = trajectory.getWayPointPtr(0);
139  p1 = trajectory.getWayPointPtr(1);
140  for (unsigned int j = 0; j < num_joints; ++j)
141  {
142  point.setVariablePosition(idx[j],
143  (9 * p0->getVariablePosition(idx[j]) + 1 * p1->getVariablePosition(idx[j])) / 10);
144  }
145  trajectory.insertWayPoint(1, point, 0.0);
146  num_points++;
147 
148  // 2nd-last point is 10% of p0, and 90% of p1
149  p0 = trajectory.getWayPointPtr(num_points - 2);
150  p1 = trajectory.getWayPointPtr(num_points - 1);
151  for (unsigned int j = 0; j < num_joints; ++j)
152  {
153  point.setVariablePosition(idx[j],
154  (1 * p0->getVariablePosition(idx[j]) + 9 * p1->getVariablePosition(idx[j])) / 10);
155  }
156  trajectory.insertWayPoint(num_points - 1, point, 0.0);
157  num_points++;
158  }
159  }
160 
161  // JointTrajectory indexes in [point][joint] order.
162  // We need [joint][point] order to solve efficiently,
163  // so convert form here.
164 
165  std::vector<SingleJointTrajectory> t2(num_joints);
166 
167  for (unsigned int j = 0; j < num_joints; ++j)
168  {
169  // Copy positions
170  t2[j].positions_.resize(num_points, 0.0);
171  for (unsigned int i = 0; i < num_points; ++i)
172  {
173  t2[j].positions_[i] = trajectory.getWayPointPtr(i)->getVariablePosition(idx[j]);
174  }
175 
176  // Initialize velocities
177  t2[j].velocities_.resize(num_points, 0.0);
178  // Copy initial/final velocities if specified
179  if (trajectory.getWayPointPtr(0)->hasVelocities())
180  t2[j].velocities_[0] = trajectory.getWayPointPtr(0)->getVariableVelocity(idx[j]);
181  if (trajectory.getWayPointPtr(num_points - 1)->hasVelocities())
182  t2[j].velocities_[num_points - 1] = trajectory.getWayPointPtr(num_points - 1)->getVariableVelocity(idx[j]);
183 
184  // Initialize accelerations
185  t2[j].accelerations_.resize(num_points, 0.0);
186  t2[j].initial_acceleration_ = 0.0;
187  t2[j].final_acceleration_ = 0.0;
188  // Copy initial/final accelerations if specified
189  if (trajectory.getWayPointPtr(0)->hasAccelerations())
190  t2[j].initial_acceleration_ = trajectory.getWayPointPtr(0)->getVariableAcceleration(idx[j]);
191  t2[j].accelerations_[0] = t2[j].initial_acceleration_;
192  if (trajectory.getWayPointPtr(num_points - 1)->hasAccelerations())
193  t2[j].final_acceleration_ = trajectory.getWayPointPtr(num_points - 1)->getVariableAcceleration(idx[j]);
194  t2[j].accelerations_[num_points - 1] = t2[j].final_acceleration_;
195 
196  // Set bounds based on model, or default limits
197  const moveit::core::VariableBounds& bounds = rmodel.getVariableBounds(vars[j]);
198  t2[j].max_velocity_ = VLIMIT;
199  t2[j].min_velocity_ = -VLIMIT;
200  if (bounds.velocity_bounded_)
201  {
202  t2[j].max_velocity_ = bounds.max_velocity_;
203  t2[j].min_velocity_ = bounds.min_velocity_;
204  if (t2[j].min_velocity_ == 0.0)
205  t2[j].min_velocity_ = -t2[j].max_velocity_;
206  }
207  t2[j].max_velocity_ *= velocity_scaling_factor;
208  t2[j].min_velocity_ *= velocity_scaling_factor;
209 
210  t2[j].max_acceleration_ = ALIMIT;
211  t2[j].min_acceleration_ = -ALIMIT;
212  if (bounds.acceleration_bounded_)
213  {
214  t2[j].max_acceleration_ = bounds.max_acceleration_;
215  t2[j].min_acceleration_ = bounds.min_acceleration_;
216  if (t2[j].min_acceleration_ == 0.0)
217  t2[j].min_acceleration_ = -t2[j].max_acceleration_;
218  }
219  t2[j].max_acceleration_ *= acceleration_scaling_factor;
220  t2[j].min_acceleration_ *= acceleration_scaling_factor;
221 
222  // Error out if bounds don't make sense
223  if (t2[j].max_velocity_ <= 0.0 || t2[j].max_acceleration_ <= 0.0)
224  {
225  RCLCPP_ERROR(LOGGER,
226  "Joint %d max velocity %f and max acceleration %f must be greater than zero "
227  "or a solution won't be found.\n",
228  j, t2[j].max_velocity_, t2[j].max_acceleration_);
229  return false;
230  }
231  if (t2[j].min_velocity_ >= 0.0 || t2[j].min_acceleration_ >= 0.0)
232  {
233  RCLCPP_ERROR(LOGGER,
234  "Joint %d min velocity %f and min acceleration %f must be less than zero "
235  "or a solution won't be found.\n",
236  j, t2[j].min_velocity_, t2[j].min_acceleration_);
237  return false;
238  }
239  }
240 
241  // Error check
242  if (num_points < 4)
243  {
244  RCLCPP_ERROR(LOGGER, "number of waypoints %d, needs to be greater than 3.\n", num_points);
245  return false;
246  }
247  for (unsigned int j = 0; j < num_joints; ++j)
248  {
249  if (t2[j].velocities_[0] > t2[j].max_velocity_ || t2[j].velocities_[0] < t2[j].min_velocity_)
250  {
251  RCLCPP_ERROR(LOGGER, "Initial velocity %f out of bounds\n", t2[j].velocities_[0]);
252  return false;
253  }
254  else if (t2[j].velocities_[num_points - 1] > t2[j].max_velocity_ ||
255  t2[j].velocities_[num_points - 1] < t2[j].min_velocity_)
256  {
257  RCLCPP_ERROR(LOGGER, "Final velocity %f out of bounds\n", t2[j].velocities_[num_points - 1]);
258  return false;
259  }
260  else if (t2[j].accelerations_[0] > t2[j].max_acceleration_ || t2[j].accelerations_[0] < t2[j].min_acceleration_)
261  {
262  RCLCPP_ERROR(LOGGER, "Initial acceleration %f out of bounds\n", t2[j].accelerations_[0]);
263  return false;
264  }
265  else if (t2[j].accelerations_[num_points - 1] > t2[j].max_acceleration_ ||
266  t2[j].accelerations_[num_points - 1] < t2[j].min_acceleration_)
267  {
268  RCLCPP_ERROR(LOGGER, "Final acceleration %f out of bounds\n", t2[j].accelerations_[num_points - 1]);
269  return false;
270  }
271  }
272 
273  // Initialize times
274  // start with valid velocities, then expand intervals
275  // epsilon to prevent divide-by-zero
276  std::vector<double> time_diff(trajectory.getWayPointCount() - 1, std::numeric_limits<double>::epsilon());
277  for (unsigned int j = 0; j < num_joints; ++j)
278  init_times(num_points, &time_diff[0], &t2[j].positions_[0], t2[j].max_velocity_, t2[j].min_velocity_);
279 
280  // Stretch intervals until close to the bounds
281  while (1)
282  {
283  int loop = 0;
284 
285  // Calculate the interval stretches due to acceleration
286  std::vector<double> time_factor(num_points - 1, 1.00);
287  for (unsigned j = 0; j < num_joints; ++j)
288  {
289  // Move points to satisfy initial/final acceleration
290  if (add_points_)
291  {
292  adjust_two_positions(num_points, &time_diff[0], &t2[j].positions_[0], &t2[j].velocities_[0],
293  &t2[j].accelerations_[0], t2[j].initial_acceleration_, t2[j].final_acceleration_);
294  }
295 
296  fit_cubic_spline(num_points, &time_diff[0], &t2[j].positions_[0], &t2[j].velocities_[0], &t2[j].accelerations_[0]);
297  for (unsigned i = 0; i < num_points; ++i)
298  {
299  const double acc = t2[j].accelerations_[i];
300  double atfactor = 1.0;
301  if (acc > t2[j].max_acceleration_)
302  atfactor = sqrt(acc / t2[j].max_acceleration_);
303  if (acc < t2[j].min_acceleration_)
304  atfactor = sqrt(acc / t2[j].min_acceleration_);
305  if (atfactor > 1.01) // within 1%
306  loop = 1;
307  atfactor = (atfactor - 1.0) / 16.0 + 1.0; // 1/16th
308  if (i > 0)
309  time_factor[i - 1] = std::max(time_factor[i - 1], atfactor);
310  if (i < num_points - 1)
311  time_factor[i] = std::max(time_factor[i], atfactor);
312  }
313  }
314 
315  if (loop == 0)
316  break; // finished
317 
318  // Stretch
319  for (unsigned i = 0; i < num_points - 1; ++i)
320  time_diff[i] *= time_factor[i];
321  }
322 
323  // Final adjustment forces the trajectory within bounds
324  globalAdjustment(t2, trajectory, time_diff);
325 
326  // Convert back to JointTrajectory form
327  for (unsigned int i = 1; i < num_points; ++i)
328  trajectory.setWayPointDurationFromPrevious(i, time_diff[i - 1]);
329  for (unsigned int i = 0; i < num_points; ++i)
330  {
331  for (unsigned int j = 0; j < num_joints; ++j)
332  {
333  trajectory.getWayPointPtr(i)->setVariableVelocity(idx[j], t2[j].velocities_[i]);
334  trajectory.getWayPointPtr(i)->setVariableAcceleration(idx[j], t2[j].accelerations_[i]);
335  }
336 
337  // Only update position of additionally inserted points (at second and next-to-last position)
338  if (add_points_ && (i == 1 || i == num_points - 2))
339  {
340  for (unsigned int j = 0; j < num_joints; ++j)
341  trajectory.getWayPointPtr(i)->setVariablePosition(idx[j], t2[j].positions_[i]);
342  trajectory.getWayPointPtr(i)->update();
343  }
344  }
345 
346  return true;
347 }

360 /*
361  Fit a 'clamped' cubic spline over a series of points.
362  A cubic spline ensures continuous function across positions,
363  1st derivative (velocities), and 2nd derivative (accelerations).
364  'Clamped' means the first derivative at the endpoints is specified.
365 
366  Fitting a cubic spline involves solving a series of linear equations.
367  The general form for each segment is:
368  (tj-t_(j-1))*x"_(j-1) + 2*(t_(j+1)-t_(j-1))*x"j + (t_(j+1)-tj)*x"_j+1) =
369  (x_(j+1)-xj)/(t_(j+1)-tj) - (xj-x_(j-1))/(tj-t_(j-1))
370  And the first and last segment equations are clamped to specified values: x1_i and x1_f.
371 
372  Represented in matrix form:
373  [ 2*(t1-t0) (t1-t0) 0 ][x0"] [(x1-x0)/(t1-t0) - t1_i ]
374  [ t1-t0 2*(t2-t0) t2-t1 ][x1"] [(x2-x1)/(t2-t1) - (x1-x0)/(t1-t0)]
375  [ t2-t1 2*(t3-t1) t3-t2 ][x2"] = 6 * [(x3-x2)/(t3/t2) - (x2-x1)/(t2-t1)]
376  [ ... ... ... ][...] [... ]
377  [ 0 tN-t_(N-1) 2*(tN-t_(N-1)) ][xN"] [t1_f - (xN-x_(N-1))/(tN-t_(N-1)) ]
378 
379  This matrix is tridiagonal, which can be solved solved in O(N) time
380  using the tridiagonal algorithm.
381  There is a forward propagation pass followed by a backsubstitution pass.
382 
383  n is the number of points
384  dt contains the time difference between each point (size=n-1)
385  x contains the positions (size=n)
386  x1 contains the 1st derivative (velocities) (size=n)
387  x1[0] and x1[n-1] MUST be specified.
388  x2 contains the 2nd derivative (accelerations) (size=n)
389  x1 and x2 are filled in by the algorithm.
390 */
391 
392 static void fit_cubic_spline(const int n, const double dt[], const double x[], double x1[], double x2[])
393 {
394  int i;
395  const double x1_i = x1[0], x1_f = x1[n - 1];
396 
397  // Tridiagonal alg - forward sweep
398  // x1 and x2 used to store the temporary coefficients c and d
399  // (will get overwritten during backsubstitution)
400  double *c = x1, *d = x2;
401  c[0] = 0.5;
402  d[0] = 3.0 * ((x[1] - x[0]) / dt[0] - x1_i) / dt[0];
403  for (i = 1; i <= n - 2; ++i)
404  {
405  const double dt2 = dt[i - 1] + dt[i];
406  const double a = dt[i - 1] / dt2;
407  const double denom = 2.0 - a * c[i - 1];
408  c[i] = (1.0 - a) / denom;
409  d[i] = 6.0 * ((x[i + 1] - x[i]) / dt[i] - (x[i] - x[i - 1]) / dt[i - 1]) / dt2;
410  d[i] = (d[i] - a * d[i - 1]) / denom;
411  }
412  const double denom = dt[n - 2] * (2.0 - c[n - 2]);
413  d[n - 1] = 6.0 * (x1_f - (x[n - 1] - x[n - 2]) / dt[n - 2]);
414  d[n - 1] = (d[n - 1] - dt[n - 2] * d[n - 2]) / denom;
415 
416  // Tridiagonal alg - backsubstitution sweep
417  // 2nd derivative
418  x2[n - 1] = d[n - 1];
419  for (i = n - 2; i >= 0; i--)
420  x2[i] = d[i] - c[i] * x2[i + 1];
421 
422  // 1st derivative
423  x1[0] = x1_i;
424  for (i = 1; i < n - 1; ++i)
425  x1[i] = (x[i + 1] - x[i]) / dt[i] - (2 * x2[i] + x2[i + 1]) * dt[i] / 6.0;
426  x1[n - 1] = x1_f;
427 }
428 
429 /*
430  Modify the value of x[1] and x[N-2]
431  so that 2nd derivative starts and ends at specified value.
432  This involves fitting the spline twice,
433  then solving for the specified value.
434 
435  x2_i and x2_f are the (initial and final) 2nd derivative at 0 and N-1
436 */
437 
438 static void adjust_two_positions(const int n, const double dt[], double x[], double x1[], double x2[],
439  const double x2_i, const double x2_f)
440 {
441  x[1] = x[0];
442  x[n - 2] = x[n - 3];
443  fit_cubic_spline(n, dt, x, x1, x2);
444  double a0 = x2[0];
445  double b0 = x2[n - 1];
446 
447  x[1] = x[2];
448  x[n - 2] = x[n - 1];
449  fit_cubic_spline(n, dt, x, x1, x2);
450  double a2 = x2[0];
451  double b2 = x2[n - 1];
452 
453  // we can solve this with linear equation (use two-point form)
454  if (a2 != a0)
455  x[1] = x[0] + ((x[2] - x[0]) / (a2 - a0)) * (x2_i - a0);
456  if (b2 != b0)
457  x[n - 2] = x[n - 3] + ((x[n - 1] - x[n - 3]) / (b2 - b0)) * (x2_f - b0);
458 }

460 /*
461  Find time required to go max velocity on each segment.
462  Increase a segment's time interval if the current time isn't long enough.
463 */
464 
465 static void init_times(const int n, double dt[], const double x[], const double max_velocity, const double min_velocity)
466 {
467  int i;
468  for (i = 0; i < n - 1; ++i)
469  {
470  double time;
471  double dx = x[i + 1] - x[i];
472  if (dx >= 0.0)
473  time = (dx / max_velocity);
474  else
475  time = (dx / min_velocity);
476  time += std::numeric_limits<double>::epsilon(); // prevent divide-by-zero
477 
478  if (dt[i] < time)
479  dt[i] = time;
480  }
481 }

484 /*
485  Fit a spline, then check each interval to see if bounds are met.
486  If all bounds met (no time adjustments made), return 0.
487  If bounds not met (time adjustments made), slightly increase the
488  surrounding time intervals and return 1.
489 
490  n is the number of points
491  dt contains the time difference between each point (size=n-1)
492  x contains the positions (size=n)
493  x1 contains the 1st derivative (velocities) (size=n)
494  x1[0] and x1[n-1] MUST be specified.
495  x2 contains the 2nd derivative (accelerations) (size=n)
496  max_velocity is the max velocity for this joint.
497  min_velocity is the min velocity for this joint.
498  max_acceleration is the max acceleration for this joint.
499  min_acceleration is the min acceleration for this joint.
500  tfactor is the time adjustment (multiplication) factor.
501  x1 and x2 are filled in by the algorithm.
502 */
503 
504 static int fit_spline_and_adjust_times(const int n, double dt[], const double x[], double x1[], double x2[],
505  const double max_velocity, const double min_velocity,
506  const double max_acceleration, const double min_acceleration,
507  const double tfactor)
508 {
509  int i, ret = 0;
510 
511  fit_cubic_spline(n, dt, x, x1, x2);
512 
513  // Instantaneous velocity is calculated at each point
514  for (i = 0; i < n - 1; ++i)
515  {
516  const double vel = x1[i];
517  const double vel2 = x1[i + 1];
518  if (vel > max_velocity || vel < min_velocity || vel2 > max_velocity || vel2 < min_velocity)
519  {
520  dt[i] *= tfactor;
521  ret = 1;
522  }
523  }
524  // Instantaneous acceleration is calculated at each point
525  if (ret == 0)
526  {
527  for (i = 0; i < n - 1; ++i)
528  {
529  const double acc = x2[i];

543 // return global expansion multiplicative factor required
544 // to force within bounds.
545 // Assumes that the spline is already fit
546 // (fit_cubic_spline must have been called before this).
547 static double global_adjustment_factor(const int n, double x1[], double x2[], const double max_velocity,
548  const double min_velocity, const double max_acceleration,
549  const double min_acceleration)
550 {
551  int i;
552  double tfactor2 = 1.00;
553 
554  for (i = 0; i < n; ++i)
555  {
556  double tfactor;
557  tfactor = x1[i] / max_velocity;
558  if (tfactor2 < tfactor)
559  tfactor2 = tfactor;
560  tfactor = x1[i] / min_velocity;
561  if (tfactor2 < tfactor)
562  tfactor2 = tfactor;
563 
564  if (x2[i] >= 0)
565  {
566  tfactor = sqrt(fabs(x2[i] / max_acceleration));
567  if (tfactor2 < tfactor)
568  tfactor2 = tfactor;
569  }
570  else
571  {
572  tfactor = sqrt(fabs(x2[i] / min_acceleration));
573  if (tfactor2 < tfactor)
574  tfactor2 = tfactor;
575  }
576  }
577 
578  return tfactor2;
