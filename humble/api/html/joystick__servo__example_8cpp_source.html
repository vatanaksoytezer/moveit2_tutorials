85 // Some axes have offsets (e.g. the default trigger position is 1.0 not 0)
86 // This will map the default values for the axes
87 std::map<Axis, double> AXIS_DEFAULTS = { { LEFT_TRIGGER, 1.0 }, { RIGHT_TRIGGER, 1.0 } };

90 // To change controls or setup a new controller, all you should to do is change the above enums and the follow 2
91 // functions
99 bool convertJoyToCmd(const std::vector<float>& axes, const std::vector<int>& buttons,
100  std::unique_ptr<geometry_msgs::msg::TwistStamped>& twist,
101  std::unique_ptr<control_msgs::msg::JointJog>& joint)
102 {
103  // Give joint jogging priority because it is only buttons
104  // If any joint jog command is requested, we are only publishing joint commands
105  if (buttons[A] || buttons[B] || buttons[X] || buttons[Y] || axes[D_PAD_X] || axes[D_PAD_Y])
106  {
107  // Map the D_PAD to the proximal joints
108  joint->joint_names.push_back("panda_joint1");
109  joint->velocities.push_back(axes[D_PAD_X]);
110  joint->joint_names.push_back("panda_joint2");
111  joint->velocities.push_back(axes[D_PAD_Y]);
112 
113  // Map the diamond to the distal joints
114  joint->joint_names.push_back("panda_joint7");
115  joint->velocities.push_back(buttons[B] - buttons[X]);
116  joint->joint_names.push_back("panda_joint6");
117  joint->velocities.push_back(buttons[Y] - buttons[A]);
118  return false;
119  }
120 
121  // The bread and butter: map buttons to twist commands
122  twist->twist.linear.z = axes[RIGHT_STICK_Y];
123  twist->twist.linear.y = axes[RIGHT_STICK_X];
124 
125  double lin_x_right = -0.5 * (axes[RIGHT_TRIGGER] - AXIS_DEFAULTS.at(RIGHT_TRIGGER));
126  double lin_x_left = 0.5 * (axes[LEFT_TRIGGER] - AXIS_DEFAULTS.at(LEFT_TRIGGER));
127  twist->twist.linear.x = lin_x_right + lin_x_left;
128 
129  twist->twist.angular.y = axes[LEFT_STICK_Y];
130  twist->twist.angular.x = axes[LEFT_STICK_X];
131 
132  double roll_positive = buttons[RIGHT_BUMPER];
133  double roll_negative = -1 * (buttons[LEFT_BUMPER]);
134  twist->twist.angular.z = roll_positive + roll_negative;
135 
136  return true;
137 }

223  void joyCB(const sensor_msgs::msg::Joy::SharedPtr msg)
224  {
225  // Create the messages we might publish
226  auto twist_msg = std::make_unique<geometry_msgs::msg::TwistStamped>();
227  auto joint_msg = std::make_unique<control_msgs::msg::JointJog>();
228 
229  // This call updates the frame for twist commands
230  updateCmdFrame(frame_to_publish_, msg->buttons);
231 
232  // Convert the joystick message to Twist or JointJog and publish
233  if (convertJoyToCmd(msg->axes, msg->buttons, twist_msg, joint_msg))
234  {
235  // publish the TwistStamped
236  twist_msg->header.frame_id = frame_to_publish_;
237  twist_msg->header.stamp = this->now();
238  twist_pub_->publish(std::move(twist_msg));
239  }
240  else
241  {
242  // publish the JointJog
243  joint_msg->header.stamp = this->now();
244  joint_msg->header.frame_id = "panda_link3";
245  joint_pub_->publish(std::move(joint_msg));
246  }
247  }
