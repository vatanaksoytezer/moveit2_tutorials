86 void KDLKinematicsPlugin::getJointWeights()
87 {
88  const std::vector<std::string>& active_names = joint_model_group_->getActiveJointModelNames();
89  std::vector<std::string> names;
90  std::vector<double> weights;
91  if (lookupParam(node_, "joint_weights.weights", weights, weights))
92  {
93  if (!lookupParam(node_, "joint_weights.names", names, names) || (names.size() != weights.size()))
94  {
95  RCLCPP_ERROR(LOGGER, "Expecting list parameter joint_weights.names of same size as list joint_weights.weights");
96  // fall back to default weights
97  weights.clear();
98  }
99  }
100  else if (lookupParam(node_, "joint_weights", weights,
101  weights)) // try reading weight lists (for all active joints) directly
102  {
103  std::size_t num_active = active_names.size();
104  if (weights.size() == num_active)
105  {
106  joint_weights_ = weights;
107  return;
108  }
109  else if (!weights.empty())
110  {
111  RCLCPP_ERROR(LOGGER, "Expecting parameter joint_weights to list weights for all active joints (%zu) in order",
112  num_active);
113  // fall back to default weights
114  weights.clear();
115  }
116  }
117 
118  // by default assign weights of 1.0 to all joints
119  joint_weights_ = std::vector<double>(active_names.size(), 1.0);
120  if (weights.empty()) // indicates default case
121  return;
122 
123  // modify weights of listed joints
124  assert(names.size() == weights.size());
125  for (size_t i = 0; i != names.size(); ++i)
126  {
127  auto it = std::find(active_names.begin(), active_names.end(), names[i]);
128  if (it == active_names.cend())
129  RCLCPP_WARN(LOGGER, "Joint '%s' is not an active joint in group '%s'", names[i].c_str(),
130  joint_model_group_->getName().c_str());
131  else if (weights[i] < 0.0)
132  RCLCPP_WARN(LOGGER, "Negative weight %f for joint '%s' will be ignored", weights[i], names[i].c_str());
133  else
134  joint_weights_[it - active_names.begin()] = weights[i];
135  }
136  RCLCPP_INFO_STREAM(
137  LOGGER, "Joint weights for group '"
138  << getGroupName() << "': \n"
139  << Eigen::Map<const Eigen::VectorXd>(joint_weights_.data(), joint_weights_.size()).transpose());
140 }

142 bool KDLKinematicsPlugin::initialize(const rclcpp::Node::SharedPtr& node, const moveit::core::RobotModel& robot_model,
143  const std::string& group_name, const std::string& base_frame,
144  const std::vector<std::string>& tip_frames, double search_discretization)
145 {
146  node_ = node;
147  storeValues(robot_model, group_name, base_frame, tip_frames, search_discretization);
148  joint_model_group_ = robot_model_->getJointModelGroup(group_name);
149  if (!joint_model_group_)
150  return false;
151 
152  if (!joint_model_group_->isChain())
153  {
154  RCLCPP_ERROR(LOGGER, "Group '%s' is not a chain", group_name.c_str());
155  return false;
156  }
157  if (!joint_model_group_->isSingleDOFJoints())
158  {
159  RCLCPP_ERROR(LOGGER, "Group '%s' includes joints that have more than 1 DOF", group_name.c_str());
160  return false;
161  }
162 
163  KDL::Tree kdl_tree;
164 
165  if (!kdl_parser::treeFromUrdfModel(*robot_model.getURDF(), kdl_tree))
166  {
167  RCLCPP_ERROR(LOGGER, "Could not initialize tree object");
168  return false;
169  }
170  if (!kdl_tree.getChain(base_frame_, getTipFrame(), kdl_chain_))
171  {
172  RCLCPP_ERROR(LOGGER, "Could not initialize chain object");
173  return false;
174  }
175 
176  dimension_ = joint_model_group_->getActiveJointModels().size() + joint_model_group_->getMimicJointModels().size();
177  for (std::size_t i = 0; i < joint_model_group_->getJointModels().size(); ++i)
178  {
179  if (joint_model_group_->getJointModels()[i]->getType() == moveit::core::JointModel::REVOLUTE ||
180  joint_model_group_->getJointModels()[i]->getType() == moveit::core::JointModel::PRISMATIC)
181  {
182  solver_info_.joint_names.push_back(joint_model_group_->getJointModelNames()[i]);
183  const std::vector<moveit_msgs::msg::JointLimits>& jvec =
184  joint_model_group_->getJointModels()[i]->getVariableBoundsMsg();
185  solver_info_.limits.insert(solver_info_.limits.end(), jvec.begin(), jvec.end());
186  }
187  }
188 
189  if (!joint_model_group_->hasLinkModel(getTipFrame()))
190  {
191  RCLCPP_ERROR(LOGGER, "Could not find tip name in joint group '%s'", group_name.c_str());
192  return false;
193  }
194  solver_info_.link_names.push_back(getTipFrame());
195 
196  joint_min_.resize(solver_info_.limits.size());
197  joint_max_.resize(solver_info_.limits.size());
198 
199  for (unsigned int i = 0; i < solver_info_.limits.size(); ++i)
200  {
201  joint_min_(i) = solver_info_.limits[i].min_position;
202  joint_max_(i) = solver_info_.limits[i].max_position;
203  }
204 
205  // Get Solver Parameters
206  lookupParam(node_, "max_solver_iterations", max_solver_iterations_, 500);
207  lookupParam(node_, "epsilon", epsilon_, 1e-5);
208  lookupParam(node_, "orientation_vs_position", orientation_vs_position_weight_, 1.0);
209 
210  bool position_ik;
211  lookupParam(node_, "position_only_ik", position_ik, false);
212  if (position_ik) // position_only_ik overrules orientation_vs_position
213  orientation_vs_position_weight_ = 0.0;
214  if (orientation_vs_position_weight_ == 0.0)
215  RCLCPP_INFO(LOGGER, "Using position only ik");
216 
217  getJointWeights();
218 
219  // Check for mimic joints
220  unsigned int joint_counter = 0;
221  for (std::size_t i = 0; i < kdl_chain_.getNrOfSegments(); ++i)
222  {
223  const moveit::core::JointModel* jm = robot_model_->getJointModel(kdl_chain_.segments[i].getJoint().getName());
224 
225  // first check whether it belongs to the set of active joints in the group
226  if (jm->getMimic() == nullptr && jm->getVariableCount() > 0)
227  {
228  JointMimic mimic_joint;
229  mimic_joint.reset(joint_counter);
230  mimic_joint.joint_name = kdl_chain_.segments[i].getJoint().getName();
231  mimic_joint.active = true;
232  mimic_joints_.push_back(mimic_joint);
233  ++joint_counter;
234  continue;
235  }
236  if (joint_model_group_->hasJointModel(jm->getName()))
237  {
238  if (jm->getMimic() && joint_model_group_->hasJointModel(jm->getMimic()->getName()))
239  {
240  JointMimic mimic_joint;
241  mimic_joint.joint_name = kdl_chain_.segments[i].getJoint().getName();
242  mimic_joint.offset = jm->getMimicOffset();
243  mimic_joint.multiplier = jm->getMimicFactor();
244  mimic_joints_.push_back(mimic_joint);
245  continue;
246  }
247  }
248  }
249  for (JointMimic& mimic_joint : mimic_joints_)
250  {
251  if (!mimic_joint.active)
252  {
253  const moveit::core::JointModel* joint_model =
254  joint_model_group_->getJointModel(mimic_joint.joint_name)->getMimic();
255  for (JointMimic& mimic_joint_recal : mimic_joints_)
256  {
257  if (mimic_joint_recal.joint_name == joint_model->getName())
258  {
259  mimic_joint.map_index = mimic_joint_recal.map_index;
260  }
261  }
262  }
263  }
264 
265  // Setup the joint state groups that we need
266  state_ = std::make_shared<moveit::core::RobotState>(robot_model_);
267 
268  fk_solver_ = std::make_unique<KDL::ChainFkSolverPos_recursive>(kdl_chain_);
269 
270  initialized_ = true;
271  RCLCPP_DEBUG(LOGGER, "KDL solver initialized");
272  return true;
273 }

325 bool KDLKinematicsPlugin::searchPositionIK(const geometry_msgs::msg::Pose& ik_pose,
326  const std::vector<double>& ik_seed_state, double timeout,
327  const std::vector<double>& consistency_limits, std::vector<double>& solution,
328  const IKCallbackFn& solution_callback,
329  moveit_msgs::msg::MoveItErrorCodes& error_code,
330  const kinematics::KinematicsQueryOptions& options) const
331 {
332  const rclcpp::Time start_time = steady_clock_.now();
333  if (!initialized_)
334  {
335  RCLCPP_ERROR(LOGGER, "kinematics solver not initialized");
336  error_code.val = error_code.NO_IK_SOLUTION;
337  return false;
338  }
339 
340  if (ik_seed_state.size() != dimension_)
341  {
342  RCLCPP_ERROR(LOGGER, "Seed state must have size %d instead of size %zu\n", dimension_, ik_seed_state.size());
343  error_code.val = error_code.NO_IK_SOLUTION;
344  return false;
345  }
346 
347  // Resize consistency limits to remove mimic joints
348  std::vector<double> consistency_limits_mimic;
349  if (!consistency_limits.empty())
350  {
351  if (consistency_limits.size() != dimension_)
352  {
353  RCLCPP_ERROR(LOGGER, "Consistency limits must be empty or have size %d instead of size %zu\n", dimension_,
354  consistency_limits.size());
355  error_code.val = error_code.NO_IK_SOLUTION;
356  return false;
357  }
358 
359  for (std::size_t i = 0; i < dimension_; ++i)
360  {
361  if (mimic_joints_[i].active)
362  consistency_limits_mimic.push_back(consistency_limits[i]);
363  }
364  }
365  Eigen::Matrix<double, 6, 1> cartesian_weights;
366  cartesian_weights.topRows<3>().setConstant(1.0);
367  cartesian_weights.bottomRows<3>().setConstant(orientation_vs_position_weight_);
368 
369  KDL::JntArray jnt_seed_state(dimension_);
370  KDL::JntArray jnt_pos_in(dimension_);
371  KDL::JntArray jnt_pos_out(dimension_);
372  jnt_seed_state.data = Eigen::Map<const Eigen::VectorXd>(ik_seed_state.data(), ik_seed_state.size());
373  jnt_pos_in = jnt_seed_state;
374 
375  KDL::ChainIkSolverVelMimicSVD ik_solver_vel(kdl_chain_, mimic_joints_, orientation_vs_position_weight_ == 0.0);
376  solution.resize(dimension_);
377 
378  KDL::Frame pose_desired;
379  tf2::fromMsg(ik_pose, pose_desired);
380 
381  RCLCPP_DEBUG_STREAM(LOGGER, "searchPositionIK: Position request pose is "
382  << ik_pose.position.x << " " << ik_pose.position.y << " " << ik_pose.position.z << " "
383  << ik_pose.orientation.x << " " << ik_pose.orientation.y << " "
384  << ik_pose.orientation.z << " " << ik_pose.orientation.w);
385 
386  unsigned int attempt = 0;
387  do
388  {
389  ++attempt;
390  if (attempt > 1) // randomly re-seed after first attempt
391  {
392  if (!consistency_limits_mimic.empty())
393  getRandomConfiguration(jnt_seed_state.data, consistency_limits_mimic, jnt_pos_in.data);
394  else
395  getRandomConfiguration(jnt_pos_in.data);
396  RCLCPP_DEBUG_STREAM(LOGGER, "New random configuration (" << attempt << "): " << jnt_pos_in);
397  }
398 
399  int ik_valid =
400  CartToJnt(ik_solver_vel, jnt_pos_in, pose_desired, jnt_pos_out, max_solver_iterations_,
401  Eigen::Map<const Eigen::VectorXd>(joint_weights_.data(), joint_weights_.size()), cartesian_weights);
402  if (ik_valid == 0 || options.return_approximate_solution) // found acceptable solution
403  {
404  if (!consistency_limits_mimic.empty() &&
405  !checkConsistency(jnt_seed_state.data, consistency_limits_mimic, jnt_pos_out.data))
406  continue;
407 
408  Eigen::Map<Eigen::VectorXd>(solution.data(), solution.size()) = jnt_pos_out.data;
409  if (!solution_callback.empty())
410  {
411  solution_callback(ik_pose, solution, error_code);
412  if (error_code.val != error_code.SUCCESS)
413  continue;
414  }
415 
416  // solution passed consistency check and solution callback
417  error_code.val = error_code.SUCCESS;
418  RCLCPP_DEBUG_STREAM(LOGGER, "Solved after " << (steady_clock_.now() - start_time).seconds() << " < " << timeout
419  << "s and " << attempt << " attempts");
420  return true;
421  }
422  } while (!timedOut(start_time, timeout));
423 
424  RCLCPP_DEBUG_STREAM(LOGGER, "IK timed out after " << (steady_clock_.now() - start_time).seconds() << " > " << timeout
425  << "s and " << attempt << " attempts");
426  error_code.val = error_code.TIMED_OUT;
427  return false;
428 }

430 // NOLINTNEXTLINE(readability-identifier-naming)
431 int KDLKinematicsPlugin::CartToJnt(KDL::ChainIkSolverVelMimicSVD& ik_solver, const KDL::JntArray& q_init,
432  const KDL::Frame& p_in, KDL::JntArray& q_out, const unsigned int max_iter,
433  const Eigen::VectorXd& joint_weights, const Twist& cartesian_weights) const
434 {
435  double last_delta_twist_norm = DBL_MAX;
436  double step_size = 1.0;
437  KDL::Frame f;
438  KDL::Twist delta_twist;
439  KDL::JntArray delta_q(q_out.rows()), q_backup(q_out.rows());
440  Eigen::ArrayXd extra_joint_weights(joint_weights.rows());
441  extra_joint_weights.setOnes();
442 
443  q_out = q_init;
444  RCLCPP_DEBUG_STREAM(LOGGER, "Input: " << q_init);
445 
446  unsigned int i;
447  bool success = false;
448  for (i = 0; i < max_iter; ++i)
449  {
450  fk_solver_->JntToCart(q_out, f);
451  delta_twist = diff(f, p_in);
452  RCLCPP_DEBUG_STREAM(LOGGER, "[" << std::setw(3) << i << "] delta_twist: " << delta_twist);
453 
454  // check norms of position and orientation errors
455  const double position_error = delta_twist.vel.Norm();
456  const double orientation_error = ik_solver.isPositionOnly() ? 0 : delta_twist.rot.Norm();
457  const double delta_twist_norm = std::max(position_error, orientation_error);
458  if (delta_twist_norm <= epsilon_)
459  {
460  success = true;
461  break;
462  }
463 
464  if (delta_twist_norm >= last_delta_twist_norm)
465  {
466  // if the error increased, we are close to a singularity -> reduce step size
467  double old_step_size = step_size;
468  step_size *= std::min(0.2, last_delta_twist_norm / delta_twist_norm); // reduce scale;
469  KDL::Multiply(delta_q, step_size / old_step_size, delta_q);
470  RCLCPP_DEBUG(LOGGER, " error increased: %f -> %f, scale: %f", last_delta_twist_norm, delta_twist_norm,
471  step_size);
472  q_out = q_backup; // restore previous unclipped joint values
473  }
474  else
475  {
476  q_backup = q_out; // remember joint values of last successful step
477  step_size = 1.0; // reset step size
478  last_delta_twist_norm = delta_twist_norm;
479 
480  ik_solver.CartToJnt(q_out, delta_twist, delta_q, extra_joint_weights * joint_weights.array(), cartesian_weights);
481  }
482 
483  clipToJointLimits(q_out, delta_q, extra_joint_weights);
484 
485  const double delta_q_norm = delta_q.data.lpNorm<1>();
486  RCLCPP_DEBUG(LOGGER, "[%3d] pos err: %f rot err: %f delta_q: %f", i, position_error, orientation_error,
487  delta_q_norm);
488  if (delta_q_norm < epsilon_) // stuck in singularity
489  {
490  if (step_size < epsilon_) // cannot reach target
491  break;
492  // wiggle joints
493  last_delta_twist_norm = DBL_MAX;
494  delta_q.data.setRandom();
495  delta_q.data *= std::min(0.1, delta_twist_norm);
496  clipToJointLimits(q_out, delta_q, extra_joint_weights);
497  extra_joint_weights.setOnes();
498  }
499 
500  KDL::Add(q_out, delta_q, q_out);
501 
502  RCLCPP_DEBUG_STREAM(LOGGER, " delta_q: " << delta_q);
503  RCLCPP_DEBUG_STREAM(LOGGER, " q: " << q_out);
504  }
505 
506  int result = (i == max_iter) ? -3 : (success ? 0 : -2);
507  RCLCPP_DEBUG_STREAM(LOGGER, "Result " << result << " after " << i << " iterations: " << q_out);
508 
509  return result;
510 }

512 void KDLKinematicsPlugin::clipToJointLimits(const KDL::JntArray& q, KDL::JntArray& q_delta,
513  Eigen::ArrayXd& weighting) const
514 {
515  weighting.setOnes();
516  for (std::size_t i = 0; i < q.rows(); ++i)
517  {
518  const double delta_max = joint_max_(i) - q(i);
519  const double delta_min = joint_min_(i) - q(i);
520  if (q_delta(i) > delta_max)
521  q_delta(i) = delta_max;
522  else if (q_delta(i) < delta_min)
523  q_delta(i) = delta_min;
524  else
525  continue;
526 
527  weighting[mimic_joints_[i].map_index] = 0.01;
528  }
529 }

531 bool KDLKinematicsPlugin::getPositionFK(const std::vector<std::string>& link_names,
532  const std::vector<double>& joint_angles,
533  std::vector<geometry_msgs::msg::Pose>& poses) const
534 {
535  if (!initialized_)
536  {
537  RCLCPP_ERROR(LOGGER, "kinematics solver not initialized");
538  return false;
539  }
540  poses.resize(link_names.size());
541  if (joint_angles.size() != dimension_)
542  {
543  RCLCPP_ERROR(LOGGER, "Joint angles vector must have size: %d", dimension_);
544  return false;
545  }
546 
547  KDL::Frame p_out;
548  KDL::JntArray jnt_pos_in(dimension_);
549  jnt_pos_in.data = Eigen::Map<const Eigen::VectorXd>(joint_angles.data(), joint_angles.size());
