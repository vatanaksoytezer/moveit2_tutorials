62 void robot_interaction::LockedRobotState::setState(const moveit::core::RobotState& state)
63 {
64  {
65  boost::mutex::scoped_lock lock(state_lock_);
66 
67  // If someone else has a reference to the state, then make a new copy.
68  // The old state is orphaned (does not change, but is now out of date).
69  if (state_.unique())
70  *state_ = state;
71  else
72  state_ = std::make_shared<moveit::core::RobotState>(state);
73 
74  state_->update();
75  }
76  robotStateChanged();
77 }
