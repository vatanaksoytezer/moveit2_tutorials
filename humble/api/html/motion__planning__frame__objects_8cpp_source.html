159 void MotionPlanningFrame::sceneScaleChanged(int value)
160 {
161  const double scaling_factor = (double)value / 100.0; // The GUI slider gives percent values
162  if (scaled_object_)
163  {
164  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
165  if (ps)
166  {
167  if (ps->getWorld()->hasObject(scaled_object_->id_))
168  {
169  ps->getWorldNonConst()->removeObject(scaled_object_->id_);
170  for (std::size_t i = 0; i < scaled_object_->shapes_.size(); ++i)
171  {
172  shapes::Shape* s = scaled_object_->shapes_[i]->clone();
173  s->scale(scaling_factor);
174 
175  Eigen::Isometry3d scaled_shape_pose = scaled_object_->shape_poses_[i];
176  scaled_shape_pose.translation() *= scaling_factor;
177 
178  ps->getWorldNonConst()->addToObject(scaled_object_->id_, scaled_object_->pose_, shapes::ShapeConstPtr(s),
179  scaled_shape_pose);
180  }
181  moveit::core::FixedTransformsMap scaled_subframes = scaled_object_->subframe_poses_;
182  for (auto& subframe_pair : scaled_subframes)
183  subframe_pair.second.translation() *= scaling_factor;
184 
185  ps->getWorldNonConst()->setSubframesOfObject(scaled_object_->id_, scaled_subframes);
186  setLocalSceneEdited();
187  scene_marker_->processMessage(createObjectMarkerMsg(ps->getWorld()->getObject(scaled_object_->id_)));
188  planning_display_->queueRenderSceneGeometry();
189  }
190  else
191  scaled_object_.reset();
192  }
193  else
194  scaled_object_.reset();
195  }
196 }

241 static QString decideStatusText(const collision_detection::CollisionEnv::ObjectConstPtr& obj)
242 {
243  QString status_text = "'" + QString::fromStdString(obj->id_) + "' is a collision object with ";
244  if (obj->shapes_.empty())
245  status_text += "no geometry";
246  else
247  {
248  std::vector<QString> shape_names;
249  for (const shapes::ShapeConstPtr& shape : obj->shapes_)
250  shape_names.push_back(QString::fromStdString(shapes::shapeStringName(shape.get())));
251  if (shape_names.size() == 1)
252  status_text += "one " + shape_names[0];
253  else
254  {
255  status_text += QString::fromStdString(std::to_string(shape_names.size())) + " shapes:";
256  for (const QString& shape_name : shape_names)
257  status_text += " " + shape_name;
258  }
259  status_text += ".";
260  }
261  if (!obj->subframe_poses_.empty())
262  {
263  status_text += subframe_poses_to_qstring(obj->subframe_poses_);
264  }
265  return status_text;
266 }

273  {

275  }
276  return status_text;

279 void MotionPlanningFrame::selectedCollisionObjectChanged()
280 {
281  QList<QListWidgetItem*> sel = ui_->collision_objects_list->selectedItems();
282  if (sel.empty())
283  {
284  bool old_state = ui_->object_x->blockSignals(true);
285  ui_->object_x->setValue(0.0);
286  ui_->object_x->blockSignals(old_state);
287 
288  old_state = ui_->object_y->blockSignals(true);
289  ui_->object_y->setValue(0.0);
290  ui_->object_y->blockSignals(old_state);
291 
292  old_state = ui_->object_z->blockSignals(true);
293  ui_->object_z->setValue(0.0);
294  ui_->object_z->blockSignals(old_state);
295 
296  old_state = ui_->object_rx->blockSignals(true);
297  ui_->object_rx->setValue(0.0);
298  ui_->object_rx->blockSignals(old_state);
299 
300  old_state = ui_->object_ry->blockSignals(true);
301  ui_->object_ry->setValue(0.0);
302  ui_->object_ry->blockSignals(old_state);
303 
304  old_state = ui_->object_rz->blockSignals(true);
305  ui_->object_rz->setValue(0.0);
306  ui_->object_rz->blockSignals(old_state);
307 
308  ui_->object_status->setText("");
309  scene_marker_.reset();
310  ui_->pose_scale_group_box->setEnabled(false);
311  }
312  else if (planning_display_->getPlanningSceneMonitor())
313  {
314  // if this is a CollisionWorld element
315  if (sel[0]->checkState() == Qt::Unchecked)
316  {
317  ui_->pose_scale_group_box->setEnabled(true);
318  bool update_scene_marker = false;
319  Eigen::Isometry3d obj_pose;
320  {
321  const planning_scene_monitor::LockedPlanningSceneRO& ps = planning_display_->getPlanningSceneRO();
322  const collision_detection::CollisionEnv::ObjectConstPtr& obj =
323  ps->getWorld()->getObject(sel[0]->text().toStdString());
324  if (obj)
325  {
326  ui_->object_status->setText(decideStatusText(obj));
327 
328  if (obj->shapes_.size() == 1)
329  {
330  obj_pose = obj->pose_; // valid isometry by contract
331  Eigen::Vector3d xyz = obj_pose.linear().eulerAngles(0, 1, 2);
332  update_scene_marker = true; // do the marker update outside locked scope to avoid deadlock
333 
334  bool old_state = ui_->object_x->blockSignals(true);
335  ui_->object_x->setValue(obj_pose.translation()[0]);
336  ui_->object_x->blockSignals(old_state);
337 
338  old_state = ui_->object_y->blockSignals(true);
339  ui_->object_y->setValue(obj_pose.translation()[1]);
340  ui_->object_y->blockSignals(old_state);
341 
342  old_state = ui_->object_z->blockSignals(true);
343  ui_->object_z->setValue(obj_pose.translation()[2]);
344  ui_->object_z->blockSignals(old_state);
345 
346  old_state = ui_->object_rx->blockSignals(true);
347  ui_->object_rx->setValue(xyz[0]);
348  ui_->object_rx->blockSignals(old_state);
349 
350  old_state = ui_->object_ry->blockSignals(true);
351  ui_->object_ry->setValue(xyz[1]);
352  ui_->object_ry->blockSignals(old_state);
353 
354  old_state = ui_->object_rz->blockSignals(true);
355  ui_->object_rz->setValue(xyz[2]);
356  ui_->object_rz->blockSignals(old_state);
357  }
358  }
359  else
360  ui_->object_status->setText("ERROR: '" + sel[0]->text() + "' should be a collision object but it is not");
361  }
362  if (update_scene_marker && ui_->tabWidget->tabText(ui_->tabWidget->currentIndex()).toStdString() == TAB_OBJECTS)
363  {
364  createSceneInteractiveMarker();
365  }
366  }
367  else
368  {
369  ui_->pose_scale_group_box->setEnabled(false);
370  // if it is an attached object
371  scene_marker_.reset();
372  const planning_scene_monitor::LockedPlanningSceneRO& ps = planning_display_->getPlanningSceneRO();
373  const moveit::core::AttachedBody* attached_body =
374  ps->getCurrentState().getAttachedBody(sel[0]->text().toStdString());
375  if (attached_body)
376  ui_->object_status->setText(decideStatusText(attached_body));
377  else
378  ui_->object_status->setText("ERROR: '" + sel[0]->text() + "' should be an attached object but it is not");
379  }
380  }

388 void MotionPlanningFrame::updateCollisionObjectPose(bool update_marker_position)
389 {
390  QList<QListWidgetItem*> sel = ui_->collision_objects_list->selectedItems();
391  if (sel.empty())
392  return;
393  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
394  if (ps)
395  {
396  collision_detection::CollisionEnv::ObjectConstPtr obj = ps->getWorld()->getObject(sel[0]->text().toStdString());
397  if (obj)
398  {
399  Eigen::Isometry3d p;
400  p.translation()[0] = ui_->object_x->value();
401  p.translation()[1] = ui_->object_y->value();
402  p.translation()[2] = ui_->object_z->value();
403 
404  p = Eigen::Translation3d(p.translation()) *
405  (Eigen::AngleAxisd(ui_->object_rx->value(), Eigen::Vector3d::UnitX()) *
406  Eigen::AngleAxisd(ui_->object_ry->value(), Eigen::Vector3d::UnitY()) *
407  Eigen::AngleAxisd(ui_->object_rz->value(), Eigen::Vector3d::UnitZ()));
408 
409  ps->getWorldNonConst()->setObjectPose(obj->id_, p);
410  planning_display_->queueRenderSceneGeometry();
411  setLocalSceneEdited();
412 
413  // Update the interactive marker pose to match the manually introduced one
414  if (update_marker_position && scene_marker_)
415  {
416  // p is isometry by construction
417  Eigen::Quaterniond eq(p.linear());
418  scene_marker_->setPose(Ogre::Vector3(ui_->object_x->value(), ui_->object_y->value(), ui_->object_z->value()),
419  Ogre::Quaternion(eq.w(), eq.x(), eq.y(), eq.z()), "");
420  }
421  }
422  }

441 /* Receives feedback from the interactive marker and updates the shape pose in the world accordingly */
442 void MotionPlanningFrame::imProcessFeedback(visualization_msgs::msg::InteractiveMarkerFeedback& feedback)
443 {
444  if (!planning_display_->getPlanningSceneRO()->knowsFrameTransform(feedback.header.frame_id))
445  {
446  RCLCPP_ERROR_STREAM(LOGGER,
447  "Frame `" << feedback.header.frame_id << "` unknown doesn't exists in the planning scene");
448  }
449  Eigen::Isometry3d fixed_frame_t_scene_marker;
450  tf2::fromMsg(feedback.pose, fixed_frame_t_scene_marker);
451  Eigen::Isometry3d model_frame_t_scene_marker =
452  planning_display_->getPlanningSceneRO()->getFrameTransform(feedback.header.frame_id) * fixed_frame_t_scene_marker;
453 
454  bool old_state = ui_->object_x->blockSignals(true);
455  ui_->object_x->setValue(model_frame_t_scene_marker.translation().x());
456  ui_->object_x->blockSignals(old_state);
457 
458  old_state = ui_->object_y->blockSignals(true);
459  ui_->object_y->setValue(model_frame_t_scene_marker.translation().y());
460  ui_->object_y->blockSignals(old_state);
461 
462  old_state = ui_->object_z->blockSignals(true);
463  ui_->object_z->setValue(model_frame_t_scene_marker.translation().z());
464  ui_->object_z->blockSignals(old_state);
465 
466  Eigen::Vector3d xyz = model_frame_t_scene_marker.linear().eulerAngles(0, 1, 2);
467 
468  old_state = ui_->object_rx->blockSignals(true);
469  ui_->object_rx->setValue(xyz[0]);
470  ui_->object_rx->blockSignals(old_state);
471 
472  old_state = ui_->object_ry->blockSignals(true);
473  ui_->object_ry->setValue(xyz[1]);
474  ui_->object_ry->blockSignals(old_state);
475 
476  old_state = ui_->object_rz->blockSignals(true);

478  ui_->object_rz->blockSignals(old_state);
479 
480  updateCollisionObjectPose(false);
481 }

483 void MotionPlanningFrame::copySelectedCollisionObject()
484 {
485  QList<QListWidgetItem*> sel = ui_->collision_objects_list->selectedItems();
486  if (sel.empty())
487  return;
488 
489  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
490  if (!ps)
491  return;
492 
493  for (const QListWidgetItem* item : sel)
494  {
495  std::string name = item->text().toStdString();
496  collision_detection::CollisionEnv::ObjectConstPtr obj = ps->getWorld()->getObject(name);
497  if (!obj)
498  continue;
499 
500  // find a name for the copy
501  name.insert(0, "Copy of ");
502  if (ps->getWorld()->hasObject(name))
503  {
504  name += " ";
505  unsigned int n = 1;
506  while (ps->getWorld()->hasObject(name + std::to_string(n)))
507  n++;
508  name += std::to_string(n);
509  }
510  ps->getWorldNonConst()->addToObject(name, obj->shapes_, obj->shape_poses_);
511  RCLCPP_DEBUG(LOGGER, "Copied collision object to '%s'", name.c_str());
512  }
513  setLocalSceneEdited();
514  planning_display_->addMainLoopJob([this] { populateCollisionObjectsList(); });
515 }

825 void MotionPlanningFrame::renameCollisionObject(QListWidgetItem* item)
826 {
827  long unsigned int version = known_collision_objects_version_;
828  if (item->text().isEmpty())
829  {
830  QMessageBox::warning(this, "Invalid object name", "Cannot set an empty object name.");
831  if (version == known_collision_objects_version_)
832  item->setText(QString::fromStdString(known_collision_objects_[item->type()].first));
833  return;
834  }
835 
836  std::string item_text = item->text().toStdString();
837  bool already_exists = planning_display_->getPlanningSceneRO()->getWorld()->hasObject(item_text);
838  if (!already_exists)
839  already_exists = planning_display_->getPlanningSceneRO()->getCurrentState().hasAttachedBody(item_text);
840  if (already_exists)
841  {
842  QMessageBox::warning(this, "Duplicate object name",
843  QString("The name '")
844  .append(item->text())
845  .append("' already exists. Not renaming object ")
846  .append((known_collision_objects_[item->type()].first.c_str())));
847  if (version == known_collision_objects_version_)
848  item->setText(QString::fromStdString(known_collision_objects_[item->type()].first));
849  return;
850  }
851 
852  if (item->checkState() == Qt::Unchecked)
853  {
854  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
855  collision_detection::CollisionEnv::ObjectConstPtr obj =
856  ps->getWorld()->getObject(known_collision_objects_[item->type()].first);
857  if (obj)
858  {
859  known_collision_objects_[item->type()].first = item_text;
860  ps->getWorldNonConst()->removeObject(obj->id_);
861  ps->getWorldNonConst()->addToObject(known_collision_objects_[item->type()].first, obj->pose_, obj->shapes_,
862  obj->shape_poses_);
863  ps->getWorldNonConst()->setSubframesOfObject(obj->id_, obj->subframe_poses_);
864  if (scene_marker_)
865  {
866  scene_marker_.reset();
867  planning_display_->addMainLoopJob([this] { createSceneInteractiveMarker(); });
868  }
869  }
870  }
871  else
872  {
873  // rename attached body
874  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
875  moveit::core::RobotState& cs = ps->getCurrentStateNonConst();
876  const moveit::core::AttachedBody* ab = cs.getAttachedBody(known_collision_objects_[item->type()].first);
877  if (ab)
878  {
879  known_collision_objects_[item->type()].first = item_text;
880  auto new_ab = std::make_unique<moveit::core::AttachedBody>(
881  ab->getAttachedLink(), known_collision_objects_[item->type()].first, ab->getPose(), ab->getShapes(),
882  ab->getShapePoses(), ab->getTouchLinks(), ab->getDetachPosture(), ab->getSubframes());
883  cs.clearAttachedBody(ab->getName());
884  cs.attachBody(std::move(new_ab));
885  }
886  }
887  setLocalSceneEdited();
888 }

890 void MotionPlanningFrame::attachDetachCollisionObject(QListWidgetItem* item)
891 {
892  long unsigned int version = known_collision_objects_version_;
893  bool checked = item->checkState() == Qt::Checked;
894  std::pair<std::string, bool> data = known_collision_objects_[item->type()];
895  moveit_msgs::msg::AttachedCollisionObject aco;
896 
897  if (checked) // we need to attach a known collision object

901  for (const std::string& link : links_std)

903  bool ok = false;
904  QString response =
905  QInputDialog::getItem(this, tr("Select Link Name"), tr("Choose the link to attach to:"), links, 0, false, &ok);
906  if (!ok)
907  {
908  if (version == known_collision_objects_version_)
909  item->setCheckState(Qt::Unchecked);
910  return;
911  }
912  aco.link_name = response.toStdString();
913  aco.object.id = data.first;
914  aco.object.operation = moveit_msgs::msg::CollisionObject::ADD;
915  }
916  else // we need to detach an attached object
917  {
918  const planning_scene_monitor::LockedPlanningSceneRO& ps = planning_display_->getPlanningSceneRO();
919  const moveit::core::AttachedBody* attached_body = ps->getCurrentState().getAttachedBody(data.first);
920  if (attached_body)
921  {
922  aco.link_name = attached_body->getAttachedLinkName();
923  aco.object.id = attached_body->getName();
924  aco.object.operation = moveit_msgs::msg::CollisionObject::REMOVE;
925  }
926  }
927 
928  moveit::core::RobotState rs(planning_display_->getRobotModel());
929  {
930  planning_scene_monitor::LockedPlanningSceneRW ps = planning_display_->getPlanningSceneRW();
931  // we loop through the list in case updates were received since the start of the function
932  for (std::pair<std::string, bool>& known_collision_object : known_collision_objects_)
933  if (known_collision_object.first == data.first)
934  {
935  known_collision_object.second = checked;
936  break;
937  }
938  ps->processAttachedCollisionObjectMsg(aco);
939  rs = ps->getCurrentState();
940  }
941 
942  selectedCollisionObjectChanged();
943  setLocalSceneEdited();
944  planning_display_->updateQueryStates(rs);
945  planning_display_->queueRenderSceneGeometry();
