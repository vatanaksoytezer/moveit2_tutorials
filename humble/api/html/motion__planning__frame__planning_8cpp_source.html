126 bool MotionPlanningFrame::computeCartesianPlan()
127 {
128  rclcpp::Time start = rclcpp::Clock().now();
129  // get goal pose
130  moveit::core::RobotState goal = *planning_display_->getQueryGoalState();
131  std::vector<geometry_msgs::msg::Pose> waypoints;
132  const std::string& link_name = move_group_->getEndEffectorLink();
133  const moveit::core::LinkModel* link = move_group_->getRobotModel()->getLinkModel(link_name);
134  if (!link)
135  {
136  RCLCPP_ERROR_STREAM(LOGGER, "Failed to determine unique end-effector link: " << link_name);
137  return false;
138  }
139  waypoints.push_back(tf2::toMsg(goal.getGlobalLinkTransform(link)));
140 
141  // setup default params
142  double cart_step_size = 0.01;
143  double cart_jump_thresh = 0.0;
144  bool avoid_collisions = true;
145 
146  // compute trajectory
147  moveit_msgs::msg::RobotTrajectory trajectory;
148  double fraction =
149  move_group_->computeCartesianPath(waypoints, cart_step_size, cart_jump_thresh, trajectory, avoid_collisions);
150 
151  if (fraction >= 1.0)
152  {
153  RCLCPP_INFO(LOGGER, "Achieved %f %% of Cartesian path", fraction * 100.);
154 
155  // Compute time parameterization to also provide velocities
156  // https://groups.google.com/forum/#!topic/moveit-users/MOoFxy2exT4
157  robot_trajectory::RobotTrajectory rt(move_group_->getRobotModel(), move_group_->getName());
158  rt.setRobotTrajectoryMsg(*move_group_->getCurrentState(), trajectory);
159  trajectory_processing::TimeOptimalTrajectoryGeneration time_parameterization;
160  bool success = time_parameterization.computeTimeStamps(rt, ui_->velocity_scaling_factor->value(),
161  ui_->acceleration_scaling_factor->value());
162  RCLCPP_INFO(LOGGER, "Computing time stamps %s", success ? "SUCCEEDED" : "FAILED");
163 
164  // Store trajectory in current_plan_
165  current_plan_ = std::make_shared<moveit::planning_interface::MoveGroupInterface::Plan>();
166  rt.getRobotTrajectoryMsg(current_plan_->trajectory_);
167  current_plan_->planning_time_ = (rclcpp::Clock().now() - start).seconds();
168  return success;
169  }
170  return false;
171 }

312 void MotionPlanningFrame::updateQueryStateHelper(moveit::core::RobotState& state, const std::string& v)
313 {
314  if (v == "<random>")
315  {
316  configureWorkspace();
317  if (const moveit::core::JointModelGroup* jmg =
318  state.getJointModelGroup(planning_display_->getCurrentPlanningGroup()))
319  state.setToRandomPositions(jmg);
320  return;
321  }
322 
323  if (v == "<random valid>")
324  {
325  configureWorkspace();
326 
327  if (const moveit::core::JointModelGroup* jmg =
328  state.getJointModelGroup(planning_display_->getCurrentPlanningGroup()))
329  {
330  // Loop until a collision free state is found
331  static const int MAX_ATTEMPTS = 100;
332  int attempt_count = 0; // prevent loop for going forever
333  while (attempt_count < MAX_ATTEMPTS)
334  {
335  // Generate random state
336  state.setToRandomPositions(jmg);
337 
338  state.update(); // prevent dirty transforms
339 
340  // Test for collision
341  if (planning_display_->getPlanningSceneRO()->isStateValid(state, "", false))
342  break;
343 
344  attempt_count++;
345  }
346  // Explain if no valid rand state found
347  if (attempt_count >= MAX_ATTEMPTS)
348  RCLCPP_WARN(LOGGER, "Unable to find a random collision free configuration after %d attempts", MAX_ATTEMPTS);
349  }
350  else
351  {
352  RCLCPP_WARN_STREAM(LOGGER, "Unable to get joint model group " << planning_display_->getCurrentPlanningGroup());
353  }
354  return;
355  }
356 
357  if (v == "<current>")
358  {
359  rclcpp::Time t = node_->now();
360  planning_display_->waitForCurrentRobotState(t);
361  const planning_scene_monitor::LockedPlanningSceneRO& ps = planning_display_->getPlanningSceneRO();
362  if (ps)
363  state = ps->getCurrentState();
364  return;
365  }
366 
367  if (v == "<same as goal>")
368  {
369  state = *planning_display_->getQueryGoalState();
370  return;
371  }
372 
373  if (v == "<same as start>")
374  {
375  state = *planning_display_->getQueryStartState();
376  return;
377  }
378 
379  if (v == "<previous>")
380  {
381  state = planning_display_->getPreviousState();
382  return;
383  }
384 
385  // maybe it is a named state
386  if (const moveit::core::JointModelGroup* jmg = state.getJointModelGroup(planning_display_->getCurrentPlanningGroup()))
387  state.setToDefaultValues(jmg, v);
388 }

417 void MotionPlanningFrame::populatePlannerDescription(const moveit_msgs::msg::PlannerInterfaceDescription& desc)
418 {
419  std::string group = planning_display_->getCurrentPlanningGroup();
420  RCLCPP_DEBUG(LOGGER, "Found %zu planners for group '%s' and pipeline '%s'", desc.planner_ids.size(), group.c_str(),
421  desc.pipeline_id.c_str());
422  ui_->planning_algorithm_combo_box->clear();
423 
424  // set the label for the planning library
425  ui_->library_label->setText(QString::fromStdString(desc.name));
426  ui_->library_label->setStyleSheet("QLabel { color : green; font: bold }");
427 
428  bool found_group = false;
429  // the name of a planner is either "GROUP[planner_id]" or "planner_id"
430  if (!group.empty())
431  {
432  for (const std::string& planner_id : desc.planner_ids)
433  {
434  RCLCPP_DEBUG(LOGGER, "planner id: %s", planner_id.c_str());
435  if (planner_id == group)
436  found_group = true;
437  else if (planner_id.substr(0, group.length()) == group)
438  {
439  if (planner_id.size() > group.length() && planner_id[group.length()] == '[')
440  {
441  std::string id = planner_id.substr(group.length());
442  if (id.size() > 2)
443  {
444  id.resize(id.length() - 1);
445  ui_->planning_algorithm_combo_box->addItem(QString::fromStdString(id.substr(1)));
446  }
447  }
448  }
449  }
450  }
451  if (ui_->planning_algorithm_combo_box->count() == 0 && !found_group)
452  for (const std::string& planner_id : desc.planner_ids)
453  ui_->planning_algorithm_combo_box->addItem(QString::fromStdString(planner_id));
454 
455  ui_->planning_algorithm_combo_box->insertItem(0, "<unspecified>");
456 
457  // retrieve default planner config from parameter server
458  const std::string& default_planner_config = move_group_->getDefaultPlannerId(found_group ? group : std::string());
459  int default_index = ui_->planning_algorithm_combo_box->findText(QString::fromStdString(default_planner_config));
460  if (default_index < 0)
461  default_index = 0; // 0 is <unspecified> fallback
462  ui_->planning_algorithm_combo_box->setCurrentIndex(default_index);
463 
464  QFont font;
465  font.setBold(true);
466  ui_->planning_algorithm_combo_box->setItemData(default_index, font, Qt::FontRole);
467 }

506 void MotionPlanningFrame::configureWorkspace()
507 {
508  moveit::core::VariableBounds bx, by, bz;
509  bx.position_bounded_ = by.position_bounded_ = bz.position_bounded_ = true;
510 
511  moveit::core::JointModel::Bounds b(3);
512  bx.min_position_ = ui_->wcenter_x->value() - ui_->wsize_x->value() / 2.0;
513  bx.max_position_ = ui_->wcenter_x->value() + ui_->wsize_x->value() / 2.0;
514  by.min_position_ = ui_->wcenter_y->value() - ui_->wsize_y->value() / 2.0;
515  by.max_position_ = ui_->wcenter_y->value() + ui_->wsize_y->value() / 2.0;
516  bz.min_position_ = ui_->wcenter_z->value() - ui_->wsize_z->value() / 2.0;
517  bz.max_position_ = ui_->wcenter_z->value() + ui_->wsize_z->value() / 2.0;
518 
519  if (move_group_)
520  move_group_->setWorkspace(bx.min_position_, by.min_position_, bz.min_position_, bx.max_position_, by.max_position_,
521  bz.max_position_);
522  planning_scene_monitor::PlanningSceneMonitorPtr psm = planning_display_->getPlanningSceneMonitor();
523  // get non-const access to the robot_model and update planar & floating joints as indicated by the workspace settings
524  if (psm && psm->getRobotModelLoader() && psm->getRobotModelLoader()->getModel())
525  {
526  const moveit::core::RobotModelPtr& robot_model = psm->getRobotModelLoader()->getModel();
527  const std::vector<moveit::core::JointModel*>& jm = robot_model->getJointModels();
528  for (moveit::core::JointModel* joint : jm)
529  if (joint->getType() == moveit::core::JointModel::PLANAR)
530  {
531  joint->setVariableBounds(joint->getName() + "/" + joint->getLocalVariableNames()[0], bx);
532  joint->setVariableBounds(joint->getName() + "/" + joint->getLocalVariableNames()[1], by);
533  }
534  else if (joint->getType() == moveit::core::JointModel::FLOATING)
535  {
536  joint->setVariableBounds(joint->getName() + "/" + joint->getLocalVariableNames()[0], bx);
537  joint->setVariableBounds(joint->getName() + "/" + joint->getLocalVariableNames()[1], by);
538  joint->setVariableBounds(joint->getName() + "/" + joint->getLocalVariableNames()[2], bz);
539  }
540  }
541 }
