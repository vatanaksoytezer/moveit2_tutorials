59 {

75 void MoveGroupMoveAction::executeMoveCallback(std::shared_ptr<MGActionGoal> goal)
76 {
77  RCLCPP_INFO(LOGGER, "executing..");
78  setMoveState(PLANNING, goal);
79  // before we start planning, ensure that we have the latest robot state received...
80  auto node = context_->moveit_cpp_->getNode();
81  context_->planning_scene_monitor_->waitForCurrentRobotState(node->get_clock()->now());
82  context_->planning_scene_monitor_->updateFrameTransforms();
83 
84  auto action_res = std::make_shared<MGAction::Result>();
85  if (goal->get_goal()->planning_options.plan_only || !context_->allow_trajectory_execution_)
86  {
87  if (!goal->get_goal()->planning_options.plan_only)
88  RCLCPP_WARN(LOGGER, "This instance of MoveGroup is not allowed to execute trajectories "
89  "but the goal request has plan_only set to false. "
90  "Only a motion plan will be computed anyway.");
91  executeMoveCallbackPlanOnly(goal, action_res);
92  }
93  else
94  executeMoveCallbackPlanAndExecute(goal, action_res);
95 
96  bool planned_trajectory_empty = trajectory_processing::isTrajectoryEmpty(action_res->planned_trajectory);
97  // @todo: Response messages
98  RCLCPP_INFO_STREAM(LOGGER, getActionResultString(action_res->error_code, planned_trajectory_empty,
99  goal->get_goal()->planning_options.plan_only));
100  if (action_res->error_code.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS)
101  goal->succeed(action_res);
102  else if (action_res->error_code.val == moveit_msgs::msg::MoveItErrorCodes::PREEMPTED)
103  goal->canceled(action_res);
104  else
105  goal->abort(action_res);
106 
107  setMoveState(IDLE, goal);
108  preempt_requested_ = false;
109 }

111 void MoveGroupMoveAction::executeMoveCallbackPlanAndExecute(const std::shared_ptr<MGActionGoal>& goal,
112  std::shared_ptr<MGAction::Result>& action_res)
113 {
114  RCLCPP_INFO(LOGGER, "Combined planning and execution request received for MoveGroup action. "
115  "Forwarding to planning and execution pipeline.");
116 
117  if (moveit::core::isEmpty(goal->get_goal()->planning_options.planning_scene_diff))
118  {
119  planning_scene_monitor::LockedPlanningSceneRO lscene(context_->planning_scene_monitor_);
120  const moveit::core::RobotState& current_state = lscene->getCurrentState();
121 
122  // check to see if the desired constraints are already met
123  for (std::size_t i = 0; i < goal->get_goal()->request.goal_constraints.size(); ++i)
124  if (lscene->isStateConstrained(
125  current_state, kinematic_constraints::mergeConstraints(goal->get_goal()->request.goal_constraints[i],
126  goal->get_goal()->request.path_constraints)))
127  {
128  RCLCPP_INFO(LOGGER, "Goal constraints are already satisfied. No need to plan or execute any motions");
129  action_res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
130  return;
131  }
132  }
133 
134  plan_execution::PlanExecution::Options opt;
135 
136  const moveit_msgs::msg::MotionPlanRequest& motion_plan_request =
137  moveit::core::isEmpty(goal->get_goal()->request.start_state) ? goal->get_goal()->request :
138  clearRequestStartState(goal->get_goal()->request);
139  const moveit_msgs::msg::PlanningScene& planning_scene_diff =
140  moveit::core::isEmpty(goal->get_goal()->planning_options.planning_scene_diff.robot_state) ?
141  goal->get_goal()->planning_options.planning_scene_diff :
142  clearSceneRobotState(goal->get_goal()->planning_options.planning_scene_diff);
143 
144  opt.replan_ = goal->get_goal()->planning_options.replan;
145  opt.replan_attempts_ = goal->get_goal()->planning_options.replan_attempts;
146  opt.replan_delay_ = goal->get_goal()->planning_options.replan_delay;
147  opt.before_execution_callback_ = [this] { startMoveExecutionCallback(); };
148 
149  opt.plan_callback_ = [this, &motion_plan_request](plan_execution::ExecutableMotionPlan& plan) {
150  return planUsingPlanningPipeline(motion_plan_request, plan);
151  };
152  if (goal->get_goal()->planning_options.look_around && context_->plan_with_sensing_)
153  {
154  opt.plan_callback_ = [plan_with_sensing = context_->plan_with_sensing_.get(), planner = opt.plan_callback_,
155  attempts = goal->get_goal()->planning_options.look_around_attempts,
156  safe_execution_cost = goal->get_goal()->planning_options.max_safe_execution_cost](
157  plan_execution::ExecutableMotionPlan& plan) {
158  return plan_with_sensing->computePlan(plan, planner, attempts, safe_execution_cost);
159  };
160 
161  context_->plan_with_sensing_->setBeforeLookCallback([this]() { return startMoveLookCallback(); });
162  }
163 
164  plan_execution::ExecutableMotionPlan plan;
165  if (preempt_requested_)
166  {
167  RCLCPP_INFO(LOGGER, "Preempt requested before the goal is planned and executed.");
168  action_res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::PREEMPTED;
169  return;
170  }
171 
172  context_->plan_execution_->planAndExecute(plan, planning_scene_diff, opt);
173 
174  convertToMsg(plan.plan_components_, action_res->trajectory_start, action_res->planned_trajectory);
175  if (plan.executed_trajectory_)
176  plan.executed_trajectory_->getRobotTrajectoryMsg(action_res->executed_trajectory);
177  action_res->error_code = plan.error_code_;
178 }

180 void MoveGroupMoveAction::executeMoveCallbackPlanOnly(const std::shared_ptr<MGActionGoal>& goal,
181  std::shared_ptr<MGAction::Result>& action_res)
182 {
183  RCLCPP_INFO(LOGGER, "Planning request received for MoveGroup action. Forwarding to planning pipeline.");
184 
185  // lock the scene so that it does not modify the world representation while diff() is called
186  planning_scene_monitor::LockedPlanningSceneRO lscene(context_->planning_scene_monitor_);
187  const planning_scene::PlanningSceneConstPtr& the_scene =
188  (moveit::core::isEmpty(goal->get_goal()->planning_options.planning_scene_diff)) ?
189  static_cast<const planning_scene::PlanningSceneConstPtr&>(lscene) :
190  lscene->diff(goal->get_goal()->planning_options.planning_scene_diff);
191  planning_interface::MotionPlanResponse res;
192 
193  if (preempt_requested_)
194  {
195  RCLCPP_INFO(LOGGER, "Preempt requested before the goal is planned.");
196  action_res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::PREEMPTED;
197  return;
198  }
199 
200  // Select planning_pipeline to handle request
201  const planning_pipeline::PlanningPipelinePtr planning_pipeline =
202  resolvePlanningPipeline(goal->get_goal()->request.pipeline_id);
203  if (!planning_pipeline)
204  {
205  action_res->error_code.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
206  return;
207  }
208 
209  try
210  {
211  planning_pipeline->generatePlan(the_scene, goal->get_goal()->request, res);
212  }
213  catch (std::exception& ex)
214  {
215  RCLCPP_ERROR(LOGGER, "Planning pipeline threw an exception: %s", ex.what());
216  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
217  }
218 
219  convertToMsg(res.trajectory_, action_res->trajectory_start, action_res->planned_trajectory);
220  action_res->error_code = res.error_code_;
221  action_res->planning_time = res.planning_time_;
222 }

224 bool MoveGroupMoveAction::planUsingPlanningPipeline(const planning_interface::MotionPlanRequest& req,
225  plan_execution::ExecutableMotionPlan& plan)
226 {
227  setMoveState(PLANNING, nullptr);
228 
229  bool solved = false;
230  planning_interface::MotionPlanResponse res;
231 
232  // Select planning_pipeline to handle request
233  const planning_pipeline::PlanningPipelinePtr planning_pipeline = resolvePlanningPipeline(req.pipeline_id);
234  if (!planning_pipeline)
235  {
236  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
237  return solved;
238  }
239 
240  planning_scene_monitor::LockedPlanningSceneRO lscene(plan.planning_scene_monitor_);
241  try
242  {
243  solved = planning_pipeline->generatePlan(plan.planning_scene_, req, res);
244  }
245  catch (std::exception& ex)
246  {
247  RCLCPP_ERROR(LOGGER, "Planning pipeline threw an exception: %s", ex.what());
248  res.error_code_.val = moveit_msgs::msg::MoveItErrorCodes::FAILURE;
249  }
250  if (res.trajectory_)
251  {
252  plan.plan_components_.resize(1);
253  plan.plan_components_[0].trajectory_ = res.trajectory_;
254  plan.plan_components_[0].description_ = "plan";
255  }
256  plan.error_code_ = res.error_code_;
257 
258  return solved;
259 }
