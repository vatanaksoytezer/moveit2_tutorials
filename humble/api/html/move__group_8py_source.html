199  def set_joint_value_target(self, arg1, arg2=None, arg3=None):
200  """
201  Specify a target joint configuration for the group.
202  - if the type of arg1 is one of the following: dict, list, JointState message, then no other arguments should be provided.
203  The dict should specify pairs of joint variable names and their target values, the list should specify all the variable values
204  for the group. The JointState message specifies the positions of some single-dof joints.
205  - if the type of arg1 is string, then arg2 is expected to be defined and be either a real value or a list of real values. This is
206  interpreted as setting a particular joint to a particular value.
207  - if the type of arg1 is Pose or PoseStamped, both arg2 and arg3 could be defined. If arg2 or arg3 are defined, their types must
208  be either string or bool. The string type argument is interpreted as the end-effector the pose is specified for (default is to use
209  the default end-effector), and the bool is used to decide whether the pose specified is approximate (default is false). This situation
210  allows setting the joint target of the group by calling IK. This does not send a pose to the planner and the planner will do no IK.
211  Instead, one IK solution will be computed first, and that will be sent to the planner.
212  """
213  if isinstance(arg1, RobotState):
214  if not self._g.set_state_value_target(conversions.msg_to_string(arg1)):
215  raise MoveItCommanderException(
216  "Error setting state target. Is the target state within bounds?"
217  )
218 
219  elif isinstance(arg1, JointState):
220  if arg2 is not None or arg3 is not None:
221  raise MoveItCommanderException("Too many arguments specified")
222  if not self._g.set_joint_value_target_from_joint_state_message(
223  conversions.msg_to_string(arg1)
224  ):
225  raise MoveItCommanderException(
226  "Error setting joint target. Is the target within bounds?"
227  )
228 
229  elif isinstance(arg1, str):
230  if arg2 is None:
231  raise MoveItCommanderException(
232  "Joint value expected when joint name specified"
233  )
234  if arg3 is not None:
235  raise MoveItCommanderException("Too many arguments specified")
236  if not self._g.set_joint_value_target(arg1, arg2):
237  raise MoveItCommanderException(
238  "Error setting joint target. Is the target within bounds?"
239  )
240 
241  elif isinstance(arg1, (Pose, PoseStamped)):
242  approx = False
243  eef = ""
244  if arg2 is not None:
245  if type(arg2) is str:
246  eef = arg2
247  else:
248  if type(arg2) is bool:
249  approx = arg2
250  else:
251  raise MoveItCommanderException("Unexpected type")
252  if arg3 is not None:
253  if type(arg3) is str:
254  eef = arg3
255  else:
256  if type(arg3) is bool:
257  approx = arg3
258  else:
259  raise MoveItCommanderException("Unexpected type")
260  r = False
261  if type(arg1) is PoseStamped:
262  r = self._g.set_joint_value_target_from_pose_stamped(
263  conversions.msg_to_string(arg1), eef, approx
264  )
265  else:
266  r = self._g.set_joint_value_target_from_pose(
267  conversions.msg_to_string(arg1), eef, approx
268  )
269  if not r:
270  if approx:
271  raise MoveItCommanderException(
272  "Error setting joint target. Does your IK solver support approximate IK?"
273  )
274  else:
275  raise MoveItCommanderException(
276  "Error setting joint target. Is the IK solver functional?"
277  )
278 
279  elif hasattr(arg1, "__iter__"):
280  if arg2 is not None or arg3 is not None:
281  raise MoveItCommanderException("Too many arguments specified")
282  if not self._g.set_joint_value_target(arg1):
283  raise MoveItCommanderException(
284  "Error setting joint target. Is the target within bounds?"
285  )
286 
287  else:
288  raise MoveItCommanderException(
289  "Unsupported argument of type %s" % type(arg1)
290  )

370  def shift_pose_target(self, axis, value, end_effector_link=""):
371  """Get the current pose of the end effector, add value to the corresponding axis (0..5: X, Y, Z, R, P, Y) and set the new pose as the pose target"""
372  if len(end_effector_link) > 0 or self.has_end_effector_link():
373  pose = self._g.get_current_pose(end_effector_link)
374  # by default we get orientation as a quaternion list
375  # if we are updating a rotation axis however, we convert the orientation to RPY
376  if axis > 2:
377  (r, p, y) = tf.transformations.euler_from_quaternion(pose[3:])
378  pose = [pose[0], pose[1], pose[2], r, p, y]
379  if axis >= 0 and axis < 6:
380  pose[axis] = pose[axis] + value
381  self.set_pose_target(pose, end_effector_link)
382  else:
383  raise MoveItCommanderException("An axis value between 0 and 5 expected")
384  else:
385  raise MoveItCommanderException("There is no end effector to set poses for")
