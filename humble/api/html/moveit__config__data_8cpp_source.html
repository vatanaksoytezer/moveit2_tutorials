510  std::find(ros_control_config.joints_.begin(), ros_control_config.joints_.end(), joint_name);

512  {
513  if (ros_control_config.type_.substr(0, 8) == "position")
514  return "hardware_interface/PositionJointInterface";

516  return "hardware_interface/VelocityJointInterface";
517  // As of writing this, available joint command interfaces are position, velocity and effort.
518  else
519  return "hardware_interface/EffortJointInterface";
520  }
521  }
522  // If the joint was not found in any controller return EffortJointInterface
523  return "hardware_interface/EffortJointInterface";

623  for (const srdf::Model::Group& group : srdf_->groups_)

629  continue;
