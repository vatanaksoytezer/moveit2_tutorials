1 """Simplify loading moveit config parameters.
2 
3 This module provides builder-pattern based class to simplify loading moveit related parameters found in
4 robot_moveit_config package generated by moveit setup assistant.
5 
6 By default it expects the following structure for the moveit configs package
7 
8 robot_name_moveit_config/
9  .setup_assistant -> Used to retrieve information about the SRDF file and
10  the URDF file used when generating the package
11  config/
12  kinematics.yaml -> IK solver's parameters
13  joint_limits.yaml -> Overriding position/velocity/acceleration limits from the URDF file
14  moveit_cpp.yaml -> MoveItCpp related parameters
15  *_planning.yaml -> planning pipelines parameters
16  cartesian_limits.yaml -> Pilz planner parameters
17  # TODO(JafarAbdi): Check to see if this is a good default value
18  robot_name_controllers.yaml -> trajectory execution manager's parameters
19  ...
20 
21 Example:
22  moveit_configs = MoveItConfigsBuilder("robot_name").to_moveit_configs()
23  ...
24  moveit_configs.package_path
25  moveit_configs.robot_description
26  moveit_configs.robot_description_semantic
27  moveit_configs.robot_description_kinematics
28  moveit_configs.planning_pipelines
29  moveit_configs.trajectory_execution
30  moveit_configs.planning_scene_monitor
31  moveit_configs.move_group_capabilities
32  moveit_configs.joint_limits
33  moveit_configs.moveit_cpp
34  moveit_configs.cartesian_limits
35  # Or to get all the parameters as a dictionary
36  moveit_configs.to_dict()
37 
38 Each function in MoveItConfigsBuilder has a file_path as an argument which is used to override the default

40

66 def get_pattern_matches(folder, pattern):
67  """Given all the files in the folder, find those that match the pattern.
68 
69  If there are groups defined, the groups are returned. Otherwise the path to the matches are returned.
70  """
71  matches = []
72  if not folder.exists():
73  return matches
74  for child in folder.iterdir():
75  if not child.is_file():
76  continue
77  m = pattern.search(child.name)
78  if m:
79  groups = m.groups()
80  if groups:
81  matches.append(groups[0])
82  else:
83  matches.append(child)
84  return matches

359  # If no pipelines are specified, search by filename

362  pipelines = get_pattern_matches(config_folder, planning_pattern)
363  if load_all:

365  if pipeline not in pipelines:
366  pipelines.append(pipeline)
367 
368  # Define default pipeline as needed
369  if not default_planning_pipeline:
370  if "ompl" in pipelines:
371  default_planning_pipeline = "ompl"
372  else:
373  default_planning_pipeline = pipelines[0]

378  f"`{','.join(pipelines)}`"
379  )
380  self.__moveit_configs.planning_pipelines = {
381  "planning_pipelines": pipelines,
382  "default_planning_pipeline": default_planning_pipeline,
383  }
