96 bool MoveItCpp::loadPlanningSceneMonitor(const PlanningSceneMonitorOptions& options)
97 {
98  planning_scene_monitor_ =
99  std::make_shared<planning_scene_monitor::PlanningSceneMonitor>(node_, options.robot_description, options.name);
100  // Allows us to synchronize to Rviz and also publish collision objects to ourselves
101  RCLCPP_DEBUG(LOGGER, "Configuring Planning Scene Monitor");
102  if (planning_scene_monitor_->getPlanningScene())
103  {
104  // Start state and scene monitors
105  RCLCPP_INFO(LOGGER, "Listening to '%s' for joint states", options.joint_state_topic.c_str());
106  // Subscribe to JointState sensor messages
107  planning_scene_monitor_->startStateMonitor(options.joint_state_topic, options.attached_collision_object_topic);
108  // Publish planning scene updates to remote monitors like RViz
109  planning_scene_monitor_->startPublishingPlanningScene(planning_scene_monitor::PlanningSceneMonitor::UPDATE_SCENE,
110  options.monitored_planning_scene_topic);
111  // Monitor and apply planning scene updates from remote publishers like the PlanningSceneInterface
112  planning_scene_monitor_->startSceneMonitor(options.publish_planning_scene_topic);
113  // Monitor requests for changes in the collision environment
114  planning_scene_monitor_->startWorldGeometryMonitor();
115  }
116  else
117  {
118  RCLCPP_ERROR(LOGGER, "Planning scene not configured");
119  return false;
120  }
121 
122  // Wait for complete state to be received
123  // TODO(henningkayser): Fix segfault in waitForCurrentState()

126  // return planning_scene_monitor_->getStateMonitor()->waitForCurrentState(node_->now(),
127  // options.wait_for_initial_state_timeout);

130  return true;

133 bool MoveItCpp::loadPlanningPipelines(const PlanningPipelineOptions& options)
134 {
135  // TODO(henningkayser): Use parent namespace for planning pipeline config lookup
136  // ros::NodeHandle node_handle(options.parent_namespace.empty() ? "~" : options.parent_namespace);
137  for (const auto& planning_pipeline_name : options.pipeline_names)
138  {
139  if (planning_pipelines_.count(planning_pipeline_name) > 0)
140  {
141  RCLCPP_WARN(LOGGER, "Skipping duplicate entry for planning pipeline '%s'.", planning_pipeline_name.c_str());
142  continue;
143  }
144  RCLCPP_INFO(LOGGER, "Loading planning pipeline '%s'", planning_pipeline_name.c_str());
145  planning_pipeline::PlanningPipelinePtr pipeline;
146  pipeline = std::make_shared<planning_pipeline::PlanningPipeline>(robot_model_, node_, planning_pipeline_name,
147  PLANNING_PLUGIN_PARAM);
148 
149  if (!pipeline->getPlannerManager())
150  {
151  RCLCPP_ERROR(LOGGER, "Failed to initialize planning pipeline '%s'.", planning_pipeline_name.c_str());
152  continue;
153  }
154 
155  planning_pipelines_[planning_pipeline_name] = pipeline;
156  }
157 
158  if (planning_pipelines_.empty())
159  {
160  RCLCPP_ERROR(LOGGER, "Failed to load any planning pipelines.");
161  return false;
162  }
163 
164  // Retrieve group/pipeline mapping for faster lookup
165  std::vector<std::string> group_names = robot_model_->getJointModelGroupNames();
166  for (const auto& pipeline_entry : planning_pipelines_)
167  {
168  for (const auto& group_name : group_names)
169  {
170  const auto& pipeline = pipeline_entry.second;
171  for (const auto& planner_configuration : pipeline->getPlannerManager()->getPlannerConfigurations())
172  {
173  if (planner_configuration.second.group == group_name)
174  {
175  groups_pipelines_map_[group_name].insert(pipeline_entry.first);
176  }
177  }
178  }
179  }
180 
181  return true;
182 }

194 bool MoveItCpp::getCurrentState(moveit::core::RobotStatePtr& current_state, double wait_seconds)
195 {
196  if (wait_seconds > 0.0 && !planning_scene_monitor_->getStateMonitor()->waitForCurrentState(node_->now(), wait_seconds))
197  {
198  RCLCPP_ERROR(LOGGER, "Did not receive robot state");
199  return false;
200  }
201  { // Lock planning scene
202  planning_scene_monitor::LockedPlanningSceneRO scene(planning_scene_monitor_);
203  current_state = std::make_shared<moveit::core::RobotState>(scene->getCurrentState());
204  } // Unlock planning scene
205  return true;

252 moveit_controller_manager::ExecutionStatus
253 MoveItCpp::execute(const std::string& group_name, const robot_trajectory::RobotTrajectoryPtr& robot_trajectory,
254  bool blocking)
255 {
256  if (!robot_trajectory)
257  {
258  RCLCPP_ERROR(LOGGER, "Robot trajectory is undefined");
259  return moveit_controller_manager::ExecutionStatus::ABORTED;
260  }
261 
262  // Check if there are controllers that can handle the execution
263  if (!trajectory_execution_manager_->ensureActiveControllersForGroup(group_name))
264  {
265  RCLCPP_ERROR(LOGGER, "Execution failed! No active controllers configured for group '%s'", group_name.c_str());
266  return moveit_controller_manager::ExecutionStatus::ABORTED;
267  }
268 
269  // Execute trajectory
270  moveit_msgs::msg::RobotTrajectory robot_trajectory_msg;
271  robot_trajectory->getRobotTrajectoryMsg(robot_trajectory_msg);
272  if (blocking)
273  {
274  trajectory_execution_manager_->push(robot_trajectory_msg);
275  trajectory_execution_manager_->execute();
276  return trajectory_execution_manager_->waitForExecution();
277  }
278  trajectory_execution_manager_->pushAndExecute(robot_trajectory_msg);
279  return moveit_controller_manager::ExecutionStatus::RUNNING;
280 }
