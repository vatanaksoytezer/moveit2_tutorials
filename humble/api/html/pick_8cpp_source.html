67 bool PickPlan::plan(const planning_scene::PlanningSceneConstPtr& planning_scene,
68  const moveit_msgs::action::PickupGoal& goal)
69 {
70  double timeout = goal.allowed_planning_time;
71  ros::WallTime endtime = ros::WallTime::now() + ros::WallDuration(timeout);
72 
73  std::string planning_group = goal.group_name;
74  std::string end_effector = goal.end_effector;
75  if (end_effector.empty() && !planning_group.empty())
76  {
77  const moveit::core::JointModelGroup* jmg = planning_scene->getRobotModel()->getJointModelGroup(planning_group);
78  if (!jmg)
79  {
80  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
81  return false;
82  }
83  const std::vector<std::string>& eefs = jmg->getAttachedEndEffectorNames();
84  if (!eefs.empty())
85  {
86  end_effector = eefs.front();
87  if (eefs.size() > 1)
88  ROS_WARN_STREAM_NAMED("manipulation", "Choice of end-effector for group '"
89  << planning_group << "' is ambiguous. Assuming '" << end_effector
90  << "'");
91  }
92  }
93  else if (!end_effector.empty() && planning_group.empty())
94  {
95  const moveit::core::JointModelGroup* jmg = planning_scene->getRobotModel()->getEndEffector(end_effector);
96  if (!jmg)
97  {
98  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
99  return false;
100  }
101  planning_group = jmg->getEndEffectorParentGroup().first;
102  if (planning_group.empty())
103  {
104  ROS_ERROR_STREAM_NAMED("manipulation", "No parent group to plan in was identified based on end-effector '"
105  << end_effector << "'. Please define a parent group in the SRDF.");
106  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
107  return false;
108  }
109  else
110  ROS_INFO_STREAM_NAMED("manipulation", "Assuming the planning group for end effector '" << end_effector << "' is '"
111  << planning_group << "'");
112  }
113  const moveit::core::JointModelGroup* eef =
114  end_effector.empty() ? nullptr : planning_scene->getRobotModel()->getEndEffector(end_effector);
115  if (!eef)
116  {
117  ROS_ERROR_NAMED("manipulation", "No end-effector specified for pick action");
118  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::INVALID_GROUP_NAME;
119  return false;
120  }
121  const std::string& ik_link = eef->getEndEffectorParentGroup().second;
122 
123  ros::WallTime start_time = ros::WallTime::now();
124 
125  // construct common data for possible manipulation plans
126  auto plan_data = std::make_shared<ManipulationPlanSharedData>();
127  ManipulationPlanSharedDataConstPtr const_plan_data = plan_data;
128  plan_data->planning_group_ = planning_scene->getRobotModel()->getJointModelGroup(planning_group);
129  plan_data->end_effector_group_ = eef;
130  plan_data->ik_link_ = planning_scene->getRobotModel()->getLinkModel(ik_link);
131  plan_data->timeout_ = endtime;
132  plan_data->path_constraints_ = goal.path_constraints;
133  plan_data->planner_id_ = goal.planner_id;
134  plan_data->minimize_object_distance_ = goal.minimize_object_distance;
135  plan_data->max_goal_sampling_attempts_ = 2;
136  moveit_msgs::msg::AttachedCollisionObject& attach_object_msg = plan_data->diff_attached_object_;
137 
138  // construct the attached object message that will change the world to what it would become after a pick
139  attach_object_msg.link_name = ik_link;
140  attach_object_msg.object.id = goal.target_name;
141  attach_object_msg.object.operation = moveit_msgs::msg::CollisionObject::ADD;
142  attach_object_msg.touch_links =
143  goal.attached_object_touch_links.empty() ? eef->getLinkModelNames() : goal.attached_object_touch_links;
144  collision_detection::AllowedCollisionMatrixPtr approach_grasp_acm(
145  new collision_detection::AllowedCollisionMatrix(planning_scene->getAllowedCollisionMatrix()));
146 
147  // we are allowed to touch the target object
148  approach_grasp_acm->setEntry(goal.target_name, attach_object_msg.touch_links, true);
149  // we are allowed to touch certain other objects with the gripper
150  approach_grasp_acm->setEntry(eef->getLinkModelNames(), goal.allowed_touch_objects, true);
151  if (!goal.support_surface_name.empty())
152  {
153  // we are allowed to have contact between the target object and the support surface before the grasp
154  approach_grasp_acm->setEntry(goal.support_surface_name, goal.target_name, true);
155 
156  // optionally, it may be allowed to touch the support surface with the gripper
157  if (goal.allow_gripper_support_collision)
158  approach_grasp_acm->setEntry(goal.support_surface_name, eef->getLinkModelNames(), true);
159  }
160 
161  // configure the manipulation pipeline
162  pipeline_.reset();
163  ManipulationStagePtr stage1(
164  new ReachableAndValidPoseFilter(planning_scene, approach_grasp_acm, pick_place_->getConstraintsSamplerManager()));
165  ManipulationStagePtr stage2 = std::make_shared<ApproachAndTranslateStage>(planning_scene, approach_grasp_acm);
166  ManipulationStagePtr stage3 = std::make_shared<PlanStage>(planning_scene, pick_place_->getPlanningPipeline());
167  pipeline_.addStage(stage1).addStage(stage2).addStage(stage3);
168 
169  initialize();
170  pipeline_.start();
171 
172  // order the grasps by quality
173  std::vector<std::size_t> grasp_order(goal.possible_grasps.size());
174  for (std::size_t i = 0; i < goal.possible_grasps.size(); ++i)
175  grasp_order[i] = i;
176  OrderGraspQuality oq(goal.possible_grasps);
177  // using stable_sort to preserve order of grasps with equal quality
178  std::stable_sort(grasp_order.begin(), grasp_order.end(), oq);
179 
180  // feed the available grasps to the stages we set up
181  for (std::size_t i = 0; i < goal.possible_grasps.size(); ++i)
182  {
183  auto p = std::make_shared<ManipulationPlan>(const_plan_data);
184  const moveit_msgs::msg::Grasp& g = goal.possible_grasps[grasp_order[i]];
185  p->approach_ = g.pre_grasp_approach;
186  p->retreat_ = g.post_grasp_retreat;
187  p->goal_pose_ = g.grasp_pose;
188  p->id_ = grasp_order[i];
189  // if no frame of reference was specified, assume the transform to be in the reference frame of the object
190  if (p->goal_pose_.header.frame_id.empty())
191  p->goal_pose_.header.frame_id = goal.target_name;
192  p->approach_posture_ = g.pre_grasp_posture;
193  p->retreat_posture_ = g.grasp_posture;
194  pipeline_.push(p);
195  }
196 
197  // wait till we're done
198  waitForPipeline(endtime);
199  pipeline_.stop();
200 
201  last_plan_time_ = (ros::WallTime::now() - start_time).toSec();
202 
203  if (!getSuccessfulManipulationPlans().empty())
204  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
205  else
206  {
207  if (last_plan_time_ > timeout)
208  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::TIMED_OUT;
209  else
210  {
211  error_code_.val = moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED;
212  if (!goal.possible_grasps.empty())
213  {
214  ROS_WARN_NAMED("manipulation", "All supplied grasps failed. Retrying last grasp in verbose mode.");
215  // everything failed. we now start the pipeline again in verbose mode for one grasp
216  initialize();
217  pipeline_.setVerbose(true);
218  pipeline_.start();
219  pipeline_.reprocessLastFailure();
220  waitForPipeline(ros::WallTime::now() + ros::WallDuration(1.0));
221  pipeline_.stop();
222  pipeline_.setVerbose(false);
223  }
224  }
225  }
226  ROS_INFO_NAMED("manipulation", "Pickup planning completed after %lf seconds", last_plan_time_);
227 
228  return error_code_.val == moveit_msgs::msg::MoveItErrorCodes::SUCCESS;
229 }

231 PickPlanPtr PickPlace::planPick(const planning_scene::PlanningSceneConstPtr& planning_scene,
232  const moveit_msgs::action::PickupGoal& goal) const
233 {
234  PickPlanPtr p(new PickPlan(shared_from_this()));
235 
236  if (moveit::core::isEmpty(goal.planning_options.planning_scene_diff))
237  p->plan(planning_scene, goal);
238  else
239  p->plan(planning_scene->diff(goal.planning_options.planning_scene_diff), goal);
240 
241  if (display_computed_motion_plans_)
242  {
243  const std::vector<pick_place::ManipulationPlanPtr>& success = p->getSuccessfulManipulationPlans();
244  if (!success.empty())
245  visualizePlan(success.back());
246  }
247 
248  if (display_grasps_)
249  {
250  const std::vector<pick_place::ManipulationPlanPtr>& success = p->getSuccessfulManipulationPlans();
251  visualizeGrasps(success);
252  const std::vector<pick_place::ManipulationPlanPtr>& failed = p->getFailedManipulationPlans();
253  visualizeGrasps(failed);
254  }
255 
256  return p;
257 }
