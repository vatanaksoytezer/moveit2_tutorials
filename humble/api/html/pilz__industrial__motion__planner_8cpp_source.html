58 bool CommandPlanner::initialize(const moveit::core::RobotModelConstPtr& model, const rclcpp::Node::SharedPtr& node,
59  const std::string& ns)
60 {
61  // Call parent class initialize
62  planning_interface::PlannerManager::initialize(model, node, ns);
63 
64  // Store the model and the namespace
65  model_ = model;
66  namespace_ = ns;
67 
68  // Obtain the aggregated joint limits
69  aggregated_limit_active_joints_ = pilz_industrial_motion_planner::JointLimitsAggregator::getAggregatedLimits(
70  node, PARAM_NAMESPACE_LIMITS, model->getActiveJointModels());
71 
72  // Obtain cartesian limits
73  cartesian_limit_ =
74  pilz_industrial_motion_planner::CartesianLimitsAggregator::getAggregatedLimits(node, PARAM_NAMESPACE_LIMITS);
75 
76  // Load the planning context loader
77  planner_context_loader = std::make_unique<pluginlib::ClassLoader<PlanningContextLoader>>(
78  "pilz_industrial_motion_planner", "pilz_industrial_motion_planner::PlanningContextLoader");
79 
80  // List available plugins
81  const std::vector<std::string>& factories = planner_context_loader->getDeclaredClasses();
82  std::stringstream ss;
83  for (const auto& factory : factories)
84  {
85  ss << factory << " ";
86  }
87 
88  RCLCPP_INFO_STREAM(LOGGER, "Available plugins: " << ss.str());
89 
90  // Load each factory
91  for (const auto& factory : factories)
92  {
93  RCLCPP_INFO_STREAM(LOGGER, "About to load: " << factory);
94  PlanningContextLoaderPtr loader_pointer(planner_context_loader->createSharedInstance(factory));
95 
96  pilz_industrial_motion_planner::LimitsContainer limits;
97  limits.setJointLimits(aggregated_limit_active_joints_);
98  limits.setCartesianLimits(cartesian_limit_);
99 
100  loader_pointer->setLimits(limits);
101  loader_pointer->setModel(model_);
102 
103  registerContextLoader(loader_pointer);
104  }
105 
106  return true;
107 }

124 planning_interface::PlanningContextPtr
125 CommandPlanner::getPlanningContext(const planning_scene::PlanningSceneConstPtr& planning_scene,
126  const moveit_msgs::msg::MotionPlanRequest& req,
127  moveit_msgs::msg::MoveItErrorCodes& error_code) const
128 {
129  // TODO(henningkayser): print req
130  // RCLCPP_DEBUG_STREAM(LOGGER, "Loading PlanningContext for request\n<request>\n" << req << "\n</request>");
131  RCLCPP_DEBUG(LOGGER, "Loading PlanningContext");
132 
133  // Check that a loaded for this request exists
134  if (!canServiceRequest(req))
135  {
136  RCLCPP_ERROR_STREAM(LOGGER, "No ContextLoader for planner_id '" << req.planner_id.c_str()
137  << "' found. Planning not possible.");
138  return nullptr;
139  }
140 
141  planning_interface::PlanningContextPtr planning_context;
142 
143  if (context_loader_map_.at(req.planner_id)->loadContext(planning_context, req.planner_id, req.group_name))
144  {
145  RCLCPP_DEBUG_STREAM(LOGGER, "Found planning context loader for " << req.planner_id << " group:" << req.group_name);
146  planning_context->setMotionPlanRequest(req);
147  planning_context->setPlanningScene(planning_scene);
148  return planning_context;
149  }
150  else
151  {
152  error_code.val = moveit_msgs::msg::MoveItErrorCodes::PLANNING_FAILED;
153  return nullptr;
154  }
155 }

162 void CommandPlanner::registerContextLoader(
163  const pilz_industrial_motion_planner::PlanningContextLoaderPtr& planning_context_loader)
164 {
165  // Only add if command is not already in list, throw exception if not
166  if (context_loader_map_.find(planning_context_loader->getAlgorithm()) == context_loader_map_.end())
167  {
168  context_loader_map_[planning_context_loader->getAlgorithm()] = planning_context_loader;
169  RCLCPP_INFO_STREAM(LOGGER, "Registered Algorithm [" << planning_context_loader->getAlgorithm() << "]");
170  }
171  else
172  {
173  throw ContextLoaderRegistrationException("The command [" + planning_context_loader->getAlgorithm() +
174  "] is already registered");
175  }
176 }
