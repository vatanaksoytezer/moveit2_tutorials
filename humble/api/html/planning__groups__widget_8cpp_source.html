280 // ******************************************************************************************
281 // Recursively Adds Groups, and subgroups to groups...
282 // ******************************************************************************************
283 void PlanningGroupsWidget::loadGroupsTreeRecursive(srdf::Model::Group& group_it, QTreeWidgetItem* parent)
284 {
285  // Fonts for tree
286  const QFont top_level_font(QFont().defaultFamily(), 11, QFont::Bold);
287  const QFont type_font(QFont().defaultFamily(), 11, QFont::Normal, QFont::StyleItalic);
288 
289  QTreeWidgetItem* group;
290 
291  // Allow a subgroup to open into a whole new group
292  if (parent == nullptr)
293  {
294  group = new QTreeWidgetItem(groups_tree_);
295  group->setText(0, group_it.name_.c_str());
296  group->setFont(0, top_level_font);
297  group->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, GROUP)));
298  groups_tree_->addTopLevelItem(group);
299  }
300  else
301  {
302  group = new QTreeWidgetItem(parent);
303  group->setText(0, group_it.name_.c_str());
304  group->setFont(0, top_level_font);
305  group->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, GROUP)));
306  parent->addChild(group);
307  }
308 
309  // Joints --------------------------------------------------------------
310  QTreeWidgetItem* joints = new QTreeWidgetItem(group);
311  joints->setText(0, "Joints");
312  joints->setFont(0, type_font);
313  joints->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, JOINT)));
314  group->addChild(joints);
315 
316  // Retrieve pointer to the shared kinematic model
317  const moveit::core::RobotModelConstPtr& model = config_data_->getRobotModel();
318 
319  // Loop through all aval. joints
320  for (std::vector<std::string>::const_iterator joint_it = group_it.joints_.begin(); joint_it != group_it.joints_.end();
321  ++joint_it)
322  {
323  QTreeWidgetItem* j = new QTreeWidgetItem(joints);
324  j->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, JOINT)));
325  std::string joint_name;
326 
327  // Get the type of joint this is
328  const moveit::core::JointModel* jm = model->getJointModel(*joint_it);
329  if (jm) // check if joint model was found
330  {
331  joint_name = *joint_it + " - " + jm->getTypeName();
332  }
333  else
334  {
335  joint_name = *joint_it;
336  }
337 
338  // Add to tree
339  j->setText(0, joint_name.c_str());
340  joints->addChild(j);
341  }
342 
343  // Links -------------------------------------------------------------
344  QTreeWidgetItem* links = new QTreeWidgetItem(group);
345  links->setText(0, "Links");
346  links->setFont(0, type_font);
347  links->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, LINK)));
348  group->addChild(links);
349 
350  // Loop through all aval. links
351  for (std::vector<std::string>::const_iterator joint_it = group_it.links_.begin(); joint_it != group_it.links_.end();
352  ++joint_it)
353  {
354  QTreeWidgetItem* j = new QTreeWidgetItem(links);
355  j->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, LINK)));
356  j->setText(0, joint_it->c_str());
357  links->addChild(j);
358  }
359 
360  // Chains -------------------------------------------------------------
361  QTreeWidgetItem* chains = new QTreeWidgetItem(group);
362  chains->setText(0, "Chain");
363  chains->setFont(0, type_font);
364  chains->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, CHAIN)));
365  group->addChild(chains);
366 
367  // Warn if there is more than 1 chain per group
368  static bool warn_once = true;
369  if (group_it.chains_.size() > 1 && warn_once)
370  {
371  warn_once = false;
372  QMessageBox::warning(this, "Group with Multiple Kinematic Chains",
373  "Warning: this MoveIt Setup Assistant is only designed to handle one kinematic chain per "
374  "group. The loaded SRDF has more than one kinematic chain for a group. A possible loss of "
375  "data may occur.");
376  }
377 
378  // Loop through all aval. chains
379  for (std::vector<std::pair<std::string, std::string> >::const_iterator chain_it = group_it.chains_.begin();
380  chain_it != group_it.chains_.end(); ++chain_it)
381  {
382  QTreeWidgetItem* j = new QTreeWidgetItem(chains);
383  j->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, CHAIN)));
384  j->setText(0, QString(chain_it->first.c_str()).append(" -> ").append(chain_it->second.c_str()));
385  chains->addChild(j);
386  }
387 
388  // Subgroups -------------------------------------------------------------
389  QTreeWidgetItem* subgroups = new QTreeWidgetItem(group);
390  subgroups->setText(0, "Subgroups");
391  subgroups->setFont(0, type_font);
392  subgroups->setData(0, Qt::UserRole, QVariant::fromValue(PlanGroupType(&group_it, SUBGROUP)));
393  group->addChild(subgroups);
394 
395  // Loop through all aval. subgroups
396  for (std::vector<std::string>::iterator subgroup_it = group_it.subgroups_.begin();
397  subgroup_it != group_it.subgroups_.end(); ++subgroup_it)
398  {
399  // Find group with this subgroups' name
400 
401  srdf::Model::Group* searched_group = nullptr; // used for holding our search results
402 
403  for (srdf::Model::Group& group : config_data_->srdf_->groups_)
404  {
405  if (group.name_ == *subgroup_it) // this is the group we are looking for
406  {
407  searched_group = &group; // convert to pointer from iterator
408  break; // we are done searching
409  }
410  }
411 
412  // Check if subgroup was found
413  if (searched_group == nullptr) // not found
414  {
415  QMessageBox::critical(this, "Error Loading SRDF",
416  QString("Subgroup '")
417  .append(subgroup_it->c_str())
418  .append("' of group '")
419  .append(group_it.name_.c_str())
420  .append("' not found. Your SRDF is invalid"));
421  return; // TODO: something better for error handling?
422  }
423 
424  // subgroup found!
425 
426  // Recurse this function for each new group
427  loadGroupsTreeRecursive(*searched_group, subgroups);
428  }
429 }

487  return;
488  }

651 // ******************************************************************************************
652 void PlanningGroupsWidget::deleteGroup()
653 {
654  std::string group = current_edit_group_;
655  if (group.empty())
656  {
657  QTreeWidgetItem* item = groups_tree_->currentItem();
658  // Check that something was actually selected
659  if (item == nullptr)
660  return;
661  // Get the user custom properties of the currently selected row
662  PlanGroupType plan_group = item->data(0, Qt::UserRole).value<PlanGroupType>();
663  if (plan_group.group_)
664  group = plan_group.group_->name_;
665  }
666  else
667  current_edit_group_.clear();
668  if (group.empty())
669  return;
670 
671  // Find the group we are editing based on the goup name string
672  srdf::Model::Group* searched_group = config_data_->findGroupByName(group);
673 
674  // Confirm user wants to delete group
675  if (QMessageBox::question(this, "Confirm Group Deletion",
676  QString("Are you sure you want to delete the planning group '")
677  .append(searched_group->name_.c_str())
678  .append("'? This will also delete all references in subgroups, robot poses and end "
679  "effectors."),
680  QMessageBox::Ok | QMessageBox::Cancel) == QMessageBox::Cancel)
681  {
682  return;
683  }
684 
685  // delete all robot poses that use that group's name
686  bool have_confirmed_group_state_deletion = false;
687  bool have_deleted_group_state = true;
688  while (have_deleted_group_state)
689  {
690  have_deleted_group_state = false;
691  for (std::vector<srdf::Model::GroupState>::iterator pose_it = config_data_->srdf_->group_states_.begin();
692  pose_it != config_data_->srdf_->group_states_.end(); ++pose_it)
693  {
694  // check if this group state depends on the currently being deleted group
695  if (pose_it->group_ == searched_group->name_)
696  {
697  if (!have_confirmed_group_state_deletion)
698  {
699  have_confirmed_group_state_deletion = true;
700 
701  // confirm the user wants to delete group states
702  if (QMessageBox::question(
703  this, "Confirm Group State Deletion",
704  QString("The group that is about to be deleted has robot poses (robot states) that depend on this "
705  "group. Are you sure you want to delete this group as well as all dependent robot poses?"),
706  QMessageBox::Ok | QMessageBox::Cancel) == QMessageBox::Cancel)
707  {
708  return;
709  }
710  }
711  config_data_->changes |= MoveItConfigData::POSES;
712 
713  // the user has confirmed, now delete this group state
714  config_data_->srdf_->group_states_.erase(pose_it);
715  have_deleted_group_state = true;
716  break; // you can only delete 1 item in vector before invalidating iterator
717  }
718  }
719  }
720 
721  // delete all end effectors that use that group's name
722  bool have_confirmed_end_effector_deletion = false;
723  bool have_deleted_end_effector = true;
724  while (have_deleted_end_effector)
725  {
726  have_deleted_end_effector = false;
727  for (std::vector<srdf::Model::EndEffector>::iterator effector_it = config_data_->srdf_->end_effectors_.begin();
728  effector_it != config_data_->srdf_->end_effectors_.end(); ++effector_it)
729  {
730  // check if this group state depends on the currently being deleted group
731  if (effector_it->component_group_ == searched_group->name_)
732  {
733  if (!have_confirmed_end_effector_deletion)
734  {
735  have_confirmed_end_effector_deletion = true;
736 
737  // confirm the user wants to delete group states
738  if (QMessageBox::question(
739  this, "Confirm End Effector Deletion",
740  QString("The group that is about to be deleted has end effectors (grippers) that depend on this "
741  "group. Are you sure you want to delete this group as well as all dependent end effectors?"),
742  QMessageBox::Ok | QMessageBox::Cancel) == QMessageBox::Cancel)
743  {
744  return;
745  }
746  }
747  config_data_->changes |= MoveItConfigData::END_EFFECTORS;
748 
749  // the user has confirmed, now delete this group state
750  config_data_->srdf_->end_effectors_.erase(effector_it);
751  have_deleted_end_effector = true;
752  break; // you can only delete 1 item in vector before invalidating iterator
753  }
754  }
755  }
756 
757  config_data_->changes |= MoveItConfigData::GROUPS;
758 
759  // delete actual group
760  for (std::vector<srdf::Model::Group>::iterator group_it = config_data_->srdf_->groups_.begin();
761  group_it != config_data_->srdf_->groups_.end(); ++group_it)
762  {
763  // check if this is the group we want to delete
764  if (group_it->name_ == group) // string match
765  {
766  config_data_->srdf_->groups_.erase(group_it);
767  break;
768  }
769  }
770 
771  // loop again to delete all subgroup references
772  for (srdf::Model::Group& group_it : config_data_->srdf_->groups_)
773  {
774  // delete all subgroup references
775  bool deleted_subgroup = true;
776  while (deleted_subgroup)
777  {
778  deleted_subgroup = false;
779 
780  // check if the subgroups reference our deleted group
781  for (std::vector<std::string>::iterator subgroup_it = group_it.subgroups_.begin();
782  subgroup_it != group_it.subgroups_.end(); ++subgroup_it)
783  {
784  // Check if that subgroup references the deletion group. if so, delete it
785  if (subgroup_it->compare(group) == 0) // same name
786  {
787  group_it.subgroups_.erase(subgroup_it); // delete
788  deleted_subgroup = true;
789  break;
790  }
791  }
792  }
793  }
794 
795  // Switch to main screen
796  showMainScreen();
797 
798  // Reload main screen table
799  loadGroupsTree();
800 
801  // Update the kinematic model with changes
802  config_data_->updateRobotModel();
803 }

952 // ******************************************************************************************
953 // Call when subgroups edit screen is done and needs to be saved
954 // ******************************************************************************************
955 void PlanningGroupsWidget::saveSubgroupsScreen()
956 {
957  // Find the group we are editing based on the goup name string
958  srdf::Model::Group* searched_group = config_data_->findGroupByName(current_edit_group_);
959 
960  // Check for cycles -------------------------------
961 
962  // Create vector index of all nodes
963  std::map<std::string, int> group_nodes;
964 
965  // Create vector of all nodes for use as id's
966  int node_id = 0;
967  for (srdf::Model::Group& group : config_data_->srdf_->groups_)
968  {
969  // Add string to vector
970  group_nodes.insert(std::pair<std::string, int>(group.name_, node_id));
971  ++node_id;
972  }
973 
974  // Create the empty graph
975  typedef boost::adjacency_list<boost::vecS, boost::vecS, boost::bidirectionalS> Graph;
976  Graph g(group_nodes.size());
977 
978  // Traverse the group list again, this time inserting subgroups into graph
979  int from_id = 0; // track the outer node we are on to reduce searches performed
980  for (srdf::Model::Group& group : config_data_->srdf_->groups_)
981  {
982  // Check if group_it is same as current group
983  if (group.name_ == searched_group->name_) // yes, same group
984  {
985  // add new subgroup list from widget, not old one. this way we can check for new cycles
986  for (int i = 0; i < subgroups_widget_->selected_data_table_->rowCount(); ++i)
987  {
988  // Get std::string of subgroup
989  const std::string to_string = subgroups_widget_->selected_data_table_->item(i, 0)->text().toStdString();
990 
991  // convert subgroup string to associated id
992  int to_id = group_nodes[to_string];
993 
994  // Add edge from from_id to to_id
995  add_edge(from_id, to_id, g);
996  }
997  }
998  else // this group is not the group we are editing, so just add subgroups from memory
999  {
1000  // add new subgroup list from widget, not old one. this way we can check for new cycles
1001  for (const std::string& to_string : group.subgroups_)
1002  {
1003  // Get std::string of subgroup
1004  // convert subgroup string to associated id
1005  int to_id = group_nodes[to_string];
1006 
1007  // Add edge from from_id to to_id
1008  add_edge(from_id, to_id, g);
1009  }
1010  }
1011 
1012  ++from_id;
1013  }
1014 
1015  // Check for cycles
1016  bool has_cycle = false;
1017  CycleDetector vis(has_cycle);
1018  boost::depth_first_search(g, visitor(vis));
1019 
1020  if (has_cycle)
1021  {
1022  // report to user the error
1023  QMessageBox::warning(this, "Error Saving", "Depth first search reveals a cycle in the subgroups");
1024  return;
1025  }
1026 
1027  // clear the old data
1028  searched_group->subgroups_.clear();
1029 
1030  // copy the data
1031  for (int i = 0; i < subgroups_widget_->selected_data_table_->rowCount(); ++i)
1032  {
1033  searched_group->subgroups_.push_back(subgroups_widget_->selected_data_table_->item(i, 0)->text().toStdString());
1034  }
1035 
1036  // Switch to main screen
1037  showMainScreen();
1038 
1039  // Reload main screen table
1040  loadGroupsTree();
1041 
1042  // Update the kinematic model with changes
1043  config_data_->updateRobotModel();
1044  config_data_->changes |= MoveItConfigData::GROUP_CONTENTS;
1045 }

1047 // ******************************************************************************************
1048 // Call when groups edit screen is done and needs to be saved
1049 // ******************************************************************************************
1050 bool PlanningGroupsWidget::saveGroupScreen()
1051 {
1052  // Get a reference to the supplied strings
1053  const std::string& group_name = group_edit_widget_->group_name_field_->text().trimmed().toStdString();
1054  const std::string& kinematics_solver = group_edit_widget_->kinematics_solver_field_->currentText().toStdString();
1055  const std::string& default_planner = group_edit_widget_->default_planner_field_->currentText().toStdString();
1056  const std::string& kinematics_resolution = group_edit_widget_->kinematics_resolution_field_->text().toStdString();
1057  const std::string& kinematics_timeout = group_edit_widget_->kinematics_timeout_field_->text().toStdString();
1058  const std::string& kinematics_parameters_file =
1059  group_edit_widget_->kinematics_parameters_file_field_->text().toStdString();
1060 
1061  // Used for editing existing groups
1062  srdf::Model::Group* searched_group = nullptr;
1063 
1064  // Check that a valid group name has been given
1065  if (group_name.empty())
1066  {
1067  QMessageBox::warning(this, "Error Saving", "A name must be given for the group!");
1068  return false;
1069  }
1070 
1071  // Check if this is an existing group
1072  if (!current_edit_group_.empty())
1073  {
1074  // Find the group we are editing based on the goup name string
1075  searched_group = config_data_->findGroupByName(current_edit_group_);
1076  }
1077 
1078  // Check that the group name is unique
1079  for (const auto& group : config_data_->srdf_->groups_)
1080  {
1081  if (group.name_.compare(group_name) == 0) // the names are the same
1082  {
1083  // is this our existing group? check if group pointers are same
1084  if (&group != searched_group)
1085  {
1086  QMessageBox::warning(this, "Error Saving", "A group already exists with that name!");
1087  return false;
1088  }
1089  }
1090  }
1091 
1092  // Check that the resolution is an double number
1093  double kinematics_resolution_double;
1094  try
1095  {
1096  kinematics_resolution_double = boost::lexical_cast<double>(kinematics_resolution);
1097  }
1098  catch (boost::bad_lexical_cast&)
1099  {
1100  QMessageBox::warning(this, "Error Saving", "Unable to convert kinematics resolution to a double number.");
1101  return false;
1102  }
1103 
1104  // Check that the timeout is a double number
1105  double kinematics_timeout_double;
1106  try
1107  {
1108  kinematics_timeout_double = boost::lexical_cast<double>(kinematics_timeout);
1109  }
1110  catch (boost::bad_lexical_cast&)
1111  {
1112  QMessageBox::warning(this, "Error Saving", "Unable to convert kinematics solver timeout to a double number.");
1113  return false;
1114  }
1115 
1116  // Check that all numbers are >0
1117  if (kinematics_resolution_double <= 0)
1118  {
1119  QMessageBox::warning(this, "Error Saving", "Kinematics solver search resolution must be greater than 0.");
1120  return false;
1121  }
1122  if (kinematics_timeout_double <= 0)
1123  {
1124  QMessageBox::warning(this, "Error Saving", "Kinematics solver search timeout must be greater than 0.");
1125  return false;
1126  }
1127 
1128  adding_new_group_ = false;
1129 
1130  // Save the new group name or create the new group
1131  if (searched_group == nullptr) // create new
1132  {
1133  srdf::Model::Group new_group;
1134  new_group.name_ = group_name;
1135  config_data_->srdf_->groups_.push_back(new_group);
1136  adding_new_group_ = true; // remember this group is new
1137  config_data_->changes |= MoveItConfigData::GROUPS;
1138  }
1139  else
1140  {
1141  // Remember old group name
1142  const std::string old_group_name = searched_group->name_;
1143 
1144  // Change group name
1145  searched_group->name_ = group_name;
1146 
1147  // Change all references to this group name in other subgroups
1148  // Loop through every group
1149  for (std::vector<srdf::Model::Group>::iterator group_it = config_data_->srdf_->groups_.begin();
1150  group_it != config_data_->srdf_->groups_.end(); ++group_it)
1151  {
1152  // Loop through every subgroup
1153  for (std::string& subgroup : group_it->subgroups_)
1154  {
1155  // Check if that subgroup references old group name. if so, update it
1156  if (subgroup.compare(old_group_name) == 0) // same name
1157  {
1158  subgroup.assign(group_name); // updated
1159  config_data_->changes |= MoveItConfigData::GROUP_CONTENTS;
1160  }
1161  }
1162  }
1163 
1164  // Change all references to this group name in the end effectors screen
1165  for (std::vector<srdf::Model::EndEffector>::iterator eef_it = config_data_->srdf_->end_effectors_.begin();
1166  eef_it != config_data_->srdf_->end_effectors_.end(); ++eef_it)
1167  {
1168  // Check if this eef's parent group references old group name. if so, update it
1169  if (eef_it->parent_group_.compare(old_group_name) == 0) // same name
1170  {
1171  ROS_DEBUG_STREAM_NAMED("setup_assistant",
1172  "Changed eef '" << eef_it->name_ << "' to new parent group name " << group_name);
1173  eef_it->parent_group_ = group_name; // updated
1174  config_data_->changes |= MoveItConfigData::END_EFFECTORS;
1175  }
1176 
1177  // Check if this eef's group references old group name. if so, update it
1178  if (eef_it->component_group_.compare(old_group_name) == 0) // same name
1179  {
1180  ROS_DEBUG_STREAM_NAMED("setup_assistant",
1181  "Changed eef '" << eef_it->name_ << "' to new group name " << group_name);
1182  eef_it->component_group_ = group_name; // updated
1183  config_data_->changes |= MoveItConfigData::END_EFFECTORS;
1184  }
1185  }
1186 
1187  // Change all references to this group name in the robot poses screen
1188  for (std::vector<srdf::Model::GroupState>::iterator state_it = config_data_->srdf_->group_states_.begin();
1189  state_it != config_data_->srdf_->group_states_.end(); ++state_it)
1190  {
1191  // Check if this eef's parent group references old group name. if so, update it
1192  if (state_it->group_.compare(old_group_name) == 0) // same name
1193  {
1194  ROS_DEBUG_STREAM_NAMED("setup_assistant", "Changed group state group '" << state_it->group_
1195  << "' to new parent group name "
1196  << group_name);
1197  state_it->group_ = group_name; // updated
1198  config_data_->changes |= MoveItConfigData::POSES;
1199  }
1200  }
1201 
1202  // Now update the robot model based on our changed to the SRDF
1203  config_data_->updateRobotModel();
1204  }
1205 
1206  // Save the group meta data
1207  config_data_->group_meta_data_[group_name].kinematics_solver_ = kinematics_solver;
1208  config_data_->group_meta_data_[group_name].kinematics_solver_search_resolution_ = kinematics_resolution_double;
1209  config_data_->group_meta_data_[group_name].kinematics_solver_timeout_ = kinematics_timeout_double;
1210  config_data_->group_meta_data_[group_name].kinematics_parameters_file_ = kinematics_parameters_file;
1211  config_data_->group_meta_data_[group_name].default_planner_ = (default_planner == "None" ? "" : default_planner);
1212  config_data_->changes |= MoveItConfigData::GROUP_KINEMATICS;
1213 
1214  // Reload main screen table
1215  loadGroupsTree();
1216 
1217  // Update the current edit group so that we can proceed to the next screen, if user desires
1218  current_edit_group_ = group_name;
1219 
1220  return true;
1221 }
