223 void PlanningScene::allocateCollisionDetector(const collision_detection::CollisionDetectorAllocatorPtr& allocator,
224  const CollisionDetectorPtr& parent_detector)
225 {
226  // Temporarily keep pointer to the previous (if any) collision detector to copy padding from
227  CollisionDetectorPtr prev_coll_detector = collision_detector_;
228 
229  // Construct a fresh CollisionDetector and store allocator
230  collision_detector_ = std::make_shared<CollisionDetector>();
231  collision_detector_->alloc_ = allocator;
232 
233  // If parent_detector is specified, copy-construct collision environments (copies link shapes and attached objects)
234  // Otherwise, construct new collision environment from world and robot model
235  if (parent_detector)
236  {
237  collision_detector_->cenv_ = collision_detector_->alloc_->allocateEnv(parent_detector->cenv_, world_);
238  collision_detector_->cenv_unpadded_ =
239  collision_detector_->alloc_->allocateEnv(parent_detector->cenv_unpadded_, world_);
240  }
241  else
242  {
243  collision_detector_->cenv_ = collision_detector_->alloc_->allocateEnv(world_, getRobotModel());
244  collision_detector_->cenv_unpadded_ = collision_detector_->alloc_->allocateEnv(world_, getRobotModel());
245 
246  // Copy padding to collision_detector_->cenv_
247  if (prev_coll_detector)
248  collision_detector_->copyPadding(*prev_coll_detector);
249  }
250 
251  // Assign const pointers
252  collision_detector_->cenv_const_ = collision_detector_->cenv_;
253  collision_detector_->cenv_unpadded_const_ = collision_detector_->cenv_unpadded_;
254 }

306 void PlanningScene::pushDiffs(const PlanningScenePtr& scene)
307 {
308  if (!parent_)
309  return;
310 
311  if (scene_transforms_)
312  scene->getTransformsNonConst().setAllTransforms(scene_transforms_->getAllTransforms());
313 
314  if (robot_state_)
315  {
316  scene->getCurrentStateNonConst() = *robot_state_;
317  // push colors and types for attached objects
318  std::vector<const moveit::core::AttachedBody*> attached_objs;
319  robot_state_->getAttachedBodies(attached_objs);
320  for (const moveit::core::AttachedBody* attached_obj : attached_objs)
321  {
322  if (hasObjectType(attached_obj->getName()))
323  scene->setObjectType(attached_obj->getName(), getObjectType(attached_obj->getName()));
324  if (hasObjectColor(attached_obj->getName()))
325  scene->setObjectColor(attached_obj->getName(), getObjectColor(attached_obj->getName()));
326  }
327  }
328 
329  if (acm_)
330  scene->getAllowedCollisionMatrixNonConst() = *acm_;
331 
332  collision_detection::CollisionEnvPtr active_cenv = scene->getCollisionEnvNonConst();
333  active_cenv->setLinkPadding(collision_detector_->cenv_->getLinkPadding());
334  active_cenv->setLinkScale(collision_detector_->cenv_->getLinkScale());
335 
336  if (world_diff_)
337  {
338  for (const std::pair<const std::string, collision_detection::World::Action>& it : *world_diff_)
339  {
340  if (it.second == collision_detection::World::DESTROY)
341  {
342  scene->world_->removeObject(it.first);
343  scene->removeObjectColor(it.first);
344  scene->removeObjectType(it.first);
345  }
346  else
347  {
348  const collision_detection::World::Object& obj = *world_->getObject(it.first);
349  scene->world_->removeObject(obj.id_);
350  scene->world_->addToObject(obj.id_, obj.pose_, obj.shapes_, obj.shape_poses_);
351  if (hasObjectColor(it.first))
352  scene->setObjectColor(it.first, getObjectColor(it.first));
353  if (hasObjectType(it.first))
354  scene->setObjectType(it.first, getObjectType(it.first));
355 
356  scene->world_->setSubframesOfObject(obj.id_, obj.subframe_poses_);
357  }
358  }
359  }
360 }

539 void PlanningScene::getPlanningSceneDiffMsg(moveit_msgs::msg::PlanningScene& scene_msg) const
540 {
541  scene_msg.name = name_;
542  scene_msg.robot_model_name = getRobotModel()->getName();
543  scene_msg.is_diff = true;
544 
545  if (scene_transforms_)
546  scene_transforms_->copyTransforms(scene_msg.fixed_frame_transforms);
547  else
548  scene_msg.fixed_frame_transforms.clear();
549 
550  if (robot_state_)
551  moveit::core::robotStateToRobotStateMsg(*robot_state_, scene_msg.robot_state);
552  else
553  {
554  scene_msg.robot_state = moveit_msgs::msg::RobotState();
555  }
556  scene_msg.robot_state.is_diff = true;
557 
558  if (acm_)
559  acm_->getMessage(scene_msg.allowed_collision_matrix);
560  else
561  scene_msg.allowed_collision_matrix = moveit_msgs::msg::AllowedCollisionMatrix();
562 
563  collision_detector_->cenv_->getPadding(scene_msg.link_padding);
564  collision_detector_->cenv_->getScale(scene_msg.link_scale);
565 
566  scene_msg.object_colors.clear();
567  if (object_colors_)
568  {
569  unsigned int i = 0;
570  scene_msg.object_colors.resize(object_colors_->size());
571  for (ObjectColorMap::const_iterator it = object_colors_->begin(); it != object_colors_->end(); ++it, ++i)
572  {
573  scene_msg.object_colors[i].id = it->first;
574  scene_msg.object_colors[i].color = it->second;
575  }
576  }
577 
578  scene_msg.world.collision_objects.clear();
579  scene_msg.world.octomap = octomap_msgs::msg::OctomapWithPose();
580 
581  if (world_diff_)
582  {
583  bool do_omap = false;
584  for (const std::pair<const std::string, collision_detection::World::Action>& it : *world_diff_)
585  {
586  if (it.first == OCTOMAP_NS)
587  do_omap = true;
588  else if (it.second == collision_detection::World::DESTROY)
589  {
590  // if object became attached, it should not be recorded as removed here
591  if (!std::count_if(scene_msg.robot_state.attached_collision_objects.cbegin(),
592  scene_msg.robot_state.attached_collision_objects.cend(),
593  [&it](const moveit_msgs::msg::AttachedCollisionObject& aco) {
594  return aco.object.id == it.first &&
595  aco.object.operation == moveit_msgs::msg::CollisionObject::ADD;
596  }))
597  {
598  moveit_msgs::msg::CollisionObject co;
599  co.header.frame_id = getPlanningFrame();
600  co.id = it.first;
601  co.operation = moveit_msgs::msg::CollisionObject::REMOVE;
602  scene_msg.world.collision_objects.push_back(co);
603  }
604  }
605  else
606  {
607  scene_msg.world.collision_objects.emplace_back();
608  getCollisionObjectMsg(scene_msg.world.collision_objects.back(), it.first);
609  }
610  }
611  if (do_omap)

909 bool PlanningScene::loadGeometryFromStream(std::istream& in, const Eigen::Isometry3d& offset)
910 {
911  if (!in.good() || in.eof())
912  {
913  RCLCPP_ERROR(LOGGER, "Bad input stream when loading scene geometry");
914  return false;
915  }
916  // Read scene name
917  std::getline(in, name_);
918 
919  // Identify scene format version for backwards compatibility of parser
920  auto pos = in.tellg(); // remember current stream position
921  std::string line;
922  do
923  {
924  std::getline(in, line);
925  } while (in.good() && !in.eof() && (line.empty() || line[0] != '*')); // read * marker
926  std::getline(in, line); // next line determines format
927  boost::algorithm::trim(line);
928  // new format: line specifies position of object, with spaces as delimiter -> spaces indicate new format
929  // old format: line specifies number of shapes
930  bool uses_new_scene_format = line.find(' ') != std::string::npos;
931  in.seekg(pos);
932 
933  Eigen::Isometry3d pose; // Transient
934  do
935  {
936  std::string marker;
937  in >> marker;
938  if (!in.good() || in.eof())
939  {
940  RCLCPP_ERROR(LOGGER, "Bad input stream when loading marker in scene geometry");
941  return false;
942  }
943  if (marker == "*") // Start of new object
944  {
945  std::string object_id;
946  std::getline(in, object_id);
947  if (!in.good() || in.eof())
948  {
949  RCLCPP_ERROR(LOGGER, "Bad input stream when loading object_id in scene geometry");
950  return false;
951  }
952  boost::algorithm::trim(object_id);
953 
954  // Read in object pose (added in the new scene format)
955  pose.setIdentity();
956  if (uses_new_scene_format && !readPoseFromText(in, pose))
957  {
958  RCLCPP_ERROR(LOGGER, "Failed to read object pose from scene file");
959  return false;
960  }
961  pose = offset * pose; // Transform pose by input pose offset
962  world_->setObjectPose(object_id, pose);
963 
964  // Read in shapes
965  unsigned int shape_count;
966  in >> shape_count;
967  for (std::size_t i = 0; i < shape_count && in.good() && !in.eof(); ++i)
968  {
969  const auto shape = shapes::ShapeConstPtr(shapes::constructShapeFromText(in));
970  if (!shape)
971  {
972  RCLCPP_ERROR(LOGGER, "Failed to load shape from scene file");
973  return false;
974  }
975  if (!readPoseFromText(in, pose))
976  {
977  RCLCPP_ERROR(LOGGER, "Failed to read pose from scene file");
978  return false;
979  }
980  float r, g, b, a;
981  if (!(in >> r >> g >> b >> a))
982  {
983  RCLCPP_ERROR(LOGGER, "Improperly formatted color in scene geometry file");
984  return false;
985  }
986  if (shape)
987  {
988  world_->addToObject(object_id, shape, pose);
989  if (r > 0.0f || g > 0.0f || b > 0.0f || a > 0.0f)
990  {
991  std_msgs::msg::ColorRGBA color;
992  color.r = r;
993  color.g = g;
994  color.b = b;
995  color.a = a;
996  setObjectColor(object_id, color);
997  }
998  }
999  }
1000 
1001  // Read in subframes (added in the new scene format)
1002  if (uses_new_scene_format)
1003  {
1004  moveit::core::FixedTransformsMap subframes;
1005  unsigned int subframe_count;
1006  in >> subframe_count;
1007  for (std::size_t i = 0; i < subframe_count && in.good() && !in.eof(); ++i)
1008  {
1009  std::string subframe_name;
1010  in >> subframe_name;
1011  if (!readPoseFromText(in, pose))
1012  {
1013  RCLCPP_ERROR(LOGGER, "Failed to read subframe pose from scene file");
1014  return false;
1015  }
1016  subframes[subframe_name] = pose;
1017  }
1018  world_->setSubframesOfObject(object_id, subframes);
1019  }
1020  }
1021  else if (marker == ".")
1022  {
1023  // Marks the end of the scene geometry;
1024  return true;
1025  }
1026  else
1027  {
1028  RCLCPP_ERROR(LOGGER, "Unknown marker in scene geometry file: %s ", marker.c_str());
1029  return false;
1030  }
1031  } while (true);
1032 }

1504  }

1509  }
