59 class PlanningSceneInterface(object):
60  """
61  Python interface for a C++ PlanningSceneInterface.
62  Uses both C++ wrapped methods and scene manipulation topics
63  to manipulate the PlanningScene managed by the PlanningSceneMonitor.
64  See wrap_python_planning_scene_interface.cpp for the wrapped methods.
65  """

67  def __init__(self, ns="", synchronous=False, service_timeout=5.0):
68  """Create a planning scene interface; it uses both C++ wrapped methods and scene manipulation topics."""
69  self._psi = _moveit_planning_scene_interface.PlanningSceneInterface(ns)
70 
71  self._pub_co = rospy.Publisher(
72  ns_join(ns, "collision_object"), CollisionObject, queue_size=100
73  )
74  self._pub_aco = rospy.Publisher(
75  ns_join(ns, "attached_collision_object"),
76  AttachedCollisionObject,
77  queue_size=100,
78  )
79  self.__synchronous = synchronous
80  if self.__synchronous:
81  self._apply_planning_scene_diff = rospy.ServiceProxy(
82  ns_join(ns, "apply_planning_scene"), ApplyPlanningScene
83  )
84  self._apply_planning_scene_diff.wait_for_service(service_timeout)

86  def __submit(self, collision_object, attach=False):
87  if self.__synchronous:
88  diff_req = self.__make_planning_scene_diff_req(collision_object, attach)
89  self._apply_planning_scene_diff.call(diff_req)
90  else:
91  if attach:
92  self._pub_aco.publish(collision_object)
93  else:
94  self._pub_co.publish(collision_object)

278  @staticmethod
279  def __make_mesh(name, pose, filename, scale=(1, 1, 1)):
280  co = CollisionObject()
281  if pyassimp is False:
282  raise MoveItCommanderException(
283  "Pyassimp needs patch https://launchpadlibrarian.net/319496602/patchPyassim.txt"
284  )
285  scene = pyassimp.load(filename)
286  if not scene.meshes or len(scene.meshes) == 0:
287  raise MoveItCommanderException("There are no meshes in the file")
288  if len(scene.meshes[0].faces) == 0:
289  raise MoveItCommanderException("There are no faces in the mesh")
290  co.operation = CollisionObject.ADD
291  co.id = name
292  co.header = pose.header
293 
294  mesh = Mesh()
295  first_face = scene.meshes[0].faces[0]
296  if hasattr(first_face, "__len__"):
297  for face in scene.meshes[0].faces:
298  if len(face) == 3:
299  triangle = MeshTriangle()
300  triangle.vertex_indices = [face[0], face[1], face[2]]
301  mesh.triangles.append(triangle)
302  elif hasattr(first_face, "indices"):
303  for face in scene.meshes[0].faces:
304  if len(face.indices) == 3:
305  triangle = MeshTriangle()
306  triangle.vertex_indices = [
307  face.indices[0],
308  face.indices[1],
309  face.indices[2],
310  ]
311  mesh.triangles.append(triangle)
312  else:
313  raise MoveItCommanderException(
314  "Unable to build triangles from mesh due to mesh object structure"
315  )
316  for vertex in scene.meshes[0].vertices:
317  point = Point()
318  point.x = vertex[0] * scale[0]
319  point.y = vertex[1] * scale[1]
320  point.z = vertex[2] * scale[2]
321  mesh.vertices.append(point)
322  co.meshes = [mesh]
323  co.mesh_poses = [pose.pose]
324  pyassimp.release(scene)
325  return co
