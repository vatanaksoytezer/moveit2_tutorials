307  publish_planning_scene_hz = parameter.as_double();

311  updatePublishSettings(publish_geometry_updates, publish_state_updates, publish_transform_updates,
312  publish_planning_scene, publish_planning_scene_hz);

360  scene_->setName(scene_->getName().substr(0, scene_->getName().length() - 1));

363  }

395 void PlanningSceneMonitor::scenePublishingThread()
396 {
397  RCLCPP_DEBUG(LOGGER, "Started scene publishing thread ...");
398 
399  // publish the full planning scene once
400  {
401  moveit_msgs::msg::PlanningScene msg;
402  {
403  collision_detection::OccMapTree::ReadLock lock;
404  if (octomap_monitor_)
405  lock = octomap_monitor_->getOcTreePtr()->reading();
406  scene_->getPlanningSceneMsg(msg);
407  }
408  planning_scene_publisher_->publish(msg);
409  RCLCPP_DEBUG(LOGGER, "Published the full planning scene: '%s'", msg.name.c_str());
410  }
411 
412  do
413  {
414  moveit_msgs::msg::PlanningScene msg;
415  bool publish_msg = false;
416  bool is_full = false;
417  rclcpp::Rate rate(publish_planning_scene_frequency_);
418  {
419  boost::unique_lock<boost::shared_mutex> ulock(scene_update_mutex_);
420  while (new_scene_update_ == UPDATE_NONE && publish_planning_scene_)
421  new_scene_update_condition_.wait(ulock);
422  if (new_scene_update_ != UPDATE_NONE)
423  {
424  if ((publish_update_types_ & new_scene_update_) || new_scene_update_ == UPDATE_SCENE)
425  {
426  if (new_scene_update_ == UPDATE_SCENE)
427  is_full = true;
428  else
429  {
430  collision_detection::OccMapTree::ReadLock lock;
431  if (octomap_monitor_)
432  lock = octomap_monitor_->getOcTreePtr()->reading();
433  scene_->getPlanningSceneDiffMsg(msg);
434  if (new_scene_update_ == UPDATE_STATE)
435  {
436  msg.robot_state.attached_collision_objects.clear();
437  msg.robot_state.is_diff = true;
438  }
439  }
440  boost::recursive_mutex::scoped_lock prevent_shape_cache_updates(shape_handles_lock_); // we don't want the
441  // transform cache to
442  // update while we are
443  // potentially changing
444  // attached bodies
445  scene_->setAttachedBodyUpdateCallback(moveit::core::AttachedBodyCallback());
446  scene_->setCollisionObjectUpdateCallback(collision_detection::World::ObserverCallbackFn());
447  scene_->pushDiffs(parent_scene_);
448  scene_->clearDiffs();
449  scene_->setAttachedBodyUpdateCallback([this](moveit::core::AttachedBody* body, bool attached) {
450  currentStateAttachedBodyUpdateCallback(body, attached);
451  });
452  scene_->setCollisionObjectUpdateCallback(
453  [this](const collision_detection::World::ObjectConstPtr& object,
454  collision_detection::World::Action action) { currentWorldObjectUpdateCallback(object, action); });
455  if (octomap_monitor_)
456  {
457  excludeAttachedBodiesFromOctree(); // in case updates have happened to the attached bodies, put them in
458  excludeWorldObjectsFromOctree(); // in case updates have happened to the attached bodies, put them in
459  }
460  if (is_full)
461  {
462  collision_detection::OccMapTree::ReadLock lock;
463  if (octomap_monitor_)
464  lock = octomap_monitor_->getOcTreePtr()->reading();
465  scene_->getPlanningSceneMsg(msg);
466  }
467  // also publish timestamp of this robot_state
468  msg.robot_state.joint_state.header.stamp = last_robot_motion_time_;
469  publish_msg = true;
470  }
471  new_scene_update_ = UPDATE_NONE;
472  }
473  }
474  if (publish_msg)
475  {
476  planning_scene_publisher_->publish(msg);
477  if (is_full)
478  RCLCPP_DEBUG(LOGGER, "Published full planning scene: '%s'", msg.name.c_str());
479  rate.sleep();
480  }
481  } while (publish_planning_scene_);
482 }

539 {

553

556  if (client->wait_for_service(std::chrono::seconds(5)))
557  {

559  auto service_result = client->async_send_request(srv);
560  const auto service_result_status = service_result.wait_for(std::chrono::seconds(5));

663 
664  SceneUpdateType upd = UPDATE_SCENE;
665  std::string old_scene_name;
666  {
667  boost::unique_lock<boost::shared_mutex> ulock(scene_update_mutex_);
668  // we don't want the transform cache to update while we are potentially changing attached bodies
669  boost::recursive_mutex::scoped_lock prevent_shape_cache_updates(shape_handles_lock_);
670 
671  last_update_time_ = rclcpp::Clock().now();
672  last_robot_motion_time_ = scene.robot_state.joint_state.header.stamp;
673  RCLCPP_DEBUG(LOGGER, "scene update %f robot stamp: %f", fmod(last_update_time_.seconds(), 10.),

688  // if we just reset the scene completely but we were maintaining diffs, we need to fix that
689  if (!scene.is_diff && parent_scene_)
690  {
691  // the scene is now decoupled from the parent, since we just reset it
692  scene_->setAttachedBodyUpdateCallback(moveit::core::AttachedBodyCallback());
693  scene_->setCollisionObjectUpdateCallback(collision_detection::World::ObserverCallbackFn());

702  currentWorldObjectUpdateCallback(object, action);
703  });

706  {
707  excludeAttachedBodiesFromOctree(); // in case updates have happened to the attached bodies, put them in
708  excludeWorldObjectsFromOctree(); // in case updates have happened to the attached bodies, put them in
709  }
710  }
711

713  if (scene.is_diff)
714  {

717  scene.link_scale.empty();

722  upd = (SceneUpdateType)(static_cast<int>(upd) | static_cast<int>(UPDATE_GEOMETRY));
723 
724  if (!scene.fixed_frame_transforms.empty())
725  upd = (SceneUpdateType)(static_cast<int>(upd) | static_cast<int>(UPDATE_TRANSFORMS));

727  if (!moveit::core::isEmpty(scene.robot_state))

733  }
734  }

1007 bool PlanningSceneMonitor::waitForCurrentRobotState(const rclcpp::Time& t, double wait_time)
1008 {
1009  if (t.nanoseconds() == 0)
1010  return false;
1011  RCLCPP_DEBUG(LOGGER, "sync robot state to: %.3fs", fmod(t.seconds(), 10.));
1012 
1013  if (current_state_monitor_)
1014  {
1015  // Wait for next robot update in state monitor.
1016  bool success = current_state_monitor_->waitForCurrentState(t, wait_time);
1017 
1018  /* As robot updates are passed to the planning scene only in throttled fashion, there might
1019  be still an update pending. If so, explicitly update the planning scene here.
1020  If waitForCurrentState failed, we didn't get any new state updates within wait_time. */
1021  if (success)
1022  {
1023  std::unique_lock<std::mutex> lock(state_pending_mutex_);
1024  if (state_update_pending_) // enforce state update
1025  {
1026  state_update_pending_ = false;
1027  last_robot_state_update_wall_time_ = std::chrono::system_clock::now();
1028  lock.unlock();
1029  updateSceneWithCurrentState();
1030  }
1031  return true;
1032  }
1033 
1034  RCLCPP_WARN(LOGGER, "Failed to fetch current robot state.");
1035  return false;
1036  }
1037 
1038  // Sometimes there is no state monitor. In this case state updates are received as part of scene updates only.
1039  // However, scene updates are only published if the robot actually moves. Hence we need a timeout!
1040  // As publishing planning scene updates is throttled (2Hz by default), a 1s timeout is a suitable default.
1041  auto start = node_->get_clock()->now();
1042  auto timeout = rclcpp::Duration::from_seconds(wait_time);
1043  boost::shared_lock<boost::shared_mutex> lock(scene_update_mutex_);
1044  rclcpp::Time prev_robot_motion_time = last_robot_motion_time_;
1045  while (last_robot_motion_time_ < t && // Wait until the state update actually reaches the scene.
1046  timeout > rclcpp::Duration(0, 0))
1047  {
1048  RCLCPP_DEBUG(LOGGER, "last robot motion: %f ago", (t - last_robot_motion_time_).seconds());
1049  new_scene_update_condition_.wait_for(lock, boost::chrono::nanoseconds(timeout.nanoseconds()));
1050  timeout = timeout - (node_->get_clock()->now() - start); // compute remaining wait_time
1051  }
1052  bool success = last_robot_motion_time_ >= t;
1053  // suppress warning if we received an update at all
1054  if (!success && prev_robot_motion_time != last_robot_motion_time_)
1055  RCLCPP_WARN(LOGGER, "Maybe failed to update robot state, time diff: %.3fs", (t - last_robot_motion_time_).seconds());
1056 
1057  RCLCPP_DEBUG(LOGGER, "sync done: robot motion: %f scene update: %f", (t - last_robot_motion_time_).seconds(),
1058  (t - last_update_time_).seconds());
1059  return success;
1060 }
