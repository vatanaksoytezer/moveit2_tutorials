155 // Check for commands going out to ros_control
156 TEST_F(PoseTrackingFixture, OutgoingMsgTest)
157 {
158  // halt Servoing when first msg to ros_control is received
159  // and test some conditions
160  trajectory_msgs::msg::JointTrajectory last_received_msg;
161  std::function<void(const trajectory_msgs::msg::JointTrajectory::ConstSharedPtr)> traj_callback =
162  [&/* this */](const trajectory_msgs::msg::JointTrajectory::ConstSharedPtr msg) {
163  EXPECT_EQ(msg->header.frame_id, "panda_link0");
164  // Check for an expected joint position command
165  // As of now, the robot doesn't actually move because there are no controllers enabled.
166  double angle_tolerance = 0.08; // rad
167  EXPECT_NEAR(msg->points[0].positions[0], 0, angle_tolerance);
168  EXPECT_NEAR(msg->points[0].positions[1], -0.785, angle_tolerance);
169  EXPECT_NEAR(msg->points[0].positions[2], 0, angle_tolerance);
170  EXPECT_NEAR(msg->points[0].positions[3], -2.360, angle_tolerance);
171  EXPECT_NEAR(msg->points[0].positions[4], 0, angle_tolerance);
172  EXPECT_NEAR(msg->points[0].positions[5], 1.571, angle_tolerance);
173  EXPECT_NEAR(msg->points[0].positions[6], 0.785, angle_tolerance);
174 
175  this->tracker_->stopMotion();
176  return;
177  };
178  auto traj_sub = node_->create_subscription<trajectory_msgs::msg::JointTrajectory>(
179  "/panda_arm_controller/joint_trajectory", 1, traj_callback);
180 
181  geometry_msgs::msg::PoseStamped target_pose;
182  target_pose.header.frame_id = "panda_link4";
183  target_pose.header.stamp = node_->now();
184  target_pose.pose.position.x = 0.2;
185  target_pose.pose.position.y = 0.2;
186  target_pose.pose.position.z = 0.2;
187  target_pose.pose.orientation.w = 1;
188 
189  // Republish the target pose in a new thread, as if the target is moving
190  std::thread target_pub_thread([&] {
191  size_t msg_count = 0;
192  rclcpp::Rate loop_rate(50);
193  while (++msg_count < 100)
194  {
195  target_pose_pub_->publish(target_pose);
196  loop_rate.sleep();
197  }
198  });
199 
200  // resetTargetPose() can be used to clear the target pose and wait for a new one, e.g. when moving between multiple waypoints
201  tracker_->resetTargetPose();
202 
203  tracker_->moveToPose(translation_tolerance_, ROTATION_TOLERANCE, 1 /* target pose timeout */);
204 
205  target_pub_thread.join();
206 }
