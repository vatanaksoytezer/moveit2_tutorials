54  if (argc > 2)

56  {

58  filename_index = 2;
59  }
60 
61  if (argc > 1)
62  {
63  rclcpp::executors::MultiThreadedExecutor executor;
64  executor.add_node(node);
65 
66  rclcpp::Publisher<moveit_msgs::msg::PlanningScene>::SharedPtr pub_scene;
67  rclcpp::Publisher<moveit_msgs::msg::PlanningSceneWorld>::SharedPtr pub_world_scene;
68 
69  if (full_scene)
70  pub_scene = node->create_publisher<moveit_msgs::msg::PlanningScene>(
71  planning_scene_monitor::PlanningSceneMonitor::DEFAULT_PLANNING_SCENE_TOPIC, 1);
72  else
73  pub_world_scene = node->create_publisher<moveit_msgs::msg::PlanningSceneWorld>(
74  planning_scene_monitor::PlanningSceneMonitor::DEFAULT_PLANNING_SCENE_WORLD_TOPIC, 1);
75 
76  robot_model_loader::RobotModelLoader::Options opt;

78  opt.load_kinematics_solvers_ = false;
79 
80  auto rml = std::make_shared<robot_model_loader::RobotModelLoader>(node, opt);
81  planning_scene::PlanningScene ps(rml->getModel());
82 
83  std::ifstream f(argv[filename_index]);
84  if (ps.loadGeometryFromStream(f))
85  {
86  RCLCPP_INFO(LOGGER, "Publishing geometry from '%s' ...", argv[filename_index]);
87  moveit_msgs::msg::PlanningScene ps_msg;
88  ps.getPlanningSceneMsg(ps_msg);
89  ps_msg.is_diff = true;
90 
91  unsigned int attempts = 0;
92  while (pub_scene->get_subscription_count() < 1 && ++attempts < 100)
93  rclcpp::sleep_for(500ms);
94 
95  if (full_scene)
96  pub_scene->publish(ps_msg);
97  else
98  pub_world_scene->publish(ps_msg.world);
99 
100  rclcpp::sleep_for(1s);
101  }
102  }
103  else
104  RCLCPP_WARN(LOGGER,
