72 bool MoveGroupQueryPlannersService::queryInterface(
73  const std::shared_ptr<rmw_request_id_t> /* unused */,
74  const std::shared_ptr<moveit_msgs::srv::QueryPlannerInterfaces::Request> /*req*/,
75  std::shared_ptr<moveit_msgs::srv::QueryPlannerInterfaces::Response> res)
76 {
77  for (const auto& planning_pipelines : context_->moveit_cpp_->getPlanningPipelines())
78  {
79  const auto& pipeline_id = planning_pipelines.first;
80  const auto& planning_pipeline = planning_pipelines.second;
81  const planning_interface::PlannerManagerPtr& planner_interface = planning_pipeline->getPlannerManager();
82  if (planner_interface)
83  {
84  std::vector<std::string> algs;
85  planner_interface->getPlanningAlgorithms(algs);
86  moveit_msgs::msg::PlannerInterfaceDescription pi_desc;
87  pi_desc.name = planner_interface->getDescription();
88  pi_desc.pipeline_id = pipeline_id;
89  planner_interface->getPlanningAlgorithms(pi_desc.planner_ids);
90  res->planner_interfaces.push_back(pi_desc);
91  }
92  }
93  return true;
94 }

96 bool MoveGroupQueryPlannersService::getParams(const std::shared_ptr<rmw_request_id_t> /* unused */,
97  const std::shared_ptr<moveit_msgs::srv::GetPlannerParams::Request> req,
98  std::shared_ptr<moveit_msgs::srv::GetPlannerParams::Response> res)
99 {
100  const planning_pipeline::PlanningPipelinePtr planning_pipeline = resolvePlanningPipeline(req->pipeline_id);
101  if (!planning_pipeline)
102  return false;
103 
104  const planning_interface::PlannerManagerPtr& planner_interface = planning_pipeline->getPlannerManager();
105  if (planner_interface)
106  {
107  std::map<std::string, std::string> config;
108 
109  const planning_interface::PlannerConfigurationMap& configs = planner_interface->getPlannerConfigurations();
110 
111  planning_interface::PlannerConfigurationMap::const_iterator it =
112  configs.find(req->planner_config); // fetch default params first
113  if (it != configs.end())
114  config.insert(it->second.config.begin(), it->second.config.end());
115 
116  if (!req->group.empty())
117  { // merge in group-specific params
118  it = configs.find(req->group + "[" + req->planner_config + "]");
119  if (it != configs.end())
120  config.insert(it->second.config.begin(), it->second.config.end());
121  }
122 
123  for (const auto& key_value_pair : config)
124  {
125  res->params.keys.push_back(key_value_pair.first);
126  res->params.values.push_back(key_value_pair.second);
127  }
128  }
129  return true;
130 }

132 bool MoveGroupQueryPlannersService::setParams(const std::shared_ptr<rmw_request_id_t> /* unused */,
133  const std::shared_ptr<moveit_msgs::srv::SetPlannerParams::Request> req,
134  std::shared_ptr<moveit_msgs::srv::SetPlannerParams::Response> /*res*/)
135 {
136  if (req->params.keys.size() != req->params.values.size())
137  return false;
138 
139  const planning_pipeline::PlanningPipelinePtr planning_pipeline = resolvePlanningPipeline(req->pipeline_id);
140  if (!planning_pipeline)
141  return false;
142 
143  const planning_interface::PlannerManagerPtr& planner_interface = planning_pipeline->getPlannerManager();
144 
145  if (planner_interface)
146  {
147  planning_interface::PlannerConfigurationMap configs = planner_interface->getPlannerConfigurations();
148  const std::string config_name =
149  req->group.empty() ? req->planner_config : req->group + "[" + req->planner_config + "]";
150 
151  planning_interface::PlannerConfigurationSettings& config = configs[config_name];
152  config.group = req->group;
153  config.name = config_name;
154  if (req->replace)
155  config.config.clear();
156  for (unsigned int i = 0, end = req->params.keys.size(); i < end; ++i)
157  config.config[req->params.keys[i]] = req->params.values[i];
158 
159  planner_interface->setPlannerConfigurations(configs);
160  }
161  return true;
162 }
