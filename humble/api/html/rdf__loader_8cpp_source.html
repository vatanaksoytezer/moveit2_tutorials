91 bool RDFLoader::loadFromStrings()
92 {
93  std::unique_ptr<urdf::Model> urdf = std::make_unique<urdf::Model>();
94  if (!urdf->initString(urdf_string_))
95  {
96  RCLCPP_INFO(LOGGER, "Unable to parse URDF");
97  return false;
98  }
99 
100  srdf::ModelSharedPtr srdf = std::make_shared<srdf::Model>();
101  if (!srdf->initString(*urdf, srdf_string_))
102  {
103  RCLCPP_ERROR(LOGGER, "Unable to parse SRDF");
104  return false;
105  }
106 
107  urdf_ = std::move(urdf);
108  srdf_ = std::move(srdf);
109  return true;
110 }

120 bool RDFLoader::loadFileToString(std::string& buffer, const std::string& path)
121 {
122  if (path.empty())
123  {
124  RCLCPP_ERROR(LOGGER, "Path is empty");
125  return false;
126  }
127 
128  if (!boost::filesystem::exists(path))
129  {
130  RCLCPP_ERROR(LOGGER, "File does not exist");
131  return false;
132  }
133 
134  std::ifstream stream(path.c_str());
135  if (!stream.good())
136  {
137  RCLCPP_ERROR(LOGGER, "Unable to load path");
138  return false;
139  }
140 
141  // Load the file to a string using an efficient memory allocation technique
142  stream.seekg(0, std::ios::end);
143  buffer.reserve(stream.tellg());
144  stream.seekg(0, std::ios::beg);
145  buffer.assign((std::istreambuf_iterator<char>(stream)), std::istreambuf_iterator<char>());
146  stream.close();
147 
148  return true;
149 }

151 bool RDFLoader::loadXacroFileToString(std::string& buffer, const std::string& path,
152  const std::vector<std::string>& xacro_args)
153 {
154  buffer.clear();
155  if (path.empty())
156  {
157  RCLCPP_ERROR(LOGGER, "Path is empty");
158  return false;
159  }
160 
161  if (!boost::filesystem::exists(path))
162  {
163  RCLCPP_ERROR(LOGGER, "File does not exist");
164  return false;
165  }
166 
167  std::string cmd = "ros2 run xacro xacro ";
168  for (const std::string& xacro_arg : xacro_args)
169  cmd += xacro_arg + " ";
170  cmd += path;
171 
172 #ifdef _WIN32
173  FILE* pipe = _popen(cmd.c_str(), "r");
174 #else
175  FILE* pipe = popen(cmd.c_str(), "r");
176 #endif
177  if (!pipe)
178  {
179  RCLCPP_ERROR(LOGGER, "Unable to load path");
180  return false;
181  }
182 
183  char pipe_buffer[128];
184  while (!feof(pipe))
185  {
186  if (fgets(pipe_buffer, 128, pipe) != nullptr)
187  buffer += pipe_buffer;
188  }
189 #ifdef _WIN32
190  _pclose(pipe);
191 #else
192  pclose(pipe);
193 #endif
194 
195  return true;
196 }
