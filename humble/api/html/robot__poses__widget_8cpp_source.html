338 // ******************************************************************************************
339 // Show the robot in the current pose
340 // ******************************************************************************************
341 void RobotPosesWidget::showPose(srdf::Model::GroupState* pose)
342 {
343  // Set the joints based on the SRDF pose
344  moveit::core::RobotState& robot_state = config_data_->getPlanningScene()->getCurrentStateNonConst();
345  for (std::map<std::string, std::vector<double> >::const_iterator value_it = pose->joint_values_.begin();
346  value_it != pose->joint_values_.end(); ++value_it)
347  {
348  robot_state.setJointPositions(value_it->first, value_it->second);
349  }
350 
351  // Update the joints
352  publishJoints();
353 
354  // Unhighlight all links
355  Q_EMIT unhighlightAll();
356 
357  // Highlight group
358  Q_EMIT highlightGroup(pose->group_);
359 }

455 // ******************************************************************************************
456 // Load the joint sliders based on group selected from combo box
457 // ******************************************************************************************
458 void RobotPosesWidget::loadJointSliders(const QString& selected)
459 {
460  // Ignore this event if the combo box is empty. This occurs when clearing the combo box and reloading with the
461  // newest groups. Also ignore if we are not on the edit screen
462  if (!group_name_field_->count() || selected.isEmpty() || stacked_widget_->currentIndex() == 0)
463  return;
464 
465  // Get group name from input
466  const std::string group_name = selected.toStdString();
467 
468  // Check that joint model exist
469  if (!config_data_->getRobotModel()->hasJointModelGroup(group_name))
470  {
471  QMessageBox::critical(this, "Error Loading",
472  QString("Unable to find joint model group for group: ")
473  .append(group_name.c_str())
474  .append(" Are you sure this group has associated joints/links?"));
475  return;
476  }
477 
478  // Delete old sliders from joint_list_layout_ if this has been loaded before
479  if (joint_list_layout_)
480  {
481  delete joint_list_layout_;
482  qDeleteAll(joint_list_widget_->children());
483  }
484 
485  // Create layout again
486  joint_list_layout_ = new QVBoxLayout();
487  joint_list_widget_->setLayout(joint_list_layout_);
488  // joint_list_widget_->setSizePolicy( QSizePolicy::Expanding, QSizePolicy::Expanding );
489  joint_list_widget_->setSizePolicy(QSizePolicy::Ignored, QSizePolicy::Ignored);
490  joint_list_widget_->setMinimumSize(50, 50); // w, h
491 
492  // Get list of associated joints
493  const moveit::core::JointModelGroup* joint_model_group =
494  config_data_->getRobotModel()->getJointModelGroup(group_name);
495  const auto& robot_state = config_data_->getPlanningScene()->getCurrentState();
496 
497  // Iterate through the joints
498  int num_joints = 0;
499  for (const moveit::core::JointModel* joint_model : joint_model_group->getJointModels())
500  {
501  if (joint_model->getVariableCount() != 1 || // only consider 1-variable joints
502  joint_model->isPassive() || // ignore passive
503  joint_model->getMimic()) // and mimic joints
504  continue;
505 
506  double init_value = robot_state.getVariablePosition(joint_model->getVariableNames()[0]);
507 
508  // For each joint in group add slider
509  SliderWidget* sw = new SliderWidget(this, joint_model, init_value);
510  joint_list_layout_->addWidget(sw);
511 
512  // Connect value change event
513  connect(sw, SIGNAL(jointValueChanged(const std::string&, double)), this,
514  SLOT(updateRobotModel(const std::string&, double)));
515 
516  ++num_joints;
517  }
518 
519  // Copy the width of column 2 and manually calculate height from number of joints
520  joint_list_widget_->resize(300, num_joints * 70); // w, h
521 
522  // Update the robot model in Rviz with newly selected joint values
523  publishJoints();
524 
525  // Highlight the planning group
526  Q_EMIT unhighlightAll();
527  Q_EMIT highlightGroup(group_name);
528 }

544  }

546 
547  return searched_state;

566  QMessageBox::Ok | QMessageBox::Cancel) == QMessageBox::Cancel)
567  {
568  return;
569  }
570 
571  // Delete pose from vector

573  pose_it != config_data_->srdf_->group_states_.end(); ++pose_it)
574  {
575  // check if this is the group we want to delete
576  if (pose_it->name_ == name && pose_it->group_ == group) // match
577  {
578  config_data_->srdf_->group_states_.erase(pose_it);
579  break;
580  }
581  }
582 
583  // Reload main screen table
584  loadDataTable();
585  config_data_->changes |= MoveItConfigData::POSES;
586 }

588 // ******************************************************************************************
589 // Save editing changes
590 // ******************************************************************************************
591 void RobotPosesWidget::doneEditing()
592 {
593  // Get a reference to the supplied strings
594  const std::string& name = pose_name_field_->text().toStdString();
595  const std::string& group = group_name_field_->currentText().toStdString();
596 
597  // Used for editing existing groups
598  srdf::Model::GroupState* searched_data = nullptr;
599 
600  // Check that name field is not empty
601  if (name.empty())
602  {
603  QMessageBox::warning(this, "Error Saving", "A name must be given for the pose!");
604  pose_name_field_->setFocus();
605  return;
606  }
607  // Check that a group was selected
608  if (group.empty())
609  {
610  QMessageBox::warning(this, "Error Saving", "A planning group must be chosen!");
611  group_name_field_->setFocus();
612  return;
613  }
614 
615  // If creating a new pose, check if the (name, group) pair already exists
616  if (!current_edit_pose_)
617  {
618  searched_data = findPoseByName(name, group);
619  if (searched_data != current_edit_pose_)
620  {
621  if (QMessageBox::warning(this, "Warning Saving", "A pose already exists with that name! Overwrite?",
622  QMessageBox::Yes | QMessageBox::No, QMessageBox::No) == QMessageBox::No)
623  return;
624  }
625  }
626  else
627  searched_data = current_edit_pose_; // overwrite edited pose
628 
629  config_data_->changes |= MoveItConfigData::POSES;
630 
631  // Save the new pose name or create the new pose ----------------------------
632  bool is_new = false;
633 
634  if (searched_data == nullptr) // create new
635  {
636  is_new = true;
637  searched_data = new srdf::Model::GroupState();
638  }
639 
640  // Copy name data ----------------------------------------------------
641  searched_data->name_ = name;
642  searched_data->group_ = group;
643 
644  // Copy joint positions ----------------------------------------
645 
646  // Clear the old values
647  searched_data->joint_values_.clear();
648 
649  const moveit::core::JointModelGroup* joint_model_group = config_data_->getRobotModel()->getJointModelGroup(group);
650  const auto& robot_state = config_data_->getPlanningScene()->getCurrentState();
651 
652  // Iterate through the current group's joints and add them to SRDF
653  for (const moveit::core::JointModel* jm : joint_model_group->getJointModels())
654  {
655  // Check that this joint only represents 1 variable.
656  if (jm->getVariableCount() == 1 && !jm->isPassive() && !jm->getMimic())
657  {
658  // Create vector for new joint values
659  std::vector<double> joint_values(jm->getVariableCount());
660  const double* const first_variable = robot_state.getVariablePositions() + jm->getFirstVariableIndex();
661  std::copy(first_variable, first_variable + joint_values.size(), joint_values.begin());
662 
663  // Add joint vector to SRDF
664  searched_data->joint_values_[jm->getName()] = std::move(joint_values);
665  }
666  }
667 
668  // Insert new poses into group state vector --------------------------
