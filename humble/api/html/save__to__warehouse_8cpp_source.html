53 void onSceneUpdate(planning_scene_monitor::PlanningSceneMonitor& psm, moveit_warehouse::PlanningSceneStorage& pss)
54 {
55  RCLCPP_INFO(LOGGER, "Received an update to the planning scene...");
56 
57  if (!psm.getPlanningScene()->getName().empty())
58  {
59  if (!pss.hasPlanningScene(psm.getPlanningScene()->getName()))
60  {
61  moveit_msgs::msg::PlanningScene psmsg;
62  psm.getPlanningScene()->getPlanningSceneMsg(psmsg);
63  pss.addPlanningScene(psmsg);
64  }
65  else
66  RCLCPP_INFO(LOGGER, "Scene '%s' was previously added. Not adding again.",
67  psm.getPlanningScene()->getName().c_str());
68  }
69  else
70  RCLCPP_INFO(LOGGER, "Scene name is empty. Not saving.");
71 }

84 void onConstraints(const moveit_msgs::msg::Constraints& msg, moveit_warehouse::ConstraintsStorage& cs)
85 {
86  if (msg.name.empty())
87  {
88  RCLCPP_INFO(LOGGER, "No name specified for constraints. Not saving.");
89  return;
90  }
91 
92  if (cs.hasConstraints(msg.name))
93  {
94  RCLCPP_INFO(LOGGER, "Replacing constraints '%s'", msg.name.c_str());
95  cs.removeConstraints(msg.name);
96  cs.addConstraints(msg);
97  }
98  else
99  {
100  RCLCPP_INFO(LOGGER, "Adding constraints '%s'", msg.name.c_str());
101  cs.addConstraints(msg);
102  }
103 }

105 void onRobotState(const moveit_msgs::msg::RobotState& msg, moveit_warehouse::RobotStateStorage& rs)
106 {
107  std::vector<std::string> names;
108  rs.getKnownRobotStates(names);
109  std::set<std::string> names_set(names.begin(), names.end());
110  std::size_t n = names.size();
111  while (names_set.find("S" + boost::lexical_cast<std::string>(n)) != names_set.end())
112  n++;
113  std::string name = "S" + boost::lexical_cast<std::string>(n);
114  RCLCPP_INFO(LOGGER, "Adding robot state '%s'", name.c_str());
115  rs.addRobotState(msg, name);
116 }

118 int main(int argc, char** argv)
119 {
120  rclcpp::init(argc, argv);
121  rclcpp::NodeOptions node_options;
122  node_options.allow_undeclared_parameters(true);
123  node_options.automatically_declare_parameters_from_overrides(true);
124  rclcpp::Node::SharedPtr node = rclcpp::Node::make_shared("save_to_warehouse", node_options);
125 
126  boost::program_options::options_description desc;
127  desc.add_options()("help", "Show help message")("host", boost::program_options::value<std::string>(),
128  "Host for the "
129  "DB.")("port", boost::program_options::value<std::size_t>(),
130  "Port for the DB.");
131 
132  boost::program_options::variables_map vm;
133  boost::program_options::store(boost::program_options::parse_command_line(argc, argv, desc), vm);
134  boost::program_options::notify(vm);
135 
136  if (vm.count("help"))
137  {
138  std::cout << desc << '\n';
139  return 1;
140  }
141  // Set up db
142  warehouse_ros::DatabaseConnection::Ptr conn = moveit_warehouse::loadDatabase(node);
143  if (vm.count("host") && vm.count("port"))
144  conn->setParams(vm["host"].as<std::string>(), vm["port"].as<std::size_t>());
145  if (!conn->connect())
146  return 1;
147 
148  planning_scene_monitor::PlanningSceneMonitor psm(node, ROBOT_DESCRIPTION);
149  if (!psm.getPlanningScene())
150  {
151  RCLCPP_ERROR(LOGGER, "Unable to initialize PlanningSceneMonitor");
152  return 1;
153  }
154 
155  psm.startSceneMonitor();
156  psm.startWorldGeometryMonitor();
157  moveit_warehouse::PlanningSceneStorage pss(conn);
158  moveit_warehouse::ConstraintsStorage cs(conn);
159  moveit_warehouse::RobotStateStorage rs(conn);
160  std::vector<std::string> names;
161  pss.getPlanningSceneNames(names);
162  if (names.empty())
163  RCLCPP_INFO(LOGGER, "There are no previously stored scenes");
164  else
165  {
166  RCLCPP_INFO(LOGGER, "Previously stored scenes:");
167  for (const std::string& name : names)
168  RCLCPP_INFO(LOGGER, " * %s", name.c_str());
169  }
170  cs.getKnownConstraints(names);
171  if (names.empty())
172  RCLCPP_INFO(LOGGER, "There are no previously stored constraints");
173  else
174  {
175  RCLCPP_INFO(LOGGER, "Previously stored constraints:");
176  for (const std::string& name : names)
177  RCLCPP_INFO(LOGGER, " * %s", name.c_str());
178  }
179  rs.getKnownRobotStates(names);
180  if (names.empty())
181  RCLCPP_INFO(LOGGER, "There are no previously stored robot states");
182  else
183  {
184  RCLCPP_INFO(LOGGER, "Previously stored robot states:");
185  for (const std::string& name : names)
186  RCLCPP_INFO(LOGGER, " * %s", name.c_str());
187  }
188 
189  psm.addUpdateCallback([&](auto&&) { return onSceneUpdate(psm, pss); });
190 
191  auto callback1 = [&](moveit_msgs::msg::MotionPlanRequest msg) -> void { return onMotionPlanRequest(msg, psm, pss); };
192  auto mplan_req_sub =
193  node->create_subscription<moveit_msgs::msg::MotionPlanRequest>("motion_plan_request", 100, callback1);
194  auto callback2 = [&](moveit_msgs::msg::Constraints msg) -> void { return onConstraints(msg, cs); };
195  auto constr_sub = node->create_subscription<moveit_msgs::msg::Constraints>("constraints", 100, callback2);
196  auto callback3 = [&](moveit_msgs::msg::RobotState msg) -> void { return onRobotState(msg, rs); };
197  auto state_sub = node->create_subscription<moveit_msgs::msg::RobotState>("robot_state", 100, callback3);
198 
199  std::vector<std::string> topics;
200  psm.getMonitoredTopics(topics);
201  RCLCPP_INFO_STREAM(LOGGER, "Listening for scene updates on topics " << boost::algorithm::join(topics, ", "));
202  RCLCPP_INFO_STREAM(LOGGER, "Listening for planning requests on topic " << mplan_req_sub->get_topic_name());
203  RCLCPP_INFO_STREAM(LOGGER, "Listening for named constraints on topic " << constr_sub->get_topic_name());
204  RCLCPP_INFO_STREAM(LOGGER, "Listening for states on topic " << state_sub->get_topic_name());
205 
206  rclcpp::spin(node);
207  return 0;
208 }
