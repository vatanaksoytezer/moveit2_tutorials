62  if (!planning_scene_monitor_->getStateMonitor()->waitForCompleteState(parameters_->move_group_name,
63  ROBOT_STATE_WAIT_TIME))
64  {
65  RCLCPP_ERROR(LOGGER, "Timeout waiting for current state");
