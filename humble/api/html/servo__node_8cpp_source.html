47 ServoNode::ServoNode(const rclcpp::NodeOptions& options)
48  : node_{ std::make_shared<rclcpp::Node>("servo_node", options) }
49 {
50  if (!options.use_intra_process_comms())
51  {
52  RCLCPP_WARN_STREAM(LOGGER, "Intra-process communication is disabled, consider enabling it by adding: "
53  "\nextra_arguments=[{'use_intra_process_comms' : True}]\nto the Servo composable node "
54  "in the launch file");
55  }
56 
57  // Set up services for interacting with Servo
58  start_servo_service_ = node_->create_service<std_srvs::srv::Trigger>(
59  "~/start_servo",
60  [this](const std::shared_ptr<std_srvs::srv::Trigger::Request> request,
61  std::shared_ptr<std_srvs::srv::Trigger::Response> response) { return startCB(request, response); });
62  stop_servo_service_ = node_->create_service<std_srvs::srv::Trigger>(
63  "~/stop_servo",
64  [this](const std::shared_ptr<std_srvs::srv::Trigger::Request> request,
65  std::shared_ptr<std_srvs::srv::Trigger::Response> response) { return stopCB(request, response); });
66  pause_servo_service_ = node_->create_service<std_srvs::srv::Trigger>(
67  "~/pause_servo",
68  [this](const std::shared_ptr<std_srvs::srv::Trigger::Request> request,
69  std::shared_ptr<std_srvs::srv::Trigger::Response> response) { return pauseCB(request, response); });
70  unpause_servo_service_ = node_->create_service<std_srvs::srv::Trigger>(
71  "~/unpause_servo",
72  [this](const std::shared_ptr<std_srvs::srv::Trigger::Request> request,
73  std::shared_ptr<std_srvs::srv::Trigger::Response> response) { return unpauseCB(request, response); });
74 
75  // Can set robot_description name from parameters
76  std::string robot_description_name = "robot_description";
77  node_->get_parameter_or("robot_description_name", robot_description_name, robot_description_name);
78 
79  // Get the servo parameters
80  auto servo_parameters = moveit_servo::ServoParameters::makeServoParameters(node_);
81  if (servo_parameters == nullptr)
82  {
83  RCLCPP_ERROR(LOGGER, "Failed to load the servo parameters");
84  throw std::runtime_error("Failed to load the servo parameters");
85  }
86 
87  // Set up planning_scene_monitor
88  planning_scene_monitor_ = std::make_shared<planning_scene_monitor::PlanningSceneMonitor>(
89  node_, robot_description_name, "planning_scene_monitor");
90  planning_scene_monitor_->startStateMonitor(servo_parameters->joint_topic);
91  planning_scene_monitor_->startSceneMonitor(servo_parameters->monitored_planning_scene_topic);
92  planning_scene_monitor_->setPlanningScenePublishingFrequency(25);
93  planning_scene_monitor_->getStateMonitor()->enableCopyDynamics(true);
94  planning_scene_monitor_->startPublishingPlanningScene(planning_scene_monitor::PlanningSceneMonitor::UPDATE_SCENE,
95  std::string(node_->get_fully_qualified_name()) +
96  "/publish_planning_scene");
97  // If the planning scene monitor in servo is the primary one we provide /get_planning_scene service so RViz displays
98  // or secondary planning scene monitors can fetch the scene, otherwise we request the planning scene from the
99  // primary planning scene monitor (e.g. move_group)
100  if (servo_parameters->is_primary_planning_scene_monitor)
101  planning_scene_monitor_->providePlanningSceneService();
102  else
103  planning_scene_monitor_->requestPlanningSceneState();
104 
105  // Create Servo
106  servo_ = std::make_unique<moveit_servo::Servo>(node_, servo_parameters, planning_scene_monitor_);
107 }
