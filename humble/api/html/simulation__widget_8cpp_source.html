129 void SimulationWidget::focusGiven()
130 {
131  if (!simulation_text_->document()->isEmpty())
132  return; // nothing to do
133 
134  simulation_text_->setVisible(true);
135  std::string text = generateGazeboCompatibleURDF();
136  config_data_->gazebo_urdf_string_ = text;
137 
138  simulation_text_->document()->setPlainText(QString::fromStdString(text));
139 
140  // Add generated Gazebo URDF to config file if not empty
141  bool have_changes = !text.empty();
142  config_data_->save_gazebo_urdf_ = have_changes;
143 
144  // GUI elements are visible only if there are URDF changes to display/edit
145  simulation_text_->setVisible(have_changes);
146  btn_overwrite_->setVisible(have_changes);
147  btn_open_->setVisible(have_changes && !qgetenv("EDITOR").isEmpty());
148  copy_urdf_->setVisible(have_changes);
149  no_changes_label_->setVisible(!have_changes);
150 
151  // Disable overwrite button if URDF originates from xacro
152  btn_overwrite_->setDisabled(config_data_->urdf_from_xacro_);
153  QString tooltip;
154  if (btn_overwrite_->isEnabled())
155  tooltip = tr("Overwrite URDF in original location:\n").append(config_data_->urdf_path_.c_str());
156  else
157  tooltip = tr("Cannot overwrite original, <i>xacro-based</i> URDF");
158  btn_overwrite_->setToolTip(tooltip);
159 
160  if (have_changes)
161  config_data_->changes |= MoveItConfigData::SIMULATION;
162  else
163  config_data_->changes &= ~MoveItConfigData::SIMULATION;
164 }

166 bool SimulationWidget::focusLost()
167 {
168  if (!config_data_->save_gazebo_urdf_)
169  return true; // saving is disabled anyway
170 
171  // validate XML
172  TiXmlDocument doc;
173  auto urdf = simulation_text_->document()->toPlainText().toStdString();
174  doc.Parse(urdf.c_str(), nullptr, TIXML_ENCODING_UTF8);
175  if (doc.Error())
176  {
177  QTextCursor cursor = simulation_text_->textCursor();
178  cursor.movePosition(QTextCursor::Start);
179  cursor.movePosition(QTextCursor::Down, QTextCursor::MoveAnchor, doc.ErrorRow());
180  cursor.movePosition(QTextCursor::Right, QTextCursor::MoveAnchor, doc.ErrorCol());
181  simulation_text_->setTextCursor(cursor);
182  QMessageBox::warning(this, tr("Gazebo URDF"), tr("Error parsing XML:\n").append(doc.ErrorDesc()));
183  simulation_text_->setFocus(Qt::OtherFocusReason);
184  return false; // reject switching
185  }
186  else
187  config_data_->gazebo_urdf_string_ = std::move(urdf);
188  return true;
189 }

194 void SimulationWidget::overwriteURDF()
195 {
196  if (!focusLost()) // validate XML
197  return;
198 
199  if (!config_data_->outputGazeboURDFFile(config_data_->urdf_path_))
200  QMessageBox::warning(this, "Gazebo URDF", tr("Failed to save to ").append(config_data_->urdf_path_.c_str()));
201  else // Display success message
202  QMessageBox::information(this, "Overwriting Successful",
203  "Original robot description URDF was successfully overwritten.");
204 
205  // Remove Gazebo URDF file from list of to-be-written config files
206  config_data_->save_gazebo_urdf_ = false;
207  config_data_->changes &= ~MoveItConfigData::SIMULATION;
208 }

224 // Generate a Gazebo-compatible robot URDF
225 std::string SimulationWidget::generateGazeboCompatibleURDF() const
226 {
227  TiXmlDocument doc;
228  doc.Parse(config_data_->urdf_string_.c_str(), nullptr, TIXML_ENCODING_UTF8);
229  auto root = doc.RootElement();
230 
231  // Normalize original urdf_string_
232  TiXmlPrinter orig_urdf;
233  doc.Accept(&orig_urdf);
234 
235  // Map existing SimpleTransmission elements to their joint name
236  std::map<std::string, TiXmlElement*> transmission_elements;
237  for (TiXmlElement* element = root->FirstChildElement("transmission"); element != nullptr;
238  element = element->NextSiblingElement(element->Value()))
239  {
240  auto type_tag = element->FirstChildElement("type");
241  auto joint_tag = element->FirstChildElement("joint");
242  if (!type_tag || !type_tag->GetText() || !joint_tag || !joint_tag->Attribute("name"))
243  continue; // ignore invalid tags
244  if (std::string(type_tag->GetText()) == "transmission_interface/SimpleTransmission")
245  transmission_elements[element->FirstChildElement("joint")->Attribute("name")] = element;
246  }
247 
248  // Loop through Link and Joint elements and add Gazebo tags if not present
249  for (TiXmlElement* element = root->FirstChildElement(); element != nullptr; element = element->NextSiblingElement())
250  {
251  const std::string tag_name(element->Value());
252  if (tag_name == "link" && element->FirstChildElement("collision"))
253  {
254  TiXmlElement* inertial = uniqueInsert(*element, "inertial");
255  uniqueInsert(*inertial, "mass", { { "value", "0.1" } });
256  uniqueInsert(*inertial, "origin", { { "xyz", "0 0 0" }, { "rpy", "0 0 0" } });
257  uniqueInsert(*inertial, "inertia",
258  { { "ixx", "0.03" },
259  { "iyy", "0.03" },
260  { "izz", "0.03" },
261  { "ixy", "0.0" },
262  { "ixz", "0.0" },
263  { "iyz", "0.0" } });
264  }
265  else if (tag_name == "joint")
266  {
267  const char* joint_type = element->Attribute("type");
268  const char* joint_name = element->Attribute("name");
269  if (!joint_type || !joint_name || strcmp(joint_type, "fixed") == 0)
270  continue; // skip invalid or fixed joints
271 
272  // find existing or create new transmission element for this joint
273  TiXmlElement* transmission;
274  auto it = transmission_elements.find(joint_name);
275  if (it != transmission_elements.end())
276  transmission = it->second;
277  else
278  {
279  transmission = root->InsertEndChild(TiXmlElement("transmission"))->ToElement();
280  transmission->SetAttribute("name", std::string("trans_") + joint_name);
281  }
282 
283  uniqueInsert(*transmission, "type", {}, "transmission_interface/SimpleTransmission");
284 
285  std::string hw_interface = config_data_->getJointHardwareInterface(joint_name);
286  auto* joint = uniqueInsert(*transmission, "joint", { { "name", joint_name } });
287  uniqueInsert(*joint, "hardwareInterface", {}, hw_interface.c_str());
288 
289  auto actuator_name = joint_name + std::string("_motor");
290  auto* actuator = uniqueInsert(*transmission, "actuator", { { "name", actuator_name.c_str() } });
291  uniqueInsert(*actuator, "hardwareInterface", {}, hw_interface.c_str());
292  uniqueInsert(*actuator, "mechanicalReduction", {}, "1");
293  }
294  }
295 
296  // Add gazebo_ros_control plugin which reads the transmission tags
297  TiXmlElement* gazebo = uniqueInsert(*root, "gazebo");
298  TiXmlElement* plugin = uniqueInsert(*gazebo, "plugin", { { "name", "gazebo_ros_control", true } });
299  uniqueInsert(*plugin, "robotNamespace", {}, "/");
300 
301  // generate new URDF
302  TiXmlPrinter new_urdf;
303  doc.Accept(&new_urdf);
304  // and return it when there are changes
305  return orig_urdf.Str() == new_urdf.Str() ? std::string() : new_urdf.Str();
306 }
