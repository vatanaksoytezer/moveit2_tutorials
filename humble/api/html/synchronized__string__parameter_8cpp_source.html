41 std::string SynchronizedStringParameter::loadInitialValue(const std::shared_ptr<rclcpp::Node>& node,
42  const std::string& name, StringCallback parent_callback,
43  bool default_continuous_value, double default_timeout)
44 {
45  node_ = node;
46  name_ = name;
47  parent_callback_ = parent_callback;
48 
49  if (getMainParameter())
50  {
51  if (shouldPublish())
52  {
53  // Transient local is similar to latching in ROS 1.
54  string_publisher_ = node_->create_publisher<std_msgs::msg::String>(name_, rclcpp::QoS(1).transient_local());
55 
56  std_msgs::msg::String msg;
57  msg.data = content_;
58  string_publisher_->publish(msg);
59  }
60  return content_;
61  }
62 
63  // Load topic parameters
64  std::string keep_open_param = name_ + "_continuous";
65  if (!node_->has_parameter(keep_open_param))
66  {
67  node_->declare_parameter(keep_open_param, rclcpp::ParameterType::PARAMETER_BOOL);
68  }
69  bool keep_open;
70  node_->get_parameter_or(keep_open_param, keep_open, default_continuous_value);
71 
72  std::string timeout_param = name_ + "_timeout";
73  if (!node_->has_parameter(timeout_param))
74  {
75  node_->declare_parameter(timeout_param, rclcpp::ParameterType::PARAMETER_DOUBLE);
76  }
77  double d_timeout;
78  node_->get_parameter_or(timeout_param, d_timeout, default_timeout); // ten second default
79  rclcpp::Duration timeout = rclcpp::Duration::from_seconds(d_timeout);
80 
81  if (!waitForMessage(timeout))
82  {
83  RCLCPP_ERROR_ONCE(node_->get_logger(),
84  "Could not find parameter %s and did not receive %s via std_msgs::msg::String subscription "
85  "within %f seconds.",
86  name_.c_str(), name_.c_str(), d_timeout);
87  }
88  if (!keep_open)
89  {
90  string_subscriber_.reset();
91  }
92  return content_;
93 }

120 bool SynchronizedStringParameter::waitForMessage(const rclcpp::Duration timeout)
121 {
122  auto const nd_name = std::string(node_->get_name()).append("_ssp_").append(name_);
123  auto const temp_node = std::make_shared<rclcpp::Node>(nd_name);
124  string_subscriber_ = temp_node->create_subscription<std_msgs::msg::String>(
125  name_, rclcpp::QoS(1).transient_local().reliable(),
126  [this](const std_msgs::msg::String::SharedPtr msg) { return stringCallback(msg); });
127 
128  rclcpp::WaitSet wait_set;
129  wait_set.add_subscription(string_subscriber_);
130 
131  auto ret = wait_set.wait(timeout.to_chrono<std::chrono::duration<double>>());
132  if (ret.kind() == rclcpp::WaitResultKind::Ready)
133  {
134  std_msgs::msg::String msg;
135  rclcpp::MessageInfo info;
136  if (string_subscriber_->take(msg, info))
137  {
138  content_ = msg.data;
139  return true;
140  }
141  }
142  return false;
143 }

145 void SynchronizedStringParameter::stringCallback(const std_msgs::msg::String::SharedPtr msg)
146 {

149  return;
