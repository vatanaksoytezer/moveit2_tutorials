65 TEST_F(LoadPlanningModelsPr2, JointConstraintsSimple)
66 {
67  moveit::core::RobotState robot_state(robot_model_);
68  robot_state.setToDefaultValues();
69  moveit::core::Transforms tf(robot_model_->getModelFrame());
70 
71  kinematic_constraints::JointConstraint jc(robot_model_);
72  moveit_msgs::msg::JointConstraint jcm;
73  jcm.joint_name = "head_pan_joint";
74  jcm.position = 0.4;
75  jcm.tolerance_above = 0.1;
76  jcm.tolerance_below = 0.05;
77 
78  EXPECT_TRUE(jc.configure(jcm));
79  // weight should have been changed to 1.0
80  EXPECT_NEAR(jc.getConstraintWeight(), 1.0, std::numeric_limits<double>::epsilon());
81 
82  // tests that the default state is outside the bounds
83  // given that the default state is at 0.0
84  EXPECT_TRUE(jc.configure(jcm));
85  kinematic_constraints::ConstraintEvaluationResult p1 = jc.decide(robot_state);
86  EXPECT_FALSE(p1.satisfied);
87  EXPECT_NEAR(p1.distance, jcm.position, 1e-6);
88 
89  // tests that when we set the state within the bounds
90  // the constraint is satisfied
91  double jval = 0.41;
92  robot_state.setJointPositions(jcm.joint_name, &jval);
93  kinematic_constraints::ConstraintEvaluationResult p2 = jc.decide(robot_state);
94  EXPECT_TRUE(p2.satisfied);
95  EXPECT_NEAR(p2.distance, 0.01, 1e-6);
96 
97  // exactly equal to the low bound is fine too
98  jval = 0.35;
99  robot_state.setJointPositions(jcm.joint_name, &jval);
100  EXPECT_TRUE(jc.decide(robot_state).satisfied);
101 
102  // and so is less than epsilon when there's no other source of error
103  // jvals[jcm.joint_name] = 0.35-std::numeric_limits<double>::epsilon();
104  jval = 0.35 - std::numeric_limits<double>::epsilon();
105 
106  robot_state.setJointPositions(jcm.joint_name, &jval);
107  EXPECT_TRUE(jc.decide(robot_state).satisfied);
108 
109  // but this is too much
110  jval = 0.35 - 3 * std::numeric_limits<double>::epsilon();
111  robot_state.setJointPositions(jcm.joint_name, &jval);
112  EXPECT_FALSE(jc.decide(robot_state).satisfied);
113 
114  // negative value makes configuration fail
115  jcm.tolerance_below = -0.05;
116  EXPECT_FALSE(jc.configure(jcm));
117 
118  jcm.tolerance_below = 0.05;
119  EXPECT_TRUE(jc.configure(jcm));
120 
121  // still satisfied at a slightly different state
122  jval = 0.46;
123  robot_state.setJointPositions(jcm.joint_name, &jval);
124  EXPECT_TRUE(jc.decide(robot_state).satisfied);
125 
126  // still satisfied at a slightly different state
127  jval = 0.501;
128  robot_state.setJointPositions(jcm.joint_name, &jval);
129  EXPECT_FALSE(jc.decide(robot_state).satisfied);
130 
131  // still satisfied at a slightly different state
132  jval = 0.39;
133  robot_state.setJointPositions(jcm.joint_name, &jval);
134  EXPECT_TRUE(jc.decide(robot_state).satisfied);
135 
136  // outside the bounds
137  jval = 0.34;
138  robot_state.setJointPositions(jcm.joint_name, &jval);
139  EXPECT_FALSE(jc.decide(robot_state).satisfied);
140 
141  // testing equality
142  kinematic_constraints::JointConstraint jc2(robot_model_);
143  EXPECT_TRUE(jc2.configure(jcm));
144  EXPECT_TRUE(jc2.enabled());
145  EXPECT_TRUE(jc.equal(jc2, 1e-12));
146 
147  // if name not equal, not equal
148  jcm.joint_name = "head_tilt_joint";
149  EXPECT_TRUE(jc2.configure(jcm));
150  EXPECT_FALSE(jc.equal(jc2, 1e-12));
151 
152  // if different, test margin behavior
153  jcm.joint_name = "head_pan_joint";
154  jcm.position = 0.3;
155  EXPECT_TRUE(jc2.configure(jcm));
156  EXPECT_FALSE(jc.equal(jc2, 1e-12));
157  // exactly equal is still false
158  EXPECT_FALSE(jc.equal(jc2, .1));
159  EXPECT_TRUE(jc.equal(jc2, .101));
160 
161  // no name makes this false
162  jcm.joint_name = "";
163  jcm.position = 0.4;
164  EXPECT_FALSE(jc2.configure(jcm));
165  EXPECT_FALSE(jc2.enabled());
166  EXPECT_FALSE(jc.equal(jc2, 1e-12));
167 
168  // no DOF makes this false
169  jcm.joint_name = "base_footprint_joint";
170  EXPECT_FALSE(jc2.configure(jcm));
171 
172  // clear means not enabled
173  jcm.joint_name = "head_pan_joint";
174  EXPECT_TRUE(jc2.configure(jcm));
175  jc2.clear();
176  EXPECT_FALSE(jc2.enabled());
177  EXPECT_FALSE(jc.equal(jc2, 1e-12));
178 }

192  jcm.tolerance_above = 0.04;
193  jcm.tolerance_below = 0.02;
194  jcm.weight = 1.0;
195 
196  EXPECT_TRUE(jc.configure(jcm));
197 
198  std::map<std::string, double> jvals;
199 
200  // state should have zeros, and work
201  EXPECT_TRUE(jc.decide(robot_state).satisfied);
202 
203  // within the above tolerance
204  jvals[jcm.joint_name] = .03;
205  robot_state.setVariablePositions(jvals);
206  robot_state.update();
207  EXPECT_TRUE(jc.decide(robot_state).satisfied);
208 
209  // outside the above tolerance
210  jvals[jcm.joint_name] = .05;
211  robot_state.setVariablePositions(jvals);
212  robot_state.update();
213  EXPECT_FALSE(jc.decide(robot_state).satisfied);
214 
215  // inside the below tolerance
216  jvals[jcm.joint_name] = -.01;
217  robot_state.setVariablePositions(jvals);
218  EXPECT_TRUE(jc.decide(robot_state).satisfied);

221  jvals[jcm.joint_name] = -.03;
222  robot_state.setVariablePositions(jvals);
223  EXPECT_FALSE(jc.decide(robot_state).satisfied);
224

226  jcm.position = 3.14;
227  EXPECT_TRUE(jc.configure(jcm));
228 
229  // testing that wrap works

231  robot_state.setVariablePositions(jvals);
232  kinematic_constraints::ConstraintEvaluationResult p1 = jc.decide(robot_state);

234  EXPECT_NEAR(p1.distance, 0.03, 1e-6);
235 
236  // testing that negative wrap works
237  jvals[jcm.joint_name] = -3.14;
238  robot_state.setVariablePositions(jvals);
239  kinematic_constraints::ConstraintEvaluationResult p2 = jc.decide(robot_state);
240  EXPECT_TRUE(p2.satisfied);
241  EXPECT_NEAR(p2.distance, 0.003185, 1e-4);
242 
243  // over bound testing
244  jvals[jcm.joint_name] = 3.19;
245  robot_state.setVariablePositions(jvals);
246  EXPECT_FALSE(jc.decide(robot_state).satisfied);
247 
248  // reverses to other direction
249  // but still tested using above tolerance
250  jvals[jcm.joint_name] = -3.11;
251  robot_state.setVariablePositions(jvals);
252  EXPECT_TRUE(jc.decide(robot_state).satisfied);
253 
254  // outside of the bound given the wrap
255  jvals[jcm.joint_name] = -3.09;
256  robot_state.setVariablePositions(jvals);
257  EXPECT_FALSE(jc.decide(robot_state).satisfied);
258 
259  // lower tolerance testing
260  // within bounds
261  jvals[jcm.joint_name] = 3.13;
262  robot_state.setVariablePositions(jvals);
263  EXPECT_TRUE(jc.decide(robot_state).satisfied);
264 
265  // within outside
266  jvals[jcm.joint_name] = 3.11;
267  robot_state.setVariablePositions(jvals);
268  EXPECT_FALSE(jc.decide(robot_state).satisfied);
269 
270  // testing the other direction
271  jcm.position = -3.14;
272  EXPECT_TRUE(jc.configure(jcm));
273 
274  // should be governed by above tolerance
275  jvals[jcm.joint_name] = -3.11;
276  robot_state.setVariablePositions(jvals);
277  EXPECT_TRUE(jc.decide(robot_state).satisfied);
278 
279  // outside upper bound
280  jvals[jcm.joint_name] = -3.09;
281  robot_state.setVariablePositions(jvals);
282  EXPECT_FALSE(jc.decide(robot_state).satisfied);
283 
284  // governed by lower bound
285  jvals[jcm.joint_name] = 3.13;
286  robot_state.setVariablePositions(jvals);
287  EXPECT_TRUE(jc.decide(robot_state).satisfied);
288 
289  // outside lower bound (but would be inside upper)
290  jvals[jcm.joint_name] = 3.12;
291  robot_state.setVariablePositions(jvals);
292  EXPECT_TRUE(jc.decide(robot_state).satisfied);
293 
294  // testing wrap
295  jcm.position = 6.28;
296  EXPECT_TRUE(jc.configure(jcm));
297 
298  // should wrap to zero
299  jvals[jcm.joint_name] = 0.0;
300  robot_state.setVariablePositions(jvals);
301  EXPECT_TRUE(jc.decide(robot_state).satisfied);
302 
303  // should wrap to close and test to be near
304  moveit_msgs::msg::JointConstraint jcm2 = jcm;
305  jcm2.position = -6.28;
306  kinematic_constraints::JointConstraint jc2(robot_model_);
307  EXPECT_TRUE(jc.configure(jcm2));
308  jc.equal(jc2, .02);
309 }

399 
400  // intentionally leaving header frame blank to test behavior

402 
403  pcm.header.frame_id = robot_model_->getModelFrame();
404  EXPECT_TRUE(pc.configure(pcm, tf));
405  EXPECT_FALSE(pc.mobileReferenceFrame());
406 
407  EXPECT_TRUE(pc.decide(robot_state).satisfied);
408
