269 TEST_F(LoadPlanningModelsPr2, ObjectPoseAndSubframes)
270 {
271  auto robot_model = std::make_shared<moveit::core::RobotModel>(urdf_model_, srdf_model_);
272 
273  moveit::core::RobotState ks(robot_model);
274  ks.setToDefaultValues();
275 
276  std::vector<shapes::ShapeConstPtr> shapes;
277  EigenSTL::vector_Isometry3d poses;
278  shapes::Shape* shape = new shapes::Box(.1, .1, .1);
279  shapes.push_back(shapes::ShapeConstPtr(shape));
280  poses.push_back(Eigen::Isometry3d::Identity());
281  std::set<std::string> touch_links;
282  Eigen::Isometry3d pose_a = Eigen::Isometry3d::Identity();
283  Eigen::Isometry3d pose_b = Eigen::Isometry3d(Eigen::Translation3d(0, 0, 1));
284  moveit::core::FixedTransformsMap subframes;
285  subframes["frame1"] = Eigen::Isometry3d(Eigen::Translation3d(0, 0, 1));
286 
287  trajectory_msgs::msg::JointTrajectory empty_state;
288  ks.attachBody(std::make_unique<moveit::core::AttachedBody>(robot_model->getLinkModel("r_gripper_palm_link"), "boxA",
289  pose_a, shapes, poses, touch_links, empty_state,
290  subframes));
291  ks.attachBody(std::make_unique<moveit::core::AttachedBody>(robot_model->getLinkModel("r_gripper_palm_link"), "boxB",
292  pose_b, shapes, poses, touch_links, empty_state,
293  subframes));
294 
295  // Check position of shape in each body
296  Eigen::Isometry3d p;
297  p = ks.getAttachedBody("boxA")->getShapePosesInLinkFrame()[0];
298  EXPECT_EQ(0.0, p(2, 3)); // check translation.z
299  p = ks.getAttachedBody("boxB")->getShapePosesInLinkFrame()[0];
300  EXPECT_EQ(1.0, p(2, 3)); // z
301 
302  // Expect the pose and the subframe to have the same effect
303  Eigen::Isometry3d p2;
304 
305  p = ks.getFrameTransform("boxA/frame1");
306  p2 = ks.getFrameTransform("boxB");
307  EXPECT_TRUE(p.isApprox(p2, EPSILON));
308 
309  // Ensure that conversion to and from message in conversions.cpp works
310  moveit_msgs::msg::RobotState msg;
311  robotStateToRobotStateMsg(ks, msg, true);
312 
313  // Add another object C that is defined in a frame that is not the link.
314  // The object will be transformed into the link's frame, which
315  // uses an otherwise inactive section of _msgToAttachedBody.
316  Eigen::Isometry3d pose_c = Eigen::Isometry3d(Eigen::Translation3d(0.1, 0.2, 0.3)) *
317  Eigen::AngleAxisd(0.1 * M_TAU, Eigen::Vector3d::UnitX()) *
318  Eigen::AngleAxisd(0.2 * M_TAU, Eigen::Vector3d::UnitY()) *
319  Eigen::AngleAxisd(0.4 * M_TAU, Eigen::Vector3d::UnitZ());
320  Eigen::Quaterniond q(pose_c.linear());
321  moveit_msgs::msg::AttachedCollisionObject new_aco = msg.attached_collision_objects[0];
322  new_aco.object.id = "boxC";
323  new_aco.object.header.frame_id = "r_shoulder_pan_link";
324  new_aco.object.pose.position.x = pose_c.translation()[0];
325  new_aco.object.pose.position.y = pose_c.translation()[1];
326  new_aco.object.pose.position.z = pose_c.translation()[2];
327  new_aco.object.pose.orientation.x = q.vec()[0];
328  new_aco.object.pose.orientation.y = q.vec()[1];
329  new_aco.object.pose.orientation.z = q.vec()[2];
330  new_aco.object.pose.orientation.w = q.w();
331  msg.attached_collision_objects.push_back(new_aco);
332 
333  // Confirm that object B is unchanged after the conversion
334  moveit::core::RobotState ks3(robot_model);
335  robotStateMsgToRobotState(msg, ks3, true);
336  Eigen::Isometry3d p_original, p_reconverted;
337  p_original = ks.getAttachedBody("boxB")->getPose();
338  p_reconverted = ks3.getAttachedBody("boxB")->getPose();
339  EXPECT_TRUE(p_original.isApprox(p_reconverted, EPSILON));
340 
341  // Confirm that the position of object C is what we expect
342  Eigen::Isometry3d p_link, p_header_frame;
343  p_link = ks3.getFrameTransform("r_gripper_palm_link");
344  p_header_frame = ks3.getFrameTransform("r_shoulder_pan_link");
345 
346  p = p_header_frame * pose_c; // Object pose in world
347  p2 = ks3.getAttachedBody("boxC")->getGlobalPose();
348  EXPECT_TRUE(p.isApprox(p2, EPSILON));
349 
350  p = p_link.inverse() * p_header_frame * pose_c; // Object pose in link frame
351  p2 = ks3.getAttachedBody("boxC")->getPose();
352  EXPECT_TRUE(p.isApprox(p2, EPSILON));
353 }
