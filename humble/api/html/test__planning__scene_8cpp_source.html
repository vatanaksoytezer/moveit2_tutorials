93 TEST(PlanningScene, LoadRestoreDiff)
94 {
95  urdf::ModelInterfaceSharedPtr urdf_model = moveit::core::loadModelInterface("pr2");
96  srdf::ModelSharedPtr srdf_model = std::make_shared<srdf::Model>();
97  auto ps = std::make_shared<planning_scene::PlanningScene>(urdf_model, srdf_model);
98 
99  collision_detection::World& world = *ps->getWorldNonConst();
100 
101  /* add one object to ps's world */
102  Eigen::Isometry3d id = Eigen::Isometry3d::Identity();
103  world.addToObject("sphere", std::make_shared<const shapes::Sphere>(0.4), id);
104 
105  /* ps can be written to and set from message */
106  moveit_msgs::msg::PlanningScene ps_msg;
107  ps_msg.robot_state.is_diff = true;
108  EXPECT_TRUE(moveit::core::isEmpty(ps_msg));
109  ps->getPlanningSceneMsg(ps_msg);
110  ps->setPlanningSceneMsg(ps_msg);
111  EXPECT_EQ(ps_msg.world.collision_objects.size(), 1u);
112  EXPECT_EQ(ps_msg.world.collision_objects[0].id, "sphere");
113  EXPECT_TRUE(world.hasObject("sphere"));
114 
115  /* test diff scene on top of ps */
116  planning_scene::PlanningScenePtr next = ps->diff();
117  /* world is inherited from ps */
118  EXPECT_TRUE(next->getWorld()->hasObject("sphere"));
119 
120  /* object in overlay is only added in overlay */
121  next->getWorldNonConst()->addToObject("sphere_in_next_only", std::make_shared<const shapes::Sphere>(0.5), id);
122  EXPECT_EQ(next->getWorld()->size(), 2u);
123  EXPECT_EQ(ps->getWorld()->size(), 1u);
124 
125  /* the worlds used for collision detection contain one and two objects, respectively */
126  EXPECT_EQ(ps->getCollisionEnv()->getWorld()->size(), 1u);
127  EXPECT_EQ(ps->getCollisionEnvUnpadded()->getWorld()->size(), 1u);
128 
129  EXPECT_EQ(next->getCollisionEnv()->getWorld()->size(), 2u);
130  EXPECT_EQ(next->getCollisionEnvUnpadded()->getWorld()->size(), 2u);
131 
132  /* maintained diff contains only overlay object */
133  next->getPlanningSceneDiffMsg(ps_msg);
134  EXPECT_EQ(ps_msg.world.collision_objects.size(), 1u);
135 
136  /* copy ps to next and apply diff */
137  next->decoupleParent();
138  moveit_msgs::msg::PlanningScene ps_msg2;
139 
140  /* diff is empty now */
141  next->getPlanningSceneDiffMsg(ps_msg2);
142  EXPECT_EQ(ps_msg2.world.collision_objects.size(), 0u);
143 
144  /* next's world contains both objects */
145  next->getPlanningSceneMsg(ps_msg);
146  EXPECT_EQ(ps_msg.world.collision_objects.size(), 2u);
147  ps->setPlanningSceneMsg(ps_msg);
148  EXPECT_EQ(ps->getWorld()->size(), 2u);
149  EXPECT_EQ(ps->getCollisionEnv()->getWorld()->size(), 2u);
150  EXPECT_EQ(ps->getCollisionEnvUnpadded()->getWorld()->size(), 2u);
151 }

302 TEST_P(CollisionDetectorTests, ClearDiff)
303 {
304  const std::string plugin_name = GetParam();
305  SCOPED_TRACE(plugin_name);
306 
307  urdf::ModelInterfaceSharedPtr urdf_model = moveit::core::loadModelInterface("pr2");
308  srdf::ModelSharedPtr srdf_model = std::make_shared<srdf::Model>();
309  // create parent scene
310  planning_scene::PlanningScenePtr parent = std::make_shared<planning_scene::PlanningScene>(urdf_model, srdf_model);
311 
312  collision_detection::CollisionPluginCache loader;
313  if (!loader.activate(plugin_name, parent))
314  {
315 #if defined(GTEST_SKIP_)
316  GTEST_SKIP_("Failed to load collision plugin");
317 #else
318  return;
319 #endif
320  }
321 
322  // create child scene
323  planning_scene::PlanningScenePtr child = parent->diff();
324 
325  // create collision request variables
326  collision_detection::CollisionRequest req;
327  collision_detection::CollisionResult res;
328  moveit::core::RobotState* state = new moveit::core::RobotState(child->getRobotModel());
329  state->setToDefaultValues();
330  state->update();
331 
332  // there should be no collision with the environment
333  res.clear();
334  parent->getCollisionEnv()->checkRobotCollision(req, res, *state, parent->getAllowedCollisionMatrix());
335  EXPECT_FALSE(res.collision);
336  res.clear();
337  child->getCollisionEnv()->checkRobotCollision(req, res, *state, child->getAllowedCollisionMatrix());
338  EXPECT_FALSE(res.collision);
339 
340  // create message to add a collision object at the world origin
341  moveit_msgs::msg::PlanningScene ps_msg;
342  ps_msg.is_diff = false;
343  moveit_msgs::msg::CollisionObject co;
344  co.header.frame_id = "base_link";
345  co.operation = moveit_msgs::msg::CollisionObject::ADD;
346  co.id = "box";
347  co.pose.orientation.w = 1.0;
348  {
349  shape_msgs::msg::SolidPrimitive sp;
350  sp.type = shape_msgs::msg::SolidPrimitive::BOX;
351  sp.dimensions = { 1., 1., 1. };
352  co.primitives.push_back(sp);
353  geometry_msgs::msg::Pose sp_pose;
354  sp_pose.orientation.w = 1.0;
355  co.primitive_poses.push_back(sp_pose);
356  }
357  ps_msg.world.collision_objects.push_back(co);
358 
359  // add object to the parent planning scene
360  parent->usePlanningSceneMsg(ps_msg);
361 
362  // the parent scene should be in collision
363  res.clear();
364  parent->getCollisionEnv()->checkRobotCollision(req, res, *state, parent->getAllowedCollisionMatrix());
365  EXPECT_TRUE(res.collision);
366 
367  // the child scene was not updated yet, so no collision
368  res.clear();
369  child->getCollisionEnv()->checkRobotCollision(req, res, *state, child->getAllowedCollisionMatrix());
370  EXPECT_FALSE(res.collision);
371 
372  // update the child scene
373  child->clearDiffs();
374 
375  // child and parent scene should be in collision
376  res.clear();
377  parent->getCollisionEnv()->checkRobotCollision(req, res, *state, parent->getAllowedCollisionMatrix());
378  EXPECT_TRUE(res.collision);
379  res.clear();
380  child->getCollisionEnv()->checkRobotCollision(req, res, *state, child->getAllowedCollisionMatrix());
381  EXPECT_TRUE(res.collision);
382 
383  child.reset();
384  parent.reset();
385 }

388 moveit_msgs::msg::PlanningScene create_planning_scene_diff(planning_scene::PlanningScene& ps,
389  const std::string& object_name, const int8_t operation,
390  const bool attach_object = false,
391  const bool create_object = true)
392 {
393  // Helper function to create an object for RobotStateDiffBug
394  auto add_object = [](const std::string& object_name, const int8_t operation) {
395  moveit_msgs::msg::CollisionObject co;
396  co.header.frame_id = "panda_link0";
397  co.id = object_name;
398  co.operation = operation;
399  co.primitives.push_back([] {
400  shape_msgs::msg::SolidPrimitive primitive;
401  primitive.type = shape_msgs::msg::SolidPrimitive::SPHERE;
402  primitive.dimensions.push_back(1.0);
403  return primitive;
404  }());
405  co.primitive_poses.push_back([] {
406  geometry_msgs::msg::Pose pose;
407  pose.orientation.w = 1.0;
408  return pose;
409  }());
410  co.pose = co.primitive_poses[0];
411  return co;
412  };
413  // Helper function to create an attached object for RobotStateDiffBug
414  auto add_attached_object = [](const std::string& object_name, const int8_t operation) {
415  moveit_msgs::msg::AttachedCollisionObject aco;
416  aco.object.operation = operation;
417  aco.object.id = object_name;
418  aco.link_name = "panda_link0";
419  return aco;
420  };
421 
422  auto new_ps = ps.diff();
423  if (operation == moveit_msgs::msg::CollisionObject::REMOVE ||
424  (operation == moveit_msgs::msg::CollisionObject::ADD && create_object))
425  new_ps->processCollisionObjectMsg(add_object(object_name, operation));
426  if (attach_object)
427  new_ps->processAttachedCollisionObjectMsg(add_attached_object(object_name, operation));
428  moveit_msgs::msg::PlanningScene scene_msg;
429  new_ps->getPlanningSceneDiffMsg(scene_msg);
430  return scene_msg;
431 }
