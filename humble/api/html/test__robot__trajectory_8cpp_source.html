66  void initTestTrajectory(robot_trajectory::RobotTrajectoryPtr& trajectory)
67  {
68  // Init a traj
69  ASSERT_TRUE(robot_model_->hasJointModelGroup(arm_jmg_name_))
70  << "Robot model does not have group: " << arm_jmg_name_;
71 
72  trajectory = std::make_shared<robot_trajectory::RobotTrajectory>(robot_model_, arm_jmg_name_);
73 
74  EXPECT_EQ(trajectory->getGroupName(), arm_jmg_name_) << "Generated trajectory group name does not match";
75  EXPECT_TRUE(trajectory->empty()) << "Generated trajectory not empty";
76 
77  double duration_from_previous = 0.1;
78  std::size_t waypoint_count = 5;
79  for (std::size_t ix = 0; ix < waypoint_count; ++ix)
80  trajectory->addSuffixWayPoint(*robot_state_, duration_from_previous);
81  // Quick check that getDuration is working correctly
82  EXPECT_EQ(trajectory->getDuration(), duration_from_previous * waypoint_count)
83  << "Generated trajectory duration incorrect";
84  EXPECT_EQ(waypoint_count, trajectory->getWayPointDurations().size())
85  << "Generated trajectory has the wrong number of waypoints";
86  EXPECT_EQ(waypoint_count, trajectory->size());

99  void modifyFirstWaypointPtrAndCheckTrajectory(robot_trajectory::RobotTrajectoryPtr& trajectory)
100  {
102  // Get the first waypoint by POINTER, modify it, and check that the value WAS updated in trajectory
104  // Get the first waypoint by shared pointer
105  moveit::core::RobotStatePtr trajectory_first_waypoint = trajectory->getWayPointPtr(0);
106  // Get the first waypoint joint values
107  std::vector<double> trajectory_first_state;
108  trajectory_first_waypoint->copyJointGroupPositions(arm_jmg_name_, trajectory_first_state);
109 
110  // Modify the first waypoint joint values
111  trajectory_first_state[0] += 0.01;
112  trajectory_first_waypoint->setJointGroupPositions(arm_jmg_name_, trajectory_first_state);
113 
114  // Check that the trajectory's first waypoint was updated
115  moveit::core::RobotStatePtr trajectory_first_waypoint_after_update = trajectory->getWayPointPtr(0);
116  std::vector<double> trajectory_first_state_after_update;
117  trajectory_first_waypoint_after_update->copyJointGroupPositions(arm_jmg_name_, trajectory_first_state_after_update);
118  EXPECT_EQ(trajectory_first_state[0], trajectory_first_state_after_update[0]);
119 
120  // Modify the first waypoint duration
121  double trajectory_first_duration_before_update = trajectory->getWayPointDurationFromPrevious(0);
122  double new_duration = trajectory_first_duration_before_update + 0.1;
123  trajectory->setWayPointDurationFromPrevious(0, new_duration);
124 
125  // Check that the trajectory's first duration was updated

256 TEST_F(RobotTrajectoryTestFixture, RobotTrajectoryIterator)
257 {
258  robot_trajectory::RobotTrajectoryPtr trajectory;
259  initTestTrajectory(trajectory);
260 
261  ASSERT_EQ(5u, trajectory->size());
262  std::vector<double> positions;
263 
264  double start_pos = 0.0;
265 
266  for (size_t i = 0; i < trajectory->size(); ++i)
267  {
268  auto waypoint = trajectory->getWayPointPtr(i);
269  // modify joint values
270  waypoint->copyJointGroupPositions(arm_jmg_name_, positions);
271  start_pos = positions[0];
272  positions[0] += 0.01 * i;
273  waypoint->setJointGroupPositions(arm_jmg_name_, positions);
274  }
275 
276  unsigned int count = 0;
277  for (const auto& waypoint_and_duration : *trajectory)
278  {
279  const auto& waypoint = waypoint_and_duration.first;
280  waypoint->copyJointGroupPositions(arm_jmg_name_, positions);
281  EXPECT_EQ(start_pos + count * 0.01, positions[0]);
282  count++;
283  }
284 
285  EXPECT_EQ(count, trajectory->size());
286 
287  // Consistency checks
288  EXPECT_EQ(trajectory->begin(), trajectory->begin());
289  EXPECT_EQ(trajectory->end(), trajectory->end());
290 
291  // trajectory has length 5; incrementing begin 5 times should reach the end
292  EXPECT_NE(trajectory->begin(), trajectory->end());
293  EXPECT_NE(++trajectory->begin(), trajectory->end());
294  EXPECT_NE(++(++trajectory->begin()), trajectory->end());
295  EXPECT_NE(++(++(++trajectory->begin())), trajectory->end());
296  EXPECT_NE(++(++(++(++trajectory->begin()))), trajectory->end());
297  EXPECT_EQ(++(++(++(++(++trajectory->begin())))), trajectory->end());
298 }
