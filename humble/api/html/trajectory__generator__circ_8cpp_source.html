73 void TrajectoryGeneratorCIRC::cmdSpecificRequestValidation(const planning_interface::MotionPlanRequest& req) const
74 {
75  if (!(req.path_constraints.name == "interim" || req.path_constraints.name == "center"))
76  {
77  std::ostringstream os;
78  os << "No path constraint named \"interim\" or \"center\" found (found "
79  "unknown constraint: "
80  << "\"req.path_constraints.name\""
81  << " instead)";
82  throw UnknownPathConstraintName(os.str());
83  }
84 
85  if (req.path_constraints.position_constraints.size() != 1)
86  {
87  throw NoPositionConstraints("CIRC trajectory generator needs valid a position constraint");
88  }
89 
90  if (req.path_constraints.position_constraints.front().constraint_region.primitive_poses.size() != 1)
91  {
92  throw NoPrimitivePose("CIRC trajectory generator needs valid a primitive pose");
93  }
94 }

96 void TrajectoryGeneratorCIRC::extractMotionPlanInfo(const planning_scene::PlanningSceneConstPtr& scene,
97  const planning_interface::MotionPlanRequest& req,
98  TrajectoryGenerator::MotionPlanInfo& info) const
99 {
100  RCLCPP_DEBUG(LOGGER, "Extract necessary information from motion plan request.");
101 
102  info.group_name = req.group_name;
103  std::string frame_id{ robot_model_->getModelFrame() };
104 
105  // goal given in joint space
106  if (!req.goal_constraints.front().joint_constraints.empty())
107  {
108  // TODO: link name from goal constraint and path constraint
109  info.link_name = req.path_constraints.position_constraints.front().link_name;
110  if (!robot_model_->hasLinkModel(info.link_name))
111  {
112  throw UnknownLinkNameOfAuxiliaryPoint("Unknown link name of CIRC auxiliary point");
113  }
114 
115  if (req.goal_constraints.front().joint_constraints.size() !=
116  robot_model_->getJointModelGroup(req.group_name)->getActiveJointModelNames().size())
117  {
118  std::ostringstream os;
119  os << "Number of joint constraint = " << req.goal_constraints.front().joint_constraints.size()
120  << " not equal to active joints of group = "
121  << robot_model_->getJointModelGroup(req.group_name)->getActiveJointModelNames().size();
122  throw NumberOfConstraintsMismatch(os.str());
123  }
124 
125  // initializing all joints of the model
126  for (const auto& joint_name : robot_model_->getVariableNames())
127  {
128  info.goal_joint_position[joint_name] = 0;
129  }
130 
131  for (const auto& joint_item : req.goal_constraints.front().joint_constraints)
132  {
133  info.goal_joint_position[joint_item.joint_name] = joint_item.position;
134  }
135 
136  computeLinkFK(robot_model_, info.link_name, info.goal_joint_position, info.goal_pose);
137  }
138  // goal given in Cartesian space
139  else
140  {
141  info.link_name = req.goal_constraints.front().position_constraints.front().link_name;
142  if (req.goal_constraints.front().position_constraints.front().header.frame_id.empty() ||
143  req.goal_constraints.front().orientation_constraints.front().header.frame_id.empty())
144  {
145  RCLCPP_WARN(LOGGER, "Frame id is not set in position/orientation constraints of "
146  "goal. Use model frame as default");
147  frame_id = robot_model_->getModelFrame();
148  }
149  else
150  {
151  frame_id = req.goal_constraints.front().position_constraints.front().header.frame_id;
152  }
153  info.goal_pose = getConstraintPose(req.goal_constraints.front());
154  }
155 
156  assert(req.start_state.joint_state.name.size() == req.start_state.joint_state.position.size());
157  for (const auto& joint_name : robot_model_->getJointModelGroup(req.group_name)->getActiveJointModelNames())
158  {
159  auto it{ std::find(req.start_state.joint_state.name.cbegin(), req.start_state.joint_state.name.cend(), joint_name) };
160  if (it == req.start_state.joint_state.name.cend())
161  {
162  std::ostringstream os;
163  os << "Could not find joint \"" << joint_name << "\" of group \"" << req.group_name
164  << "\" in start state of request";
165  throw CircJointMissingInStartState(os.str());
166  }
167  size_t index = it - req.start_state.joint_state.name.cbegin();
168  info.start_joint_position[joint_name] = req.start_state.joint_state.position[index];
169  }
170 
171  computeLinkFK(robot_model_, info.link_name, info.start_joint_position, info.start_pose);
172 
173  // check goal pose ik before Cartesian motion plan starts
174  std::map<std::string, double> ik_solution;
175  if (!computePoseIK(scene, info.group_name, info.link_name, info.goal_pose, frame_id, info.start_joint_position,
176  ik_solution))
177  {
178  // LCOV_EXCL_START
179  std::ostringstream os;
180  os << "Failed to compute inverse kinematics for link: " << info.link_name << " of goal pose";
181  throw CircInverseForGoalIncalculable(os.str());
182  // LCOV_EXCL_STOP // not able to trigger here since lots of checks before
183  // are in place
184  }
185  info.circ_path_point.first = req.path_constraints.name;
186  if (!req.goal_constraints.front().position_constraints.empty())
187  {
188  const moveit_msgs::msg::Constraints& goal = req.goal_constraints.front();
189  info.circ_path_point.second =
190  getConstraintPose(
191  req.path_constraints.position_constraints.front().constraint_region.primitive_poses.front().position,
192  goal.orientation_constraints.front().orientation, goal.position_constraints.front().target_point_offset)
193  .translation();
194  }
195  else
196  {
197  Eigen::Vector3d circ_path_point;
198  tf2::fromMsg(req.path_constraints.position_constraints.front().constraint_region.primitive_poses.front().position,
199  circ_path_point);
200  info.circ_path_point.second = circ_path_point;
201  }
202 }

204 void TrajectoryGeneratorCIRC::plan(const planning_scene::PlanningSceneConstPtr& scene,
205  const planning_interface::MotionPlanRequest& req, const MotionPlanInfo& plan_info,
206  const double& sampling_time, trajectory_msgs::msg::JointTrajectory& joint_trajectory)
207 {
208  std::unique_ptr<KDL::Path> cart_path(setPathCIRC(plan_info));
209  std::unique_ptr<KDL::VelocityProfile> vel_profile(
210  cartesianTrapVelocityProfile(req.max_velocity_scaling_factor, req.max_acceleration_scaling_factor, cart_path));
211 
212  // combine path and velocity profile into Cartesian trajectory
213  // with the third parameter set to false, KDL::Trajectory_Segment does not
214  // take
215  // the ownship of Path and Velocity Profile
216  KDL::Trajectory_Segment cart_trajectory(cart_path.get(), vel_profile.get(), false);
217 
218  moveit_msgs::msg::MoveItErrorCodes error_code;
219  // sample the Cartesian trajectory and compute joint trajectory using inverse
220  // kinematics
221  if (!generateJointTrajectory(scene, planner_limits_.getJointLimitContainer(), cart_trajectory, plan_info.group_name,
222  plan_info.link_name, plan_info.start_joint_position, sampling_time, joint_trajectory,
223  error_code))
224  {
225  throw CircTrajectoryConversionFailure("Failed to generate valid joint trajectory from the Cartesian path",
226  error_code.val);
227  }

230 std::unique_ptr<KDL::Path> TrajectoryGeneratorCIRC::setPathCIRC(const MotionPlanInfo& info) const
231 {
232  RCLCPP_DEBUG(LOGGER, "Set Cartesian path for CIRC command.");
233 
234  KDL::Frame start_pose, goal_pose;
235  tf2::transformEigenToKDL(info.start_pose, start_pose);
236  tf2::transformEigenToKDL(info.goal_pose, goal_pose);
237 
238  const auto& eigen_path_point = info.circ_path_point.second;
239  const KDL::Vector path_point{ eigen_path_point.x(), eigen_path_point.y(), eigen_path_point.z() };
240 
241  // pass the ratio of translational by rotational velocity as equivalent radius
242  // to get a trajectory with rotational speed, if no (or very little)
243  // translational distance
244  // The KDL::Path implementation chooses the motion with the longer duration
245  // (translation vs. rotation)
246  // and uses eqradius as scaling factor between the distances.
247  double eqradius = planner_limits_.getCartesianLimits().getMaxTranslationalVelocity() /
248  planner_limits_.getCartesianLimits().getMaxRotationalVelocity();
249 
250  try
251  {
252  if (info.circ_path_point.first == "center")
253  {
254  return PathCircleGenerator::circleFromCenter(start_pose, goal_pose, path_point, eqradius);
255  }
256  else // if (info.circ_path_point.first == "interim")
257  {
258  return PathCircleGenerator::circleFromInterim(start_pose, goal_pose, path_point, eqradius);
259  }
260  }
261  catch (KDL::Error_MotionPlanning_Circle_No_Plane& e)
262  {
263  std::ostringstream os;
264  os << "Failed to create path object for circle." << e.Description();
265  throw CircleNoPlane(os.str());
266  }
267  catch (KDL::Error_MotionPlanning_Circle_ToSmall& e)
268  {
269  std::ostringstream os;
270  os << "Failed to create path object for circle." << e.Description();
271  throw CircleToSmall(os.str());
272  }
273  catch (ErrorMotionPlanningCenterPointDifferentRadius& e)
274  {
275  std::ostringstream os;
276  os << "Failed to create path object for circle." << e.Description();
277  throw CenterPointDifferentRadius(os.str());
278  }
279 
280  return nullptr;
281 }
