73 void TrajectoryGeneratorLIN::extractMotionPlanInfo(const planning_scene::PlanningSceneConstPtr& scene,
74  const planning_interface::MotionPlanRequest& req,
75  TrajectoryGenerator::MotionPlanInfo& info) const
76 {
77  RCLCPP_DEBUG(LOGGER, "Extract necessary information from motion plan request.");
78 
79  info.group_name = req.group_name;
80  std::string frame_id{ robot_model_->getModelFrame() };
81 
82  // goal given in joint space
83  if (!req.goal_constraints.front().joint_constraints.empty())
84  {
85  info.link_name = robot_model_->getJointModelGroup(req.group_name)->getSolverInstance()->getTipFrame();
86 
87  if (req.goal_constraints.front().joint_constraints.size() !=
88  robot_model_->getJointModelGroup(req.group_name)->getActiveJointModelNames().size())
89  {
90  std::ostringstream os;
91  os << "Number of joints in goal does not match number of joints of group "
92  "(Number joints goal: "
93  << req.goal_constraints.front().joint_constraints.size() << " | Number of joints of group: "
94  << robot_model_->getJointModelGroup(req.group_name)->getActiveJointModelNames().size() << ")";
95  throw JointNumberMismatch(os.str());
96  }
97  // initializing all joints of the model
98  for (const auto& joint_name : robot_model_->getVariableNames())
99  {
100  info.goal_joint_position[joint_name] = 0;
101  }
102 
103  for (const auto& joint_item : req.goal_constraints.front().joint_constraints)
104  {
105  info.goal_joint_position[joint_item.joint_name] = joint_item.position;
106  }
107 
108  // Ignored return value because at this point the function should always
109  // return 'true'.
110  computeLinkFK(robot_model_, info.link_name, info.goal_joint_position, info.goal_pose);
111  }
112  // goal given in Cartesian space
113  else
114  {
115  info.link_name = req.goal_constraints.front().position_constraints.front().link_name;
116  if (req.goal_constraints.front().position_constraints.front().header.frame_id.empty() ||
117  req.goal_constraints.front().orientation_constraints.front().header.frame_id.empty())
118  {
119  RCLCPP_WARN(LOGGER, "Frame id is not set in position/orientation constraints of "
120  "goal. Use model frame as default");
121  frame_id = robot_model_->getModelFrame();
122  }
123  else
124  {
125  frame_id = req.goal_constraints.front().position_constraints.front().header.frame_id;
126  }
127  info.goal_pose = getConstraintPose(req.goal_constraints.front());
128  }
129 
130  assert(req.start_state.joint_state.name.size() == req.start_state.joint_state.position.size());
131  for (const auto& joint_name : robot_model_->getJointModelGroup(req.group_name)->getActiveJointModelNames())
132  {
133  auto it{ std::find(req.start_state.joint_state.name.cbegin(), req.start_state.joint_state.name.cend(), joint_name) };
134  if (it == req.start_state.joint_state.name.cend())
135  {
136  std::ostringstream os;
137  os << "Could not find joint \"" << joint_name << "\" of group \"" << req.group_name
138  << "\" in start state of request";
139  throw LinJointMissingInStartState(os.str());
140  }
141  size_t index = it - req.start_state.joint_state.name.cbegin();
142  info.start_joint_position[joint_name] = req.start_state.joint_state.position[index];
143  }
144 
145  // Ignored return value because at this point the function should always
146  // return 'true'.
147  computeLinkFK(robot_model_, info.link_name, info.start_joint_position, info.start_pose);
148 
149  // check goal pose ik before Cartesian motion plan starts
150  std::map<std::string, double> ik_solution;
151  if (!computePoseIK(scene, info.group_name, info.link_name, info.goal_pose, frame_id, info.start_joint_position,
152  ik_solution))
153  {
154  std::ostringstream os;
155  os << "Failed to compute inverse kinematics for link: " << info.link_name << " of goal pose";
156  throw LinInverseForGoalIncalculable(os.str());
157  }
158 }

160 void TrajectoryGeneratorLIN::plan(const planning_scene::PlanningSceneConstPtr& scene,
161  const planning_interface::MotionPlanRequest& req, const MotionPlanInfo& plan_info,
162  const double& sampling_time, trajectory_msgs::msg::JointTrajectory& joint_trajectory)
163 {
164  // create Cartesian path for lin
165  std::unique_ptr<KDL::Path> path(setPathLIN(plan_info.start_pose, plan_info.goal_pose));
166 
167  // create velocity profile
168  std::unique_ptr<KDL::VelocityProfile> vp(
169  cartesianTrapVelocityProfile(req.max_velocity_scaling_factor, req.max_acceleration_scaling_factor, path));
170 
171  // combine path and velocity profile into Cartesian trajectory
172  // with the third parameter set to false, KDL::Trajectory_Segment does not
173  // take
174  // the ownship of Path and Velocity Profile
175  KDL::Trajectory_Segment cart_trajectory(path.get(), vp.get(), false);
176 
177  moveit_msgs::msg::MoveItErrorCodes error_code;
178  // sample the Cartesian trajectory and compute joint trajectory using inverse
179  // kinematics
180  if (!generateJointTrajectory(scene, planner_limits_.getJointLimitContainer(), cart_trajectory, plan_info.group_name,
181  plan_info.link_name, plan_info.start_joint_position, sampling_time, joint_trajectory,
182  error_code))
183  {
184  std::ostringstream os;
185  os << "Failed to generate valid joint trajectory from the Cartesian path";
186  throw LinTrajectoryConversionFailure(os.str(), error_code.val);
187  }
188 }

190 std::unique_ptr<KDL::Path> TrajectoryGeneratorLIN::setPathLIN(const Eigen::Affine3d& start_pose,
191  const Eigen::Affine3d& goal_pose) const
192 {
193  RCLCPP_DEBUG(LOGGER, "Set Cartesian path for LIN command.");
194 
195  KDL::Frame kdl_start_pose, kdl_goal_pose;
196  tf2::transformEigenToKDL(start_pose, kdl_start_pose);
197  tf2::transformEigenToKDL(goal_pose, kdl_goal_pose);
198  double eqradius = planner_limits_.getCartesianLimits().getMaxTranslationalVelocity() /
199  planner_limits_.getCartesianLimits().getMaxRotationalVelocity();
200  KDL::RotationalInterpolation* rot_interpo = new KDL::RotationalInterpolation_SingleAxis();
201 
202  return std::unique_ptr<KDL::Path>(
203  std::make_unique<KDL::Path_Line>(kdl_start_pose, kdl_goal_pose, rot_interpo, eqradius, true));
204 }
