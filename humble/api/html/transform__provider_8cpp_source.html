94 bool TransformProvider::getTransform(mesh_filter::MeshHandle handle, Eigen::Isometry3d& transform) const
95 {
96  auto context_it = handle2context_.find(handle);
97 
98  if (context_it == handle2context_.end())
99  {
100  ROS_ERROR("Unable to find mesh with handle %d", handle);
101  return false;
102  }
103  context_it->second->mutex_.lock();
104  transform = context_it->second->transformation_;
105  context_it->second->mutex_.unlock();
106  return !(transform.matrix().isZero(0));
107 }

126 void TransformProvider::updateTransforms()
127 {
128  // Don't bother if frame_id_ is empty (true initially)
129  if (frame_id_.empty())
130  {
131  ROS_DEBUG_THROTTLE(2., "Not updating transforms because frame_id_ is empty.");
132  return;
133  }
134 
135  static tf2::Stamped<Eigen::Isometry3d> input_transform, output_transform;
136  static moveit::core::RobotStatePtr robot_state;
137  robot_state = psm_->getStateMonitor()->getCurrentState();
138  try
139  {
140  geometry_msgs::TransformStamped common_tf =
141  tf_buffer_->lookupTransform(frame_id_, psm_->getPlanningScene()->getPlanningFrame(), ros::Time(0.0));
142  input_transform.stamp_ = common_tf.header.stamp;
143  }
144  catch (tf2::TransformException& ex)
145  {
146  ROS_ERROR("TF Problem: %s", ex.what());
147  return;
148  }
149  input_transform.frame_id_ = psm_->getPlanningScene()->getPlanningFrame();
150 
151  for (auto& context_it : handle2context_)
152  {
153  try
154  {
155  // TODO: check logic here - which global collision body's transform should be used?
156  input_transform.setData(robot_state->getGlobalLinkTransform(context_it.second->frame_id_));
157  tf_buffer_->transform(input_transform, output_transform, frame_id_);
158  }
159  catch (const tf2::TransformException& ex)
160  {
161  handle2context_[context_it.first]->mutex_.lock();
162  handle2context_[context_it.first]->transformation_.matrix().setZero();
163  handle2context_[context_it.first]->mutex_.unlock();
164  continue;
165  }
166  catch (std::exception& ex)
167  {
168  ROS_ERROR("Caught %s while updating transforms", ex.what());
169  }
170  handle2context_[context_it.first]->mutex_.lock();
171  handle2context_[context_it.first]->transformation_ = static_cast<Eigen::Isometry3d>(output_transform);
172  handle2context_[context_it.first]->mutex_.unlock();
173  }
174 }
