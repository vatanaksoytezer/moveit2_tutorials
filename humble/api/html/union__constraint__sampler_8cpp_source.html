45 struct OrderSamplers
46 {
47  bool operator()(const ConstraintSamplerPtr& a, const ConstraintSamplerPtr& b) const
48  {
49  const std::vector<std::string>& alinks = a->getJointModelGroup()->getUpdatedLinkModelNames();
50  const std::vector<std::string>& blinks = b->getJointModelGroup()->getUpdatedLinkModelNames();
51  std::set<std::string> a_updates(alinks.begin(), alinks.end());
52  std::set<std::string> b_updates(blinks.begin(), blinks.end());
53 
54  bool a_contains_b = std::includes(a_updates.begin(), a_updates.end(), b_updates.begin(), b_updates.end());
55 
56  bool b_contains_a = std::includes(b_updates.begin(), b_updates.end(), a_updates.begin(), a_updates.end());
57 
58  // a contains b and sets are not equal
59  if (a_contains_b && !b_contains_a)
60  return true;
61  if (b_contains_a && !a_contains_b)
62  return false;
63 
64  // sets are equal or disjoint
65  bool a_depends_on_b = false;
66  bool b_depends_on_a = false;
67  const std::vector<std::string>& fda = a->getFrameDependency();
68  const std::vector<std::string>& fdb = b->getFrameDependency();
69  for (std::size_t i = 0; i < fda.size() && !a_depends_on_b; ++i)
70  for (const std::string& blink : blinks)
71  if (blink == fda[i])
72  {
73  a_depends_on_b = true;
74  break;
75  }
76  for (std::size_t i = 0; i < fdb.size() && !b_depends_on_a; ++i)
77  for (const std::string& alink : alinks)
78  if (alink == fdb[i])
79  {
80  b_depends_on_a = true;
81  break;
82  }
83  if (b_depends_on_a && a_depends_on_b)
84  {
85  RCLCPP_WARN(LOGGER,
86  "Circular frame dependency! "
87  "Sampling will likely produce invalid results (sampling for groups '%s' and '%s')",
88  a->getJointModelGroup()->getName().c_str(), b->getJointModelGroup()->getName().c_str());
89  return true;
90  }
91  if (b_depends_on_a && !a_depends_on_b)
92  return true;
93  if (a_depends_on_b && !b_depends_on_a)
94  return false;
95 
96  // prefer sampling JointConstraints first
97  JointConstraintSampler* ja = dynamic_cast<JointConstraintSampler*>(a.get());
98  JointConstraintSampler* jb = dynamic_cast<JointConstraintSampler*>(b.get());
99  if (ja && jb == nullptr)
100  return true;
101  if (jb && ja == nullptr)
102  return false;
103 
104  // neither depends on either, so break ties based on group name
105  return (a->getJointModelGroup()->getName() < b->getJointModelGroup()->getName());
106  }
107 };

109 UnionConstraintSampler::UnionConstraintSampler(const planning_scene::PlanningSceneConstPtr& scene,
110  const std::string& group_name,
111  const std::vector<ConstraintSamplerPtr>& samplers)
112  : ConstraintSampler(scene, group_name), samplers_(samplers)
113 {
114  // using stable sort to preserve order of equivalents
115  std::stable_sort(samplers_.begin(), samplers_.end(), OrderSamplers());
116 
117  for (ConstraintSamplerPtr& sampler : samplers_)
118  {
119  const std::vector<std::string>& fds = sampler->getFrameDependency();
120  for (const std::string& fd : fds)
121  frame_depends_.push_back(fd);
122 
123  RCLCPP_DEBUG(LOGGER, "Union sampler for group '%s' includes sampler for group '%s'", jmg_->getName().c_str(),
124  sampler->getJointModelGroup()->getName().c_str());
125  }

128 bool UnionConstraintSampler::sample(moveit::core::RobotState& state, const moveit::core::RobotState& reference_state,
129  unsigned int max_attempts)
130 {
131  state = reference_state;
132  state.setToRandomPositions(jmg_);
133 
134  if (!samplers_.empty())
135  {
136  if (!samplers_[0]->sample(state, reference_state, max_attempts))
137  return false;
138  }
139 
140  for (std::size_t i = 1; i < samplers_.size(); ++i)
141  {
142  // ConstraintSampler::sample returns states with dirty link transforms (because it only writes values)
143  // but requires a state with clean link transforms as input. This means that we need to clean the link
144  // transforms between calls to ConstraintSampler::sample.
145  state.updateLinkTransforms();
146  if (!samplers_[i]->sample(state, state, max_attempts))
147  return false;
148  }
149  return true;
150 }

158  // transforms between calls to ConstraintSampler::sample.

160  if (!sampler->project(state, max_attempts))
161  return false;
162  }
163  return true;
