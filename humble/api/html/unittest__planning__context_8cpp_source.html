135  planning_interface::MotionPlanRequest getValidRequest(const std::string& context_name) const
136  {
137  planning_interface::MotionPlanRequest req;
138 
139  req.planner_id =
140  std::string(context_name).erase(0, std::string("pilz_industrial_motion_planner::PlanningContext").length());
141  req.group_name = this->planning_group_;
142  req.max_velocity_scaling_factor = 0.01;
143  req.max_acceleration_scaling_factor = 0.01;
144 
145  // start state
146  moveit::core::RobotState rstate(this->robot_model_);
147  rstate.setToDefaultValues();
148  // state state in joint space, used as initial positions, since IK does not
149  // work at zero positions
150  rstate.setJointGroupPositions(this->planning_group_,
151  std::vector<double>{ 4.430233957464225e-12, 0.007881892504574495, -1.8157263253868452,
152  1.1801525390026025e-11, 1.8236082178909834,
153  8.591793942969161e-12 });
154  Eigen::Isometry3d start_pose(Eigen::Isometry3d::Identity());
155  start_pose.translation() = Eigen::Vector3d(0.3, 0, 0.65);
156  rstate.setFromIK(this->robot_model_->getJointModelGroup(this->planning_group_), start_pose);
157  moveit::core::robotStateToRobotStateMsg(rstate, req.start_state, false);
158 
159  // goal constraint
160  Eigen::Isometry3d goal_pose(Eigen::Isometry3d::Identity());
161  goal_pose.translation() = Eigen::Vector3d(0, 0.3, 0.65);
162  Eigen::Matrix3d goal_rotation;
163  goal_rotation = Eigen::AngleAxisd(0 * M_PI, Eigen::Vector3d::UnitZ());
164  goal_pose.linear() = goal_rotation;
165  rstate.setFromIK(this->robot_model_->getJointModelGroup(this->planning_group_), goal_pose);
166  req.goal_constraints.push_back(kinematic_constraints::constructGoalConstraints(
167  rstate, this->robot_model_->getJointModelGroup(this->planning_group_)));
168 
169  // path constraint
170  req.path_constraints.name = "center";
171  moveit_msgs::msg::PositionConstraint center_point;
172  center_point.link_name = this->target_link_;
173  geometry_msgs::msg::Pose center_position;
174  center_position.position.x = 0.0;
175  center_position.position.y = 0.0;
176  center_position.position.z = 0.65;
177  center_point.constraint_region.primitive_poses.push_back(center_position);
178  req.path_constraints.position_constraints.push_back(center_point);
179 
180  return req;
181  }
