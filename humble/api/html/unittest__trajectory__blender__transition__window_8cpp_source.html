137  std::vector<planning_interface::MotionPlanResponse> generateLinTrajs(const Sequence& seq, size_t num_cmds)
138  {
139  std::vector<planning_interface::MotionPlanResponse> responses(num_cmds);
140  for (size_t index = 0; index < num_cmds; ++index)
141  {
142  planning_interface::MotionPlanRequest req{ seq.getCmd<LinCart>(index).toRequest() };
143  // Set start state of request to end state of previous trajectory (except
144  // for first)
145  if (index > 0)
146  {
147  moveit::core::robotStateToRobotStateMsg(responses[index - 1].trajectory_->getLastWayPoint(), req.start_state);
148  }
149  // generate trajectory
150  planning_interface::MotionPlanResponse resp;
151  if (!lin_generator_->generate(planning_scene_, req, resp, sampling_time_))
152  {
153  std::runtime_error("Failed to generate trajectory.");
154  }
155  responses.at(index) = resp;
156  }
157  return responses;
158  }

625 TEST_F(TrajectoryBlenderTransitionWindowTest, testNonLinearBlending)
626 {
627  const double sine_scaling_factor{ 0.01 };
628  const double time_scaling_factor{ 10 };
629 
630  Sequence seq{ data_loader_->getSequence("SimpleSequence") };
631 
632  std::vector<planning_interface::MotionPlanResponse> res{ generateLinTrajs(seq, 2) };
633 
634  // prepare looping over trajectories
635  std::vector<robot_trajectory::RobotTrajectoryPtr> sine_trajs(2);
636 
637  for (size_t traj_index = 0; traj_index < 2; ++traj_index)
638  {
639  auto lin_traj{ res.at(traj_index).trajectory_ };
640 
641  CartesianTrajectory cart_traj;
642  trajectory_msgs::msg::JointTrajectory joint_traj;
643  const double duration{ lin_traj->getWayPointDurationFromStart(lin_traj->getWayPointCount()) };
644  // time from start zero does not work
645  const double time_from_start_offset{ time_scaling_factor * lin_traj->getWayPointDurations().back() };
646 
647  // generate modified cartesian trajectory
648  for (size_t i = 0; i < lin_traj->getWayPointCount(); ++i)
649  {
650  // transform time to interval [0, 4*pi]
651  const double sine_arg{ 4 * M_PI * lin_traj->getWayPointDurationFromStart(i) / duration };
652 
653  // get pose
654  CartesianTrajectoryPoint waypoint;
655  const Eigen::Isometry3d eigen_pose{ lin_traj->getWayPointPtr(i)->getFrameTransform(target_link_) };
656  geometry_msgs::msg::Pose waypoint_pose = tf2::toMsg(eigen_pose);
657 
658  // add scaled sine function
659  waypoint_pose.position.x += sine_scaling_factor * sin(sine_arg);
660  waypoint_pose.position.y += sine_scaling_factor * sin(sine_arg);
661  waypoint_pose.position.z += sine_scaling_factor * sin(sine_arg);
662 
663  // add to trajectory
664  waypoint.pose = waypoint_pose;
665  waypoint.time_from_start = rclcpp::Duration::from_seconds(
666  time_from_start_offset + time_scaling_factor * lin_traj->getWayPointDurationFromStart(i));
667  cart_traj.points.push_back(waypoint);
668  }
669 
670  // prepare ik
671  std::map<std::string, double> initial_joint_position, initial_joint_velocity;
672  for (const std::string& joint_name :
673  lin_traj->getFirstWayPointPtr()->getJointModelGroup(planning_group_)->getActiveJointModelNames())
674  {
675  if (traj_index == 0)
676  {
677  initial_joint_position[joint_name] = lin_traj->getFirstWayPoint().getVariablePosition(joint_name);
678  initial_joint_velocity[joint_name] = lin_traj->getFirstWayPoint().getVariableVelocity(joint_name);
679  }
680  else
681  {
682  initial_joint_position[joint_name] =
683  sine_trajs[traj_index - 1]->getLastWayPoint().getVariablePosition(joint_name);
684  initial_joint_velocity[joint_name] =
685  sine_trajs[traj_index - 1]->getLastWayPoint().getVariableVelocity(joint_name);
686  }
687  }
688 
689  moveit_msgs::msg::MoveItErrorCodes error_code;
690  if (!generateJointTrajectory(planning_scene_, planner_limits_.getJointLimitContainer(), cart_traj, planning_group_,
691  target_link_, initial_joint_position, initial_joint_velocity, joint_traj, error_code,
692  true))
693  {
694  std::runtime_error("Failed to generate trajectory.");
695  }
696 
697  joint_traj.points.back().velocities.assign(joint_traj.points.back().velocities.size(), 0.0);
698  joint_traj.points.back().accelerations.assign(joint_traj.points.back().accelerations.size(), 0.0);
699 
700  // convert trajectory_msgs::JointTrajectory to robot_trajectory::RobotTrajectory
701  sine_trajs[traj_index] = std::make_shared<robot_trajectory::RobotTrajectory>(robot_model_, planning_group_);
702  sine_trajs.at(traj_index)->setRobotTrajectoryMsg(lin_traj->getFirstWayPoint(), joint_traj);
703  }
704 
705  TrajectoryBlendRequest blend_req;
706  TrajectoryBlendResponse blend_res;
707 
708  blend_req.group_name = planning_group_;
709  blend_req.link_name = target_link_;
710  blend_req.blend_radius = seq.getBlendRadius(0);
