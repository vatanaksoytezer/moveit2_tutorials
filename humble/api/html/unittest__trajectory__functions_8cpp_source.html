158 bool TrajectoryFunctionsTestBase::tfNear(const Eigen::Isometry3d& pose1, const Eigen::Isometry3d& pose2,
159  const double& epsilon)
160 {
161  for (std::size_t i = 0; i < 3; ++i)
162  for (std::size_t j = 0; j < 4; ++j)
163  {
164  if (fabs(pose1(i, j) - pose2(i, j)) > fabs(epsilon))
165  return false;
166  }
167  return true;
168 }

224  const kinematics::KinematicsBaseConstPtr& solver = jmg->getSolverInstance();
225

227  moveit::core::RobotState rstate(robot_model_);

230  {
231  // sample random robot state
232  rstate.setToRandomPositions(jmg, rng_);
233  rstate.update();
234  geometry_msgs::msg::Pose pose_expect = tf2::toMsg(rstate.getFrameTransform(ik_fast_link_));
235 
236  // prepare inverse kinematics
237  std::vector<geometry_msgs::msg::Pose> ik_poses;

247  }
248 
249  std::vector<std::vector<double>> ik_solutions;
250  kinematics::KinematicsResult ik_result;
251  moveit_msgs::msg::MoveItErrorCodes err_code;
252  kinematics::KinematicsQueryOptions options = kinematics::KinematicsQueryOptions();
253 
254  // compute all ik solutions
255  EXPECT_TRUE(solver->getPositionIK(ik_poses, ik_seed, ik_solutions, ik_result, options));
256 
257  // compute one ik solution
258  EXPECT_TRUE(solver->getPositionIK(pose_expect, ik_seed, ik_actual, err_code));
259 
260  ASSERT_EQ(ik_expect.size(), ik_actual.size());
261 
262  for (std::size_t i = 0; i < ik_expect.size(); ++i)
263  {
264  EXPECT_NEAR(ik_actual.at(i), ik_expect.at(i), 4 * IK_SEED_OFFSET);
265  }
266 
267  --random_test_number_;
268  }
269 }
