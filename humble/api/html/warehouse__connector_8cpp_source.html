70 bool WarehouseConnector::connectToDatabase(const std::string& dirname)
71 {
72  if (child_pid_ != 0)
73  kill(child_pid_, SIGTERM);
74 
75  child_pid_ = fork();
76  if (child_pid_ == -1)
77  {
78  RCLCPP_ERROR(LOGGER, "Error forking process.");
79  child_pid_ = 0;
80  return false;
81  }
82 
83  if (child_pid_ == 0)
84  {
85  std::size_t exec_file_pos = dbexec_.find_last_of("/\\");
86  if (exec_file_pos != std::string::npos)
87  {
88  char** argv = new char*[4];
89  std::size_t exec_length = 1 + dbexec_.length() - exec_file_pos;
90  argv[0] = new char[1 + exec_length];
91  snprintf(argv[0], exec_length, "%s", dbexec_.substr(exec_file_pos + 1).c_str());
92 
93  argv[1] = new char[16];
94  snprintf(argv[1], 15, "--dbpath");
95 
96  argv[2] = new char[1024];
97  snprintf(argv[2], 1023, "%s", dirname.c_str());
98 
99  argv[3] = nullptr;
100 
101  int code = execv(dbexec_.c_str(), argv);
102  delete[] argv[0];
103  delete[] argv[1];
104  delete[] argv[2];
105  delete[] argv;
106  RCLCPP_ERROR_STREAM(LOGGER,
107  "execv() returned " << code << ", errno=" << errno << " string errno = " << strerror(errno));
108  }
109  return false;
110  }
111  else
112  {
113  // sleep so mongod has time to come up
114  using namespace std::chrono_literals;
115  rclcpp::sleep_for(1s);
116  }
117  return true;
118 }
