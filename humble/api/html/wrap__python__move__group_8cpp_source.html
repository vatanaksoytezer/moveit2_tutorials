312  void convertListToArrayOfPoses(const bp::list& poses, std::vector<geometry_msgs::Pose>& msg)
313  {
314  int l = bp::len(poses);
315  for (int i = 0; i < l; ++i)
316  {
317  const bp::list& pose = bp::extract<bp::list>(poses[i]);
318  std::vector<double> v = py_bindings_tools::doubleFromList(pose);
319  if (v.size() == 6 || v.size() == 7)
320  {
321  Eigen::Isometry3d p;
322  if (v.size() == 6)
323  {
324  tf2::Quaternion tq;
325  tq.setRPY(v[3], v[4], v[5]);
326  Eigen::Quaterniond eq;
327  tf2::convert(tq, eq);
328  p = Eigen::Isometry3d(eq);
329  }
330  else
331  p = Eigen::Isometry3d(Eigen::Quaterniond(v[6], v[3], v[4], v[5]));
332  p.translation() = Eigen::Vector3d(v[0], v[1], v[2]);
333  geometry_msgs::Pose pm = tf2::toMsg(p);
334  msg.push_back(pm);
335  }
336  else
337  ROS_WARN("Incorrect number of values for a pose: %u", (unsigned int)v.size());
338  }
339  }

380  bool setPoseTargetPython(bp::list& pose, const std::string& end_effector_link = "")
381  {
382  std::vector<double> v = py_bindings_tools::doubleFromList(pose);
383  geometry_msgs::Pose msg;
384  if (v.size() == 6)
385  {
386  tf2::Quaternion q;
387  q.setRPY(v[3], v[4], v[5]);
388  tf2::convert(q, msg.orientation);
389  }
390  else if (v.size() == 7)
391  {
392  msg.orientation.x = v[3];
393  msg.orientation.y = v[4];
394  msg.orientation.z = v[5];
395  msg.orientation.w = v[6];
396  }
397  else
398  {
399  ROS_ERROR("Pose description expected to consist of either 6 or 7 values");
400  return false;
401  }
402  msg.position.x = v[0];
403  msg.position.y = v[1];
404  msg.position.z = v[2];
405  return setPoseTarget(msg, end_effector_link);

573  py_bindings_tools::ByteString retimeTrajectory(const py_bindings_tools::ByteString& ref_state_str,
574  const py_bindings_tools::ByteString& traj_str,
575  double velocity_scaling_factor, double acceleration_scaling_factor,
576  const std::string& algorithm)
577  {
578  // Convert reference state message to object
579  moveit_msgs::msg::RobotState ref_state_msg;
580  py_bindings_tools::deserializeMsg(ref_state_str, ref_state_msg);
581  moveit::core::RobotState ref_state_obj(getRobotModel());
582  if (moveit::core::robotStateMsgToRobotState(ref_state_msg, ref_state_obj, true))
583  {
584  // Convert trajectory message to object
585  moveit_msgs::msg::RobotTrajectory traj_msg;
586  py_bindings_tools::deserializeMsg(traj_str, traj_msg);
587  bool algorithm_found = true;
588  {
589  GILReleaser gr;
590  robot_trajectory::RobotTrajectory traj_obj(getRobotModel(), getName());
591  traj_obj.setRobotTrajectoryMsg(ref_state_obj, traj_msg);
592 
593  // Do the actual retiming
594  if (algorithm == "iterative_time_parameterization")
595  {
596  trajectory_processing::IterativeParabolicTimeParameterization time_param;
597  time_param.computeTimeStamps(traj_obj, velocity_scaling_factor, acceleration_scaling_factor);
598  }
