60 double countSamplesPerSecond(const ConstraintSamplerPtr& sampler, const moveit::core::RobotState& reference_state)
61 {
62  if (!sampler)
63  {
64  RCLCPP_ERROR(LOGGER, "No sampler specified for counting samples per second");
65  return 0.0;
66  }
67  moveit::core::RobotState ks(reference_state);
68  unsigned long int valid = 0;
69  unsigned long int total = 0;
70  rclcpp::Time end = rclcpp::Clock().now() + rclcpp::Duration::from_seconds(1);
71  do
72  {
73  static const unsigned int N = 10;
74  total += N;
75  for (unsigned int i = 0; i < N; ++i)
76  {
77  if (sampler->sample(ks, 1))
78  valid++;
79  }
80  } while (rclcpp::Clock().now() < end);
81  return (double)valid / (double)total;
82 }

84 void visualizeDistribution(const ConstraintSamplerPtr& sampler, const moveit::core::RobotState& reference_state,
85  const std::string& link_name, unsigned int sample_count,
86  visualization_msgs::msg::MarkerArray& markers)
87 {
88  if (!sampler)
89  {
90  RCLCPP_ERROR(LOGGER, "No sampler specified for visualizing distribution of samples");
91  return;
92  }
93  const moveit::core::LinkModel* lm = reference_state.getLinkModel(link_name);
94  if (!lm)
95  return;
96  moveit::core::RobotState ks(reference_state);
97  std_msgs::msg::ColorRGBA color;
98  color.r = 1.0f;
99  color.g = 0.0f;
100  color.b = 0.0f;
101  color.a = 1.0f;
102  for (unsigned int i = 0; i < sample_count; ++i)
103  {
104  if (!sampler->sample(ks))
105  continue;
106  const Eigen::Vector3d& pos = ks.getGlobalLinkTransform(lm).translation();
107  visualization_msgs::msg::Marker mk;
108  mk.header.stamp = rclcpp::Clock(RCL_ROS_TIME).now();
109  mk.header.frame_id = sampler->getJointModelGroup()->getParentModel().getModelFrame();
110  mk.ns = "constraint_samples";
111  mk.id = i;
112  mk.type = visualization_msgs::msg::Marker::SPHERE;
113  mk.action = visualization_msgs::msg::Marker::ADD;
114  mk.pose.position.x = pos.x();
115  mk.pose.position.y = pos.y();
116  mk.pose.position.z = pos.z();
117  mk.pose.orientation.w = 1.0;
118  mk.scale.x = mk.scale.y = mk.scale.z = 0.035;
119  mk.color = color;
120  mk.lifetime = rclcpp::Duration::from_seconds(30);
121  markers.markers.push_back(mk);
122  }
123 }
