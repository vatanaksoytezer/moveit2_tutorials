108 TEST(PlanningScene, rememberMetadataWhenAttached)
109 {
110  moveit::core::RobotModelPtr robot_model(moveit::core::RobotModelBuilder("empty_robot", "base_link").build());
111  planning_scene::PlanningScene scene(robot_model);
112 
113  // prepare planning scene message to add a colored object
114  moveit_msgs::msg::PlanningScene scene_msg;
115  scene_msg.robot_model_name = robot_model->getName();
116  scene_msg.is_diff = true;
117  scene_msg.robot_state.is_diff = true;
118 
119  moveit_msgs::msg::CollisionObject co;
120  co.header.frame_id = robot_model->getModelFrame();
121  co.id = "blue_sphere";
122  co.operation = moveit_msgs::msg::CollisionObject::ADD;
123  co.pose.orientation.w = 1.0;
124  makeSphere(co);
125 
126  // meta-data 1: object type
127  co.type.key = "blue_sphere_type";
128  co.type.db = "{'type':'CustomDB'}";
129  scene_msg.world.collision_objects.push_back(co);
130 
131  // meta-data 2: object color
132  moveit_msgs::msg::ObjectColor color;
133  color.id = co.id;
134  color.color.b = 1.0;
135  color.color.a = 1.0;
136  scene_msg.object_colors.push_back(color);
137 
138  EXPECT_FALSE(scene.hasObjectColor(co.id)) << "scene knows color before adding it(?)";
139  EXPECT_FALSE(scene.hasObjectType(co.id)) << "scene knows type before adding it(?)";
140 
141  // add object to scene
142  scene.usePlanningSceneMsg(scene_msg);
143 
144  EXPECT_TRUE(scene.hasObjectColor(co.id)) << "scene failed to add object color";
145  EXPECT_EQ(scene.getObjectColor(co.id), color.color) << "scene added wrong object color";
146  EXPECT_TRUE(scene.hasObjectType(co.id)) << "scene failed to add object type";
147  EXPECT_EQ(scene.getObjectType(co.id), co.type) << "scene added wrong object type";
148 
149  // attach object
150  moveit_msgs::msg::AttachedCollisionObject aco;
151  aco.object.operation = moveit_msgs::msg::CollisionObject::ADD;
152  aco.object.id = co.id;
153  aco.link_name = robot_model->getModelFrame();
154  scene.processAttachedCollisionObjectMsg(aco);
155 
156  EXPECT_EQ(scene.getObjectColor(co.id), color.color) << "scene forgot object color after it got attached";
157  EXPECT_EQ(scene.getObjectType(co.id), co.type) << "scene forgot object type after it got attached";
158 
159  // trying to remove object from the scene while it is attached is expected to fail
160  co.operation = moveit_msgs::msg::CollisionObject::REMOVE;
161  EXPECT_FALSE(scene.processCollisionObjectMsg(co))
162  << "scene removed attached object from collision world (although it's not there)";
163 
164  // detach again right away
165  aco.object.operation = moveit_msgs::msg::CollisionObject::REMOVE;
166  scene.processAttachedCollisionObjectMsg(aco);
167 
168  EXPECT_EQ(scene.getObjectColor(co.id), color.color) << "scene forgot specified color after attach/detach";
169  EXPECT_EQ(scene.getObjectType(co.id), co.type) << "scene forgot specified type after attach/detach";
170 }
